91class Linker(IdGeneratorMixIn, utils.LocalsVisitor):
92 """Walk on the project tree and resolve relationships.
93
94 According to options the following attributes may be
95 added to visited nodes:
96
97 * uid,
98 a unique identifier for the node (on astroid.Project, astroid.Module,
99 astroid.Class and astroid.locals_type). Only if the linker
100 has been instantiated with tag=True parameter (False by default).
101
102 * Function
103 a mapping from locals names to their bounded value, which may be a
104 constant like a string or an integer, or an astroid node
105 (on astroid.Module, astroid.Class and astroid.Function).
106
107 * instance_attrs_type
108 as locals_type but for klass member attributes (only on astroid.Class)
109
110 * associations_type
111 as instance_attrs_type but for association relationships
112
113 * aggregations_type
114 as instance_attrs_type but for aggregations relationships
115 """
116
117 def __init__(self, project: Project, tag: bool = False) -> None:
118 IdGeneratorMixIn.__init__(self)
119 utils.LocalsVisitor.__init__(self)
120 # tag nodes or not
121 self.tag = tag
122 # visited project
123 self.project = project
124 self.associations_handler = AggregationsHandler()
125 self.associations_handler.set_next(OtherAssociationsHandler())
126
127 def visit_project(self, node: Project) -> None:
128 """Visit a pyreverse.utils.Project node.
129
130 * optionally tag the node with a unique id
131 """
132 if self.tag:
133 node.uid = self.generate_id()
134 for module in node.modules:
135 self.visit(module)
136
137 def visit_module(self, node: nodes.Module) -> None:
138 """Visit an astroid.Module node.
139
140 * set the locals_type mapping
141 * set the depends mapping
142 * optionally tag the node with a unique id
143 """
144 if hasattr(node, "locals_type"):
145 return
146 node.locals_type = collections.defaultdict(list)
147 node.depends = []
148 node.type_depends = []
149 if self.tag:
150 node.uid = self.generate_id()
151
152 def visit_classdef(self, node: nodes.ClassDef) -> None:
153 """Visit an astroid.Class node.
154
155 * set the locals_type and instance_attrs_type mappings
156 * optionally tag the node with a unique id
157 """
158 if hasattr(node, "locals_type"):
159 return
160 node.locals_type = collections.defaultdict(list)
161 if self.tag:
162 node.uid = self.generate_id()
163 # resolve ancestors
164 for baseobj in node.ancestors(recurs=False):
165 specializations = getattr(baseobj, "specializations", [])
166 specializations.append(node)
167 baseobj.specializations = specializations
168 # resolve instance attributes
169 node.instance_attrs_type = collections.defaultdict(list)
170 node.aggregations_type = collections.defaultdict(list)
171 node.associations_type = collections.defaultdict(list)
172 for assignattrs in tuple(node.instance_attrs.values()):
173 for assignattr in assignattrs:
174 if not isinstance(assignattr, nodes.Unknown):
175 self.associations_handler.handle(assignattr, node)
176 self.handle_assignattr_type(assignattr, node)
177
178 def visit_functiondef(self, node: nodes.FunctionDef) -> None:
179 """Visit an astroid.Function node.
180
181 * set the locals_type mapping
182 * optionally tag the node with a unique id
183 """
184 if hasattr(node, "locals_type"):
185 return
186 node.locals_type = collections.defaultdict(list)
187 if self.tag:
188 node.uid = self.generate_id()
189
190 def visit_assignname(self, node: nodes.AssignName) -> None:
191 """Visit an astroid.AssignName node.
192
193 handle locals_type
194 """
195 # avoid double parsing done by different Linkers.visit
196 # running over the same project:
197 if hasattr(node, "_handled"):
198 return
199 node._handled = True
200 if node.name in node.frame():
201 frame = node.frame()
202 else:
203 # the name has been defined as 'global' in the frame and belongs
204 # there.
205 frame = node.root()
206 if not hasattr(frame, "locals_type"):
207 # If the frame doesn't have a locals_type yet,
208 # it means it wasn't yet visited. Visit it now
209 # to add what's missing from it.
210 if isinstance(frame, nodes.ClassDef):
211 self.visit_classdef(frame)
212 elif isinstance(frame, nodes.FunctionDef):
213 self.visit_functiondef(frame)
214 else:
215 self.visit_module(frame)
216
217 current = frame.locals_type[node.name]
218 frame.locals_type[node.name] = list(set(current) | utils.infer_node(node))
219
220 @staticmethod
221 def handle_assignattr_type(node: nodes.AssignAttr, parent: nodes.ClassDef) -> None:
222 """Handle an astroid.assignattr node.
223
224 handle instance_attrs_type
225 """
226 current = set(parent.instance_attrs_type[node.attrname])
227 parent.instance_attrs_type[node.attrname] = list(
228 current | utils.infer_node(node)
229 )
230
231 def visit_import(self, node: nodes.Import) -> None:
232 """Visit an astroid.Import node.
233
234 resolve module dependencies
235 """
236 context_file = node.root().file
237 for name in node.names:
238 relative = astroid.modutils.is_relative(name[0], context_file)
239 self._imported_module(node, name[0], relative)
240
241 def visit_importfrom(self, node: nodes.ImportFrom) -> None:
242 """Visit an astroid.ImportFrom node.
243
244 resolve module dependencies
245 """
246 basename = node.modname
247 context_file = node.root().file
248 if context_file is not None:
249 relative = astroid.modutils.is_relative(basename, context_file)
250 else:
251 relative = False
252 for name in node.names:
253 if name[0] == "*":
254 continue
255 # analyze dependencies
256 fullname = f"{basename}.{name[0]}"
257 if fullname.find(".") > -1:
258 try:
259 fullname = astroid.modutils.get_module_part(fullname, context_file)
260 except ImportError:
261 continue
262 if fullname != basename:
263 self._imported_module(node, fullname, relative)
264
265 def compute_module(self, context_name: str, mod_path: str) -> bool:
266 """Should the module be added to dependencies ?"""
267 package_dir = os.path.dirname(self.project.path)
268 if context_name == mod_path:
269 return False
270 # astroid does return a boolean but is not typed correctly yet
271
272 return astroid.modutils.module_in_path(mod_path, (package_dir,)) # type: ignore[no-any-return]
273
274 def _imported_module(
275 self, node: nodes.Import | nodes.ImportFrom, mod_path: str, relative: bool
276 ) -> None:
277 """Notify an imported module, used to analyze dependencies."""
278 module = node.root()
279 context_name = module.name
280 if relative:
281 mod_path = f"{'.'.join(context_name.split('.')[:-1])}.{mod_path}"
282 if self.compute_module(context_name, mod_path):
283 # handle dependencies
284 if not hasattr(module, "depends"):
285 module.depends = []
286 mod_paths = module.depends
287 if mod_path not in mod_paths:
288 mod_paths.append(mod_path)

303class AbstractAssociationHandler(AssociationHandlerInterface):
304 """
305 Chain of Responsibility for handling types of association, useful
306 to expand in the future if we want to add more distinct associations.
307
308 Every link of the chain checks if it's a certain type of association.
309 If no association is found it's set as a generic association in `associations_type`.
310
311 The default chaining behavior is implemented inside the base handler
312 class.
313 """
314
315 _next_handler: AssociationHandlerInterface
316
317 def set_next(
318 self, handler: AssociationHandlerInterface
319 ) -> AssociationHandlerInterface:
320 self._next_handler = handler
321 return handler
322
323 @abstractmethod
324 def handle(self, node: nodes.AssignAttr, parent: nodes.ClassDef) -> None:
325 if self._next_handler:
326 self._next_handler.handle(node, parent)

329class AggregationsHandler(AbstractAssociationHandler):
330 def handle(self, node: nodes.AssignAttr, parent: nodes.ClassDef) -> None:
331 if isinstance(node.parent, (nodes.AnnAssign, nodes.Assign)) and isinstance(
332 node.parent.value, astroid.node_classes.Name
333 ):
334 current = set(parent.aggregations_type[node.attrname])
335 parent.aggregations_type[node.attrname] = list(
336 current | utils.infer_node(node)
337 )
338 else:
339 super().handle(node, parent)
340

342class OtherAssociationsHandler(AbstractAssociationHandler):
343 def handle(self, node: nodes.AssignAttr, parent: nodes.ClassDef) -> None:
344 current = set(parent.associations_type[node.attrname])
345 parent.associations_type[node.attrname] = list(current | utils.infer_node(node))

348def project_from_files(
349 files: list[str],
350 func_wrapper: _WrapperFuncT = _astroid_wrapper,
351 project_name: str = "no name",
352 black_list: tuple[str, ...] = constants.DEFAULT_IGNORE_LIST,
353 verbose: bool = False,
354) -> Project:
355 """Return a Project from a list of files or modules."""
356 # build the project representation
357 astroid_manager = astroid.MANAGER
358 project = Project(project_name)
359 for something in files:
360 if not os.path.exists(something):
361 fpath = astroid.modutils.file_from_modpath(something.split("."))
362 elif os.path.isdir(something):
363 fpath = os.path.join(something, "__init__.py")
364 else:
365 fpath = something
366 ast = func_wrapper(astroid_manager.ast_from_file, fpath, verbose)
367 if ast is None:
368 continue
369 project.path = project.path or ast.file
370 project.add_module(ast)
371 base_name = ast.name
372 # recurse in package except if __init__ was explicitly given
373 if ast.package and something.find("__init__") == -1:
374 # recurse on others packages / modules if this is a package
375 for fpath in astroid.modutils.get_module_files(
376 os.path.dirname(ast.file), black_list
377 ):
378 ast = func_wrapper(astroid_manager.ast_from_file, fpath, verbose)
379 if ast is None or ast.name == base_name:
380 continue
381 project.add_module(ast)
382 return project
