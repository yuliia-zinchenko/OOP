32 def __init__(self, config: argparse.Namespace) -> None:

42 def write(self, diadefs: Iterable[ClassDiagram | PackageDiagram]) -> None:
43 """Write files for <project> according to <diadefs>."""
44 for diagram in diadefs:
45 basename = diagram.title.strip().replace("/", "_").replace(" ", "_")
46 file_name = f"{basename}.{self.config.output_format}"
47 if os.path.exists(self.config.output_directory):
48 file_name = os.path.join(self.config.output_directory, file_name)
49 self.set_printer(file_name, basename)
50 if isinstance(diagram, PackageDiagram):
51 self.write_packages(diagram)
52 else:
53 self.write_classes(diagram)
54 self.save()

56 def write_packages(self, diagram: PackageDiagram) -> None:
57 """Write a package diagram."""
58 module_info: dict[str, dict[str, int]] = {}
59
60 # sorted to get predictable (hence testable) results
61 for module in sorted(diagram.modules(), key=lambda x: x.title):
62 module.fig_id = module.node.qname()
63
64 if self.config.no_standalone and not any(
65 module in (rel.from_object, rel.to_object)
66 for rel in diagram.get_relationships("depends")
67 ):
68 continue
69
70 self.printer.emit_node(
71 module.fig_id,
72 type_=NodeType.PACKAGE,
73 properties=self.get_package_properties(module),
74 )
75
76 module_info[module.fig_id] = {
77 "imports": 0,
78 "imported": 0,
79 }
80
81 # package dependencies
82 for rel in diagram.get_relationships("depends"):
83 from_id = rel.from_object.fig_id
84 to_id = rel.to_object.fig_id
85
86 self.printer.emit_edge(
87 from_id,
88 to_id,
89 type_=EdgeType.USES,
90 )
91
92 module_info[from_id]["imports"] += 1
93 module_info[to_id]["imported"] += 1
94
95 for rel in diagram.get_relationships("type_depends"):
96 from_id = rel.from_object.fig_id
97 to_id = rel.to_object.fig_id
98
99 self.printer.emit_edge(
100 from_id,
101 to_id,
102 type_=EdgeType.TYPE_DEPENDENCY,
103 )
104
105 module_info[from_id]["imports"] += 1
106 module_info[to_id]["imported"] += 1
107
108 print(
109 f"Analysed {len(module_info)} modules with a total "
110 f"of {sum(mod['imports'] for mod in module_info.values())} imports"
111 )

113 def write_classes(self, diagram: ClassDiagram) -> None:
114 """Write a class diagram."""
115 # sorted to get predictable (hence testable) results
116 for obj in sorted(diagram.objects, key=lambda x: x.title):
117 obj.fig_id = obj.node.qname()
118 if self.config.no_standalone and not any(
119 obj in (rel.from_object, rel.to_object)
120 for rel_type in ("specialization", "association", "aggregation")
121 for rel in diagram.get_relationships(rel_type)
122 ):
123 continue
124
125 self.printer.emit_node(
126 obj.fig_id,
127 type_=NodeType.CLASS,
128 properties=self.get_class_properties(obj),
129 )
130 # inheritance links
131 for rel in diagram.get_relationships("specialization"):
132 self.printer.emit_edge(
133 rel.from_object.fig_id,
134 rel.to_object.fig_id,
135 type_=EdgeType.INHERITS,
136 )
137 associations: dict[str, set[str]] = defaultdict(set)
138 # generate associations
139 for rel in diagram.get_relationships("association"):
140 associations[rel.from_object.fig_id].add(rel.to_object.fig_id)
141 self.printer.emit_edge(
142 rel.from_object.fig_id,
143 rel.to_object.fig_id,
144 label=rel.name,
145 type_=EdgeType.ASSOCIATION,
146 )
147 # generate aggregations
148 for rel in diagram.get_relationships("aggregation"):
149 if rel.to_object.fig_id in associations[rel.from_object.fig_id]:
150 continue
151 self.printer.emit_edge(
152 rel.from_object.fig_id,
153 rel.to_object.fig_id,
154 label=rel.name,
155 type_=EdgeType.AGGREGATION,
156 )

163 def get_package_properties(self, obj: PackageEntity) -> NodeProperties:
164 """Get label and shape for packages."""
165 return NodeProperties(
166 label=obj.title,
167 color=self.get_shape_color(obj) if self.config.colorized else "black",
168 )

170 def get_class_properties(self, obj: ClassEntity) -> NodeProperties:
171 """Get label and shape for classes."""
172 properties = NodeProperties(
173 label=obj.title,
174 attrs=obj.attrs if not self.config.only_classnames else None,
175 methods=obj.methods if not self.config.only_classnames else None,
176 fontcolor="red" if is_exception(obj.node) else "black",
177 color=self.get_shape_color(obj) if self.config.colorized else "black",
178 )
179 return properties
180
181 def get_shape_color(self, obj: DiagramEntity) -> str:
182 """Get shape color."""
183 qualified_name = obj.node.qname()
184 if modutils.is_stdlib_module(qualified_name.split(".", maxsplit=1)[0]):
185 return "grey"
186 if isinstance(obj.node, nodes.ClassDef):
187 package = qualified_name.rsplit(".", maxsplit=2)[0]
188 elif obj.node.package:
189 package = qualified_name
190 else:
191 package = qualified_name.rsplit(".", maxsplit=1)[0]
192 base_name = ".".join(package.split(".", self.depth)[: self.depth])
193 if base_name not in self.used_colors:
194 self.used_colors[base_name] = next(self.available_colors)
195 return self.used_colors[base_name]
