81def _signature_from_call(call: nodes.Call) -> _CallSignature:
82 kws = {}
83 args = []
84 starred_kws = []
85 starred_args = []
86 for keyword in call.keywords or []:
87 arg, value = keyword.arg, keyword.value
88 if arg is None and isinstance(value, nodes.Name):
89 # Starred node, and we are interested only in names,
90 # otherwise some transformation might occur for the parameter.
91 starred_kws.append(value.name)
92 elif isinstance(value, nodes.Name):
93 kws[arg] = value.name
94 else:
95 kws[arg] = None
96
97 for arg in call.args:
98 if isinstance(arg, nodes.Starred) and isinstance(arg.value, nodes.Name):
99 # Positional variadic and a name, otherwise some transformation
100 # might have occurred.
101 starred_args.append(arg.value.name)
102 elif isinstance(arg, nodes.Name):
103 args.append(arg.name)
104 else:
105 args.append(None)
106
107 return _CallSignature(args, kws, starred_args, starred_kws)

110def _signature_from_arguments(arguments: nodes.Arguments) -> _ParameterSignature:

122def _definition_equivalent_to_call(
123 definition: _ParameterSignature, call: _CallSignature
124) -> bool:
125 """Check if a definition signature is equivalent to a call."""
126 if definition.kwargs:
127 if definition.kwargs not in call.starred_kws:
128 return False
129 elif call.starred_kws:
130 return False
131 if definition.varargs:
132 if definition.varargs not in call.starred_args:
133 return False
134 elif call.starred_args:
135 return False
136 if any(kw not in call.kws for kw in definition.kwonlyargs):
137 return False
138 if definition.args != call.args:
139 return False
140
141 # No extra kwargs in call.
142 return all(kw in call.args or kw in definition.kwonlyargs for kw in call.kws)

145def _is_trivial_super_delegation(function: nodes.FunctionDef) -> bool:
146 """Check whether a function definition is a method consisting only of a
147 call to the same function on the superclass.
148 """
149 if (
150 not function.is_method()
151 # Adding decorators to a function changes behavior and
152 # constitutes a non-trivial change.
153 or function.decorators
154 ):
155 return False
156
157 body = function.body
158 if len(body) != 1:
159 # Multiple statements, which means this overridden method
160 # could do multiple things we are not aware of.
161 return False
162
163 statement = body[0]
164 if not isinstance(statement, (nodes.Expr, nodes.Return)):
165 # Doing something else than what we are interested in.
166 return False
167
168 call = statement.value
169 if (
170 not isinstance(call, nodes.Call)
171 # Not a super() attribute access.
172 or not isinstance(call.func, nodes.Attribute)
173 ):
174 return False
175
176 # Anything other than a super call is non-trivial.
177 super_call = safe_infer(call.func.expr)
178 if not isinstance(super_call, astroid.objects.Super):
179 return False
180
181 # The name should be the same.
182 if call.func.attrname != function.name:
183 return False
184
185 # Should be a super call with the MRO pointer being the
186 # current class and the type being the current instance.
187 current_scope = function.parent.scope()
188 if (
189 super_call.mro_pointer != current_scope
190 or not isinstance(super_call.type, astroid.Instance)
191 or super_call.type.name != current_scope.name
192 ):
193 return False
194
195 return True

215def _has_different_parameters_default_value(
216 original: nodes.Arguments, overridden: nodes.Arguments
217) -> bool:
218 """Check if original and overridden methods arguments have different default values.
219
220 Return True if one of the overridden arguments has a default
221 value different from the default value of the original argument
222 If one of the method doesn't have argument (.args is None)
223 return False
224 """
225 if original.args is None or overridden.args is None:
226 return False
227
228 for param in chain(original.args, original.kwonlyargs):
229 try:
230 original_default = original.default_value(param.name)
231 except astroid.exceptions.NoDefault:
232 original_default = _DEFAULT_MISSING
233 try:
234 overridden_default = overridden.default_value(param.name)
235 if original_default is _DEFAULT_MISSING:
236 # Only the original has a default.
237 return True
238 except astroid.exceptions.NoDefault:
239 if original_default is _DEFAULT_MISSING:
240 # Both have a default, no difference
241 continue
242 # Only the override has a default.
243 return True
244
245 original_type = type(original_default)
246 if not isinstance(overridden_default, original_type):
247 # Two args with same name but different types
248 return True
249 is_same_fn: Callable[[Any, Any], bool] | None = ASTROID_TYPE_COMPARATORS.get(
250 original_type
251 )
252 if is_same_fn is None:
253 # If the default value comparison is unhandled, assume the value is different
254 return True
255 if not is_same_fn(original_default, overridden_default):
256 # Two args with same type but different values
257 return True
258 return False

261def _has_different_parameters(
262 original: list[nodes.AssignName],
263 overridden: list[nodes.AssignName],
264 dummy_parameter_regex: Pattern[str],
265) -> list[str]:
266 result: list[str] = []
267 zipped = zip_longest(original, overridden)
268 for original_param, overridden_param in zipped:
269 if not overridden_param:
270 return ["Number of parameters "]
271
272 if not original_param:
273 try:
274 overridden_param.parent.default_value(overridden_param.name)
275 continue
276 except astroid.NoDefault:
277 return ["Number of parameters "]
278
279 # check for the arguments' name
280 names = [param.name for param in (original_param, overridden_param)]
281 if any(dummy_parameter_regex.match(name) for name in names):
282 continue
283 if original_param.name != overridden_param.name:
284 result.append(
285 f"Parameter '{original_param.name}' has been renamed "
286 f"to '{overridden_param.name}' in"
287 )
288
289 return result

292def _has_different_keyword_only_parameters(
293 original: list[nodes.AssignName],
294 overridden: list[nodes.AssignName],
295) -> list[str]:
296 """Determine if the two methods have different keyword only parameters."""
297 original_names = [i.name for i in original]
298 overridden_names = [i.name for i in overridden]
299
300 if any(name not in overridden_names for name in original_names):
301 return ["Number of parameters "]
302
303 for name in overridden_names:
304 if name in original_names:
305 continue
306
307 try:
308 overridden[0].parent.default_value(name)
309 except astroid.NoDefault:
310 return ["Number of parameters "]
311
312 return []

315def _different_parameters(
316 original: nodes.FunctionDef,
317 overridden: nodes.FunctionDef,
318 dummy_parameter_regex: Pattern[str],
319) -> list[str]:
320 """Determine if the two methods have different parameters.
321
322 They are considered to have different parameters if:
323
324 * they have different positional parameters, including different names
325
326 * one of the methods is having variadics, while the other is not
327
328 * they have different keyword only parameters.
329 """
330 output_messages = []
331 original_parameters = _positional_parameters(original)
332 overridden_parameters = _positional_parameters(overridden)
333
334 # Copy kwonlyargs list so that we don't affect later function linting
335 original_kwonlyargs = original.args.kwonlyargs
336
337 # Allow positional/keyword variadic in overridden to match against any
338 # positional/keyword argument in original.
339 # Keep any arguments that are found separately in overridden to satisfy
340 # later tests
341 if overridden.args.vararg:
342 overridden_names = [v.name for v in overridden_parameters]
343 original_parameters = [
344 v for v in original_parameters if v.name in overridden_names
345 ]
346
347 if overridden.args.kwarg:
348 overridden_names = [v.name for v in overridden.args.kwonlyargs]
349 original_kwonlyargs = [
350 v for v in original.args.kwonlyargs if v.name in overridden_names
351 ]
352
353 different_positional = _has_different_parameters(
354 original_parameters, overridden_parameters, dummy_parameter_regex
355 )
356 different_kwonly = _has_different_keyword_only_parameters(
357 original_kwonlyargs, overridden.args.kwonlyargs
358 )
359 if different_kwonly and different_positional:
360 if "Number " in different_positional[0] and "Number " in different_kwonly[0]:
361 output_messages.append("Number of parameters ")
362 output_messages += different_positional[1:]
363 output_messages += different_kwonly[1:]
364 else:
365 output_messages += different_positional
366 output_messages += different_kwonly
367 else:
368 if different_positional:
369 output_messages += different_positional
370 if different_kwonly:
371 output_messages += different_kwonly
372
373 # Arguments will only violate LSP if there are variadics in the original
374 # that are then removed from the overridden
375 kwarg_lost = original.args.kwarg and not overridden.args.kwarg
376 vararg_lost = original.args.vararg and not overridden.args.vararg
377
378 if kwarg_lost or vararg_lost:
379 output_messages += ["Variadics removed in"]
380
381 if original.name in PYMETHODS:
382 # Ignore the difference for special methods. If the parameter
383 # numbers are different, then that is going to be caught by
384 # unexpected-special-method-signature.
385 # If the names are different, it doesn't matter, since they can't
386 # be used as keyword arguments anyway.
387 output_messages.clear()
388
389 return output_messages

397 attributes = cls.getattr(attr)

415def _called_in_methods(
416 func: LocalsDictNodeNG,
417 klass: nodes.ClassDef,
418 methods: Sequence[str],
419) -> bool:
420 """Check if the func was called in any of the given methods,
421 belonging to the *klass*.
422
423 Returns True if so, False otherwise.
424 """
425 if not isinstance(func, nodes.FunctionDef):
426 return False
427 for method in methods:
428 try:
429 inferred = klass.getattr(method)
430 except astroid.NotFoundError:
431 continue
432 for infer_method in inferred:
433 for call in infer_method.nodes_of_class(nodes.Call):
434 try:
435 bound = next(call.func.infer())
436 except (astroid.InferenceError, StopIteration):
437 continue
438 if not isinstance(bound, astroid.BoundMethod):
439 continue
440 func_obj = bound._proxied
441 if isinstance(func_obj, astroid.UnboundMethod):
442 func_obj = func_obj._proxied
443 if func_obj.name == func.name:
444 return True
445 return False

448def _is_attribute_property(name: str, klass: nodes.ClassDef) -> bool:
449 """Check if the given attribute *name* is a property in the given *klass*.
450
451 It will look for `property` calls or for functions
452 with the given name, decorated by `property` or `property`
453 subclasses.
454 Returns ``True`` if the name is a property in the given klass,
455 ``False`` otherwise.
456 """
457 try:
458 attributes = klass.getattr(name)
459 except astroid.NotFoundError:
460 return False
461 property_name = "builtins.property"
462 for attr in attributes:
463 if isinstance(attr, util.UninferableBase):
464 continue
465 try:
466 inferred = next(attr.infer())
467 except astroid.InferenceError:
468 continue
469 if isinstance(inferred, nodes.FunctionDef) and decorated_with_property(
470 inferred
471 ):
472 return True
473 if inferred.pytype() != property_name:
474 continue
475
476 cls = node_frame_class(inferred)
477 if cls == klass.declared_metaclass():
478 continue
479 return True
480 return False

1372 def _check_useless_super_delegation(self, function: nodes.FunctionDef) -> None:
1373 """Check if the given function node is an useless method override.
1374
1375 We consider it *useless* if it uses the super() builtin, but having
1376 nothing additional whatsoever than not implementing the method at all.
1377 If the method uses super() to delegate an operation to the rest of the MRO,
1378 and if the method called is the same as the current one, the arguments
1379 passed to super() are the same as the parameters that were passed to
1380 this method, then the method could be removed altogether, by letting
1381 other implementation to take precedence.
1382 """
1383 if not _is_trivial_super_delegation(function):
1384 return
1385
1386 call: nodes.Call = function.body[0].value
1387
1388 # Classes that override __eq__ should also override
1389 # __hash__, even a trivial override is meaningful
1390 if function.name == "__hash__":
1391 for other_method in function.parent.mymethods():
1392 if other_method.name == "__eq__":
1393 return
1394
1395 # Check values of default args
1396 klass = function.parent.frame()
1397 meth_node = None
1398 for overridden in klass.local_attr_ancestors(function.name):
1399 # get astroid for the searched method
1400 try:
1401 meth_node = overridden[function.name]
1402 except KeyError:
1403 # we have found the method but it's not in the local
1404 # dictionary.
1405 # This may happen with astroid build from living objects
1406 continue
1407 if (
1408 not isinstance(meth_node, nodes.FunctionDef)
1409 # If the method have an ancestor which is not a
1410 # function then it is legitimate to redefine it
1411 or _has_different_parameters_default_value(
1412 meth_node.args, function.args
1413 )
1414 # arguments to builtins such as Exception.__init__() cannot be inspected
1415 or (meth_node.args.args is None and function.argnames() != ["self"])
1416 ):
1417 return
1418 break
1419
1420 # Detect if the parameters are the same as the call's arguments.
1421 params = _signature_from_arguments(function.args)
1422 args = _signature_from_call(call)
1423
1424 if meth_node is not None:
1425 # Detect if the super method uses varargs and the function doesn't or makes some of those explicit
1426 if meth_node.args.vararg and (
1427 not function.args.vararg
1428 or len(function.args.args) > len(meth_node.args.args)
1429 ):
1430 return

1432 def form_annotations(arguments: nodes.Arguments) -> list[str]:
1433 annotations = chain(
1434 (arguments.posonlyargs_annotations or []), arguments.annotations
1435 )

1437

1897 def _check_protected_attribute_access(
1898 self, node: nodes.Attribute | nodes.AssignAttr
1899 ) -> None:
1900 """Given an attribute access node (set or get), check if attribute
1901 access is legitimate.
1902
1903 Call _check_first_attr with node before calling
1904 this method. Valid cases are:
1905 * self._attr in a method or cls._attr in a classmethod. Checked by
1906 _check_first_attr.
1907 * Klass._attr inside "Klass" class.
1908 * Klass2._attr inside "Klass" class when Klass2 is a base class of
1909 Klass.

1911 attrname = node.attrname
1912
1913 if (
1914 not is_attr_protected(attrname)
1915 or attrname in self.linter.config.exclude_protected
1916 ):

1919 # Typing annotations in function definitions can include protected members
1920 if utils.is_node_in_type_annotation_context(node):

1924 # and is listed in `exclude-protected`.

1927 inferred

1998 and self._is_called_inside_special_method(node)

2053 def _check_accessed_members(

2056 """Check that accessed members are defined."""
2057 excs = ("AttributeError", "Exception", "BaseException")

2060 # is it a class attribute ?
2061 node.local_attr(attr)
2062 # yes, stop here
2063 continue
2064 except astroid.NotFoundError:

2066 # is it an instance attribute of a parent class ?
2067 try:
2068 next(node.instance_attr_ancestors(attr))

2083 # triggered by the typecheck checker

2086 # several assignments in the same scope

2089 stmt
