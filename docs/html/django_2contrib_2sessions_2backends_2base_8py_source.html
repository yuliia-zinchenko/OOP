268 def get_expiry_age(self, **kwargs):
269 """Get the number of seconds until the session expires.
270
271 Optionally, this function accepts `modification` and `expiry` keyword
272 arguments specifying the modification and expiry of the session.
273 """
274 try:
275 modification = kwargs["modification"]
276 except KeyError:
277 modification = timezone.now()
278 # Make the difference between "expiry=None passed in kwargs" and
279 # "expiry not passed in kwargs", in order to guarantee not to trigger
280 # self.load() when expiry is provided.
281 try:
282 expiry = kwargs["expiry"]
283 except KeyError:
284 expiry = self.get("_session_expiry")
285
286 if not expiry: # Checks both None and 0 cases
287 return self.get_session_cookie_age()
288 if not isinstance(expiry, (datetime, str)):
289 return expiry
290 if isinstance(expiry, str):
291 expiry = datetime.fromisoformat(expiry)
292 delta = expiry - modification
293 return delta.days * 86400 + delta.seconds
294

354 def set_expiry(self, value):
355 """
356 Set a custom expiration for the session. ``value`` can be an integer,
357 a Python ``datetime`` or ``timedelta`` object or ``None``.
358
359 If ``value`` is an integer, the session will expire after that many
360 seconds of inactivity. If set to ``0`` then the session will expire on
361 browser close.
362
363 If ``value`` is a ``datetime`` or ``timedelta`` object, the session
364 will expire at that specific future time.
365
366 If ``value`` is ``None``, the session uses the global session expiry
367 policy.
368 """
369 if value is None:
370 # Remove any custom expiration for this session.
371 try:
372 del self["_session_expiry"]
373 except KeyError:
374 pass
375 return
376 if isinstance(value, timedelta):
377 value = timezone.now() + value
378 if isinstance(value, datetime):
379 value = value.isoformat()
380 self["_session_expiry"] = value
381
382 async def aset_expiry(self, value):
383 if value is None:
384 # Remove any custom expiration for this session.
385 try:
386 await self.apop("_session_expiry")
387 except KeyError:
388 pass
389 return
390 if isinstance(value, timedelta):
391 value = timezone.now() + value
392 if isinstance(value, datetime):
393 value = value.isoformat()
394 await self.aset("_session_expiry", value)
