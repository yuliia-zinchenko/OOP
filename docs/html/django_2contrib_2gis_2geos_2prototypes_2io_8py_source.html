157 def read(self, wkt):

163class _WKBReader(IOBase):
164 _constructor = wkb_reader_create
165 ptr_type = WKB_READ_PTR
166 destructor = wkb_reader_destroy
167
168 def read(self, wkb):
169 "Return a _pointer_ to C GEOS Geometry object from the given WKB."
170 if isinstance(wkb, memoryview):
171 wkb_s = bytes(wkb)
172 return wkb_reader_read(self.ptr, wkb_s, len(wkb_s))
173 elif isinstance(wkb, bytes):
174 return wkb_reader_read_hex(self.ptr, wkb, len(wkb))
175 elif isinstance(wkb, str):
176 wkb_s = wkb.encode()
177 return wkb_reader_read_hex(self.ptr, wkb_s, len(wkb_s))
178 else:
179 raise TypeError

257 def _handle_empty_point(self, geom):
258 from django.contrib.gis.geos import Point
259
260 if isinstance(geom, Point) and geom.empty:
261 if self.srid:
262 # PostGIS uses POINT(NaN NaN) for WKB representation of empty
263 # points. Use it for EWKB as it's a PostGIS specific format.
264 # https://trac.osgeo.org/postgis/ticket/3181
265 geom = Point(float("NaN"), float("NaN"), srid=geom.srid)
266 else:
267 raise ValueError("Empty point is not representable in WKB.")
268 return geom

270 def write(self, geom):
271 "Return the WKB representation of the given geometry."
272 geom = self._handle_empty_point(geom)
273 wkb = wkb_writer_write(self.ptr, geom.ptr, byref(c_size_t()))
274 return memoryview(wkb)

278 geom = self._handle_empty_point(geom)
