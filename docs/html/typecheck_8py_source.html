106def _is_owner_ignored(
107 owner: SuccessfulInferenceResult,
108 attrname: str | None,
109 ignored_classes: Iterable[str],
110 ignored_modules: Iterable[str],
111) -> bool:
112 """Check if the given owner should be ignored.
113
114 This will verify if the owner's module is in *ignored_modules*
115 or the owner's module fully qualified name is in *ignored_modules*
116 or if the *ignored_modules* contains a pattern which catches
117 the fully qualified name of the module.
118
119 Also, similar checks are done for the owner itself, if its name
120 matches any name from the *ignored_classes* or if its qualified
121 name can be found in *ignored_classes*.
122 """
123 if is_module_ignored(owner.root().qname(), ignored_modules):
124 return True
125
126 # Match against ignored classes.
127 ignored_classes = set(ignored_classes)
128 qname = owner.qname() if hasattr(owner, "qname") else ""
129 return any(ignore in (attrname, qname) for ignore in ignored_classes)

153def _string_distance(seq1: str, seq2: str) -> int:
154 seq2_length = len(seq2)
155
156 row = [*list(range(1, seq2_length + 1)), 0]
157 for seq1_index, seq1_char in enumerate(seq1):
158 last_row = row
159 row = [0] * seq2_length + [seq1_index + 1]
160
161 for seq2_index, seq2_char in enumerate(seq2):
162 row[seq2_index] = min(
163 last_row[seq2_index] + 1,
164 row[seq2_index - 1] + 1,
165 last_row[seq2_index - 1] + (seq1_char != seq2_char),
166 )
167
168 return row[seq2_length - 1]

171def _similar_names(
172 owner: SuccessfulInferenceResult,
173 attrname: str | None,
174 distance_threshold: int,
175 max_choices: int,
176) -> list[str]:
177 """Given an owner and a name, try to find similar names.
178
179 The similar names are searched given a distance metric and only
180 a given number of choices will be returned.
181 """
182 possible_names: list[tuple[str, int]] = []
183 names = _node_names(owner)
184
185 for name in names:
186 if name == attrname:
187 continue
188
189 distance = _string_distance(attrname or "", name)
190 if distance <= distance_threshold:
191 possible_names.append((name, distance))
192
193 # Now get back the values with a minimum, up to the given
194 # limit or choices.
195 picked = [
196 name
197 for (name, _) in heapq.nsmallest(
198 max_choices, possible_names, key=operator.itemgetter(1)
199 )
200 ]
201 return sorted(picked)

427def _emit_no_member(
428 node: nodes.Attribute | nodes.AssignAttr | nodes.DelAttr,
429 owner: InferenceResult,
430 owner_name: str | None,
431 mixin_class_rgx: Pattern[str],
432 ignored_mixins: bool = True,
433 ignored_none: bool = True,
434) -> bool:
435 """Try to see if no-member should be emitted for the given owner.
436
437 The following cases are ignored:
438
439 * the owner is a function and it has decorators.
440 * the owner is an instance and it has __getattr__, __getattribute__ implemented
441 * the module is explicitly ignored from no-member checks
442 * the owner is a class and the name can be found in its metaclass.
443 * The access node is protected by an except handler, which handles
444 AttributeError, Exception or bare except.
445 * The node is guarded behind and `IF` or `IFExp` node
446 """
447 # pylint: disable = too-many-return-statements, too-many-branches
448 if node_ignores_exception(node, AttributeError):
449 return False
450 if ignored_none and isinstance(owner, nodes.Const) and owner.value is None:
451 return False
452 if is_super(owner) or getattr(owner, "type", None) == "metaclass":
453 return False
454 if owner_name and ignored_mixins and mixin_class_rgx.match(owner_name):
455 return False
456 if isinstance(owner, nodes.FunctionDef) and (
457 owner.decorators or owner.is_abstract()
458 ):
459 return False
460 if isinstance(owner, (astroid.Instance, nodes.ClassDef)):
461 # Issue #2565: Don't ignore enums, as they have a `__getattr__` but it's not
462 # invoked at this point.
463 try:
464 metaclass = owner.metaclass()
465 except astroid.MroError:
466 pass
467 else:
468 # Renamed in Python 3.10 to `EnumType`
469 if metaclass and metaclass.qname() in {"enum.EnumMeta", "enum.EnumType"}:
470 return not _enum_has_attribute(owner, node)
471 if owner.has_dynamic_getattr():
472 return False
473 if not has_known_bases(owner):
474 return False
475
476 # Exclude typed annotations, since these might actually exist
477 # at some point during the runtime of the program.
478 if utils.is_attribute_typed_annotation(owner, node.attrname):
479 return False
480 if isinstance(owner, astroid.objects.Super):
481 # Verify if we are dealing with an invalid Super object.
482 # If it is invalid, then there's no point in checking that
483 # it has the required attribute. Also, don't fail if the
484 # MRO is invalid.
485 try:
486 owner.super_mro()
487 except (astroid.MroError, astroid.SuperError):
488 return False
489 if not all(has_known_bases(base) for base in owner.type.mro()):
490 return False
491 if isinstance(owner, nodes.Module):
492 try:
493 owner.getattr("__getattr__")
494 return False
495 except astroid.NotFoundError:
496 pass
497 if owner_name and node.attrname.startswith("_" + owner_name):
498 # Test if an attribute has been mangled ('private' attribute)
499 unmangled_name = node.attrname.split("_" + owner_name)[-1]
500 try:
501 if owner.getattr(unmangled_name, context=None) is not None:
502 return False
503 except astroid.NotFoundError:
504 return True
505
506 # Don't emit no-member if guarded behind `IF` or `IFExp`
507 # * Walk up recursively until if statement is found.
508 # * Check if condition can be inferred as `Const`,
509 # would evaluate as `False`,
510 # and whether the node is part of the `body`.
511 # * Continue checking until scope of node is reached.
512 scope: nodes.NodeNG = node.scope()
513 node_origin: nodes.NodeNG = node
514 parent: nodes.NodeNG = node.parent
515 while parent != scope:
516 if isinstance(parent, (nodes.If, nodes.IfExp)):
517 inferred = safe_infer(parent.test)
518 if ( # pylint: disable=too-many-boolean-expressions
519 isinstance(inferred, nodes.Const)
520 and inferred.bool_value() is False
521 and (
522 isinstance(parent, nodes.If)
523 and node_origin in parent.body
524 or isinstance(parent, nodes.IfExp)
525 and node_origin == parent.body
526 )
527 ):
528 return False
529 node_origin, parent = parent, parent.parent
530
531 return True

534def _get_all_attribute_assignments(
535 node: nodes.FunctionDef, name: str | None = None
536) -> set[str]:
537 attributes: set[str] = set()
538 for child in node.nodes_of_class((nodes.Assign, nodes.AnnAssign)):
539 targets = []
540 if isinstance(child, nodes.Assign):
541 targets = child.targets
542 elif isinstance(child, nodes.AnnAssign):
543 targets = [child.target]
544 for assign_target in targets:
545 if isinstance(assign_target, nodes.Tuple):
546 targets.extend(assign_target.elts)
547 continue
548 if (
549 isinstance(assign_target, nodes.AssignAttr)
550 and isinstance(assign_target.expr, nodes.Name)
551 and (name is None or assign_target.expr.name == name)
552 ):
553 attributes.add(assign_target.attrname)
554 return attributes

557def _enum_has_attribute(
558 owner: astroid.Instance | nodes.ClassDef, node: nodes.Attribute
559) -> bool:
560 if isinstance(owner, astroid.Instance):
561 enum_def = next(
562 (b.parent for b in owner.bases if isinstance(b.parent, nodes.ClassDef)),
563 None,
564 )
565
566 if enum_def is None:
567 # We don't inherit from anything, so try to find the parent
568 # class definition and roll with that
569 enum_def = node
570 while enum_def is not None and not isinstance(enum_def, nodes.ClassDef):
571 enum_def = enum_def.parent
572
573 # If this blows, something is clearly wrong
574 assert enum_def is not None, "enum_def unexpectedly None"
575 else:
576 enum_def = owner
577
578 # Find __new__ and __init__
579 dunder_new = next((m for m in enum_def.methods() if m.name == "__new__"), None)
580 dunder_init = next((m for m in enum_def.methods() if m.name == "__init__"), None)
581
582 enum_attributes: set[str] = set()
583
584 # Find attributes defined in __new__
585 if dunder_new:
586 # Get the object returned in __new__
587 returned_obj_name = next(
588 (c.value for c in dunder_new.get_children() if isinstance(c, nodes.Return)),
589 None,
590 )
591 if isinstance(returned_obj_name, nodes.Name):
592 # Find all attribute assignments to the returned object
593 enum_attributes |= _get_all_attribute_assignments(
594 dunder_new, returned_obj_name.name
595 )
596
597 # Find attributes defined in __init__
598 if dunder_init and dunder_init.body and dunder_init.args:
599 # Grab the name referring to `self` from the function def
600 enum_attributes |= _get_all_attribute_assignments(
601 dunder_init, dunder_init.args.arguments[0].name
602 )
603
604 return node.attrname in enum_attributes

607def _determine_callable(
608 callable_obj: nodes.NodeNG,
609) -> tuple[CallableObjects, int, str]:
610 # TODO: The typing of the second return variable is actually Literal[0,1]
611 # We need typing on astroid.NodeNG.implicit_parameters for this
612 # TODO: The typing of the third return variable can be narrowed to a Literal
613 # We need typing on astroid.NodeNG.type for this
614
615 # Ordering is important, since BoundMethod is a subclass of UnboundMethod,
616 # and Function inherits Lambda.
617 parameters = 0
618 if hasattr(callable_obj, "implicit_parameters"):
619 parameters = callable_obj.implicit_parameters()
620 if isinstance(callable_obj, bases.BoundMethod):
621 # Bound methods have an extra implicit 'self' argument.
622 return callable_obj, parameters, callable_obj.type
623 if isinstance(callable_obj, bases.UnboundMethod):
624 return callable_obj, parameters, "unbound method"
625 if isinstance(callable_obj, nodes.FunctionDef):
626 return callable_obj, parameters, callable_obj.type
627 if isinstance(callable_obj, nodes.Lambda):
628 return callable_obj, parameters, "lambda"
629 if isinstance(callable_obj, nodes.ClassDef):
630 # Class instantiation, lookup __new__ instead.
631 # If we only find object.__new__, we can safely check __init__
632 # instead. If __new__ belongs to builtins, then we look
633 # again for __init__ in the locals, since we won't have
634 # argument information for the builtin __new__ function.
635 try:
636 # Use the last definition of __new__.
637 new = callable_obj.local_attr("__new__")[-1]
638 except astroid.NotFoundError:
639 new = None
640
641 from_object = new and new.parent.scope().name == "object"
642 from_builtins = new and new.root().name in sys.builtin_module_names
643
644 if not new or from_object or from_builtins:
645 try:
646 # Use the last definition of __init__.
647 callable_obj = callable_obj.local_attr("__init__")[-1]
648 except astroid.NotFoundError as e:
649 raise ValueError from e
650 else:
651 callable_obj = new
652
653 if not isinstance(callable_obj, nodes.FunctionDef):
654 raise ValueError
655 # both have an extra implicit 'cls'/'self' argument.
656 return callable_obj, parameters, "constructor"
657
658 raise ValueError

697def _no_context_variadic(
698 node: nodes.Call,
699 variadic_name: str | None,
700 variadic_type: nodes.Keyword | nodes.Starred,
701 variadics: list[nodes.Keyword | nodes.Starred],
702) -> bool:
703 """Verify if the given call node has variadic nodes without context.
704
705 This is a workaround for handling cases of nested call functions
706 which don't have the specific call context at hand.
707 Variadic arguments (variable positional arguments and variable
708 keyword arguments) are inferred, inherently wrong, by astroid
709 as a Tuple, respectively a Dict with empty elements.
710 This can lead pylint to believe that a function call receives
711 too few arguments.
712 """
713 scope = node.scope()
714 is_in_lambda_scope = not isinstance(scope, nodes.FunctionDef) and isinstance(
715 scope, nodes.Lambda
716 )

758def _infer_from_metaclass_constructor(
759 cls: nodes.ClassDef, func: nodes.FunctionDef
760) -> InferenceResult | None:
761 """Try to infer what the given *func* constructor is building.
762
763 :param astroid.FunctionDef func:
764 A metaclass constructor. Metaclass definitions can be
765 functions, which should accept three arguments, the name of
766 the class, the bases of the class and the attributes.
767 The function could return anything, but usually it should
768 be a proper metaclass.
769 :param astroid.ClassDef cls:
770 The class for which the *func* parameter should generate
771 a metaclass.
772 :returns:
773 The class generated by the function or None,
774 if we couldn't infer it.
775 :rtype: astroid.ClassDef
776 """
777 context = astroid.context.InferenceContext()
778
779 class_bases = nodes.List()
780 class_bases.postinit(elts=cls.bases)
781
782 attrs = nodes.Dict(
783 lineno=0, col_offset=0, parent=None, end_lineno=0, end_col_offset=0
784 )
785 local_names = [(name, values[-1]) for name, values in cls.locals.items()]
786 attrs.postinit(local_names)
787
788 builder_args = nodes.Tuple()
789 builder_args.postinit([cls.name, class_bases, attrs])
790
791 context.callcontext = astroid.context.CallContext(builder_args)
792 try:
793 inferred = next(func.infer_call_result(func, context), None)
794 except astroid.InferenceError:
795 return None
796 return inferred or None

1376 def _check_argument_order(
1377 self,
1378 node: nodes.Call,
1379 call_site: arguments.CallSite,
1380 called: CallableObjects,
1381 called_param_names: list[str | None],
1382 ) -> None:
1383 """Match the supplied argument names against the function parameters.
1384
1385 Warn if some argument names are not in the same order as they are in
1386 the function signature.
1387 """
1388 # Check for called function being an object instance function
1389 # If so, ignore the initial 'self' argument in the signature
1390 try:
1391 is_classdef = isinstance(called.parent, nodes.ClassDef)
1392 if is_classdef and called_param_names[0] == "self":
1393 called_param_names = called_param_names[1:]
1394 except IndexError:
1395 return
1396
1397 try:
1398 # extract argument names, if they have names
1399 calling_parg_names = [p.name for p in call_site.positional_arguments]
1400
1401 # Additionally, get names of keyword arguments to use in a full match
1402 # against parameters
1403 calling_kwarg_names = [
1404 arg.name for arg in call_site.keyword_arguments.values()
1405 ]
1406 except AttributeError:
1407 # the type of arg does not provide a `.name`. In this case we
1408 # stop checking for out-of-order arguments because it is only relevant
1409 # for named variables.
1410 return
1411
1412 # Don't check for ordering if there is an unmatched arg or param
1413 arg_set = set(calling_parg_names) | set(calling_kwarg_names)
1414 param_set = set(called_param_names)
1415 if arg_set != param_set:

1453 # pylint: disable = too-many-branches, too-many-locals, too-many-statements
1454 def visit_call(self, node: nodes.Call) -> None:
1455 """Check that called functions/methods are inferred to callable objects,
1456 and that passed arguments match the parameters in the inferred function.
1457 """
1458 called = safe_infer(node.func, compare_constructors=True)
1459
1460 self._check_not_callable(node, called)
1461
1462 try:
1463 called, implicit_args, callable_name = _determine_callable(called)
1464 except ValueError:
1465 # Any error occurred during determining the function type, most of
1466 # those errors are handled by different warnings.
1467 return
1468
1469 if called.args.args is None:
1470 if called.name == "isinstance":
1471 # Verify whether second argument of isinstance is a valid type
1472 self._check_isinstance_args(node, callable_name)
1473 # Built-in functions have no argument information.
1474 return
1475
1476 if len(called.argnames()) != len(set(called.argnames())):
1477 # Duplicate parameter name (see duplicate-argument). We can't really
1478 # make sense of the function call in this case, so just return.
1479 return
1480
1481 # Build the set of keyword arguments, checking for duplicate keywords,
1482 # and count the positional arguments.
1483 call_site = astroid.arguments.CallSite.from_call(node)
1484
1485 # Warn about duplicated keyword arguments, such as `f=24, **{'f': 24}`
1486 for keyword in call_site.duplicated_keywords:
1487 self.add_message("repeated-keyword", node=node, args=(keyword,))
1488
1489 if call_site.has_invalid_arguments() or call_site.has_invalid_keywords():
1490 # Can't make sense of this.
1491 return
1492
1493 # Has the function signature changed in ways we cannot reliably detect?
1494 if hasattr(called, "decorators") and decorated_with(
1495 called, self.linter.config.signature_mutators
1496 ):
1497 return
1498
1499 num_positional_args = len(call_site.positional_arguments)
1500 keyword_args = list(call_site.keyword_arguments.keys())
1501 overload_function = is_overload_stub(called)
1502
1503 # Determine if we don't have a context for our call and we use variadics.
1504 node_scope = node.scope()
1505 if isinstance(node_scope, (nodes.Lambda, nodes.FunctionDef)):
1506 has_no_context_positional_variadic = _no_context_variadic_positional(
1507 node, node_scope
1508 )
1509 has_no_context_keywords_variadic = _no_context_variadic_keywords(
1510 node, node_scope
1511 )
1512 else:
1513 has_no_context_positional_variadic = has_no_context_keywords_variadic = (
1514 False
1515 )
1516
1517 # These are coming from the functools.partial implementation in astroid
1518 already_filled_positionals = getattr(called, "filled_positionals", 0)
1519 already_filled_keywords = getattr(called, "filled_keywords", {})
1520
1521 keyword_args += list(already_filled_keywords)
1522 num_positional_args += implicit_args + already_filled_positionals
1523
1524 # Decrement `num_positional_args` by 1 when a function call is assigned to a class attribute
1525 # inside the class where the function is defined.
1526 # This avoids emitting `too-many-function-args` since `num_positional_args`
1527 # includes an implicit `self` argument which is not present in `called.args`.
1528 if (
1529 isinstance(node.frame(), nodes.ClassDef)
1530 and isinstance(called, nodes.FunctionDef)
1531 and called in node.frame().body
1532 and num_positional_args > 0
1533 and "builtins.staticmethod" not in called.decoratornames()
1534 ):
1535 num_positional_args -= 1
1536
1537 # Analyze the list of formal parameters.
1538 args = list(itertools.chain(called.args.posonlyargs or (), called.args.args))
1539 num_mandatory_parameters = len(args) - len(called.args.defaults)
1540 parameters: list[tuple[tuple[str | None, nodes.NodeNG | None], bool]] = []
1541 parameter_name_to_index = {}
1542 for i, arg in enumerate(args):
1543 name = arg.name
1544 parameter_name_to_index[name] = i
1545 if i >= num_mandatory_parameters:
1546 defval = called.args.defaults[i - num_mandatory_parameters]
1547 else:
1548 defval = None
1549 parameters.append(((name, defval), False))
1550
1551 kwparams = {}
1552 for i, arg in enumerate(called.args.kwonlyargs):
1553 if isinstance(arg, nodes.Keyword):
1554 name = arg.arg
1555 else:
1556 assert isinstance(arg, nodes.AssignName)
1557 name = arg.name
1558 kwparams[name] = [called.args.kw_defaults[i], False]
1559
1560 self._check_argument_order(
1561 node, call_site, called, [p[0][0] for p in parameters]
1562 )
1563
1564 # 1. Match the positional arguments.
1565 for i in range(num_positional_args):
1566 if i < len(parameters):
1567 parameters[i] = (parameters[i][0], True)
1568 elif called.args.vararg is not None:
1569 # The remaining positional arguments get assigned to the *args
1570 # parameter.
1571 break
1572 elif not overload_function:
1573 # Too many positional arguments.
1574 self.add_message(
1575 "too-many-function-args",
1576 node=node,
1577 args=(callable_name,),
1578 )
1579 break
1580
1581 # 2. Match the keyword arguments.
1582 for keyword in keyword_args:
1583 # Skip if `keyword` is the same name as a positional-only parameter
1584 # and a `**kwargs` parameter exists.
1585 if called.args.kwarg and keyword in [
1586 arg.name for arg in called.args.posonlyargs
1587 ]:
1588 self.add_message(
1589 "kwarg-superseded-by-positional-arg",
1590 node=node,
1591 args=(keyword, f"**{called.args.kwarg}"),
1592 confidence=HIGH,
1593 )
1594 continue
1595 if keyword in parameter_name_to_index:
1596 i = parameter_name_to_index[keyword]
1597 if parameters[i][1]:
1598 # Duplicate definition of function parameter.
1599
1600 # Might be too hard-coded, but this can actually
1601 # happen when using str.format and `self` is passed
1602 # by keyword argument, as in `.format(self=self)`.
1603 # It's perfectly valid to so, so we're just skipping
1604 # it if that's the case.
1605 if not (keyword == "self" and called.qname() in STR_FORMAT):
1606 self.add_message(
1607 "redundant-keyword-arg",
1608 node=node,
1609 args=(keyword, callable_name),
1610 )
1611 else:
1612 parameters[i] = (parameters[i][0], True)
1613 elif keyword in kwparams:
1614 if kwparams[keyword][1]:
1615 # Duplicate definition of function parameter.
1616 self.add_message(
1617 "redundant-keyword-arg",
1618 node=node,
1619 args=(keyword, callable_name),
1620 )
1621 else:
1622 kwparams[keyword][1] = True
1623 elif called.args.kwarg is not None:
1624 # The keyword argument gets assigned to the **kwargs parameter.
1625 pass
1626 elif isinstance(
1627 called, nodes.FunctionDef
1628 ) and self._keyword_argument_is_in_all_decorator_returns(called, keyword):
1629 pass
1630 elif not overload_function:
1631 # Unexpected keyword argument.
1632 self.add_message(
1633 "unexpected-keyword-arg", node=node, args=(keyword, callable_name)
1634 )
1635
1636 # 3. Match the **kwargs, if any.
1637 if node.kwargs:
1638 for i, [(name, _defval), _assigned] in enumerate(parameters):
1639 # Assume that *kwargs provides values for all remaining
1640 # unassigned named parameters.
1641 if name is not None:
1642 parameters[i] = (parameters[i][0], True)
1643 else:
1644 # **kwargs can't assign to tuples.
1645 pass
1646
1647 # Check that any parameters without a default have been assigned
1648 # values.
1649 for [(name, defval), assigned] in parameters:
1650 if (defval is None) and not assigned:
1651 display_name = "<tuple>" if name is None else repr(name)
1652 if not has_no_context_positional_variadic and not overload_function:
1653 self.add_message(
1654 "no-value-for-parameter",
1655 node=node,
1656 args=(display_name, callable_name),
1657 )
1658
1659 for name, val in kwparams.items():
1660 defval, assigned = val
1661 if (
1662 defval is None
1663 and not assigned
1664 and not has_no_context_keywords_variadic
1665 and not overload_function
1666 ):
1667 self.add_message(
1668 "missing-kwoa",
1669 node=node,
1670 args=(name, callable_name),
1671 confidence=INFERENCE,
1672 )

1714 def _check_invalid_sequence_index(self, subscript: nodes.Subscript) -> None:
1715 # Look for index operations where the parent is a sequence type.
1716 # If the types can be determined, only allow indices to be int,
1717 # slice or instances with __index__.
1718 parent_type = safe_infer(subscript.value)
1719 if not isinstance(
1720 parent_type, (nodes.ClassDef, astroid.Instance)
1721 ) or not has_known_bases(parent_type):
1722 return None
1723
1724 # Determine what method on the parent this index will use
1725 # The parent of this node will be a Subscript, and the parent of that
1726 # node determines if the Subscript is a get, set, or delete operation.
1727 if subscript.ctx is astroid.Context.Store:
1728 methodname = "__setitem__"
1729 elif subscript.ctx is astroid.Context.Del:
1730 methodname = "__delitem__"
1731 else:
1732 methodname = "__getitem__"
1733
1734 # Check if this instance's __getitem__, __setitem__, or __delitem__, as
1735 # appropriate to the statement, is implemented in a builtin sequence
1736 # type. This way we catch subclasses of sequence types but skip classes
1737 # that override __getitem__ and which may allow non-integer indices.
1738 try:
1739 methods = astroid.interpreter.dunder_lookup.lookup(parent_type, methodname)
1740 if isinstance(methods, util.UninferableBase):
1741 return None
1742 itemmethod = methods[0]
1743 except (
1744 astroid.AttributeInferenceError,
1745 IndexError,
1746 ):
1747 return None
1748 if (
1749 not isinstance(itemmethod, nodes.FunctionDef)
1750 or itemmethod.root().name != "builtins"
1751 or not itemmethod.parent
1752 or itemmethod.parent.frame().name not in SEQUENCE_TYPES
1753 ):
1754 return None
1755
1756 index_type = safe_infer(subscript.slice)
1757 if index_type is None or isinstance(index_type, util.UninferableBase):
1758 return None
1759 # Constants must be of type int
1760 if isinstance(index_type, nodes.Const):
1761 if isinstance(index_type.value, int):
1762 return None
1763 # Instance values must be int, slice, or have an __index__ method
1764 elif isinstance(index_type, astroid.Instance):
1765 if index_type.pytype() in {"builtins.int", "builtins.slice"}:
1766 return None
1767 try:
1768 index_type.getattr("__index__")
1769 return None
1770 except astroid.NotFoundError:
1771 pass
1772 elif isinstance(index_type, nodes.Slice):
1773 # A slice can be present
1774 # here after inferring the index node, which could
1775 # be a `slice(...)` call for instance.
1776 return self._check_invalid_slice_index(index_type)
1777
1778 # Anything else is an error
1779 self.add_message("invalid-sequence-index", node=subscript)
1780 return None
