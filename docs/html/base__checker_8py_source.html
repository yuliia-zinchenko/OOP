47 def __init__(self, linter: PyLinter) -> None:
48 """Checker instances should have the linter as argument."""
49 if self.name is not None:
50 self.name = self.name.lower()
51 self.linter = linter

54 def __gt__(self, other: Any) -> bool:
55 """Permits sorting checkers for stable doc and tests.
56
57 The main checker is always the first one, then builtin checkers in alphabetical
58 order, then extension checkers in alphabetical order.
59 """
60 if not isinstance(other, BaseChecker):
61 return False
62 if self.name == MAIN_CHECKER_NAME:
63 return False
64 if other.name == MAIN_CHECKER_NAME:
65 return True
66 self_is_builtin = type(self).__module__.startswith("pylint.checkers")
67 if self_is_builtin ^ type(other).__module__.startswith("pylint.checkers"):
68 return not self_is_builtin
69 return self.name > other.name
70
71 def __eq__(self, other: object) -> bool:
72 """Permit to assert Checkers are equal."""
73 if not isinstance(other, BaseChecker):
74 return False
75 return f"{self.name}{self.msgs}" == f"{other.name}{other.msgs}"
76
77 def __hash__(self) -> int:
78 """Make Checker hashable."""
79 return hash(f"{self.name}{self.msgs}")

81 def __repr__(self) -> str:
82 status = "Checker" if self.enabled else "Disabled checker"
83 msgs = "', '".join(self.msgs.keys())

96 def get_full_documentation(
97 self,
98 msgs: dict[str, MessageDefinitionTuple],
99 options: Iterable[tuple[str, OptionDict, Any]],
100 reports: Sequence[tuple[str, str, ReportsCallable]],
101 doc: str | None = None,
102 module: str | None = None,
103 show_options: bool = True,
104 ) -> str:
105 result = ""
106 checker_title = f"{self.name.replace('_', ' ').title()} checker"
107 if module:
108 # Provide anchor to link against
109 result += f".. _{module}:\n\n"
110 result += f"{get_rst_title(checker_title, '~')}\n"
111 if module:
112 result += f"This checker is provided by ``{module}``.\n"
113 result += f"Verbatim name of the checker is ``{self.name}``.\n\n"
114 if doc:
115 # Provide anchor to link against
116 result += get_rst_title(f"{checker_title} Documentation", "^")
117 result += f"{cleandoc(doc)}\n\n"
118 # options might be an empty generator and not be False when cast to boolean
119 options_list = list(options)
120 if options_list:
121 if show_options:
122 result += get_rst_title(f"{checker_title} Options", "^")
123 result += f"{get_rst_section(None, options_list)}\n"
124 else:
125 result += f"See also :ref:`{self.name} checker's options' documentation <{self.name}-options>`\n\n"
126 if msgs:
127 result += get_rst_title(f"{checker_title} Messages", "^")
128 for msgid, msg in sorted(
129 msgs.items(), key=lambda kv: (_MSG_ORDER.index(kv[0][0]), kv[1])
130 ):
131 msg_def = self.create_message_definition_from_tuple(msgid, msg)
132 result += f"{msg_def.format_help(checkerref=False)}\n"
133 result += "\n"
134 if reports:
135 result += get_rst_title(f"{checker_title} Reports", "^")
136 for report in reports:
137 result += f":{report[0]}: {report[1]}\n"
138 result += "\n"
139 result += "\n"
140 return result

157 def check_consistency(self) -> None:
158 """Check the consistency of msgid.
159
160 msg ids for a checker should be a string of len 4, where the two first
161 characters are the checker id and the two last the msg id in this
162 checker.
163
164 :raises InvalidMessageError: If the checker id in the messages are not
165 always the same.
166 """
167 checker_id = None
168 existing_ids = []
169 for message in self.messages:
170 # Id's for shared messages such as the 'deprecated-*' messages
171 # can be inconsistent with their checker id.
172 if message.shared:
173 continue
174 if checker_id is not None and checker_id != message.msgid[1:3]:
175 error_msg = "Inconsistent checker part in message id "
176 error_msg += f"'{message.msgid}' (expected 'x{checker_id}xx' "
177 error_msg += f"because we already had {existing_ids})."
178 raise InvalidMessageError(error_msg)
179 checker_id = message.msgid[1:3]
180 existing_ids.append(message.msgid)

182 def create_message_definition_from_tuple(
183 self, msgid: str, msg_tuple: MessageDefinitionTuple
184 ) -> MessageDefinition:
185 if isinstance(self, (BaseTokenChecker, BaseRawFileChecker)):
186 default_scope = WarningScope.LINE
187 else:
188 default_scope = WarningScope.NODE
189 options: ExtraMessageOptions = {}
190 if len(msg_tuple) == 4:
191 (msg, symbol, descr, msg_options) = msg_tuple
192 options = ExtraMessageOptions(**msg_options)
193 elif len(msg_tuple) == 3:
194 (msg, symbol, descr) = msg_tuple
195 else:
196 error_msg = """Messages should have a msgid, a symbol and a description. Something like this :
197
198"W1234": (
199 "message",
200 "message-symbol",
201 "Message description with detail.",
202 ...
203),
204"""
205 raise InvalidMessageError(error_msg)
206 options.setdefault("scope", default_scope)
207 return MessageDefinition(self, msgid, msg, descr, symbol, **options)
