17class ForLoopSimulator:
18 """
19 Simulates a forloop tag, precisely::
20
21 {% for form in formset.forms %}
22
23 If `{% crispy %}` is rendering a formset with a helper, We inject a `ForLoopSimulator` object
24 in the context as `forloop` so that formset forms can do things like::
25
26 Fieldset("Item {{ forloop.counter }}", [...])
27 HTML("{% if forloop.first %}First form text{% endif %}"
28 """
29
30 def __init__(self, formset):
31 self.len_values = len(formset.forms)
32
33 # Shortcuts for current loop iteration number.
34 self.counter = 1
35 self.counter0 = 0
36 # Reverse counter iteration numbers.
37 self.revcounter = self.len_values
38 self.revcounter0 = self.len_values - 1
39 # Boolean values designating first and last times through loop.
40 self.first = True
41 self.last = 0 == self.len_values - 1
42
43 def iterate(self):
44 """
45 Updates values as if we had iterated over the for
46 """
47 self.counter += 1
48 self.counter0 += 1
49 self.revcounter -= 1
50 self.revcounter0 -= 1
51 self.first = False
52 self.last = self.revcounter0 == self.len_values - 1

55class BasicNode(template.Node):
56 """
57 Basic Node object that we can rely on for Node objects in normal
58 template tags. I created this because most of the tags we'll be using
59 will need both the form object and the helper string. This handles
60 both the form object and parses out the helper string into attributes
61 that templates can easily handle.
62 """
63
64 def __init__(self, form, helper, template_pack=None):
65 self.form = form
66 self.helper = helper
67 self.template_pack = template_pack or get_template_pack()
68
69 def get_render(self, context):
70 """
71 Returns a `Context` object with all the necessary stuff for rendering the form
72
73 :param context: `django.template.Context` variable holding the context for the node
74
75 `self.form` and `self.helper` are resolved into real Python objects resolving them
76 from the `context`. The `actual_form` can be a form or a formset. If it's a formset
77 `is_formset` is set to True. If the helper has a layout we use it, for rendering the
78 form or the formset's forms.
79 """
80 # Nodes are not thread safe in multithreaded environments
81 # https://docs.djangoproject.com/en/dev/howto/custom-template-tags/#thread-safety-considerations
82 if self not in context.render_context:
83 context.render_context[self] = (
84 template.Variable(self.form),
85 template.Variable(self.helper) if self.helper else None,
86 )
87 form, helper = context.render_context[self]
88
89 actual_form = form.resolve(context)
90 if self.helper is not None:
91 helper = helper.resolve(context)
92 else:
93 # If the user names the helper within the form `helper` (standard), we use it
94 # This allows us to have simplified tag syntax: {% crispy form %}
95 helper = FormHelper() if not hasattr(actual_form, "helper") else actual_form.helper
96
97 # use template_pack from helper, if defined
98 try:
99 if helper.template_pack:
100 self.template_pack = helper.template_pack
101 except AttributeError:
102 pass
103
104 self.actual_helper = helper
105
106 # We get the response dictionary
107 is_formset = isinstance(actual_form, BaseFormSet)
108 response_dict = self.get_response_dict(helper, context, is_formset)
109 node_context = context.__copy__()
110 node_context.update({"is_bound": actual_form.is_bound})
111 node_context.update(response_dict)
112 final_context = node_context.__copy__()
113
114 # If we have a helper's layout we use it, for the form or the formset's forms
115 if helper and helper.layout:
116 if not is_formset:
117 actual_form.form_html = helper.render_layout(
118 actual_form, node_context, template_pack=self.template_pack
119 )
120 else:
121 forloop = ForLoopSimulator(actual_form)
122 helper.render_hidden_fields = True
123 for form in actual_form:
124 node_context.update({"forloop": forloop})
125 node_context.update({"formset_form": form})
126 form.form_html = helper.render_layout(form, node_context, template_pack=self.template_pack)
127 forloop.iterate()
128
129 if is_formset:
130 final_context["formset"] = actual_form
131 else:
132 final_context["form"] = actual_form
133
134 return final_context

136 def get_response_dict(self, helper, context, is_formset):
137 """
138 Returns a dictionary with all the parameters necessary to render the form/formset in a template.
139
140 :param context: `django.template.Context` for the node
141 :param is_formset: Boolean value. If set to True, indicates we are working with a formset.
142 """
143 if not isinstance(helper, FormHelper):
144 raise TypeError("helper object provided to {% crispy %} tag must be a crispy.helper.FormHelper object.")
145
146 attrs = helper.get_attributes(template_pack=self.template_pack)
147 form_type = "form"
148 if is_formset:
149 form_type = "formset"
150
151 # We take form/formset parameters from attrs if they are set, otherwise we use defaults
152 response_dict = {
153 "%s_action" % form_type: attrs["attrs"].get("action", ""),
154 "%s_attrs" % form_type: attrs.get("attrs", ""),
155 "%s_class" % form_type: attrs["attrs"].get("class", ""),
156 "%s_id" % form_type: attrs["attrs"].get("id", ""),
157 "%s_method" % form_type: attrs.get("form_method", "post"),
158 "%s_tag" % form_type: attrs.get("form_tag", True),
159 "disable_csrf": attrs.get("disable_csrf", False),
160 "error_text_inline": attrs.get("error_text_inline", True),
161 "field_class": attrs.get("field_class", ""),
162 "field_template": attrs.get("field_template", ""),
163 "flat_attrs": attrs.get("flat_attrs", ""),
164 "form_error_title": attrs.get("form_error_title", None),
165 "form_show_errors": attrs.get("form_show_errors", True),
166 "form_show_labels": attrs.get("form_show_labels", True),
167 "formset_error_title": attrs.get("formset_error_title", None),
168 "help_text_inline": attrs.get("help_text_inline", False),
169 "include_media": attrs.get("include_media", True),
170 "inputs": attrs.get("inputs", []),
171 "is_formset": is_formset,
172 "label_class": attrs.get("label_class", ""),
173 "template_pack": self.template_pack,
174 }
175
176 # Handles custom attributes added to helpers
177 for attribute_name, value in attrs.items():
178 if attribute_name not in response_dict:
179 response_dict[attribute_name] = value
180
181 if "csrf_token" in context:
182 response_dict["csrf_token"] = context["csrf_token"]
183
184 return response_dict

213def do_uni_form(parser, token):
214 """
215 You need to pass in at least the form/formset object, and can also pass in the
216 optional `crispy_forms.helpers.FormHelper` object.
217
218 helper (optional): A `crispy_forms.helper.FormHelper` object.
219
220 Usage::
221
222 {% load crispy_tags %}
223 {% crispy form form.helper %}
224
225 You can also provide the template pack as the third argument::
226
227 {% crispy form form.helper 'bootstrap' %}
228
229 If the `FormHelper` attribute is named `helper` you can simply do::
230
231 {% crispy form %}
232 {% crispy form 'bootstrap' %}
233 """
234 tokens = token.split_contents()
235 form = tokens.pop(1)
236
237 helper = None
238 template_pack = "'%s'" % get_template_pack()
239
240 # {% crispy form helper %}
241 try:
242 helper = tokens.pop(1)
243 except IndexError:
244 pass
245
246 # {% crispy form helper 'bootstrap' %}
247 try:
248 template_pack = tokens.pop(1)
249 except IndexError:
250 pass
251
252 # {% crispy form 'bootstrap' %}
253 if helper is not None and isinstance(helper, str) and ("'" in helper or '"' in helper):
254 template_pack = helper
255 helper = None
256
257 if template_pack is not None:
258 template_pack = template_pack[1:-1]
259 ALLOWED_TEMPLATE_PACKS = getattr(
260 settings, "CRISPY_ALLOWED_TEMPLATE_PACKS", ("uni_form", "bootstrap3", "bootstrap4")
261 )
262 if template_pack not in ALLOWED_TEMPLATE_PACKS:
263 raise template.TemplateSyntaxError(
264 "crispy tag's template_pack argument should be in %s" % str(ALLOWED_TEMPLATE_PACKS)
265 )
266
267 return CrispyFormNode(form, helper, template_pack=template_pack)
