1"""
2This is the Django template system.
3
4How it works:
5
6The Lexer.tokenize() method converts a template string (i.e., a string
7containing markup with custom template tags) to tokens, which can be either
8plain text (TokenType.TEXT), variables (TokenType.VAR), or block statements
9(TokenType.BLOCK).
10
11The Parser() class takes a list of tokens in its constructor, and its parse()
12method returns a compiled template -- which is, under the hood, a list of
13Node objects.
14
15Each Node is responsible for creating some sort of output -- e.g. simple text
16(TextNode), variable values in a given context (VariableNode), results of basic
17logic (IfNode), results of looping (ForNode), or anything else. The core Node
18types are TextNode, VariableNode, IfNode and ForNode, but plugin modules can
19define their own custom node types.
20
21Each Node has a render() method, which takes a Context and returns a string of
22the rendered node. For example, the render() method of a Variable Node returns
23the variable's value as a string. The render() method of a ForNode returns the
24rendered output of whatever was inside the loop, recursively.
25
26The Template class is a convenient wrapper that takes care of template
27compilation and rendering.
28
29Usage:
30
31The only thing you should ever use directly in this file is the Template class.
32Create a compiled template object with a template_string, then call render()
33with a context. In the compilation stage, the TemplateSyntaxError exception
34will be raised if the template doesn't have proper syntax.
35
36Sample code:
37
38>>> from django import template
39>>> s = '<html>{% if test %}<h1>{{ varvalue }}</h1>{% endif %}</html>'
40>>> t = template.Template(s)
41
42(t is now a compiled template, and its render() method can be called multiple
43times with multiple contexts)
44
45>>> c = template.Context({'test':True, 'varvalue': 'Hello'})
46>>> t.render(c)
47'<html><h1>Hello</h1></html>'
48>>> c = template.Context({'test':False, 'varvalue': 'Hello'})
49>>> t.render(c)
50'<html></html>'
51"""

204 def get_exception_info(self, exception, token):
205 """
206 Return a dictionary containing contextual line information of where
207 the exception occurred in the template. The following information is
208 provided:
209
210 message
211 The message of the exception raised.
212
213 source_lines
214 The lines before, after, and including the line the exception
215 occurred on.
216
217 line
218 The line number the exception occurred on.
219
220 before, during, after
221 The line the exception occurred on split into three parts:
222 1. The content before the token that raised the error.
223 2. The token that raised the error.
224 3. The content after the token that raised the error.
225
226 total
227 The number of lines in source_lines.
228
229 top
230 The line number where source_lines starts.
231
232 bottom
233 The line number where source_lines ends.
234
235 start
236 The start position of the token in the template source.
237
238 end
239 The end position of the token in the template source.
240 """
241 start, end = token.position
242 context_lines = 10
243 line = 0
244 upto = 0
245 source_lines = []
246 before = during = after = ""
247 for num, next in enumerate(linebreak_iter(self.source)):
248 if start >= upto and end <= next:
249 line = num
250 before = escape(self.source[upto:start])
251 during = escape(self.source[start:end])
252 after = escape(self.source[end:next])
253 source_lines.append((num, escape(self.source[upto:next])))
254 upto = next
255 total = len(source_lines)
256
257 top = max(1, line - context_lines)
258 bottom = min(total, line + 1 + context_lines)
259
260 # In some rare cases exc_value.args can be empty or an invalid
261 # string.
262 try:
263 message = str(exception.args[0])
264 except (IndexError, UnicodeDecodeError):
265 message = "(Could not get exception message)"
266
267 return {
268 "message": message,
269 "source_lines": source_lines[top:bottom],
270 "before": before,
271 "during": during,
272 "after": after,
273 "top": top,
274 "bottom": bottom,
275 "total": total,
276 "line": line,
277 "name": self.origin.name,
278 "start": start,
279 "end": end,
280 }

324 def split_contents(self):
325 split = []
326 bits = smart_split(self.contents)
327 for bit in bits:
328 # Handle translation-marked template pieces
329 if bit.startswith(('_("', "_('")):
330 sentinel = bit[2] + ")"
331 trans_bit = [bit]
332 while not bit.endswith(sentinel):
333 bit = next(bits)
334 trans_bit.append(bit)
335 bit = " ".join(trans_bit)
336 split.append(bit)
337 return split

366 def create_token(self, token_string, position, lineno, in_tag):
367 """
368 Convert the given token string into a new Token object and return it.
369 If in_tag is True, we are processing something that matched a tag,
370 otherwise it should be treated as a literal string.
371 """
372 if in_tag:
373 # The [0:2] and [2:-2] ranges below strip off *_TAG_START and
374 # *_TAG_END. The 2's are hard-coded for performance. Using
375 # len(BLOCK_TAG_START) would permit BLOCK_TAG_START to be
376 # different, but it's not likely that the TAG_START values will
377 # change anytime soon.
378 token_start = token_string[0:2]
379 if token_start == BLOCK_TAG_START:
380 content = token_string[2:-2].strip()
381 if self.verbatim:
382 # Then a verbatim block is being processed.
383 if content != self.verbatim:
384 return Token(TokenType.TEXT, token_string, position, lineno)
385 # Otherwise, the current verbatim block is ending.
386 self.verbatim = False
387 elif content[:9] in ("verbatim", "verbatim "):
388 # Then a verbatim block is starting.
389 self.verbatim = "end%s" % content
390 return Token(TokenType.BLOCK, content, position, lineno)
391 if not self.verbatim:
392 content = token_string[2:-2].strip()
393 if token_start == VARIABLE_TAG_START:
394 return Token(TokenType.VAR, content, position, lineno)
395 # BLOCK_TAG_START was handled above.
396 assert token_start == COMMENT_TAG_START
397 return Token(TokenType.COMMENT, content, position, lineno)
398 return Token(TokenType.TEXT, token_string, position, lineno)

436 def __init__(self, tokens, libraries=None, builtins=None, origin=None):
437 # Reverse the tokens so delete_first_token(), prepend_token(), and
438 # next_token() can operate at the end of the list in constant time.
439 self.tokens = list(reversed(tokens))
440 self.tags = {}

442 self.command_stack = []

445 # will be made available on the template instance. Library authors
446 # should use a key to namespace any added data. The 'django' namespace
447 # is reserved for internal use.

463 def parse(self, parse_until=None):
464 """
465 Iterate through the parser tokens and compiles each one into a node.
466
467 If parse_until is provided, parsing will stop once one of the
468 specified tokens has been reached. This is formatted as a list of
469 tokens, e.g. ['elif', 'else', 'endif']. If no matching token is
470 reached, raise an exception with the unclosed block tag details.
471 """
472 if parse_until is None:
473 parse_until = []
474 nodelist = NodeList()
475 while self.tokens:
476 token = self.next_token()
477 # Use the raw values here for TokenType.* for a tiny performance boost.
478 token_type = token.token_type.value
479 if token_type == 0: # TokenType.TEXT
480 self.extend_nodelist(nodelist, TextNode(token.contents), token)
481 elif token_type == 1: # TokenType.VAR
482 if not token.contents:
483 raise self.error(
484 token, "Empty variable tag on line %d" % token.lineno
485 )
486 try:
487 filter_expression = self.compile_filter(token.contents)
488 except TemplateSyntaxError as e:
489 raise self.error(token, e)
490 var_node = VariableNode(filter_expression)
491 self.extend_nodelist(nodelist, var_node, token)
492 elif token_type == 2: # TokenType.BLOCK
493 try:
494 command = token.contents.split()[0]
495 except IndexError:
496 raise self.error(token, "Empty block tag on line %d" % token.lineno)
497 if command in parse_until:
498 # A matching token has been reached. Return control to
499 # the caller. Put the token back on the token list so the
500 # caller knows where it terminated.
501 self.prepend_token(token)
502 return nodelist
503 # Add the token to the command stack. This is used for error
504 # messages if further parsing fails due to an unclosed block
505 # tag.
506 self.command_stack.append((command, token))
507 # Get the tag callback function from the ones registered with
508 # the parser.
509 try:
510 compile_func = self.tags[command]
511 except KeyError:
512 self.invalid_block_tag(token, command, parse_until)
513 # Compile the callback into a node object and add it to
514 # the node list.
515 try:
516 compiled_result = compile_func(self, token)
517 except Exception as e:
518 raise self.error(token, e)
519 self.extend_nodelist(nodelist, compiled_result, token)
520 # Compile success. Remove the token from the command stack.
521 self.command_stack.pop()
522 if parse_until:
523 self.unclosed_block_tag(parse_until)
524 return nodelist

652class FilterExpression:
653 """
654 Parse a variable token and its optional filters (all as a single string),
655 and return a list of tuples of the filter name and arguments.
656 Sample::
657
658 >>> token = 'variable|default:"Default value"|date:"Y-m-d"'
659 >>> p = Parser('')
660 >>> fe = FilterExpression(token, p)
661 >>> len(fe.filters)
662 2
663 >>> fe.var
664 <Variable: 'variable'>
665 """
666
667 __slots__ = ("token", "filters", "var", "is_var")
668
669 def __init__(self, token, parser):
670 self.token = token
671 matches = filter_re.finditer(token)
672 var_obj = None
673 filters = []
674 upto = 0
675 for match in matches:
676 start = match.start()
677 if upto != start:
678 raise TemplateSyntaxError(
679 "Could not parse some characters: "
680 "%s|%s|%s" % (token[:upto], token[upto:start], token[start:])
681 )
682 if var_obj is None:
683 if constant := match["constant"]:
684 try:
685 var_obj = Variable(constant).resolve({})
686 except VariableDoesNotExist:
687 var_obj = None
688 elif (var := match["var"]) is None:
689 raise TemplateSyntaxError(
690 "Could not find variable at start of %s." % token
691 )
692 else:
693 var_obj = Variable(var)
694 else:
695 filter_name = match["filter_name"]
696 args = []
697 if constant_arg := match["constant_arg"]:
698 args.append((False, Variable(constant_arg).resolve({})))
699 elif var_arg := match["var_arg"]:
700 args.append((True, Variable(var_arg)))
701 filter_func = parser.find_filter(filter_name)
702 self.args_check(filter_name, filter_func, args)
703 filters.append((filter_func, args))
704 upto = match.end()
705 if upto != len(token):
706 raise TemplateSyntaxError(
707 "Could not parse the remainder: '%s' "
708 "from '%s'" % (token[upto:], token)
709 )
710
711 self.filters = filters
712 self.var = var_obj
713 self.is_var = isinstance(var_obj, Variable)
714
715 def resolve(self, context, ignore_failures=False):
716 if self.is_var:
717 try:
718 obj = self.var.resolve(context)
719 except VariableDoesNotExist:
720 if ignore_failures:
721 obj = None
722 else:
723 string_if_invalid = context.template.engine.string_if_invalid
724 if string_if_invalid:
725 if "%s" in string_if_invalid:
726 return string_if_invalid % self.var
727 else:
728 return string_if_invalid
729 else:
730 obj = string_if_invalid
731 else:
732 obj = self.var
733 for func, args in self.filters:
734 arg_vals = []
735 for lookup, arg in args:
736 if not lookup:
737 arg_vals.append(mark_safe(arg))
738 else:
739 arg_vals.append(arg.resolve(context))
740 if getattr(func, "expects_localtime", False):
741 obj = template_localtime(obj, context.use_tz)
742 if getattr(func, "needs_autoescape", False):
743 new_obj = func(obj, autoescape=context.autoescape, *arg_vals)
744 else:
745 new_obj = func(obj, *arg_vals)
746 if getattr(func, "is_safe", False) and isinstance(obj, SafeData):
747 obj = mark_safe(new_obj)
748 else:
749 obj = new_obj
750 return obj

752 def args_check(name, func, provided):
753 provided = list(provided)
754 # First argument, filter input, is implied.
755 plen = len(provided) + 1
756 # Check to see if a decorator is providing the real function.
757 func = inspect.unwrap(func)

779class Variable:
780 """
781 A template variable, resolvable against a given context. The variable may
782 be a hard-coded string (if it begins and ends with single or double quote
783 marks)::
784
785 >>> c = {'article': {'section':'News'}}
786 >>> Variable('article.section').resolve(c)
787 'News'
788 >>> Variable('article').resolve(c)
789 {'section': 'News'}
790 >>> class AClass: pass
791 >>> c = AClass()
792 >>> c.article = AClass()
793 >>> c.article.section = 'News'
794
795 (The example assumes VARIABLE_ATTRIBUTE_SEPARATOR is '.')
796 """
797
798 __slots__ = ("var", "literal", "lookups", "translate", "message_context")
799
800 def __init__(self, var):
801 self.var = var
802 self.literal = None
803 self.lookups = None
804 self.translate = False
805 self.message_context = None
806
807 if not isinstance(var, str):
808 raise TypeError("Variable must be a string or number, got %s" % type(var))
809 try:
810 # First try to treat this variable as a number.
811 #
812 # Note that this could cause an OverflowError here that we're not
813 # catching. Since this should only happen at compile time, that's
814 # probably OK.
815
816 # Try to interpret values containing a period or an 'e'/'E'
817 # (possibly scientific notation) as a float; otherwise, try int.
818 if "." in var or "e" in var.lower():
819 self.literal = float(var)
820 # "2." is invalid
821 if var[-1] == ".":
822 raise ValueError
823 else:
824 self.literal = int(var)
825 except ValueError:
826 # A ValueError means that the variable isn't a number.
827 if var[0:2] == "_(" and var[-1] == ")":
828 # The result of the lookup should be translated at rendering
829 # time.
830 self.translate = True
831 var = var[2:-1]
832 # If it's wrapped with quotes (single or double), then
833 # we're also dealing with a literal.
834 try:
835 self.literal = mark_safe(unescape_string_literal(var))
836 except ValueError:
837 # Otherwise we'll set self.lookups so that resolve() knows we're
838 # dealing with a bonafide variable
839 if VARIABLE_ATTRIBUTE_SEPARATOR + "_" in var or var[0] == "_":
840 raise TemplateSyntaxError(
841 "Variables and attributes may "
842 "not begin with underscores: '%s'" % var
843 )
844 self.lookups = tuple(var.split(VARIABLE_ATTRIBUTE_SEPARATOR))
845
846 def resolve(self, context):
847 """Resolve this variable against a given context."""
848 if self.lookups is not None:
849 # We're dealing with a variable that needs to be resolved
850 value = self._resolve_lookup(context)
851 else:
852 # We're dealing with a literal, so it's already been "resolved"
853 value = self.literal
854 if self.translate:
855 is_safe = isinstance(value, SafeData)
856 msgid = value.replace("%", "%%")
857 msgid = mark_safe(msgid) if is_safe else msgid
858 if self.message_context:
859 return pgettext_lazy(self.message_context, msgid)
860 else:
861 return gettext_lazy(msgid)
862 return value

864 def __repr__(self):
865 return "<%s: %r>" % (self.__class__.__name__, self.var)
866
867 def __str__(self):
868 return self.var

870 def _resolve_lookup(self, context):
871 """
872 Perform resolution of a real variable (i.e. not a literal) against the
873 given context.
874
875 As indicated by the method's name, this method is an implementation
876 detail and shouldn't be called by external code. Use Variable.resolve()
877 instead.
878 """
879 current = context
880 try: # catch-all for silent variable failures
881 for bit in self.lookups:
882 try: # dictionary lookup
883 current = current[bit]
884 # ValueError/IndexError are for numpy.array lookup on
885 # numpy < 1.9 and 1.9+ respectively
886 except (TypeError, AttributeError, KeyError, ValueError, IndexError):
887 try: # attribute lookup
888 # Don't return class attributes if the class is the context:
889 if isinstance(current, BaseContext) and getattr(
890 type(current), bit
891 ):
892 raise AttributeError
893 current = getattr(current, bit)
894 except (TypeError, AttributeError):
895 # Reraise if the exception was raised by a @property
896 if not isinstance(current, BaseContext) and bit in dir(current):
897 raise
898 try: # list-index lookup
899 current = current[int(bit)]
900 except (
901 IndexError, # list index out of range
902 ValueError, # invalid literal for int()
903 KeyError, # current is a dict without `int(bit)` key
904 TypeError,
905 ): # unsubscriptable object
906 raise VariableDoesNotExist(
907 "Failed lookup for key [%s] in %r",
908 (bit, current),
909 ) # missing attribute
910 if callable(current):
911 if getattr(current, "do_not_call_in_templates", False):
912 pass
913 elif getattr(current, "alters_data", False):
914 current = context.template.engine.string_if_invalid
915 else:
916 try: # method call (assuming no args required)
917 current = current()
918 except TypeError:
919 try:
920 signature = inspect.signature(current)
921 except ValueError: # No signature found.
922 current = context.template.engine.string_if_invalid
923 else:
924 try:
925 signature.bind()
926 except TypeError: # Arguments *were* required.
927 # Invalid method call.
928 current = context.template.engine.string_if_invalid
929 else:
930 raise
931 except Exception as e:
932 template_name = getattr(context, "template_name", None) or "unknown"
933 logger.debug(
934 "Exception while resolving variable '%s' in template '%s'.",
935 bit,
936 template_name,
937 exc_info=True,
938 )
939
940 if getattr(e, "silent_variable_failure", False):
941 current = context.template.engine.string_if_invalid
942 else:
943 raise
944
945 return current

1080def token_kwargs(bits, parser, support_legacy=False):
1081 """
1082 Parse token keyword arguments and return a dictionary of the arguments
1083 retrieved from the ``bits`` token list.
1084
1085 `bits` is a list containing the remainder of the token (split by spaces)
1086 that is to be checked for arguments. Valid arguments are removed from this
1087 list.
1088
1089 `support_legacy` - if True, the legacy format ``1 as foo`` is accepted.
1090 Otherwise, only the standard ``foo=1`` format is allowed.
1091
1092 There is no requirement for all remaining token ``bits`` to be keyword
1093 arguments, so return the dictionary as soon as an invalid argument format
1094 is reached.
1095 """
1096 if not bits:
1097 return {}
1098 match = kwarg_re.match(bits[0])
1099 kwarg_format = match and match[1]
1100 if not kwarg_format:
1101 if not support_legacy:
1102 return {}
1103 if len(bits) < 3 or bits[1] != "as":
1104 return {}
1105
1106 kwargs = {}
1107 while bits:
1108 if kwarg_format:
1109 match = kwarg_re.match(bits[0])
1110 if not match or not match[1]:
1111 return kwargs
1112 key, value = match.groups()
1113 del bits[:1]
1114 else:
1115 if len(bits) < 3 or bits[1] != "as":
1116 return kwargs
1117 key, value = bits[2], bits[0]
1118 del bits[:3]
1119 kwargs[key] = parser.compile_filter(value)
1120 if bits and not kwarg_format:
1121 if bits[0] != "and":
1122 return kwargs
1123 del bits[:1]
1124 return kwargs
