32def get_srid_info(srid, connection):
33 """
34 Return the units, unit name, and spheroid WKT associated with the
35 given SRID from the `spatial_ref_sys` (or equivalent) spatial database
36 table for the given database connection. These results are cached.
37 """
38 from django.contrib.gis.gdal import SpatialReference
39
40 global _srid_cache
41
42 try:
43 # The SpatialRefSys model for the spatial backend.
44 SpatialRefSys = connection.ops.spatial_ref_sys()
45 except NotImplementedError:
46 SpatialRefSys = None
47
48 alias, get_srs = (
49 (
50 connection.alias,
51 lambda srid: SpatialRefSys.objects.using(connection.alias)
52 .get(srid=srid)
53 .srs,
54 )
55 if SpatialRefSys
56 else (None, SpatialReference)
57 )
58 if srid not in _srid_cache[alias]:
59 srs = get_srs(srid)
60 units, units_name = srs.units
61 _srid_cache[alias][srid] = SRIDCacheEntry(
62 units=units,
63 units_name=units_name,
64 spheroid='SPHEROID["%s",%s,%s]'
65 % (srs["spheroid"], srs.semi_major, srs.inverse_flattening),
66 geodetic=srs.geographic,
67 )
68
69 return _srid_cache[alias][srid]

85 """
86 The initialization function for base spatial fields. Takes the following

90 The spatial reference system identifier, an OGC standard.
91 Defaults to 4326 (WGS84).
92
93 spatial_index:
94 Indicates whether to create a spatial index. Defaults to True.
95 Set this instead of 'db_index' for geographic fields since index

147 def get_srid(self, obj):
148 """
149 Return the default SRID for the given geometry or raster, taking into
150 account the SRID set for the field. For example, if the input geometry
151 or raster doesn't have an SRID, then the SRID of the field will be
152 returned.
153 """
154 srid = obj.srid # SRID of given geometry.
155 if srid is None or self.srid == -1 or (srid == -1 and self.srid != -1):
156 return self.srid
157 else:
158 return srid

172 def get_raster_prep_value(self, value, is_candidate):
173 """
174 Return a GDALRaster if conversion is successful, otherwise return None.
175 """
176 if isinstance(value, gdal.GDALRaster):
177 return value
178 elif is_candidate:
179 try:
180 return gdal.GDALRaster(value)
181 except GDALException:
182 pass
183 elif isinstance(value, dict):
184 try:
185 return gdal.GDALRaster(value)
186 except GDALException:
187 raise ValueError(
188 "Couldn't create spatial object from lookup value '%s'." % value
189 )
190
191 def get_prep_value(self, value):
192 obj = super().get_prep_value(value)
193 if obj is None:
194 return None
195 # When the input is not a geometry or raster, attempt to construct one
196 # from the given string input.
197 if isinstance(obj, GEOSGeometry):
198 pass
199 else:
200 # Check if input is a candidate for conversion to raster or geometry.
201 is_candidate = isinstance(obj, (bytes, str)) or hasattr(
202 obj, "__geo_interface__"
203 )
204 # Try to convert the input to raster.
205 raster = self.get_raster_prep_value(obj, is_candidate)
206
207 if raster:
208 obj = raster
209 elif is_candidate:
210 try:
211 obj = GEOSGeometry(obj)
212 except (GEOSException, GDALException):
213 raise ValueError(
214 "Couldn't create spatial object from lookup value '%s'." % obj
215 )
216 else:
217 raise ValueError(
218 "Cannot use object with type %s for a spatial lookup parameter."
219 % type(obj).__name__
220 )
221
222 # Assigning the SRID value.
223 obj.srid = self.get_srid(obj)
224 return obj
