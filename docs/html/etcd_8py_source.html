138 def _get_jobs(self):
139 jobs = []
140 failed_job_ids = []
141 all_ids = list(self.client.get_prefix(self.path))
142
143 for doc, _ in all_ids:
144 try:
145 content = pickle.loads(doc)
146 job_record = {
147 "next_run_time": content["next_run_time"],
148 "job": self._reconstitute_job(content["job_state"]),
149 }
150 jobs.append(job_record)
151 except BaseException:
152 content = pickle.loads(doc)
153 failed_id = content["job_state"]["id"]
154 failed_job_ids.append(failed_id)
155 self._logger.exception(
156 'Unable to restore job "%s" -- removing it', failed_id
157 )
158
159 if failed_job_ids:
160 for failed_id in failed_job_ids:
161 self.remove_job(failed_id)
162 paused_sort_key = datetime(9999, 12, 31, tzinfo=timezone.utc)
163 return sorted(
164 jobs,
165 key=lambda job_record: job_record["job"].next_run_time or paused_sort_key,
