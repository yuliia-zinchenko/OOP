24def get_expected_or_default(
25 tested_configuration_file: str | Path,
26 suffix: str,
27 default: str,
28) -> str:
29 """Return the expected value from the file if it exists, or the given default."""
30 expected = default
31 path = Path(tested_configuration_file)
32 expected_result_path = path.parent / f"{path.stem}.{suffix}"
33 if expected_result_path.exists():
34 with open(expected_result_path, encoding="utf8") as f:
35 expected = f.read()
36 # logging is helpful to realize your file is not taken into
37 # account after a misspelling of the file name. The output of the
38 # program is checked during the test so printing messes with the result.
39 logging.info("%s exists.", expected_result_path)
40 else:
41 logging.info("%s not found, using '%s'.", expected_result_path, default)
42 return expected

49def get_expected_configuration(
50 configuration_path: str, default_configuration: PylintConfiguration
51) -> PylintConfiguration:
52 """Get the expected parsed configuration of a configuration functional test."""
53 result = copy.deepcopy(default_configuration)
54 config_as_json = get_expected_or_default(
55 configuration_path, suffix="result.json", default="{}"
56 )
57 to_override = json.loads(config_as_json)
58 for key, value in to_override.items():
59 if key == EXPECTED_CONF_APPEND_KEY:
60 for fkey, fvalue in value.items():
61 result[fkey] += fvalue
62 elif key == EXPECTED_CONF_REMOVE_KEY:
63 for fkey, fvalue in value.items():
64 new_value = []
65 for old_value in result[fkey]:
66 if old_value not in fvalue:
67 new_value.append(old_value)
68 result[fkey] = new_value
69 else:
70 result[key] = value
71 return result

74def get_related_files(
75 tested_configuration_file: str | Path, suffix_filter: str
76) -> list[Path]:
77 """Return all the file related to a test conf file ending with a suffix."""
78 conf_path = Path(tested_configuration_file)
79 return [
80 p
81 for p in conf_path.parent.iterdir()
82 if str(p.stem).startswith(conf_path.stem) and str(p).endswith(suffix_filter)
83 ]

86def get_expected_output(
87 configuration_path: str | Path, user_specific_path: Path
88) -> tuple[int, str]:
89 """Get the expected output of a functional test."""
90 exit_code = 0
91 msg = (
92 "we expect a single file of the form 'filename.32.out' where 'filename' represents "
93 "the name of the configuration file, and '32' the expected error code."
94 )
95 possible_out_files = get_related_files(configuration_path, suffix_filter="out")
96 if len(possible_out_files) > 1:
97 logging.error(
98 "Too much .out files for %s %s.",
99 configuration_path,
100 msg,
101 )
102 return -1, "out file is broken"
103 if not possible_out_files:
104 # logging is helpful to see what the expected exit code is and why.
105 # The output of the program is checked during the test so printing
106 # messes with the result.
107 logging.info(".out file does not exists, so the expected exit code is 0")
108 return 0, ""
109 path = possible_out_files[0]
110 try:
111 exit_code = int(str(path.stem).rsplit(".", maxsplit=1)[-1])
112 except Exception as e: # pylint: disable=broad-except
113 logging.error(
114 "Wrong format for .out file name for %s %s: %s",
115 configuration_path,
116 msg,
117 e,
118 )
119 return -1, "out file is broken"
120
121 output = get_expected_or_default(
122 configuration_path, suffix=f"{exit_code}.out", default=""
123 )
124 logging.info(
125 "Output exists for %s so the expected exit code is %s",
126 configuration_path,
127 exit_code,
128 )
129 return exit_code, output.format(
130 abspath=configuration_path,
131 relpath=Path(configuration_path).relative_to(user_specific_path),
132 )

135def run_using_a_configuration_file(
136 configuration_path: Path | str, file_to_lint: str = __file__
137) -> Run:
138 """Simulate a run with a configuration without really launching the checks."""
139 configuration_path = str(configuration_path)
140 args = ["--rcfile", configuration_path, file_to_lint]
141 # Do not actually run checks, that could be slow. We don't mock
142 # `PyLinter.check`: it calls `PyLinter.initialize` which is
143 # needed to properly set up messages inclusion/exclusion
144 # in `_msg_states`, used by `is_message_enabled`.
145 check = "pylint.lint.pylinter.check_parallel"
146 with unittest.mock.patch(check):
147 runner = Run(args, exit=False)
148 return runner
