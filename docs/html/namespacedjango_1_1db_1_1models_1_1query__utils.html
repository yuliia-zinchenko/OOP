Detailed Description

Various data structures used in query construction.

Factored out from django.db.models.query to avoid making the main module very
large and/or so that they can be used by other modules without getting into
circular import difficulties.

422 # If the field is a primary key, then doing a query against the field's
423 # model is ok, too. Consider the case:
424 # class Restaurant(models.Model):
425 # place = OneToOneField(Place, primary_key=True):
426 # Restaurant.objects.filter(pk__in=Restaurant.objects.all()).

356def select_related_descend(field, restricted, requested, select_mask):
357 """
358 Return whether `field` should be used to descend deeper for
359 `select_related()` purposes.
360
361 Arguments:
362 * `field` - the field to be checked. Can be either a `Field` or
363 `ForeignObjectRel` instance.
364 * `restricted` - a boolean field, indicating if the field list has been
365 manually restricted using a select_related() clause.
366 * `requested` - the select_related() dictionary.
367 * `select_mask` - the dictionary of selected fields.
368 """
369 # Only relationships can be descended.
370 if not field.remote_field:
371 return False
372 # Forward MTI parent links should not be explicitly descended as they are
373 # always JOIN'ed against (unless excluded by `select_mask`).
374 if getattr(field.remote_field, "parent_link", False):
375 return False
376 # When `select_related()` is used without a `*requested` mask all
377 # relationships are descended unless they are nullable.
378 if not restricted:
379 return not field.null
380 # When `select_related(*requested)` is used only fields that are part of
381 # `requested` should be descended.
382 if field.name not in requested:
383 return False
384 # Prevent invalid usages of `select_related()` and `only()`/`defer()` such
385 # as `select_related("a").only("b")` and `select_related("a").defer("a")`.
386 if select_mask and field not in select_mask:
387 raise FieldError(
388 f"Field {field.model._meta.object_name}.{field.name} cannot be both "
389 "deferred and traversed using select_related at the same time."
390 )
391 return True
