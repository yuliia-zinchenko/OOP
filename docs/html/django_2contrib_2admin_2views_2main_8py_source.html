357 """

359 field_name to use for ordering. field_name may either be the name of a
360 proper model field, possibly across relations, or the name of a method

385 def get_ordering(self, request, queryset):
386 """
387 Return the list of ordering fields for the change list.
388 First check the get_ordering() method in model admin, then check
389 the object's default ordering. Then, any manually-specified ordering
390 from the query string overrides anything. Finally, a deterministic
391 order is guaranteed by calling _get_deterministic_ordering() with the
392 constructed ordering.
393 """
394 params = self.params
395 ordering = list(
396 self.model_admin.get_ordering(request) or self._get_default_ordering()
397 )
398 if ORDER_VAR in params:
399 # Clear ordering and used params
400 ordering = []
401 order_params = params[ORDER_VAR].split(".")
402 for p in order_params:
403 try:
404 none, pfx, idx = p.rpartition("-")
405 field_name = self.list_display[int(idx)]
406 order_field = self.get_ordering_field(field_name)
407 if not order_field:
408 continue # No 'admin_order_field', skip it
409 if isinstance(order_field, OrderBy):
410 if pfx == "-":
411 order_field = order_field.copy()
412 order_field.reverse_ordering()
413 ordering.append(order_field)
414 elif hasattr(order_field, "resolve_expression"):
415 # order_field is an expression.
416 ordering.append(
417 order_field.desc() if pfx == "-" else order_field.asc()
418 )
419 # reverse order if order_field has already "-" as prefix
420 elif pfx == "-" and order_field.startswith(pfx):
421 ordering.append(order_field.removeprefix(pfx))
422 else:
423 ordering.append(pfx + order_field)
424 except (IndexError, ValueError):
425 continue # Invalid ordering specified, skip it.
426
427 # Add the given query's ordering fields, if any.
428 ordering.extend(queryset.query.order_by)
429
430 return self._get_deterministic_ordering(ordering)

432 def _get_deterministic_ordering(self, ordering):
433 """
434 Ensure a deterministic order across all database backends. Search for a
435 single field or unique together set of fields providing a total
436 ordering. If these are missing, augment the ordering with a descendant
437 primary key.
438 """
439 ordering = list(ordering)
440 ordering_fields = set()
441 total_ordering_fields = {"pk"} | {
442 field.attname
443 for field in self.lookup_opts.fields
444 if field.unique and not field.null
445 }
446 for part in ordering:
447 # Search for single field providing a total ordering.
448 field_name = None
449 if isinstance(part, str):
450 field_name = part.lstrip("-")
451 elif isinstance(part, F):
452 field_name = part.name
453 elif isinstance(part, OrderBy) and isinstance(part.expression, F):
454 field_name = part.expression.name
455 if field_name:
456 # Normalize attname references by using get_field().
457 try:
458 field = self.lookup_opts.get_field(field_name)
459 except FieldDoesNotExist:
460 # Could be "?" for random ordering or a related field
461 # lookup. Skip this part of introspection for now.
462 continue
463 # Ordering by a related field name orders by the referenced
464 # model's ordering. Skip this part of introspection for now.
465 if field.remote_field and field_name == field.name:
466 continue
467 if field.attname in total_ordering_fields:
468 break
469 ordering_fields.add(field.attname)
470 else:
471 # No single total ordering field, try unique_together and total
472 # unique constraints.
473 constraint_field_names = (
474 *self.lookup_opts.unique_together,
475 *(
476 constraint.fields
477 for constraint in self.lookup_opts.total_unique_constraints
478 ),
479 )
480 for field_names in constraint_field_names:
481 # Normalize attname references by using get_field().
482 fields = [
483 self.lookup_opts.get_field(field_name) for field_name in field_names
484 ]
485 # Composite unique constraints containing a nullable column
486 # cannot ensure total ordering.
487 if any(field.null for field in fields):
488 continue
489 if ordering_fields.issuperset(field.attname for field in fields):
490 break
491 else:
492 # If no set of unique fields is present in the ordering, rely
493 # on the primary key to provide total ordering.
494 ordering.append("-pk")
495 return ordering

537 def get_queryset(self, request, exclude_parameters=None):
538 # First, we collect all the declared list filters.
539 (
540 self.filter_specs,
541 self.has_filters,
542 remaining_lookup_params,
543 filters_may_have_duplicates,
544 self.has_active_filters,
545 ) = self.get_filters(request)
546 # Then, we let every list filter modify the queryset to its liking.
547 qs = self.root_queryset
548 for filter_spec in self.filter_specs:
549 if (
550 exclude_parameters is None
551 or filter_spec.expected_parameters() != exclude_parameters
552 ):
553 new_qs = filter_spec.queryset(request, qs)
554 if new_qs is not None:
555 qs = new_qs
556
557 try:
558 # Finally, we apply the remaining lookup parameters from the query
559 # string (i.e. those that haven't already been processed by the
560 # filters).
561 q_object = build_q_object_from_lookup_parameters(remaining_lookup_params)
562 qs = qs.filter(q_object)
563 except (SuspiciousOperation, ImproperlyConfigured):
564 # Allow certain types of errors to be re-raised as-is so that the
565 # caller can treat them in a special way.
566 raise
567 except Exception as e:

569 # have any other way of validating lookup parameters. They might be
570 # invalid if the keyword arguments are incorrect, or if the values

572 # ValueError, ValidationError, or ?.

579 ordering = self.get_ordering(request, qs)
