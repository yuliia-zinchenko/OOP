43 @classmethod
44 def from_msg(cls, msg: Message, check_endline: bool = True) -> OutputLine:
45 """Create an OutputLine from a Pylint Message."""
46 column = cls._get_column(msg.column)
47 end_line = cls._get_end_line_and_end_col(msg.end_line, check_endline)

74 @classmethod
75 def from_csv(
76 cls, row: Sequence[str] | str, check_endline: bool = True
77 ) -> OutputLine:
78 """Create an OutputLine from a comma separated list (the functional tests
79 expected output .txt files).
80 """
81 if isinstance(row, str):
82 row = row.split(",")
83 try:
84 line = int(row[1])
85 column = cls._get_column(row[2])
86 end_line = cls._value_to_optional_int(
87 cls._get_end_line_and_end_col(row[3], check_endline)
88 )
89 end_column = cls._value_to_optional_int(
90 cls._get_end_line_and_end_col(row[4], check_endline)
91 )
92 # symbol, line, column, end_line, end_column, node, msg, confidences
93 assert len(row) == 8
94 return cls(
95 row[0], line, column, end_line, end_column, row[5], row[6], row[7]
96 )
97 except Exception: # pylint: disable=broad-except
98 # We need this to not fail for the update script to work.
99 return cls("", 0, 0, None, None, "", "", "")
