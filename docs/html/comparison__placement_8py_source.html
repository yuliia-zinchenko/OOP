20REVERSED_COMPS = {"<": ">", "<=": ">=", ">": "<", ">=": "<="}
21COMPARISON_OPERATORS = frozenset(("==", "!=", "<", ">", "<=", ">="))

24class MisplacedComparisonConstantChecker(BaseChecker):
25 """Checks the placement of constants in comparisons."""
26
27 # configuration section name

34 "of a comparison. It is usually clearer in intent to "
35 "place it in the right hand side of the comparison.",
36 {"old_names": [("C0122", "old-misplaced-comparison-constant")]},

42 def _check_misplaced_constant(
43 self,
44 node: nodes.Compare,
45 left: nodes.NodeNG,
46 right: nodes.NodeNG,
47 operator: str,
48 ) -> None:
49 if isinstance(right, nodes.Const):
50 return
51 operator = REVERSED_COMPS.get(operator, operator)
52 suggestion = f"{right.as_string()} {operator} {left.value!r}"
53 self.add_message("misplaced-comparison-constant", node=node, args=(suggestion,))
54
55 @utils.only_required_for_messages("misplaced-comparison-constant")
56 def visit_compare(self, node: nodes.Compare) -> None:
57 # NOTE: this checker only works with binary comparisons like 'x == 42'
58 # but not 'x == y == 42'
59 if len(node.ops) != 1:
60 return
61
62 left = node.left
63 operator, right = node.ops[0]
64 if operator in COMPARISON_OPERATORS and isinstance(left, nodes.Const):
65 self._check_misplaced_constant(node, left, right, operator)
