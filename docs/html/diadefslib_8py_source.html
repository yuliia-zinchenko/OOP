24class DiaDefGenerator:
25 """Handle diagram generation options."""
26
27 def __init__(self, linker: Linker, handler: DiadefsHandler) -> None:
28 """Common Diagram Handler initialization."""
29 self.config = handler.config
30 self.module_names: bool = False
31 self._set_default_options()
32 self.linker = linker
33 self.classdiagram: ClassDiagram # defined by subclasses
34
35 def get_title(self, node: nodes.ClassDef) -> str:
36 """Get title for objects."""
37 title = node.name
38 if self.module_names:
39 title = f"{node.root().name}.{title}"
40 return title # type: ignore[no-any-return]
41
42 def _set_option(self, option: bool | None) -> bool:
43 """Activate some options if not explicitly deactivated."""
44 # if we have a class diagram, we want more information by default;
45 # so if the option is None, we return True
46 if option is None:
47 return bool(self.config.classes)
48 return option
49
50 def _set_default_options(self) -> None:
51 """Set different default options with _default dictionary."""
52 self.module_names = self._set_option(self.config.module_names)
53 all_ancestors = self._set_option(self.config.all_ancestors)
54 all_associated = self._set_option(self.config.all_associated)
55 anc_level, association_level = (0, 0)
56 if all_ancestors:
57 anc_level = -1
58 if all_associated:
59 association_level = -1
60 if self.config.show_ancestors is not None:
61 anc_level = self.config.show_ancestors
62 if self.config.show_associated is not None:
63 association_level = self.config.show_associated
64 self.anc_level, self.association_level = anc_level, association_level
65
66 def _get_levels(self) -> tuple[int, int]:
67 """Help function for search levels."""
68 return self.anc_level, self.association_level
69
70 def show_node(self, node: nodes.ClassDef) -> bool:
71 """Determine if node should be shown based on config."""
72 if node.root().name == "builtins":
73 return self.config.show_builtin # type: ignore[no-any-return]
74
75 if is_stdlib_module(node.root().name):
76 return self.config.show_stdlib # type: ignore[no-any-return]
77
78 return True
79
80 def add_class(self, node: nodes.ClassDef) -> None:
81 """Visit one class and add it to diagram."""
82 self.linker.visit(node)
83 self.classdiagram.add_object(self.get_title(node), node)
84
85 def get_ancestors(
86 self, node: nodes.ClassDef, level: int
87 ) -> Generator[nodes.ClassDef]:
88 """Return ancestor nodes of a class node."""
89 if level == 0:
90 return
91 for ancestor in node.ancestors(recurs=False):
92 if not self.show_node(ancestor):
93 continue
94 yield ancestor
95
96 def get_associated(
97 self, klass_node: nodes.ClassDef, level: int
98 ) -> Generator[nodes.ClassDef]:
99 """Return associated nodes of a class node."""
100 if level == 0:
101 return
102 for association_nodes in list(klass_node.instance_attrs_type.values()) + list(
103 klass_node.locals_type.values()
104 ):
105 for node in association_nodes:
106 if isinstance(node, astroid.Instance):
107 node = node._proxied
108 if not (isinstance(node, nodes.ClassDef) and self.show_node(node)):
109 continue
110 yield node
111
112 def extract_classes(
113 self, klass_node: nodes.ClassDef, anc_level: int, association_level: int
114 ) -> None:
115 """Extract recursively classes related to klass_node."""
116 if self.classdiagram.has_node(klass_node) or not self.show_node(klass_node):
117 return
118 self.add_class(klass_node)
119
120 for ancestor in self.get_ancestors(klass_node, anc_level):
121 self.extract_classes(ancestor, anc_level - 1, association_level)
122
123 for node in self.get_associated(klass_node, association_level):
124 self.extract_classes(node, anc_level, association_level - 1)

127class DefaultDiadefGenerator(LocalsVisitor, DiaDefGenerator):
128 """Generate minimum diagram definition for the project :
129
130 * a package diagram including project's modules
131 * a class diagram including project's classes
132 """
133
134 def __init__(self, linker: Linker, handler: DiadefsHandler) -> None:
135 DiaDefGenerator.__init__(self, linker, handler)
136 LocalsVisitor.__init__(self)
137
138 def visit_project(self, node: Project) -> None:
139 """Visit a pyreverse.utils.Project node.
140
141 create a diagram definition for packages
142 """
143 mode = self.config.mode
144 if len(node.modules) > 1:
145 self.pkgdiagram: PackageDiagram | None = PackageDiagram(
146 f"packages {node.name}", mode
147 )
148 else:
149 self.pkgdiagram = None
150 self.classdiagram = ClassDiagram(f"classes {node.name}", mode)
151
152 def leave_project(self, _: Project) -> Any:
153 """Leave the pyreverse.utils.Project node.
154
155 return the generated diagram definition
156 """
157 if self.pkgdiagram:
158 return self.pkgdiagram, self.classdiagram
159 return (self.classdiagram,)
160
161 def visit_module(self, node: nodes.Module) -> None:
162 """Visit an astroid.Module node.
163
164 add this class to the package diagram definition
165 """
166 if self.pkgdiagram:
167 self.linker.visit(node)
168 self.pkgdiagram.add_object(node.name, node)
169
170 def visit_classdef(self, node: nodes.ClassDef) -> None:
171 """Visit an astroid.Class node.
172
173 add this class to the class diagram definition
174 """
175 anc_level, association_level = self._get_levels()
176 self.extract_classes(node, anc_level, association_level)
177
178 def visit_importfrom(self, node: nodes.ImportFrom) -> None:
179 """Visit astroid.ImportFrom and catch modules for package diagram."""
180 if self.pkgdiagram:
181 self.pkgdiagram.add_from_depend(node, node.modname)

184class ClassDiadefGenerator(DiaDefGenerator):
185 """Generate a class diagram definition including all classes related to a
186 given class.
187 """
188
189 def class_diagram(self, project: Project, klass: nodes.ClassDef) -> ClassDiagram:
190 """Return a class diagram definition for the class and related classes."""
191 self.classdiagram = ClassDiagram(klass, self.config.mode)
192 if len(project.modules) > 1:
193 module, klass = klass.rsplit(".", 1)
194 module = project.get_module(module)
195 else:
196 module = project.modules[0]
197 klass = klass.split(".")[-1]
198 klass = next(module.ilookup(klass))
199
200 anc_level, association_level = self._get_levels()
201 self.extract_classes(klass, anc_level, association_level)
202 return self.classdiagram

208class DiadefsHandler:
209 """Get diagram definitions from user (i.e. xml files) or generate them."""
210
211 def __init__(self, config: argparse.Namespace) -> None:
212 self.config = config
213
214 def get_diadefs(self, project: Project, linker: Linker) -> list[ClassDiagram]:
215 """Get the diagram's configuration data.
216
217 :param project:The pyreverse project
218 :type project: pyreverse.utils.Project
219 :param linker: The linker
220 :type linker: pyreverse.inspector.Linker(IdGeneratorMixIn, LocalsVisitor)
221
222 :returns: The list of diagram definitions
223 :rtype: list(:class:`pylint.pyreverse.diagrams.ClassDiagram`)
224 """
225 # read and interpret diagram definitions (Diadefs)
226 diagrams = []
227 generator = ClassDiadefGenerator(linker, self)
228 for klass in self.config.classes:
229 diagrams.append(generator.class_diagram(project, klass))
230 if not diagrams:
231 diagrams = DefaultDiadefGenerator(linker, self).visit(project)
232 for diagram in diagrams:
233 diagram.extract_relationships()
234 return diagrams
