13def _get_tests_info(
14 input_dir: str, msg_dir: str, prefix: str, suffix: str
15) -> list[tuple[str, str]]:
16 """Get python input examples and output messages.
17
18 We use following conventions for input files and messages:
19 for different inputs:
20 test for python >= x.y -> input = <name>_pyxy.py
21 test for python < x.y -> input = <name>_py_xy.py
22 for one input and different messages:
23 message for python >= x.y -> message = <name>_pyxy.txt
24 lower versions -> message with highest num
25 """
26 result = []
27 for fname in glob(join(input_dir, prefix + "*" + suffix)):
28 infile = basename(fname)
29 fbase = splitext(infile)[0]
30 # filter input files :
31 pyrestr = fbase.rsplit("_py", 1)[-1] # like _26 or 26
32 if pyrestr.isdigit(): # '24', '25'...
33 if pyrestr.isdigit() and int(SYS_VERS_STR) < int(pyrestr):
34 continue
35 if pyrestr.startswith("_") and pyrestr[1:].isdigit():
36 # skip test for higher python versions
37 if pyrestr[1:].isdigit() and int(SYS_VERS_STR) >= int(pyrestr[1:]):
38 continue
39 messages = glob(join(msg_dir, fbase + "*.txt"))
40 # the last one will be without ext, i.e. for all or upper versions:
41 if messages:
42 for outfile in sorted(messages, reverse=True):
43 py_rest = outfile.rsplit("_py", 1)[-1][:-4]
44 if py_rest.isdigit() and int(SYS_VERS_STR) >= int(py_rest):
45 break
46 else:
47 # This will provide an error message indicating the missing filename.
48 outfile = join(msg_dir, fbase + ".txt")
49 result.append((infile, outfile))
50 return result
