139def convert_to_datetime(input, tz, arg_name):
140 """
141 Converts the given object to a timezone aware datetime object.
142
143 If a timezone aware datetime object is passed, it is returned unmodified.
144 If a native datetime object is passed, it is given the specified timezone.
145 If the input is a string, it is parsed as a datetime with the given timezone.
146
147 Date strings are accepted in three different forms: date only (Y-m-d), date with
148 time (Y-m-d H:M:S) or with date+time with microseconds (Y-m-d H:M:S.micro).
149 Additionally you can override the time zone by giving a specific offset in the
150 format specified by ISO 8601: Z (UTC), +HH:MM or -HH:MM.
151
152 :param str|datetime input: the datetime or string to convert to a timezone aware
153 datetime
154 :param datetime.tzinfo tz: timezone to interpret ``input`` in
155 :param str arg_name: the name of the argument (used in an error message)
156 :rtype: datetime
157
158 """
159 if input is None:
160 return
161 elif isinstance(input, datetime):
162 datetime_ = input
163 elif isinstance(input, date):
164 datetime_ = datetime.combine(input, time())
165 elif isinstance(input, str):
166 m = _DATE_REGEX.match(input)
167 if not m:
168 raise ValueError("Invalid date string")
169
170 values = m.groupdict()
171 tzname = values.pop("timezone")
172 if tzname == "Z":
173 tz = timezone.utc
174 elif tzname:
175 hours, minutes = (int(x) for x in tzname[1:].split(":"))
176 sign = 1 if tzname[0] == "+" else -1
177 tz = timezone(sign * timedelta(hours=hours, minutes=minutes))
178
179 values = {k: int(v or 0) for k, v in values.items()}
180 datetime_ = datetime(**values)
181 else:
182 raise TypeError(f"Unsupported type for {arg_name}: {input.__class__.__name__}")
183
184 if datetime_.tzinfo is not None:
185 return datetime_
186 if tz is None:
187 raise ValueError(
188 f'The "tz" argument must be specified if {arg_name} has no timezone information'
189 )
190 if isinstance(tz, str):
191 tz = astimezone(tz)
192
193 return localize(datetime_, tz)

254def get_callable_name(func):
255 """
256 Returns the best available display name for the given function/callable.
257
258 :rtype: str
259
260 """
261 if ismethod(func):
262 self = func.__self__
263 cls = self if isclass(self) else type(self)
264 return f"{cls.__qualname__}.{func.__name__}"
265 elif isclass(func) or isfunction(func) or isbuiltin(func):
266 return func.__qualname__
267 elif hasattr(func, "__call__") and callable(func.__call__):
268 # instance of a class with a __call__ method
269 return type(func).__qualname__
270
271 raise TypeError(
272 f"Unable to determine a name for {func!r} -- maybe it is not a callable?"
273 )

276def obj_to_ref(obj):
277 """
278 Returns the path to the given callable.
279
280 :rtype: str
281 :raises TypeError: if the given object is not callable
282 :raises ValueError: if the given object is a :class:`~functools.partial`, lambda or a nested
283 function
284
285 """
286 if isinstance(obj, partial):
287 raise ValueError("Cannot create a reference to a partial()")
288
289 name = get_callable_name(obj)
290 if "<lambda>" in name:
291 raise ValueError("Cannot create a reference to a lambda")
292 if "<locals>" in name:
293 raise ValueError("Cannot create a reference to a nested function")
294
295 if ismethod(obj):
296 module = obj.__self__.__module__
297 else:
298 module = obj.__module__
299
300 return f"{module}:{name}"

303def ref_to_obj(ref):
304 """
305 Returns the object pointed to by ``ref``.
306
307 :type ref: str
308
309 """
310 if not isinstance(ref, str):
311 raise TypeError("References must be strings")
312 if ":" not in ref:
313 raise ValueError("Invalid reference")
314
315 modulename, rest = ref.split(":", 1)
316 try:
317 obj = __import__(modulename, fromlist=[rest])
318 except ImportError as exc:
319 raise LookupError(
320 f"Error resolving reference {ref}: could not import module"
321 ) from exc
322
323 try:
324 for name in rest.split("."):
325 obj = getattr(obj, name)
326 return obj
327 except Exception:
328 raise LookupError(f"Error resolving reference {ref}: error looking up object")

342def check_callable_args(func, args, kwargs):
343 """
344 Ensures that the given callable can be called with the given arguments.
345
346 :type args: tuple
347 :type kwargs: dict
348
349 """
350 pos_kwargs_conflicts = [] # parameters that have a match in both args and kwargs
351 positional_only_kwargs = [] # positional-only parameters that have a match in kwargs
352 unsatisfied_args = [] # parameters in signature that don't have a match in args or kwargs
353 unsatisfied_kwargs = [] # keyword-only arguments that don't have a match in kwargs
354 unmatched_args = list(
355 args
356 ) # args that didn't match any of the parameters in the signature
357 # kwargs that didn't match any of the parameters in the signature
358 unmatched_kwargs = list(kwargs)
359 # indicates if the signature defines *args and **kwargs respectively
360 has_varargs = has_var_kwargs = False
361
362 try:
363 sig = signature(func, follow_wrapped=False)
364 except ValueError:
365 # signature() doesn't work against every kind of callable
366 return
367
368 for param in sig.parameters.values():
369 if param.kind == param.POSITIONAL_OR_KEYWORD:
370 if param.name in unmatched_kwargs and unmatched_args:
371 pos_kwargs_conflicts.append(param.name)
372 elif unmatched_args:
373 del unmatched_args[0]
374 elif param.name in unmatched_kwargs:
375 unmatched_kwargs.remove(param.name)
376 elif param.default is param.empty:
377 unsatisfied_args.append(param.name)
378 elif param.kind == param.POSITIONAL_ONLY:
379 if unmatched_args:
380 del unmatched_args[0]
381 elif param.name in unmatched_kwargs:
382 unmatched_kwargs.remove(param.name)
383 positional_only_kwargs.append(param.name)
384 elif param.default is param.empty:
385 unsatisfied_args.append(param.name)
386 elif param.kind == param.KEYWORD_ONLY:
387 if param.name in unmatched_kwargs:
388 unmatched_kwargs.remove(param.name)
389 elif param.default is param.empty:
390 unsatisfied_kwargs.append(param.name)
391 elif param.kind == param.VAR_POSITIONAL:
392 has_varargs = True
393 elif param.kind == param.VAR_KEYWORD:
394 has_var_kwargs = True
395
396 # Make sure there are no conflicts between args and kwargs
397 if pos_kwargs_conflicts:
398 raise ValueError(
399 "The following arguments are supplied in both args and kwargs: {}".format(
400 ", ".join(pos_kwargs_conflicts)
401 )
402 )
403
404 # Check if keyword arguments are being fed to positional-only parameters
405 if positional_only_kwargs:
406 raise ValueError(
407 "The following arguments cannot be given as keyword arguments: {}".format(
408 ", ".join(positional_only_kwargs)
409 )
410 )
411
412 # Check that the number of positional arguments minus the number of matched kwargs
413 # matches the argspec
414 if unsatisfied_args:
415 raise ValueError(
416 "The following arguments have not been supplied: {}".format(
417 ", ".join(unsatisfied_args)
418 )
419 )
420
421 # Check that all keyword-only arguments have been supplied
422 if unsatisfied_kwargs:
423 raise ValueError(
424 "The following keyword-only arguments have not been supplied in kwargs: "
425 "{}".format(", ".join(unsatisfied_kwargs))
426 )
427
428 # Check that the callable can accept the given number of positional arguments
429 if not has_varargs and unmatched_args:
430 raise ValueError(
431 f"The list of positional arguments is longer than the target callable can "
432 f"handle (allowed: {len(args) - len(unmatched_args)}, given in args: "
433 f"{len(args)})"
434 )
435
436 # Check that the callable can accept the given keyword arguments
437 if not has_var_kwargs and unmatched_kwargs:
438 raise ValueError(
439 "The target callable does not accept the following keyword arguments: "
440 "{}".format(", ".join(unmatched_kwargs))
441 )
