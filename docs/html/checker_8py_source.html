66def _get_properties(config: argparse.Namespace) -> tuple[set[str], set[str]]:
67 """Returns a tuple of property classes and names.
68
69 Property classes are fully qualified, such as 'abc.abstractproperty' and
70 property names are the actual names, such as 'abstract_property'.
71 """
72 property_classes = {BUILTIN_PROPERTY}
73 property_names: set[str] = set() # Not returning 'property', it has its own check.
74 if config is not None:
75 property_classes.update(config.property_classes)
76 property_names.update(
77 prop.rsplit(".", 1)[-1] for prop in config.property_classes
78 )
79 return property_classes, property_names

82def _redefines_import(node: nodes.AssignName) -> bool:
83 """Detect that the given node (AssignName) is inside an
84 exception handler and redefines an import from the tryexcept body.
85
86 Returns True if the node redefines an import, False otherwise.
87 """
88 current = node
89 while current and not isinstance(current.parent, nodes.ExceptHandler):
90 current = current.parent
91 if not current or not utils.error_of_type(current.parent, ImportError):
92 return False
93 try_block = current.parent.parent
94 for import_node in try_block.nodes_of_class((nodes.ImportFrom, nodes.Import)):
95 for name, alias in import_node.names:
96 if alias:
97 if alias == node.name:
98 return True
99 elif name == node.name:
100 return True
101 return False

104def _determine_function_name_type(
105 node: nodes.FunctionDef, config: argparse.Namespace
106) -> str:
107 """Determine the name type whose regex the function's name should match.
108
109 :param node: A function node.
110 :param config: Configuration from which to pull additional property classes.
111
112 :returns: One of ('function', 'method', 'attr')
113 """
114 property_classes, property_names = _get_properties(config)
115 if not node.is_method():
116 return "function"
117
118 if is_property_setter(node) or is_property_deleter(node):
119 # If the function is decorated using the prop_method.{setter,getter}
120 # form, treat it like an attribute as well.
121 return "attr"
122
123 decorators = node.decorators.nodes if node.decorators else []
124 for decorator in decorators:
125 # If the function is a property (decorated with @property
126 # or @abc.abstractproperty), the name type is 'attr'.
127 if isinstance(decorator, nodes.Name) or (
128 isinstance(decorator, nodes.Attribute)
129 and decorator.attrname in property_names
130 ):
131 inferred = utils.safe_infer(decorator)
132 if (
133 inferred
134 and hasattr(inferred, "qname")
135 and inferred.qname() in property_classes
136 ):
137 return "attr"
138 return "method"

301 def _create_naming_rules(self) -> tuple[dict[str, Pattern[str]], dict[str, str]]:
302 regexps: dict[str, Pattern[str]] = {}
303 hints: dict[str, str] = {}
304
305 for name_type in KNOWN_NAME_TYPES:
306 if name_type in KNOWN_NAME_TYPES_WITH_STYLE:
307 naming_style_name = getattr(
308 self.linter.config, f"{name_type}_naming_style"
309 )
310 regexps[name_type] = NAMING_STYLES[naming_style_name].get_regex(
311 name_type
312 )
313 else:
314 naming_style_name = "predefined"
315 regexps[name_type] = DEFAULT_PATTERNS[name_type]
316
317 custom_regex_setting_name = f"{name_type}_rgx"
318 custom_regex = getattr(self.linter.config, custom_regex_setting_name, None)
319 if custom_regex is not None:
320 regexps[name_type] = custom_regex
321
322 if custom_regex is not None:
323 hints[name_type] = f"{custom_regex.pattern!r} pattern"
324 else:
325 hints[name_type] = f"{naming_style_name} naming style"
326
327 return regexps, hints

334 def leave_module(self, _: nodes.Module) -> None:
335 for all_groups in self._bad_names.values():
336 if len(all_groups) < 2:
337 continue
338 groups: collections.defaultdict[int, list[list[_BadNamesTuple]]] = (
339 collections.defaultdict(list)
340 )
341 min_warnings = sys.maxsize
342 prevalent_group, _ = max(all_groups.items(), key=lambda item: len(item[1]))
343 for group in all_groups.values():
344 groups[len(group)].append(group)
345 min_warnings = min(len(group), min_warnings)
346 if len(groups[min_warnings]) > 1:
347 by_line = sorted(
348 groups[min_warnings],
349 key=lambda group: min(
350 warning[0].lineno
351 for warning in group
352 if warning[0].lineno is not None
353 ),
354 )
355 warnings: Iterable[_BadNamesTuple] = itertools.chain(*by_line[1:])
356 else:
357 warnings = groups[min_warnings][0]
358 for args in warnings:
359 self._raise_name_warning(prevalent_group, *args)

395 @utils.only_required_for_messages(
396 "disallowed-name",
397 "invalid-name",
398 "typevar-name-incorrect-variance",
399 "typevar-double-variance",

401 )
402 def visit_assignname( # pylint: disable=too-many-branches
403 self, node: nodes.AssignName
404 ) -> None:
405 """Check module level assigned names."""
406 frame = node.frame()
407 assign_type = node.assign_type()
408
409 # Check names defined in comprehensions
410 if isinstance(assign_type, nodes.Comprehension):
411 self._check_name("inlinevar", node.name, node)
412
413 elif isinstance(assign_type, nodes.TypeVar):
414 self._check_name("typevar", node.name, node)
415
416 elif isinstance(assign_type, nodes.TypeAlias):
417 self._check_name("typealias", node.name, node)
418
419 # Check names defined in module scope
420 elif isinstance(frame, nodes.Module):
421 # Check names defined in Assign nodes
422 if isinstance(assign_type, nodes.Assign):
423 inferred_assign_type = utils.safe_infer(assign_type.value)
424
425 # Check TypeVar's and TypeAliases assigned alone or in tuple assignment
426 if isinstance(node.parent, nodes.Assign):
427 if self._assigns_typevar(assign_type.value):
428 self._check_name("typevar", assign_type.targets[0].name, node)
429 return
430 if self._assigns_typealias(assign_type.value):
431 self._check_name("typealias", assign_type.targets[0].name, node)
432 return
433
434 if (
435 isinstance(node.parent, nodes.Tuple)
436 and isinstance(assign_type.value, nodes.Tuple)
437 # protect against unbalanced tuple unpacking
438 and node.parent.elts.index(node) < len(assign_type.value.elts)
439 ):
440 assigner = assign_type.value.elts[node.parent.elts.index(node)]
441 if self._assigns_typevar(assigner):
442 self._check_name(
443 "typevar",
444 assign_type.targets[0]
445 .elts[node.parent.elts.index(node)]
446 .name,
447 node,
448 )
449 return
450 if self._assigns_typealias(assigner):
451 self._check_name(
452 "typealias",
453 assign_type.targets[0]
454 .elts[node.parent.elts.index(node)]
455 .name,
456 node,
457 )
458 return
459
460 # Check classes (TypeVar's are classes so they need to be excluded first)
461 elif isinstance(inferred_assign_type, nodes.ClassDef):
462 self._check_name("class", node.name, node)
463
464 # Don't emit if the name redefines an import in an ImportError except handler.
465 elif not _redefines_import(node) and isinstance(
466 inferred_assign_type, nodes.Const
467 ):
468 self._check_name("const", node.name, node)
469 else:
470 self._check_name(
471 "variable", node.name, node, disallowed_check_only=True
472 )
473
474 # Check names defined in AnnAssign nodes
475 elif isinstance(assign_type, nodes.AnnAssign):
476 if utils.is_assign_name_annotated_with(node, "Final"):
477 self._check_name("const", node.name, node)
478 elif self._assigns_typealias(assign_type.annotation):
479 self._check_name("typealias", node.name, node)
480
481 # Check names defined in function scopes
482 elif isinstance(frame, nodes.FunctionDef):
483 # global introduced variable aren't in the function locals
484 if node.name in frame and node.name not in frame.argnames():
485 if not _redefines_import(node):
486 if isinstance(
487 assign_type, nodes.AnnAssign
488 ) and self._assigns_typealias(assign_type.annotation):
489 self._check_name("typealias", node.name, node)
490 else:
491 self._check_name("variable", node.name, node)
492
493 # Check names defined in class scopes
494 elif isinstance(frame, nodes.ClassDef) and not any(
495 frame.local_attr_ancestors(node.name)
496 ):
497 if utils.is_enum_member(node) or utils.is_assign_name_annotated_with(
498 node, "Final"
499 ):
500 self._check_name("class_const", node.name, node)
501 else:
502 self._check_name("class_attribute", node.name, node)

504 def _recursive_check_names(self, args: list[nodes.AssignName]) -> None:
505 """Check names in a possibly recursive list <arg>."""
506 for arg in args:
507 self._check_name("argument", arg.name, arg)

512 def _raise_name_warning(
513 self,
514 prevalent_group: str | None,
515 node: nodes.NodeNG,
516 node_type: str,
517 name: str,
518 confidence: interfaces.Confidence,
519 warning: str = "invalid-name",
520 ) -> None:
521 type_label = constants.HUMAN_READABLE_TYPES[node_type]
522 hint = self._name_hints[node_type]
523 if prevalent_group:
524 # This happens in the multi naming match case. The expected
525 # prevalent group needs to be spelled out to make the message
526 # correct.
527 hint = f"the `{prevalent_group}` group in the {hint}"
528 if self.linter.config.include_naming_hint:
529 hint += f" ({self._name_regexps[node_type].pattern!r} pattern)"
530 args = (
531 (type_label.capitalize(), name, hint)
532 if warning == "invalid-name"
533 else (type_label.capitalize(), name)
534 )
535
536 self.add_message(warning, node=node, args=args, confidence=confidence)
537 self.linter.stats.increase_bad_name(node_type, 1)

549 def _check_name(
550 self,
551 node_type: str,
552 name: str,
553 node: nodes.NodeNG,
554 confidence: interfaces.Confidence = interfaces.HIGH,
555 disallowed_check_only: bool = False,

559 def _should_exempt_from_invalid_name(node: nodes.NodeNG) -> bool:
560 if node_type == "variable":
561 inferred = utils.safe_infer(node)
562 if isinstance(inferred, nodes.ClassDef):
563 return True
564 return False
565
566 if self._name_allowed_by_regex(name=name):
567 return
568 if self._name_disallowed_by_regex(name=name):
569 self.linter.stats.increase_bad_name(node_type, 1)
570 self.add_message(
571 "disallowed-name", node=node, args=name, confidence=interfaces.HIGH
572 )
573 return
574 regexp = self._name_regexps[node_type]
575 match = regexp.match(name)
576
577 if _is_multi_naming_match(match, node_type, confidence):
578 name_group = self._find_name_group(node_type)
579 bad_name_group = self._bad_names.setdefault(name_group, {})
580 # Ignored because this is checked by the if statement
581 warnings = bad_name_group.setdefault(match.lastgroup, []) # type: ignore[union-attr, arg-type]
582 warnings.append((node, node_type, name, confidence))
583
584 if (
585 match is None
586 and not disallowed_check_only
587 and not _should_exempt_from_invalid_name(node)
588 ):
589 self._raise_name_warning(None, node, node_type, name, confidence)
590
591 # Check TypeVar names for variance suffixes
592 if node_type == "typevar":
593 self._check_typevar(name, node)

607 @staticmethod
608 def _assigns_typealias(node: nodes.NodeNG | None) -> bool:
609 """Check if a node is assigning a TypeAlias."""
610 inferred = utils.safe_infer(node)
611 if isinstance(inferred, nodes.ClassDef):
612 qname = inferred.qname()
613 if qname == "typing.TypeAlias":
614 return True
615 if qname == ".Union":

617 # or as a type annotation. We only want to check the former.
618 assert node is not None
619 return not isinstance(node.parent, nodes.AnnAssign)
620 elif isinstance(inferred, nodes.FunctionDef):
621 # TODO: when py3.12 is minimum, remove this condition
622 # TypeAlias became a class in python 3.12
623 if inferred.qname() == "typing.TypeAlias":
624 return True
625 return False

627 def _check_typevar(self, name: str, node: nodes.AssignName) -> None:
628 """Check for TypeVar lint violations."""
629 variance: TypeVarVariance = TypeVarVariance.invariant
630 if isinstance(node.parent, nodes.Assign):
631 keywords = node.assign_type().value.keywords
632 args = node.assign_type().value.args
633 elif isinstance(node.parent, nodes.Tuple):
634 keywords = (
635 node.assign_type().value.elts[node.parent.elts.index(node)].keywords
636 )
637 args = node.assign_type().value.elts[node.parent.elts.index(node)].args
638 else: # PEP 695 generic type nodes
639 keywords = ()
640 args = ()
641 variance = TypeVarVariance.inferred
642
643 name_arg = None
644 for kw in keywords:
645 if variance == TypeVarVariance.double_variant:
646 pass
647 elif kw.arg == "covariant" and kw.value.value:
648 variance = (
649 TypeVarVariance.covariant
650 if variance != TypeVarVariance.contravariant
651 else TypeVarVariance.double_variant
652 )
653 elif kw.arg == "contravariant" and kw.value.value:
654 variance = (
655 TypeVarVariance.contravariant
656 if variance != TypeVarVariance.covariant
657 else TypeVarVariance.double_variant
658 )
659
660 if kw.arg == "name" and isinstance(kw.value, nodes.Const):
661 name_arg = kw.value.value
662
663 if name_arg is None and args and isinstance(args[0], nodes.Const):
664 name_arg = args[0].value
665
666 if variance == TypeVarVariance.inferred:
667 # Ignore variance check for PEP 695 type parameters.
668 # The variance is inferred by the type checker.
669 # Adding _co or _contra suffix can help to reason about TypeVar.
670 pass
671 elif variance == TypeVarVariance.double_variant:
672 self.add_message(
673 "typevar-double-variance",
674 node=node,
675 confidence=interfaces.INFERENCE,
676 )
677 self.add_message(
678 "typevar-name-incorrect-variance",
679 node=node,
680 args=("",),
681 confidence=interfaces.INFERENCE,
682 )
683 elif variance == TypeVarVariance.covariant and not name.endswith("_co"):
684 suggest_name = f"{re.sub('_contra$', '', name)}_co"
685 self.add_message(
686 "typevar-name-incorrect-variance",
687 node=node,
688 args=(f'. "{name}" is covariant, use "{suggest_name}" instead'),
689 confidence=interfaces.INFERENCE,
690 )
691 elif variance == TypeVarVariance.contravariant and not name.endswith("_contra"):
692 suggest_name = f"{re.sub('_co$', '', name)}_contra"
