129 def ast_from_file(
130 self,
131 filepath: str,
132 modname: str | None = None,
133 fallback: bool = True,
134 source: bool = False,
135 ) -> nodes.Module:
136 """Given a module name, return the astroid object."""
137 if modname is None:
138 try:
139 modname = ".".join(modpath_from_file(filepath))
140 except ImportError:
141 modname = filepath
142 if (
143 modname in self.astroid_cache
144 and self.astroid_cache[modname].file == filepath
145 ):
146 return self.astroid_cache[modname]
147 # Call get_source_file() only after a cache miss,
148 # since it calls os.path.exists().
149 try:
150 filepath = get_source_file(
151 filepath, include_no_ext=True, prefer_stubs=self.prefer_stubs
152 )
153 source = True
154 except NoSourceFile:
155 pass
156 # Second attempt on the cache after get_source_file().
157 if (
158 modname in self.astroid_cache
159 and self.astroid_cache[modname].file == filepath
160 ):
161 return self.astroid_cache[modname]
162 if source:
163 # pylint: disable=import-outside-toplevel; circular import
164 from astroid.builder import AstroidBuilder
165
166 return AstroidBuilder(self).file_build(filepath, modname)
167 if fallback and modname:
168 return self.ast_from_module_name(modname)
169 raise AstroidBuildingError("Unable to build an AST for {path}.", path=filepath)

205 def ast_from_module_name( # noqa: C901
206 self,
207 modname: str | None,
208 context_file: str | None = None,
209 use_cache: bool = True,
210 ) -> nodes.Module:
211 """Given a module name, return the astroid object."""
212 if modname is None:
213 raise AstroidBuildingError("No module name given.")
214 # Sometimes we don't want to use the cache. For example, when we're
215 # importing a module with the same name as the file that is importing
216 # we want to fallback on the import system to make sure we get the correct
217 # module.
218 if modname in self.module_denylist:
219 raise AstroidImportError(f"Skipping ignored module {modname!r}")
220 if modname in self.astroid_cache and use_cache:
221 return self.astroid_cache[modname]
222 if modname == "__main__":
223 return self._build_stub_module(modname)
224 if context_file:
225 old_cwd = os.getcwd()
226 os.chdir(os.path.dirname(context_file))
227 try:
228 found_spec = self.file_from_module_name(modname, context_file)
229 if found_spec.type == spec.ModuleType.PY_ZIPMODULE:
230 module = self.zip_import_data(found_spec.location)
231 if module is not None:
232 return module
233
234 elif found_spec.type in (
235 spec.ModuleType.C_BUILTIN,
236 spec.ModuleType.C_EXTENSION,
237 ):
238 if (
239 found_spec.type == spec.ModuleType.C_EXTENSION
240 and not self._can_load_extension(modname)
241 ):
242 return self._build_stub_module(modname)
243 try:
244 named_module = load_module_from_name(modname)
245 except Exception as e:
246 raise AstroidImportError(
247 "Loading {modname} failed with:\n{error}",
248 modname=modname,
249 path=found_spec.location,
250 ) from e
251 return self.ast_from_module(named_module, modname)
252
253 elif found_spec.type == spec.ModuleType.PY_COMPILED:
254 raise AstroidImportError(
255 "Unable to load compiled module {modname}.",
256 modname=modname,
257 path=found_spec.location,
258 )
259
260 elif found_spec.type == spec.ModuleType.PY_NAMESPACE:
261 return self._build_namespace_module(
262 modname, found_spec.submodule_search_locations or []
263 )
264 elif found_spec.type == spec.ModuleType.PY_FROZEN:
265 if found_spec.location is None:
266 return self._build_stub_module(modname)
267 # For stdlib frozen modules we can determine the location and
268 # can therefore create a module from the source file
269 return self.ast_from_file(found_spec.location, modname, fallback=False)
270
271 if found_spec.location is None:
272 raise AstroidImportError(
273 "Can't find a file for module {modname}.", modname=modname
274 )
275
276 return self.ast_from_file(found_spec.location, modname, fallback=False)
277 except AstroidBuildingError as e:
278 for hook in self._failed_import_hooks:
279 try:
280 return hook(modname)
281 except AstroidBuildingError:
282 pass
283 raise e
284 finally:
285 if context_file:
286 os.chdir(old_cwd)

288 def zip_import_data(self, filepath: str) -> nodes.Module | None:
289 if zipimport is None:
290 return None
291
292 # pylint: disable=import-outside-toplevel; circular import
293 from astroid.builder import AstroidBuilder
294
295 builder = AstroidBuilder(self)
296 for ext in ZIP_IMPORT_EXTS:
297 try:
298 eggpath, resource = filepath.rsplit(ext + os.path.sep, 1)
299 except ValueError:
300 continue
301 try:
302 importer = zipimport.zipimporter(eggpath + ext)
303 zmodname = resource.replace(os.path.sep, ".")
304 if importer.is_package(resource):
305 zmodname = zmodname + ".__init__"
306 module = builder.string_build(
307 importer.get_source(resource), zmodname, filepath
308 )
309 return module
310 except Exception: # pylint: disable=broad-except
311 continue
312 return None

314 def file_from_module_name(
315 self, modname: str, contextfile: str | None
316 ) -> spec.ModuleSpec:
317 try:
318 value = self._mod_file_cache[(modname, contextfile)]
319 except KeyError:
320 try:
321 value = file_info_from_modpath(
322 modname.split("."), context_file=contextfile
323 )
324 except ImportError as e:
325 value = AstroidImportError(
326 "Failed to import module {modname} with error:\n{error}.",
327 modname=modname,
328 # we remove the traceback here to save on memory usage (since these exceptions are cached)
329 error=e.with_traceback(None),
330 )
331 self._mod_file_cache[(modname, contextfile)] = value
332 if isinstance(value, AstroidBuildingError):
333 # we remove the traceback here to save on memory usage (since these exceptions are cached)
334 raise value.with_traceback(None) # pylint: disable=no-member
335 return value

337 def ast_from_module(
338 self, module: types.ModuleType, modname: str | None = None
339 ) -> nodes.Module:
340 """Given an imported module, return the astroid object."""
341 modname = modname or module.__name__
342 if modname in self.astroid_cache:
343 return self.astroid_cache[modname]
344 try:
345 # some builtin modules don't have __file__ attribute
346 filepath = module.__file__
347 if is_python_source(filepath):
348 # Type is checked in is_python_source
349 return self.ast_from_file(filepath, modname) # type: ignore[arg-type]
350 except AttributeError:
351 pass
352
353 # pylint: disable=import-outside-toplevel; circular import
354 from astroid.builder import AstroidBuilder
355
356 return AstroidBuilder(self).module_build(module, modname)

358 def ast_from_class(self, klass: type, modname: str | None = None) -> nodes.ClassDef:
359 """Get astroid for the given class."""
360 if modname is None:
361 try:
362 modname = klass.__module__
363 except AttributeError as exc:
364 raise AstroidBuildingError(
365 "Unable to get module for class {class_name}.",
366 cls=klass,
367 class_repr=safe_repr(klass),
368 modname=modname,
369 ) from exc
370 modastroid = self.ast_from_module_name(modname)
371 ret = modastroid.getattr(klass.__name__)[0]
372 assert isinstance(ret, nodes.ClassDef)
373 return ret

375 def infer_ast_from_something(
376 self, obj: object, context: InferenceContext | None = None
377 ) -> Iterator[InferenceResult]:
378 """Infer astroid for the given class."""
379 if hasattr(obj, "__class__") and not isinstance(obj, type):
380 klass = obj.__class__
381 elif isinstance(obj, type):
382 klass = obj
383 else:
384 raise AstroidBuildingError( # pragma: no cover
385 "Unable to get type for {class_repr}.",
386 cls=None,
387 class_repr=safe_repr(obj),
388 )
389 try:
390 modname = klass.__module__
391 except AttributeError as exc:
392 raise AstroidBuildingError(
393 "Unable to get module for {class_repr}.",
394 cls=klass,
395 class_repr=safe_repr(klass),
396 ) from exc
397 except Exception as exc:
398 raise AstroidImportError(
399 "Unexpected error while retrieving module for {class_repr}:\n"
400 "{error}",
401 cls=klass,
402 class_repr=safe_repr(klass),
403 ) from exc
404 try:
405 name = klass.__name__
406 except AttributeError as exc:
407 raise AstroidBuildingError(
408 "Unable to get name for {class_repr}:\n",
409 cls=klass,
410 class_repr=safe_repr(klass),
411 ) from exc
412 except Exception as exc:
413 raise AstroidImportError(
414 "Unexpected error while retrieving name for {class_repr}:\n{error}",
415 cls=klass,
416 class_repr=safe_repr(klass),
417 ) from exc
418 # take care, on living object __module__ is regularly wrong :(
419 modastroid = self.ast_from_module_name(modname)
420 if klass is obj:
421 yield from modastroid.igetattr(name, context)
422 else:
423 for inferred in modastroid.igetattr(name, context):
424 yield inferred.instantiate_class()
