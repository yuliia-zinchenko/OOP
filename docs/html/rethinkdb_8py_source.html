144 def _get_jobs(self, predicate=None):
145 jobs = []
146 failed_job_ids = []
147 query = (
148 self.table.filter(self.r.row["next_run_time"] != None).filter(predicate)
149 if predicate
150 else self.table
151 )
152 query = query.order_by("next_run_time", "id").pluck("id", "job_state")
153
154 for document in query.run(self.conn):
155 try:
156 jobs.append(self._reconstitute_job(document["job_state"]))
157 except Exception:
158 self._logger.exception(
159 'Unable to restore job "%s" -- removing it', document["id"]
160 )
161 failed_job_ids.append(document["id"])
162
163 # Remove all the jobs we failed to restore
164 if failed_job_ids:
165 self.r.expr(failed_job_ids).for_each(
166 lambda job_id: self.table.get_all(job_id).delete()
167 ).run(self.conn)
