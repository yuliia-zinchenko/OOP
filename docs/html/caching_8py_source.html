18def _get_pdata_path(
19 base_name: Path, recurs: int, pylint_home: Path = PYLINT_HOME_AS_PATH
20) -> Path:
21 # We strip all characters that can't be used in a filename. Also strip '/' and
22 # '\\' because we want to create a single file, not sub-directories.
23 underscored_name = "_".join(
24 str(p.replace(":", "_").replace("/", "_").replace("\\", "_"))
25 for p in base_name.parts
26 )
27 return pylint_home / f"{underscored_name}_{recurs}.stats"

30def load_results(
31 base: str | Path, pylint_home: str | Path = PYLINT_HOME
32) -> LinterStats | None:
33 base = Path(base)
34 pylint_home = Path(pylint_home)
35 data_file = _get_pdata_path(base, 1, pylint_home)
36
37 if not data_file.exists():
38 return None
39
40 try:
41 with open(data_file, "rb") as stream:
42 data = pickle.load(stream)
43 if not isinstance(data, LinterStats):
44 warnings.warn(
45 "You're using an old pylint cache with invalid data following "
46 f"an upgrade, please delete '{data_file}'.",
47 UserWarning,
48 stacklevel=2,
49 )
50 raise TypeError
51 return data
52 except Exception: # pylint: disable=broad-except
53 # There's an issue with the cache but we just continue as if it isn't there
54 return None

57def save_results(
58 results: LinterStats, base: str | Path, pylint_home: str | Path = PYLINT_HOME
59) -> None:
60 base = Path(base)
61 pylint_home = Path(pylint_home)
62 try:
63 pylint_home.mkdir(parents=True, exist_ok=True)
64 except OSError: # pragma: no cover
65 print(f"Unable to create directory {pylint_home}", file=sys.stderr)
66 data_file = _get_pdata_path(base, 1)
67 try:
68 with open(data_file, "wb") as stream:
69 pickle.dump(results, stream)
70 except OSError as ex: # pragma: no cover
71 print(f"Unable to create file {data_file}: {ex}", file=sys.stderr)
