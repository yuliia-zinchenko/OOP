1543 connection.alias,

1623class FSFilesHandler(WSGIHandler):
1624 """
1625 WSGI middleware that intercepts calls to a directory, as defined by one of
1626 the *_ROOT settings, and serves those files, publishing them under *_URL.
1627 """

1632 super().__init__()
1633
1634 def _should_handle(self, path):
1635 """
1636 Check if the path should be handled. Ignore the path if:
1637 * the host is provided as part of the base_url
1638 * the request's path isn't under the media path (or equal)
1639 """
1640 return path.startswith(self.base_url[2]) and not self.base_url[1]
1641
1642 def file_path(self, url):
1643 """Return the relative path to the file on disk for the given URL."""
1644 relative_url = url.removeprefix(self.base_url[2])
1645 return url2pathname(relative_url)
1646
1647 def get_response(self, request):
1648 from django.http import Http404
1649
1650 if self._should_handle(request.path):
1651 try:
1652 return self.serve(request)
1653 except Http404:
1654 pass
1655 return super().get_response(request)
1656
1657 def serve(self, request):
1658 os_rel_path = self.file_path(request.path)
1659 os_rel_path = posixpath.normpath(unquote(os_rel_path))
1660 # Emulate behavior of django.contrib.staticfiles.views.serve() when it
1661 # invokes staticfiles' finders functionality.
1662 # TODO: Modify if/when that internal API is refactored
1663 final_rel_path = os_rel_path.replace("\\", "/").lstrip("/")
1664 return serve(request, final_rel_path, document_root=self.get_base_dir())
1665
1666 def __call__(self, environ, start_response):

1668 return self.application(environ, start_response)

1715 HTTP requests.
