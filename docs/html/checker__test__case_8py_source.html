40 @contextlib.contextmanager
41 def assertAddsMessages(
42 self, *messages: MessageTest, ignore_position: bool = False
43 ) -> Generator[None]:
44 """Assert that exactly the given method adds the given messages.
45
46 The list of messages must exactly match *all* the messages added by the
47 method. Additionally, we check to see whether the args in each message can
48 actually be substituted into the message string.
49
50 Using the keyword argument `ignore_position`, all checks for position
51 arguments (line, col_offset, ...) will be skipped. This can be used to
52 just test messages for the correct node.
53 """
54 yield
55 got = self.linter.release_messages()
56 no_msg = "No message."
57 expected = "\n".join(repr(m) for m in messages) or no_msg
58 got_str = "\n".join(repr(m) for m in got) or no_msg
59 msg = (
60 "Expected messages did not match actual.\n"
61 f"\nExpected:\n{expected}\n\nGot:\n{got_str}\n"
62 )
63
64 assert len(messages) == len(got), msg
65
66 for expected_msg, gotten_msg in zip(messages, got):
67 assert expected_msg.msg_id == gotten_msg.msg_id, msg
68 assert expected_msg.node == gotten_msg.node, msg
69 assert expected_msg.args == gotten_msg.args, msg
70 assert expected_msg.confidence == gotten_msg.confidence, msg
71
72 if ignore_position:
73 # Do not check for line, col_offset etc...
74 continue
75
76 assert expected_msg.line == gotten_msg.line, msg
77 assert expected_msg.col_offset == gotten_msg.col_offset, msg
78 assert expected_msg.end_line == gotten_msg.end_line, msg
79 assert expected_msg.end_col_offset == gotten_msg.end_col_offset, msg
