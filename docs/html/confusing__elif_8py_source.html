18class ConfusingConsecutiveElifChecker(BaseChecker):
19 """Checks if "elif" is used right after an indented block that finishes with "if" or
20 "elif" itself.
21 """
22
23 name = "confusing_elif"
24 msgs = {
25 "R5601": (
26 "Consecutive elif with differing indentation level, consider creating a function to separate the inner"
27 " elif",
28 "confusing-consecutive-elif",
29 "Used when an elif statement follows right after an indented block which itself ends with if or elif. "
30 "It may not be obvious if the elif statement was willingly or mistakenly unindented. "
31 "Extracting the indented if statement into a separate function might avoid confusion and prevent "
32 "errors.",
33 )
34 }

36 @only_required_for_messages("confusing-consecutive-elif")
37 def visit_if(self, node: nodes.If) -> None:
38 body_ends_with_if = isinstance(
39 node.body[-1], nodes.If
40 ) and self._has_no_else_clause(node.body[-1])
41 if node.has_elif_block() and body_ends_with_if:
42 self.add_message("confusing-consecutive-elif", node=node.orelse[0])
43
44 @staticmethod
45 def _has_no_else_clause(node: nodes.If) -> bool:
46 orelse = node.orelse
47 while orelse and isinstance(orelse[0], nodes.If):
48 orelse = orelse[0].orelse
49 if not orelse or isinstance(orelse[0], nodes.If):
50 return True
51 return False
