3"""
4This module contains a base type which provides list-style mutations
5without specific data storage methods.
6
7See also http://static.aryehleib.com/oldsite/MutableLists.html
8
9Author: Aryeh Leib Taurog.
10"""
11from functools import total_ordering
12
13
14@total_ordering
15class ListMixin:
16 """
17 A base class which provides complete list interface.
18 Derived classes must call ListMixin's __init__() function
19 and implement the following:
20
21 function _get_single_external(self, i):
22 Return single item with index i for general use.
23 The index i will always satisfy 0 <= i < len(self).
24
25 function _get_single_internal(self, i):
26 Same as above, but for use within the class [Optional]
27 Note that if _get_single_internal and _get_single_internal return
28 different types of objects, _set_list must distinguish
29 between the two and handle each appropriately.
30
31 function _set_list(self, length, items):
32 Recreate the entire object.
33
34 NOTE: items may be a generator which calls _get_single_internal.
35 Therefore, it is necessary to cache the values in a temporary:
36 temp = list(items)
37 before clobbering the original storage.
38
39 function _set_single(self, i, value):
40 Set the single item at index i to value [Optional]
41 If left undefined, all mutations will result in rebuilding
42 the object using _set_list.
43
44 function __len__(self):
45 Return the length
46
47 int _minlength:
48 The minimum legal length [Optional]
49
50 int _maxlength:
51 The maximum legal length [Optional]
52
53 type or tuple _allowed:
54 A type or tuple of allowed item types [Optional]
55 """
56
57 _minlength = 0
58 _maxlength = None
59
60 # ### Python initialization and special list interface methods ###
61
62 def __init__(self, *args, **kwargs):
63 if not hasattr(self, "_get_single_internal"):
64 self._get_single_internal = self._get_single_external
65
66 if not hasattr(self, "_set_single"):
67 self._set_single = self._set_single_rebuild
68 self._assign_extended_slice = self._assign_extended_slice_rebuild
69
70 super().__init__(*args, **kwargs)
71
72 def __getitem__(self, index):
73 "Get the item(s) at the specified index/slice."
74 if isinstance(index, slice):
75 return [
76 self._get_single_external(i) for i in range(*index.indices(len(self)))
77 ]
78 else:
79 index = self._checkindex(index)
80 return self._get_single_external(index)
81
82 def __delitem__(self, index):
83 "Delete the item(s) at the specified index/slice."
84 if not isinstance(index, (int, slice)):
85 raise TypeError("%s is not a legal index" % index)
86
87 # calculate new length and dimensions
88 origLen = len(self)
89 if isinstance(index, int):
90 index = self._checkindex(index)
91 indexRange = [index]
92 else:
93 indexRange = range(*index.indices(origLen))
94
95 newLen = origLen - len(indexRange)
96 newItems = (
97 self._get_single_internal(i) for i in range(origLen) if i not in indexRange
98 )
99
100 self._rebuild(newLen, newItems)
101
102 def __setitem__(self, index, val):
103 "Set the item(s) at the specified index/slice."
104 if isinstance(index, slice):
105 self._set_slice(index, val)
106 else:
107 index = self._checkindex(index)
108 self._check_allowed((val,))
109 self._set_single(index, val)

111 # ### Special methods for arithmetic operations ###
112 def __add__(self, other):
113 "add another list-like object"
114 return self.__class__([*self, *other])
115
116 def __radd__(self, other):
117 "add to another list-like object"
118 return other.__class__([*other, *self])
119
120 def __iadd__(self, other):
121 "add another list-like object to self"
122 self.extend(other)
123 return self
124
125 def __mul__(self, n):
126 "multiply"
127 return self.__class__(list(self) * n)
128
129 def __rmul__(self, n):
130 "multiply"
131 return self.__class__(list(self) * n)
132
133 def __imul__(self, n):
134 "multiply"
135 if n <= 0:
136 del self[:]
137 else:
138 cache = list(self)
139 for i in range(n - 1):
140 self.extend(cache)
141 return self
142
143 def __eq__(self, other):
144 olen = len(other)
145 for i in range(olen):
146 try:
147 c = self[i] == other[i]
148 except IndexError:
149 # self must be shorter
150 return False
151 if not c:
152 return False
153 return len(self) == olen
154
155 def __lt__(self, other):
156 olen = len(other)
157 for i in range(olen):
158 try:
159 c = self[i] < other[i]
160 except IndexError:
161 # self must be shorter
162 return True
163 if c:
164 return c
165 elif other[i] < self[i]:
166 return False
167 return len(self) < olen

169 # ### Public list interface Methods ###
170 # ## Non-mutating ##
171 def count(self, val):
172 "Standard list count method"
173 count = 0
174 for i in self:
175 if val == i:
176 count += 1
177 return count
178
179 def index(self, val):
180 "Standard list index method"
181 for i in range(0, len(self)):
182 if self[i] == val:
183 return i
184 raise ValueError("%s not found in object" % val)

186 # ## Mutating ##
187 def append(self, val):
188 "Standard list append method"
189 self[len(self) :] = [val]
190
191 def extend(self, vals):
192 "Standard list extend method"
193 self[len(self) :] = vals
194
195 def insert(self, index, val):
196 "Standard list insert method"
197 if not isinstance(index, int):

199 self[index:index] = [val]

201 def pop(self, index=-1):
202 "Standard list pop method"
203 result = self[index]

208 "Standard list remove method"

211 def reverse(self):

219 # ### Private routines ###
220 def _rebuild(self, newLen, newItems):
221 if newLen and newLen < self._minlength:
222 raise ValueError("Must have at least %d items" % self._minlength)
223 if self._maxlength is not None and newLen > self._maxlength:
224 raise ValueError("Cannot have more than %d items" % self._maxlength)
225
226 self._set_list(newLen, newItems)
227
228 def _set_single_rebuild(self, index, value):
229 self._set_slice(slice(index, index + 1, 1), [value])
230
231 def _checkindex(self, index):
232 length = len(self)
233 if 0 <= index < length:
234 return index
235 if -length <= index < 0:
236 return index + length
237 raise IndexError("invalid index: %s" % index)
238
239 def _check_allowed(self, items):
240 if hasattr(self, "_allowed"):
241 if False in [isinstance(val, self._allowed) for val in items]:
242 raise TypeError("Invalid type encountered in the arguments.")
243
244 def _set_slice(self, index, values):
245 "Assign values to a slice of the object"
246 try:
247 valueList = list(values)
248 except TypeError:
249 raise TypeError("can only assign an iterable to a slice")
250
251 self._check_allowed(valueList)
252
253 origLen = len(self)
254 start, stop, step = index.indices(origLen)
255
256 # CAREFUL: index.step and step are not the same!
257 # step will never be None
258 if index.step is None:
259 self._assign_simple_slice(start, stop, valueList)
260 else:
261 self._assign_extended_slice(start, stop, step, valueList)
262
263 def _assign_extended_slice_rebuild(self, start, stop, step, valueList):
264 "Assign an extended slice by rebuilding entire list"
265 indexList = range(start, stop, step)
266 # extended slice, only allow assigning slice of same size
267 if len(valueList) != len(indexList):
268 raise ValueError(
269 "attempt to assign sequence of size %d "
270 "to extended slice of size %d" % (len(valueList), len(indexList))
271 )
272
273 # we're not changing the length of the sequence
274 newLen = len(self)
275 newVals = dict(zip(indexList, valueList))
276
277 def newItems():
278 for i in range(newLen):
279 if i in newVals:
280 yield newVals[i]
281 else:
282 yield self._get_single_internal(i)
283
284 self._rebuild(newLen, newItems())
285
286 def _assign_extended_slice(self, start, stop, step, valueList):
287 "Assign an extended slice by re-assigning individual items"
288 indexList = range(start, stop, step)
289 # extended slice, only allow assigning slice of same size
290 if len(valueList) != len(indexList):
291 raise ValueError(
292 "attempt to assign sequence of size %d "
293 "to extended slice of size %d" % (len(valueList), len(indexList))
294 )
295
296 for i, val in zip(indexList, valueList):
297 self._set_single(i, val)
298
299 def _assign_simple_slice(self, start, stop, valueList):
300 "Assign a simple slice; Can assign slice of any length"
301 origLen = len(self)
302 stop = max(start, stop)
303 newLen = origLen - stop + start + len(valueList)
304
305 def newItems():
306 for i in range(origLen + 1):
307 if i == start:
308 yield from valueList
309
310 if i < origLen:
311 if i < start or i >= stop:
312 yield self._get_single_internal(i)
313
314 self._rebuild(newLen, newItems())
