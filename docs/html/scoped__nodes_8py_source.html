76def _c3_merge(sequences, cls, context):
77 """Merges MROs in *sequences* to a single MRO using the C3 algorithm.
78
79 Adapted from http://www.python.org/download/releases/2.3/mro/.
80
81 """
82 result = []
83 while True:
84 sequences = [s for s in sequences if s] # purge empty sequences
85 if not sequences:
86 return result
87 for s1 in sequences: # find merge candidates among seq heads
88 candidate = s1[0]
89 for s2 in sequences:
90 if candidate in s2[1:]:
91 candidate = None
92 break # reject the current head, it appears later
93 else:
94 break
95 if not candidate:
96 # Show all the remaining bases, which were considered as
97 # candidates for the next mro sequence.
98 raise InconsistentMroError(
99 message="Cannot create a consistent method resolution order "
100 "for MROs {mros} of class {cls!r}.",
101 mros=sequences,
102 cls=cls,
103 context=context,
104 )
105
106 result.append(candidate)
107 # remove the chosen candidate
108 for seq in sequences:
109 if seq[0] == candidate:
110 del seq[0]
111 return None

114def clean_typing_generic_mro(sequences: list[list[ClassDef]]) -> None:
115 """A class can inherit from typing.Generic directly, as base,
116 and as base of bases. The merged MRO must however only contain the last entry.
117 To prepare for _c3_merge, remove some typing.Generic entries from
118 sequences if multiple are present.
119
120 This method will check if Generic is in inferred_bases and also
121 part of bases_mro. If true, remove it from inferred_bases
122 as well as its entry the bases_mro.
123
124 Format sequences: [[self]] + bases_mro + [inferred_bases]
125 """
126 bases_mro = sequences[1:-1]
127 inferred_bases = sequences[-1]
128 # Check if Generic is part of inferred_bases
129 for i, base in enumerate(inferred_bases):
130 if base.qname() == "typing.Generic":
131 position_in_inferred_bases = i
132 break
133 else:
134 return
135 # Check if also part of bases_mro
136 # Ignore entry for typing.Generic
137 for i, seq in enumerate(bases_mro):
138 if i == position_in_inferred_bases:
139 continue
140 if any(base.qname() == "typing.Generic" for base in seq):
141 break
142 else:
143 return
144 # Found multiple Generics in mro, remove entry from inferred_bases
145 # and the corresponding one from bases_mro
146 inferred_bases.pop(position_in_inferred_bases)
147 bases_mro.pop(position_in_inferred_bases)

150def clean_duplicates_mro(
151 sequences: list[list[ClassDef]],
152 cls: ClassDef,
153 context: InferenceContext | None,
154) -> list[list[ClassDef]]:
155 for sequence in sequences:
156 seen = set()
157 for node in sequence:
158 lineno_and_qname = (node.lineno, node.qname())
159 if lineno_and_qname in seen:
160 raise DuplicateBasesError(
161 message="Duplicates found in MROs {mros} for {cls!r}.",
162 mros=sequences,
163 cls=cls,
164 context=context,
165 )
166 seen.add(lineno_and_qname)
167 return sequences

476 def relative_to_absolute_name(self, modname: str, level: int | None) -> str:
477 """Get the absolute module name for a relative import.
478
479 The relative import can be implicit or explicit.
480
481 :param modname: The module name to convert.
482
483 :param level: The level of relative import.
484
485 :returns: The absolute module name.
486
487 :raises TooManyLevelsError: When the relative import refers to a
488 module too far above this one.
489 """
490 # XXX this returns non sens when called on an absolute import
491 # like 'pylint.checkers.astroid.utils'
492 # XXX doesn't return absolute name if self.name isn't absolute name
493 if self.absolute_import_activated() and level is None:
494 return modname
495 if level:
496 if self.package:
497 level = level - 1
498 package_name = self.name.rsplit(".", level)[0]
499 elif (
500 self.path
501 and not os.path.exists(os.path.dirname(self.path[0]) + "/__init__.py")
502 and os.path.exists(
503 os.path.dirname(self.path[0]) + "/" + modname.split(".")[0]
504 )
505 ):
506 level = level - 1
507 package_name = ""
508 else:
509 package_name = self.name.rsplit(".", level)[0]
510 if level and self.name.count(".") < level:
511 raise TooManyLevelsError(level=level, name=self.name)
512
513 elif self.package:
514 package_name = self.name
515 else:
516 package_name = self.name.rsplit(".", 1)[0]
517
518 if package_name:
519 if not modname:
520 return package_name
521 return f"{package_name}.{modname}"
522 return modname

524 def wildcard_import_names(self):
525 """The list of imported names when this module is 'wildcard imported'.
526
527 It doesn't include the '__builtins__' name which is added by the
528 current CPython implementation of wildcard imports.
529
530 :returns: The list of imported names.
531 :rtype: list(str)
532 """
533 # We separate the different steps of lookup in try/excepts
534 # to avoid catching too many Exceptions
535 default = [name for name in self.keys() if not name.startswith("_")]
536 try:
537 all_values = self["__all__"]
538 except KeyError:
539 return default
540
541 try:
542 explicit = next(all_values.assigned_stmts())
543 except (InferenceError, StopIteration):
544 return default
545 except AttributeError:
546 # not an assignment node
547 # XXX infer?
548 return default
549
550 # Try our best to detect the exported name.
551 inferred = []
552 try:
553 explicit = next(explicit.infer())
554 except (InferenceError, StopIteration):
555 return default
556 if not isinstance(explicit, (node_classes.Tuple, node_classes.List)):
557 return default
558
559 def str_const(node) -> bool:
560 return isinstance(node, node_classes.Const) and isinstance(node.value, str)
561
562 for node in explicit.elts:
563 if str_const(node):
564 inferred.append(node.value)
565 else:
566 try:
567 inferred_node = next(node.infer())
568 except (InferenceError, StopIteration):
569 continue
570 if str_const(inferred_node):
571 inferred.append(inferred_node.value)
572 return inferred

1513 def _infer(

1515 ) -> Generator[objects.Property | FunctionDef, None, InferenceErrorInfo]:
1516 from astroid import objects # pylint: disable=import-outside-toplevel
1517
1518 if not self.decorators or not bases._is_property(self):
1519 yield self

1521
1522 # When inferring a property, we instantiate a new `objects.Property` object,
1523 # which in turn, because it inherits from `FunctionDef`, sets itself in the locals
1524 # of the wrapping frame. This means that every time we infer a property, the locals
1525 # are mutated with a new instance of the property. To avoid this, we detect this
1526 # scenario and avoid passing the `parent` argument to the constructor.
1527 if not self.parent:
1528 raise ParentMissingError(target=self)
1529 parent_frame = self.parent.frame()
1530 property_already_in_parent_locals = self.name in parent_frame.locals and any(
1531 isinstance(val, objects.Property) for val in parent_frame.locals[self.name]
1532 )
1533 # We also don't want to pass parent if the definition is within a Try node
1534 if isinstance(
1535 self.parent,
1536 (node_classes.Try, node_classes.If),
1537 ):
1538 property_already_in_parent_locals = True
1539
1540 prop_func = objects.Property(
1541 function=self,
1542 name=self.name,
1543 lineno=self.lineno,
1544 parent=self.parent if not property_already_in_parent_locals else None,
1545 col_offset=self.col_offset,
1546 )
1547 if property_already_in_parent_locals:
1548 prop_func.parent = self.parent
1549 prop_func.postinit(body=[], args=self.args, doc_node=self.doc_node)
1550 yield prop_func

2386 found, but a ``__getattr__`` or ``__getattribute__`` method is defined.

2388 attribute is accessed from a class context,
2389 e.g. ClassDef.attribute, otherwise it might have been accessed

2402 if not name:

2406 values: list[InferenceResult] = list(self.locals.get(name, []))
2407 for classnode in self.ancestors(recurs=True, context=context):
2408 values += classnode.locals.get(name, [])
2409

2411 result = [self.special_attributes.lookup(name)]

2414 # and we need to return all the values.
2415 result += values

2499 # to the attribute happening *after* the attribute's definition (e.g. AugAssigns on lists)
