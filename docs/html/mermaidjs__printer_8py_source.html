13class MermaidJSPrinter(Printer):
14 """Printer for MermaidJS diagrams."""
15
16 DEFAULT_COLOR = "black"
17
18 NODES: dict[NodeType, str] = {
19 NodeType.CLASS: "class",
20 NodeType.PACKAGE: "class",
21 }
22 ARROWS: dict[EdgeType, str] = {
23 EdgeType.INHERITS: "--|>",
24 EdgeType.ASSOCIATION: "--*",
25 EdgeType.AGGREGATION: "--o",
26 EdgeType.USES: "-->",
27 EdgeType.TYPE_DEPENDENCY: "..>",
28 }
29
30 def _open_graph(self) -> None:
31 """Emit the header lines."""
32 self.emit("classDiagram")
33 self._inc_indent()
34
35 def emit_node(
36 self,
37 name: str,
38 type_: NodeType,
39 properties: NodeProperties | None = None,
40 ) -> None:
41 """Create a new node.
42
43 Nodes can be classes, packages, participants etc.
44 """
45 # pylint: disable=duplicate-code
46 if properties is None:
47 properties = NodeProperties(label=name)
48 nodetype = self.NODES[type_]
49 body = []
50 if properties.attrs:
51 body.extend(properties.attrs)
52 if properties.methods:
53 for func in properties.methods:
54 args = self._get_method_arguments(func)
55 line = f"{func.name}({', '.join(args)})"
56 line += "*" if func.is_abstract() else ""
57 if func.returns:
58 line += f" {get_annotation_label(func.returns)}"
59 body.append(line)
60 name = name.split(".")[-1]
61 self.emit(f"{nodetype} {name} {{")
62 self._inc_indent()
63 for line in body:
64 self.emit(line)
65 self._dec_indent()
66 self.emit("}")
67
68 def emit_edge(
69 self,
70 from_node: str,
71 to_node: str,
72 type_: EdgeType,
73 label: str | None = None,
74 ) -> None:
75 """Create an edge from one node to another to display relationships."""
76 from_node = from_node.split(".")[-1]
77 to_node = to_node.split(".")[-1]
78 edge = f"{from_node} {self.ARROWS[type_]} {to_node}"
79 if label:
80 edge += f" : {label}"
81 self.emit(edge)
82
83 def _close_graph(self) -> None:
84 """Emit the lines needed to properly close the graph."""
85 self._dec_indent()
