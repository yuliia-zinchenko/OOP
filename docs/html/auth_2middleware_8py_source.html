91class RemoteUserMiddleware(MiddlewareMixin):
92 """
93 Middleware for utilizing web-server-provided authentication.
94
95 If request.user is not authenticated, then this middleware attempts to
96 authenticate the username passed in the ``REMOTE_USER`` request header.
97 If authentication is successful, the user is automatically logged in to
98 persist the user in the session.
99
100 The header used is configurable and defaults to ``REMOTE_USER``. Subclass
101 this class and change the ``header`` attribute if you need to use a
102 different header.
103 """
104
105 # Name of request header to grab username from. This will be the key as
106 # used in the request.META dictionary, i.e. the normalization of headers to
107 # all uppercase and the addition of "HTTP_" prefix apply.
108 header = "REMOTE_USER"
109 force_logout_if_no_header = True
110
111 def process_request(self, request):
112 # AuthenticationMiddleware is required so that request.user exists.
113 if not hasattr(request, "user"):
114 raise ImproperlyConfigured(
115 "The Django remote user auth middleware requires the"
116 " authentication middleware to be installed. Edit your"
117 " MIDDLEWARE setting to insert"
118 " 'django.contrib.auth.middleware.AuthenticationMiddleware'"
119 " before the RemoteUserMiddleware class."
120 )
121 try:
122 username = request.META[self.header]
123 except KeyError:
124 # If specified header doesn't exist then remove any existing
125 # authenticated remote-user, or return (leaving request.user set to
126 # AnonymousUser by the AuthenticationMiddleware).
127 if self.force_logout_if_no_header and request.user.is_authenticated:
128 self._remove_invalid_user(request)
129 return
130 # If the user is already authenticated and that user is the user we are
131 # getting passed in the headers, then the correct user is already
132 # persisted in the session and we don't need to continue.
133 if request.user.is_authenticated:
134 if request.user.get_username() == self.clean_username(username, request):
135 return
136 else:
137 # An authenticated user is associated with the request, but
138 # it does not match the authorized user in the header.
139 self._remove_invalid_user(request)
140
141 # We are seeing this user for the first time in this session, attempt
142 # to authenticate the user.
143 user = auth.authenticate(request, remote_user=username)
144 if user:
145 # User is valid. Set request.user and persist user in the session
146 # by logging the user in.
147 request.user = user
148 auth.login(request, user)
149
150 def clean_username(self, username, request):
151 """
152 Allow the backend to clean the username, if the backend defines a
153 clean_username method.
154 """
155 backend_str = request.session[auth.BACKEND_SESSION_KEY]
156 backend = auth.load_backend(backend_str)
157 try:
158 username = backend.clean_username(username)
159 except AttributeError: # Backend has no clean_username method.
160 pass
161 return username
162
163 def _remove_invalid_user(self, request):
164 """
165 Remove the current authenticated user in the request which is invalid
166 but only if the user is authenticated via the RemoteUserBackend.
167 """
168 try:
169 stored_backend = load_backend(
170 request.session.get(auth.BACKEND_SESSION_KEY, "")
171 )
172 except ImportError:
173 # backend failed to load
174 auth.logout(request)
175 else:
176 if isinstance(stored_backend, RemoteUserBackend):
177 auth.logout(request)
