383 @async_unsafe
384 def savepoint(self):
385 """
386 Create a savepoint inside the current transaction. Return an
387 identifier for the savepoint that will be used for the subsequent
388 rollback or commit. Do nothing if savepoints are not supported.
389 """
390 if not self._savepoint_allowed():
391 return
392
393 thread_ident = _thread.get_ident()
394 tid = str(thread_ident).replace("-", "")
395
396 self.savepoint_state += 1
397 sid = "s%s_x%d" % (tid, self.savepoint_state)
398
399 self.validate_thread_sharing()
400 self._savepoint(sid)
401
402 return sid
403
404 @async_unsafe
405 def savepoint_rollback(self, sid):
406 """
407 Roll back to a savepoint. Do nothing if savepoints are not supported.
408 """
409 if not self._savepoint_allowed():
410 return
411
412 self.validate_thread_sharing()
413 self._savepoint_rollback(sid)
414
415 # Remove any callbacks registered while this savepoint was active.
416 self.run_on_commit = [
417 (sids, func, robust)
418 for (sids, func, robust) in self.run_on_commit
419 if sid not in sids

457 def set_autocommit(
458 self, autocommit, force_begin_transaction_with_broken_autocommit=False
459 ):
460 """
461 Enable or disable autocommit.
462
463 The usual way to start a transaction is to turn autocommit off.
464 SQLite does not properly start a transaction when disabling
465 autocommit. To avoid this buggy behavior and to actually enter a new
466 transaction, an explicit BEGIN is required. Using
467 force_begin_transaction_with_broken_autocommit=True will issue an
468 explicit BEGIN with SQLite. This option will be ignored for other
469 backends.
470 """
471 self.validate_no_atomic_block()
472 self.close_if_health_check_failed()
473 self.ensure_connection()
474
475 start_transaction_under_autocommit = (
476 force_begin_transaction_with_broken_autocommit
477 and not autocommit
478 and hasattr(self, "_start_transaction_under_autocommit")
479 )
480
481 if start_transaction_under_autocommit:
482 self._start_transaction_under_autocommit()
483 elif autocommit:
484 self._set_autocommit(autocommit)
485 else:
486 with debug_transaction(self, "BEGIN"):
487 self._set_autocommit(autocommit)
488 self.autocommit = autocommit
489
490 if autocommit and self.run_commit_hooks_on_set_autocommit_on:
491 self.run_and_clear_commit_hooks()
492 self.run_commit_hooks_on_set_autocommit_on = False

583 or self.health_check_done

591 def close_if_unusable_or_obsolete(self):
592 """
593 Close the current connection if unrecoverable errors have occurred
594 or if it outlived its maximum age.
595 """
596 if self.connection is not None:
597 self.health_check_done = False
598 # If the application didn't restore the original autocommit setting,
599 # don't take chances, drop the connection.
600 if self.get_autocommit() != self.settings_dict["AUTOCOMMIT"]:
601 self.close()
602 return
603
604 # If an exception other than DataError or IntegrityError occurred
605 # since the last commit / rollback, check if the connection works.
606 if self.errors_occurred:
607 if self.is_usable():
608 self.errors_occurred = False
609 self.health_check_done = True
610 else:
611 self.close()
612 return
613
614 if self.close_at is not None and time.monotonic() >= self.close_at:
615 self.close()
616 return

727 def on_commit(self, func, robust=False):
728 if not callable(func):
729 raise TypeError("on_commit()'s callback must be a callable.")
730 if self.in_atomic_block:
731 # Transaction in progress; save for execution on commit.
732 self.run_on_commit.append((set(self.savepoint_ids), func, robust))
733 elif not self.get_autocommit():
734 raise TransactionManagementError(
735 "on_commit() cannot be used in manual transaction management"
736 )
737 else:
738 # No transaction in progress and in autocommit mode; execute
739 # immediately.
740 if robust:
741 try:
742 func()
743 except Exception as e:
744 logger.error(
745 f"Error calling {func.__qualname__} in on_commit() (%s).",
746 e,
747 exc_info=True,
748 )
749 else:
750 func()

752 def run_and_clear_commit_hooks(self):
753 self.validate_no_atomic_block()
