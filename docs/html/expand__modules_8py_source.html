27def discover_package_path(modulepath: str, source_roots: Sequence[str]) -> str:
28 """Discover package path from one its modules and source roots."""
29 dirname = os.path.realpath(os.path.expanduser(modulepath))
30 if not os.path.isdir(dirname):
31 dirname = os.path.dirname(dirname)
32
33 # Look for a source root that contains the module directory
34 for source_root in source_roots:
35 source_root = os.path.realpath(os.path.expanduser(source_root))
36 if os.path.commonpath([source_root, dirname]) in [dirname, source_root]:
37 return source_root
38
39 # Fall back to legacy discovery by looking for __init__.py upwards as
40 # it's the only way given that source root was not found or was not provided
41 while True:
42 if not os.path.exists(os.path.join(dirname, "__init__.py")):
43 return dirname
44 old_dirname = dirname
45 dirname = os.path.dirname(dirname)
46 if old_dirname == dirname:
47 return os.getcwd()

50def _is_in_ignore_list_re(element: str, ignore_list_re: list[Pattern[str]]) -> bool:
51 """Determines if the element is matched in a regex ignore-list."""

71def expand_modules(
72 files_or_modules: Sequence[str],
73 source_roots: Sequence[str],
74 ignore_list: list[str],
75 ignore_list_re: list[Pattern[str]],
76 ignore_list_paths_re: list[Pattern[str]],
77) -> tuple[dict[str, ModuleDescriptionDict], list[ErrorDescriptionDict]]:
78 """Take a list of files/modules/packages and return the list of tuple
79 (file, module name) which have to be actually checked.
80 """
81 result: dict[str, ModuleDescriptionDict] = {}
82 errors: list[ErrorDescriptionDict] = []
83 path = sys.path.copy()
84
85 for something in files_or_modules:
86 basename = os.path.basename(something)
87 if _is_ignored_file(
88 something, ignore_list, ignore_list_re, ignore_list_paths_re
89 ):
90 result[something] = {
91 "path": something,
92 "name": "",
93 "isarg": False,
94 "basepath": something,
95 "basename": "",
96 "isignored": True,
97 }
98 continue
99 module_package_path = discover_package_path(something, source_roots)
100 additional_search_path = [".", module_package_path, *path]
101 if os.path.exists(something):
102 # this is a file or a directory
103 try:
104 modname = ".".join(
105 modutils.modpath_from_file(something, path=additional_search_path)
106 )
107 except ImportError:
108 modname = os.path.splitext(basename)[0]
109 if os.path.isdir(something):
110 filepath = os.path.join(something, "__init__.py")
111 else:
112 filepath = something
113 else:
114 # suppose it's a module or package
115 modname = something
116 try:
117 filepath = modutils.file_from_modpath(
118 modname.split("."), path=additional_search_path
119 )
120 if filepath is None:
121 continue
122 except ImportError as ex:
123 errors.append({"key": "fatal", "mod": modname, "ex": ex})
124 continue
125 filepath = os.path.normpath(filepath)
126 modparts = (modname or something).split(".")
127 try:
128 spec = modutils.file_info_from_modpath(
129 modparts, path=additional_search_path
130 )
131 except ImportError:
132 # Might not be acceptable, don't crash.
133 is_namespace = not os.path.exists(filepath)
134 is_directory = os.path.isdir(something)
135 else:
136 is_namespace = modutils.is_namespace(spec)
137 is_directory = modutils.is_directory(spec)
138 if not is_namespace:
139 if filepath in result:
140 # Always set arg flag if module explicitly given.
141 result[filepath]["isarg"] = True
142 else:
143 result[filepath] = {
144 "path": filepath,
145 "name": modname,
146 "isarg": True,
147 "basepath": filepath,
148 "basename": modname,
149 "isignored": False,
150 }
151 has_init = (
152 not (modname.endswith(".__init__") or modname == "__init__")
153 and os.path.basename(filepath) == "__init__.py"
154 )
155 if has_init or is_namespace or is_directory:
156 for subfilepath in modutils.get_module_files(
157 os.path.dirname(filepath) or ".", ignore_list, list_all=is_namespace
158 ):
159 subfilepath = os.path.normpath(subfilepath)
160 if filepath == subfilepath:
161 continue
162 if _is_in_ignore_list_re(
163 os.path.basename(subfilepath), ignore_list_re
164 ) or _is_in_ignore_list_re(subfilepath, ignore_list_paths_re):
165 result[subfilepath] = {
166 "path": subfilepath,
167 "name": "",
168 "isarg": False,
169 "basepath": subfilepath,
170 "basename": "",
171 "isignored": True,
172 }
173 continue
174
175 modpath = _modpath_from_file(
176 subfilepath, is_namespace, path=additional_search_path
177 )
178 submodname = ".".join(modpath)
179 # Preserve arg flag if module is also explicitly given.
180 isarg = subfilepath in result and result[subfilepath]["isarg"]
181 result[subfilepath] = {
182 "path": subfilepath,
183 "name": submodname,
184 "isarg": isarg,
185 "basepath": filepath,
186 "basename": modname,
187 "isignored": False,
188 }
189 return result, errors
