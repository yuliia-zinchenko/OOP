45 def start_object(self, obj):
46 """
47 Called as each object is handled.
48 """
49 if not hasattr(obj, "_meta"):
50 raise base.SerializationError(
51 "Non-model object (%s) encountered during serialization" % type(obj)
52 )
53
54 self.indent(1)
55 attrs = {"model": str(obj._meta)}
56 if not self.use_natural_primary_keys or not hasattr(obj, "natural_key"):
57 obj_pk = obj.pk
58 if obj_pk is not None:
59 attrs["pk"] = str(obj_pk)
60
61 self.xml.startElement("object", attrs)

63 def end_object(self, obj):

65 Called after handling all fields for an object.
66 """

70 def handle_field(self, obj, field):
71 """
72 Handle each field on an object (except for ForeignKeys and
73 ManyToManyFields).
74 """
75 self.indent(2)
76 self.xml.startElement(
77 "field",
78 {
79 "name": field.name,
80 "type": field.get_internal_type(),
81 },
82 )
83
84 # Get a "string version" of the object's data.
85 if getattr(obj, field.name) is not None:
86 value = field.value_to_string(obj)
87 if field.get_internal_type() == "JSONField":
88 # Dump value since JSONField.value_to_string() doesn't output
89 # strings.
90 value = json.dumps(value, cls=field.encoder)
91 try:
92 self.xml.characters(value)
93 except UnserializableContentError:
94 raise ValueError(
95 "%s.%s (pk:%s) contains unserializable characters"
96 % (obj.__class__.__name__, field.name, obj.pk)
97 )
98 else:
99 self.xml.addQuickElement("None")
100
101 self.xml.endElement("field")

103 def handle_fk_field(self, obj, field):
104 """
105 Handle a ForeignKey (they need to be treated slightly
106 differently from regular fields).
107 """
108 self._start_relational_field(field)
109 related_att = getattr(obj, field.attname)
110 if related_att is not None:
111 if self.use_natural_foreign_keys and hasattr(
112 field.remote_field.model, "natural_key"
113 ):
114 related = getattr(obj, field.name)
115 # If related object has a natural key, use it
116 related = related.natural_key()
117 # Iterable natural keys are rolled out as subelements
118 for key_value in related:
119 self.xml.startElement("natural", {})
120 self.xml.characters(str(key_value))
121 self.xml.endElement("natural")
122 else:
123 self.xml.characters(str(related_att))
124 else:
125 self.xml.addQuickElement("None")
126 self.xml.endElement("field")

128 def handle_m2m_field(self, obj, field):
129 """
130 Handle a ManyToManyField. Related objects are only serialized as
131 references to the object's PK (i.e. the related *data* is not dumped,
132 just the relation).
133 """
134 if field.remote_field.through._meta.auto_created:
135 self._start_relational_field(field)
136 if self.use_natural_foreign_keys and hasattr(
137 field.remote_field.model, "natural_key"
138 ):
139 # If the objects in the m2m have a natural key, use it
140 def handle_m2m(value):
141 natural = value.natural_key()
142 # Iterable natural keys are rolled out as subelements
143 self.xml.startElement("object", {})
144 for key_value in natural:
145 self.xml.startElement("natural", {})
146 self.xml.characters(str(key_value))
147 self.xml.endElement("natural")
148 self.xml.endElement("object")

196 ignorenonexistent=False,

199 super().__init__(stream_or_string, **options)
200 self.handle_forward_references = options.pop("handle_forward_references", False)
201 self.event_stream = pulldom.parse(self.stream, self._make_parser())

205 def _make_parser(self):
206 """Create a hardened XML parser (no custom/external entities)."""
207 return DefusedExpatParser()

209 def __next__(self):
210 for event, node in self.event_stream:
211 if event == "START_ELEMENT" and node.nodeName == "object":
212 self.event_stream.expandNode(node)
213 return self._handle_object(node)
214 raise StopIteration

216 def _handle_object(self, node):
217 """Convert an <object> node to a DeserializedObject."""
218 # Look up the model using the model loading mechanism. If this fails,
219 # bail.
220 Model = self._get_model_from_node(node, "model")
221
222 # Start building a data dictionary from the object.
223 data = {}
224 if node.hasAttribute("pk"):
225 data[Model._meta.pk.attname] = Model._meta.pk.to_python(
226 node.getAttribute("pk")
227 )
228
229 # Also start building a dict of m2m data (this is saved as
230 # {m2m_accessor_attribute : [list_of_related_objects]})
231 m2m_data = {}
232 deferred_fields = {}
233
234 field_names = {f.name for f in Model._meta.get_fields()}
235 # Deserialize each field.
236 for field_node in node.getElementsByTagName("field"):
237 # If the field is missing the name attribute, bail (are you
238 # sensing a pattern here?)
239 field_name = field_node.getAttribute("name")
240 if not field_name:
241 raise base.DeserializationError(
242 "<field> node is missing the 'name' attribute"
243 )
244
245 # Get the field from the Model. This will raise a
246 # FieldDoesNotExist if, well, the field doesn't exist, which will
247 # be propagated correctly unless ignorenonexistent=True is used.
248 if self.ignore and field_name not in field_names:
249 continue
250 field = Model._meta.get_field(field_name)
251
252 # As is usually the case, relation fields get the special treatment.
253 if field.remote_field and isinstance(
254 field.remote_field, models.ManyToManyRel
255 ):
256 value = self._handle_m2m_field_node(field_node, field)
257 if value == base.DEFER_FIELD:
258 deferred_fields[field] = [
259 [
260 getInnerText(nat_node).strip()
261 for nat_node in obj_node.getElementsByTagName("natural")
262 ]
263 for obj_node in field_node.getElementsByTagName("object")
264 ]
265 else:
266 m2m_data[field.name] = value
267 elif field.remote_field and isinstance(
268 field.remote_field, models.ManyToOneRel
269 ):
270 value = self._handle_fk_field_node(field_node, field)
271 if value == base.DEFER_FIELD:
272 deferred_fields[field] = [
273 getInnerText(k).strip()
274 for k in field_node.getElementsByTagName("natural")
275 ]
276 else:
277 data[field.attname] = value
278 else:
279 if field_node.getElementsByTagName("None"):
280 value = None
281 else:
282 value = field.to_python(getInnerText(field_node).strip())
283 # Load value since JSONField.to_python() outputs strings.
284 if field.get_internal_type() == "JSONField":
285 value = json.loads(value, cls=field.decoder)
286 data[field.name] = value
287
288 obj = base.build_instance(Model, data, self.db)
289
290 # Return a DeserializedObject so that the m2m data has a place to live.
291 return base.DeserializedObject(obj, m2m_data, deferred_fields)

293 def _handle_fk_field_node(self, node, field):
294 """
295 Handle a <field> node for a ForeignKey
296 """
297 # Check if there is a child node named 'None', returning None if so.
298 if node.getElementsByTagName("None"):
299 return None
300 else:
301 model = field.remote_field.model
302 if hasattr(model._default_manager, "get_by_natural_key"):
303 keys = node.getElementsByTagName("natural")
304 if keys:
305 # If there are 'natural' subelements, it must be a natural key
306 field_value = [getInnerText(k).strip() for k in keys]
307 try:
308 obj = model._default_manager.db_manager(
309 self.db
310 ).get_by_natural_key(*field_value)
311 except ObjectDoesNotExist:
312 if self.handle_forward_references:
313 return base.DEFER_FIELD
314 else:
315 raise
316 obj_pk = getattr(obj, field.remote_field.field_name)
317 # If this is a natural foreign key to an object that
318 # has a FK/O2O as the foreign key, use the FK value
319 if field.remote_field.model._meta.pk.remote_field:
320 obj_pk = obj_pk.pk
321 else:
322 # Otherwise, treat like a normal PK
323 field_value = getInnerText(node).strip()
324 obj_pk = model._meta.get_field(
325 field.remote_field.field_name
326 ).to_python(field_value)
327 return obj_pk
328 else:
329 field_value = getInnerText(node).strip()
330 return model._meta.get_field(field.remote_field.field_name).to_python(
331 field_value
332 )

334 def _handle_m2m_field_node(self, node, field):
335 """
336 Handle a <field> node for a ManyToManyField.
337 """
338 model = field.remote_field.model
339 default_manager = model._default_manager
340 if hasattr(default_manager, "get_by_natural_key"):
341
342 def m2m_convert(n):
343 keys = n.getElementsByTagName("natural")
344 if keys:
345 # If there are 'natural' subelements, it must be a natural key
346 field_value = [getInnerText(k).strip() for k in keys]
347 obj_pk = (
348 default_manager.db_manager(self.db)
349 .get_by_natural_key(*field_value)
350 .pk
351 )
352 else:
353 # Otherwise, treat like a normal PK value.
354 obj_pk = model._meta.pk.to_python(n.getAttribute("pk"))
355 return obj_pk
356
357 else:
358
359 def m2m_convert(n):
360 return model._meta.pk.to_python(n.getAttribute("pk"))
361
362 values = []
363 try:
364 for c in node.getElementsByTagName("object"):
365 values.append(m2m_convert(c))
366 except Exception as e:
367 if isinstance(e, ObjectDoesNotExist) and self.handle_forward_references:
368 return base.DEFER_FIELD
369 else:
370 raise base.M2MDeserializationError(e, c)
371 else:
372 return values

374 def _get_model_from_node(self, node, attr):
375 """
376 Look up a model from a <object model=...> or a <field rel=... to=...>
377 node.
378 """
379 model_identifier = node.getAttribute(attr)
380 if not model_identifier:
381 raise base.DeserializationError(
382 "<%s> node is missing the required '%s' attribute"
383 % (node.nodeName, attr)

385 try:
386 return apps.get_model(model_identifier)
387 except (LookupError, TypeError):
388 raise base.DeserializationError(
389 "<%s> node has invalid model identifier: '%s'"
390 % (node.nodeName, model_identifier)

394def getInnerText(node):
395 """Get all the inner text of a DOM node (recursively)."""
396 # inspired by https://mail.python.org/pipermail/xml-sig/2005-March/011022.html
397 inner_text = []
398 for child in node.childNodes:
399 if (
400 child.nodeType == child.TEXT_NODE
401 or child.nodeType == child.CDATA_SECTION_NODE
402 ):
403 inner_text.append(child.data)
404 elif child.nodeType == child.ELEMENT_NODE:
405 inner_text.extend(getInnerText(child))
406 else:
407 pass
408 return "".join(inner_text)
