32class Lexer:
33 """
34 A generic multi-state regex-based lexer.
35 """
36
37 def __init__(self, states, first):
38 self.regexes = {}
39 self.toks = {}
40
41 for state, rules in states.items():
42 parts = []
43 for tok in rules:
44 groupid = "t%d" % tok.id
45 self.toks[groupid] = tok
46 parts.append("(?P<%s>%s)" % (groupid, tok.regex))
47 self.regexes[state] = re.compile("|".join(parts), re.MULTILINE | re.VERBOSE)
48
49 self.state = first
50
51 def lex(self, text):
52 """
53 Lexically analyze `text`.
54
55 Yield pairs (`name`, `tokentext`).
56 """
57 end = len(text)
58 state = self.state
59 regexes = self.regexes
60 toks = self.toks
61 start = 0
62
63 while start < end:
64 for match in regexes[state].finditer(text, start):
65 name = match.lastgroup
66 tok = toks[name]
67 toktext = match[name]
68 start += len(toktext)
69 yield (tok.name, toktext)
70
71 if tok.next:
72 state = tok.next
73 break
74
75 self.state = state

78class JsLexer(Lexer):
79 """
80 A JavaScript lexer
81
82 >>> lexer = JsLexer()

85
86 This doesn't properly handle non-ASCII characters in the JavaScript source.
87 """

214def prepare_js_for_gettext(js):
215 """
216 Convert the JavaScript source `js` into something resembling C for
217 xgettext.
218
219 What actually happens is that all the regex literals are replaced with
220 "REGEX".
221 """
222
223 def escape_quotes(m):
224 """Used in a regex to properly escape double quotes."""
225 s = m[0]
226 if s == '"':
227 return r"\""
228 else:
229 return s
230
231 lexer = JsLexer()
232 c = []
233 for name, tok in lexer.lex(js):
234 if name == "regex":
235 # C doesn't grok regexes, and they aren't needed for gettext,
236 # so just output a string instead.
237 tok = '"REGEX"'
238 elif name == "string":
239 # C doesn't have single-quoted strings, so make all strings
240 # double-quoted.
241 if tok.startswith("'"):
242 guts = re.sub(r"\\.|.", escape_quotes, tok[1:-1])
243 tok = '"' + guts + '"'
244 elif name == "id":
245 # C can't deal with Unicode escapes in identifiers. We don't
246 # need them for gettext anyway, so replace them with something
247 # innocuous
248 tok = tok.replace("\\", "U")
249 c.append(tok)
250 return "".join(c)
