2This module houses the GeoIP2 object, a wrapper for the MaxMind GeoIP2(R)

8For IP-based geolocation, this module requires the GeoLite2 Country and City

60 def __init__(self, path=None, cache=0, country=None, city=None):
61 """
62 Initialize the GeoIP object. No parameters are required to use default
63 settings. Keyword arguments may be passed in to customize the locations
64 of the GeoIP datasets.
65
66 * path: Base directory to where GeoIP data is located or the full path
67 to where the city or country data files (*.mmdb) are located.
68 Assumes that both the city and country data sets are located in
69 this directory; overrides the GEOIP_PATH setting.
70
71 * cache: The cache settings when opening up the GeoIP datasets. May be
72 an integer in (0, 1, 2, 4, 8) corresponding to the MODE_AUTO,
73 MODE_MMAP_EXT, MODE_MMAP, MODE_FILE, and MODE_MEMORY,
74 `GeoIPOptions` C API settings, respectively. Defaults to 0,
75 meaning MODE_AUTO.
76
77 * country: The name of the GeoIP country data file. Defaults to
78 'GeoLite2-Country.mmdb'; overrides the GEOIP_COUNTRY setting.
79
80 * city: The name of the GeoIP city data file. Defaults to
81 'GeoLite2-City.mmdb'; overrides the GEOIP_CITY setting.
82 """
83 if cache not in self.cache_options:
84 raise GeoIP2Exception("Invalid GeoIP caching option: %s" % cache)
85
86 path = path or getattr(settings, "GEOIP_PATH", None)
87 city = city or getattr(settings, "GEOIP_CITY", "GeoLite2-City.mmdb")
88 country = country or getattr(settings, "GEOIP_COUNTRY", "GeoLite2-Country.mmdb")
89
90 if not path:
91 raise GeoIP2Exception(
92 "GeoIP path must be provided via parameter or the GEOIP_PATH setting."
93 )
94
95 path = to_path(path)
96
97 # Try the path first in case it is the full path to a database.
98 for path in (path, path / city, path / country):
99 if path.is_file():
100 self._path = path
101 self._reader = geoip2.database.Reader(path, mode=cache)
102 break
103 else:
104 raise GeoIP2Exception(
105 "Path must be a valid database or directory containing databases."
106 )
107
108 database_type = self._metadata.database_type
109 if not database_type.endswith(("City", "Country")):
110 raise GeoIP2Exception(f"Unable to handle database edition: {database_type}")
111
112 def __del__(self):
113 # Cleanup any GeoIP file handles lying around.
114 if self._reader:
115 self._reader.close()

126 def _query(self, query, *, require_city=False):
127 if not isinstance(query, (str, ipaddress.IPv4Address, ipaddress.IPv6Address)):
128 raise TypeError(
129 "GeoIP query must be a string or instance of IPv4Address or "
130 "IPv6Address, not type %s" % type(query).__name__,
131 )
132
133 is_city = self._metadata.database_type.endswith("City")
134
135 if require_city and not is_city:
136 raise GeoIP2Exception(f"Invalid GeoIP city data file: {self._path}")
137
138 try:
139 validate_ipv46_address(query)
140 except ValidationError:
141 # GeoIP2 only takes IP addresses, so try to resolve a hostname.
142 query = socket.gethostbyname(query)
143
144 function = self._reader.city if is_city else self._reader.country
145 return function(query)

147 def city(self, query):
148 """
149 Return a dictionary of city information for the given IP address or
150 Fully Qualified Domain Name (FQDN). Some information in the dictionary
151 may be undefined (None).
152 """
153 response = self._query(query, require_city=True)
154 region = response.subdivisions[0] if response.subdivisions else None
155 return {
156 "accuracy_radius": response.location.accuracy_radius,
157 "city": response.city.name,
158 "continent_code": response.continent.code,
159 "continent_name": response.continent.name,
160 "country_code": response.country.iso_code,
161 "country_name": response.country.name,
162 "is_in_european_union": response.country.is_in_european_union,
163 "latitude": response.location.latitude,
164 "longitude": response.location.longitude,
165 "metro_code": response.location.metro_code,
166 "postal_code": response.postal.code,
167 "region_code": region.iso_code if region else None,
168 "region_name": region.name if region else None,
169 "time_zone": response.location.time_zone,
170 # Kept for backward compatibility.
171 "dma_code": response.location.metro_code,
172 "region": region.iso_code if region else None,
173 }
