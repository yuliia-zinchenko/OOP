21class DeclarativeFieldsMetaclass(MediaDefiningClass):
22 """Collect Fields declared on the base classes."""
23
24 def __new__(mcs, name, bases, attrs):
25 # Collect fields from current class and remove them from attrs.
26 attrs["declared_fields"] = {
27 key: attrs.pop(key)
28 for key, value in list(attrs.items())
29 if isinstance(value, Field)
30 }
31
32 new_class = super().__new__(mcs, name, bases, attrs)
33
34 # Walk through the MRO.
35 declared_fields = {}
36 for base in reversed(new_class.__mro__):
37 # Collect fields from base class.
38 if hasattr(base, "declared_fields"):
39 declared_fields.update(base.declared_fields)
40
41 # Field shadowing.
42 for attr, value in base.__dict__.items():
43 if value is None and attr in declared_fields:
44 declared_fields.pop(attr)
45
46 new_class.base_fields = declared_fields
47 new_class.declared_fields = declared_fields
48
49 return new_class

127 def order_fields(self, field_order):
128 """
129 Rearrange the fields according to field_order.
130
131 field_order is a list of field names specifying the order. Append fields
132 not included in the list in the default order for backward compatibility
133 with subclasses not overriding field_order. If field_order is None,
134 keep all fields in the order defined in the class. Ignore unknown
135 fields in field_order to allow disabling fields in form subclasses
136 without redefining ordering.
137 """
138 if field_order is None:
139 return
140 fields = {}
141 for key in field_order:
142 try:
143 fields[key] = self.fields.pop(key)
144 except KeyError: # ignore unknown fields
145 pass
146 fields.update(self.fields) # add remaining fields in original order
147 self.fields = fields

255 def add_error(self, field, error):
256 """
257 Update the content of `self._errors`.
258
259 The `field` argument is the name of the field to which the errors
260 should be added. If it's None, treat the errors as NON_FIELD_ERRORS.
261
262 The `error` argument can be a single error, a list of errors, or a
263 dictionary that maps field names to lists of errors. An "error" can be
264 either a simple string or an instance of ValidationError with its
265 message attribute set and a "list or dictionary" can be an actual
266 `list` or `dict` or an instance of ValidationError with its
267 `error_list` or `error_dict` attribute set.
268
269 If `error` is a dictionary, the `field` argument *must* be None and
270 errors will be added to the fields that correspond to the keys of the
271 dictionary.
272 """
273 if not isinstance(error, ValidationError):
274 # Normalize to ValidationError and let its constructor
275 # do the hard work of making sense of the input.
276 error = ValidationError(error)
277
278 if hasattr(error, "error_dict"):
279 if field is not None:
280 raise TypeError(
281 "The argument `field` must be `None` when the `error` "
282 "argument contains errors for multiple fields."
283 )
284 else:
285 error = error.error_dict
286 else:
287 error = {field or NON_FIELD_ERRORS: error.error_list}
288
289 for field, error_list in error.items():
290 if field not in self.errors:
291 if field != NON_FIELD_ERRORS and field not in self.fields:
292 raise ValueError(
293 "'%s' has no field named '%s'."
294 % (self.__class__.__name__, field)
295 )
296 if field == NON_FIELD_ERRORS:
297 self._errors[field] = self.error_class(
298 error_class="nonfield", renderer=self.renderer
299 )
300 else:
301 self._errors[field] = self.error_class(renderer=self.renderer)
302 self._errors[field].extend(error_list)
303 if field in self.cleaned_data:
304 del self.cleaned_data[field]

312 def full_clean(self):
313 """
314 Clean all of self.data and populate self._errors and self.cleaned_data.
315 """
316 self._errors = ErrorDict()
317 if not self.is_bound: # Stop further processing.
318 return
319 self.cleaned_data = {}
320 # If the form is permitted to be empty, and none of the form data has
321 # changed from the initial data, short circuit any validation.
322 if self.empty_permitted and not self.has_changed():
323 return
324
325 self._clean_fields()
326 self._clean_form()
327 self._post_clean()

329 def _clean_fields(self):
330 for name, bf in self._bound_items():
331 field = bf.field
332 try:
333 self.cleaned_data[name] = field._clean_bound_field(bf)
334 if hasattr(self, "clean_%s" % name):
335 value = getattr(self, "clean_%s" % name)()
336 self.cleaned_data[name] = value
337 except ValidationError as e:
