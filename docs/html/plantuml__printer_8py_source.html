30 def _open_graph(self) -> None:
31 """Emit the header lines."""
32 self.emit("@startuml " + self.title)
33 if not self.use_automatic_namespace:
34 self.emit("set namespaceSeparator none")
35 if self.layout:
36 if self.layout is Layout.LEFT_TO_RIGHT:
37 self.emit("left to right direction")
38 elif self.layout is Layout.TOP_TO_BOTTOM:
39 self.emit("top to bottom direction")
40 else:
41 raise ValueError(
42 f"Unsupported layout {self.layout}. PlantUmlPrinter only "
43 "supports left to right and top to bottom layout."
44 )

46 def emit_node(
47 self,
48 name: str,
49 type_: NodeType,
50 properties: NodeProperties | None = None,
51 ) -> None:
52 """Create a new node.
53
54 Nodes can be classes, packages, participants etc.
55 """
56 if properties is None:
57 properties = NodeProperties(label=name)
58 nodetype = self.NODES[type_]
59 if properties.color and properties.color != self.DEFAULT_COLOR:
60 color = f" #{properties.color.lstrip('#')}"
61 else:
62 color = ""
63 body = []
64 if properties.attrs:
65 body.extend(properties.attrs)
66 if properties.methods:
67 for func in properties.methods:
68 args = self._get_method_arguments(func)
69 line = "{abstract}" if func.is_abstract() else ""
70 line += f"{func.name}({', '.join(args)})"
71 if func.returns:
72 line += " -> " + get_annotation_label(func.returns)
73 body.append(line)
74 label = properties.label if properties.label is not None else name
75 if properties.fontcolor and properties.fontcolor != self.DEFAULT_COLOR:
76 label = f"<color:{properties.fontcolor}>{label}</color>"
77 self.emit(f'{nodetype} "{label}" as {name}{color} {{')
78 self._inc_indent()
79 for line in body:
80 self.emit(line)
81 self._dec_indent()
82 self.emit("}")

84 def emit_edge(
85 self,
86 from_node: str,
87 to_node: str,
88 type_: EdgeType,
89 label: str | None = None,
90 ) -> None:
91 """Create an edge from one node to another to display relationships."""
92 edge = f"{from_node} {self.ARROWS[type_]} {to_node}"
93 if label:
94 edge += f" : {label}"
95 self.emit(edge)
