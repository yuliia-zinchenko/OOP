21class MessageDefinitionStore:
22 """The messages store knows information about every possible message definition but
23 has no particular state during analysis.
24 """
25
26 def __init__(
27 self, py_version: tuple[int, ...] | sys._version_info = sys.version_info
28 ) -> None:
29 self.message_id_store: MessageIdStore = MessageIdStore()
30 # Primary registry for all active messages definitions.
31 # It contains the 1:1 mapping from msgid to MessageDefinition.
32 # Keys are msgid, values are MessageDefinition
33 self._messages_definitions: dict[str, MessageDefinition] = {}
34 # MessageDefinition kept by category
35 self._msgs_by_category: dict[str, list[str]] = collections.defaultdict(list)
36 self.py_version = py_version
37
38 @property
39 def messages(self) -> ValuesView[MessageDefinition]:
40 """The list of all active messages."""
41 return self._messages_definitions.values()
42
43 def register_messages_from_checker(self, checker: BaseChecker) -> None:
44 """Register all messages definitions from a checker."""
45 checker.check_consistency()
46 for message in checker.messages:
47 self.register_message(message)
48
49 def register_message(self, message: MessageDefinition) -> None:
50 """Register a MessageDefinition with consistency in mind."""
51 self.message_id_store.register_message_definition(
52 message.msgid, message.symbol, message.old_names
53 )
54 self._messages_definitions[message.msgid] = message
55 self._msgs_by_category[message.msgid[0]].append(message.msgid)
56
57 # Since MessageDefinitionStore is only initialized once
58 # and the arguments are relatively small we do not run the
59 # risk of creating a large memory leak.
60 # See discussion in: https://github.com/pylint-dev/pylint/pull/5673
61 @cache # pylint: disable=method-cache-max-size-none # noqa: B019
62 def get_message_definitions(self, msgid_or_symbol: str) -> list[MessageDefinition]:
63 """Returns the Message definition for either a numeric or symbolic id.
64
65 The cache has no limit as its size will likely stay minimal. For each message we store
66 about 1000 characters, so even if we would have 1000 messages the cache would only
67 take up ~= 1 Mb.
68 """
69 return [
70 self._messages_definitions[m]
71 for m in self.message_id_store.get_active_msgids(msgid_or_symbol)
72 ]
73
74 def get_msg_display_string(self, msgid_or_symbol: str) -> str:
75 """Generates a user-consumable representation of a message."""
76 message_definitions = self.get_message_definitions(msgid_or_symbol)
77 if len(message_definitions) == 1:
78 return repr(message_definitions[0].symbol)
79 return repr([md.symbol for md in message_definitions])
80
81 def help_message(self, msgids_or_symbols: Sequence[str]) -> None:
82 """Display help messages for the given message identifiers."""
83 for msgids_or_symbol in msgids_or_symbols:
84 try:
85 for message_definition in self.get_message_definitions(
86 msgids_or_symbol
87 ):
88 print(message_definition.format_help(checkerref=True))
89 print("")
90 except UnknownMessageError as ex:
91 print(ex)
92 print("")
93 continue
94
95 def list_messages(self) -> None:
96 """Output full messages list documentation in ReST format."""
97 emittable, non_emittable = self.find_emittable_messages()
98 print("Emittable messages with current interpreter:")
99 for msg in emittable:
100 print(msg.format_help(checkerref=False))
101 print("\nNon-emittable messages with current interpreter:")
102 for msg in non_emittable:
103 print(msg.format_help(checkerref=False))
104 print("")
105
106 def find_emittable_messages(
107 self,
108 ) -> tuple[list[MessageDefinition], list[MessageDefinition]]:
109 """Finds all emittable and non-emittable messages."""
110 messages = sorted(self._messages_definitions.values(), key=lambda m: m.msgid)
111 emittable = []
112 non_emittable = []
113 for message in messages:
114 if message.may_be_emitted(self.py_version):
115 emittable.append(message)
116 else:
117 non_emittable.append(message)
118 return emittable, non_emittable
