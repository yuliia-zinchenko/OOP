29class MessageStyle(NamedTuple):
30 """Styling of a message."""
31
32 color: str | None
33 """The color name (see `ANSI_COLORS` for available values)
34 or the color number when 256 colors are available.
35 """
36 style: tuple[str, ...] = ()
37 """Tuple of style strings (see `ANSI_COLORS` for available values)."""
38
39 def __get_ansi_code(self) -> str:
40 """Return ANSI escape code corresponding to color and style.
41
42 :raise KeyError: if a nonexistent color or style identifier is given
43
44 :return: the built escape code
45 """
46 ansi_code = [ANSI_STYLES[effect] for effect in self.style]
47 if self.color:
48 if self.color.isdigit():
49 ansi_code.extend(["38", "5"])
50 ansi_code.append(self.color)
51 else:
52 ansi_code.append(ANSI_COLORS[self.color])
53 if ansi_code:
54 return ANSI_PREFIX + ";".join(ansi_code) + ANSI_END
55 return ""
56
57 def _colorize_ansi(self, msg: str) -> str:
58 if self.color is None and len(self.style) == 0:
59 # If both color and style are not defined, then leave the text as is.
60 return msg
61 escape_code = self.__get_ansi_code()
62 # If invalid (or unknown) color, don't wrap msg with ANSI codes
63 if escape_code:
64 return f"{escape_code}{msg}{ANSI_RESET}"
65 return msg

72ANSI_PREFIX = "\033["
73ANSI_END = "m"
74ANSI_RESET = "\033[0m"
75ANSI_STYLES = {
76 "reset": "0",
77 "bold": "1",
78 "italic": "3",

109class TextReporter(BaseReporter):
110 """Reports messages and layouts in plain text."""
111
112 name = "text"
113 extension = "txt"
114 line_format = "{path}:{line}:{column}: {msg_id}: {msg} ({symbol})"
115
116 def __init__(self, output: TextIO | None = None) -> None:
117 super().__init__(output)
118 self._modules: set[str] = set()
119 self._template = self.line_format
120 self._fixed_template = self.line_format
121 """The output format template with any unrecognized arguments removed."""
122
123 def on_set_current_module(self, module: str, filepath: str | None) -> None:
124 """Set the format template to be used and check for unrecognized arguments."""
125 template = str(self.linter.config.msg_template or self._template)
126
127 # Return early if the template is the same as the previous one
128 if template == self._template:
129 return
130
131 # Set template to the currently selected template
132 self._template = template
133
134 # Check to see if all parameters in the template are attributes of the Message
135 arguments = re.findall(r"\{(\w+?)(:.*)?\}", template)
136 for argument in arguments:
137 if argument[0] not in MESSAGE_FIELDS:
138 warnings.warn(
139 f"Don't recognize the argument '{argument[0]}' in the --msg-template. "
140 "Are you sure it is supported on the current version of pylint?",
141 stacklevel=2,
142 )
143 template = re.sub(r"\{" + argument[0] + r"(:.*?)?\}", "", template)
144 self._fixed_template = template
145
146 def write_message(self, msg: Message) -> None:
147 """Convenience method to write a formatted message with class default
148 template.
149 """
150 self_dict = asdict(msg)
151 for key in ("end_line", "end_column"):
152 self_dict[key] = self_dict[key] or ""
153
154 self.writeln(self._fixed_template.format(**self_dict))
155
156 def handle_message(self, msg: Message) -> None:
157 """Manage message of different type and in the context of path."""
158 if msg.module not in self._modules:
159 self.writeln(make_header(msg))
160 self._modules.add(msg.module)
161 self.write_message(msg)
162
163 def _display(self, layout: Section) -> None:
164 """Launch layouts display."""
165 print(file=self.out)
166 TextWriter().format(layout, self.out)

207class ColorizedTextReporter(TextReporter):
208 """Simple TextReporter that colorizes text output."""
209
210 name = "colorized"
211 COLOR_MAPPING: ColorMappingDict = {
212 "I": MessageStyle("green"),
213 "C": MessageStyle(None, ("bold",)),
214 "R": MessageStyle("magenta", ("bold", "italic")),
215 "W": MessageStyle("magenta"),
216 "E": MessageStyle("red", ("bold",)),
217 "F": MessageStyle("red", ("bold", "underline")),
218 "S": MessageStyle("yellow", ("inverse",)), # S stands for module Separator
219 }
220
221 def __init__(
222 self,
223 output: TextIO | None = None,
224 color_mapping: ColorMappingDict | None = None,
225 ) -> None:
226 super().__init__(output)
227 self.color_mapping = color_mapping or ColorizedTextReporter.COLOR_MAPPING
228 ansi_terms = ["xterm-16color", "xterm-256color"]
229 if os.environ.get("TERM") not in ansi_terms:
230 if sys.platform == "win32":
231 # pylint: disable=import-outside-toplevel
232 import colorama
233
234 self.out = colorama.AnsiToWin32(self.out)
235
236 def _get_decoration(self, msg_id: str) -> MessageStyle:
237 """Returns the message style as defined in self.color_mapping."""
238 return self.color_mapping.get(msg_id[0]) or MessageStyle(None)
239
240 def handle_message(self, msg: Message) -> None:
241 """Manage message of different types, and colorize output
242 using ANSI escape codes.
243 """
244 if msg.module not in self._modules:
245 msg_style = self._get_decoration("S")
246 modsep = colorize_ansi(make_header(msg), msg_style)
247 self.writeln(modsep)
248 self._modules.add(msg.module)
249 msg_style = self._get_decoration(msg.C)
250
251 msg.msg = colorize_ansi(msg.msg, msg_style)
252 msg.symbol = colorize_ansi(msg.symbol, msg_style)
253 msg.category = colorize_ansi(msg.category, msg_style)
254 msg.C = colorize_ansi(msg.C, msg_style)
255 self.write_message(msg)
