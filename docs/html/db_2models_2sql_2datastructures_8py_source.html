30class Join:
31 """
32 Used by sql.Query and sql.SQLCompiler to generate JOIN clauses into the
33 FROM entry. For example, the SQL generated could be
34 LEFT OUTER JOIN "sometable" T1
35 ON ("othertable"."sometable_id" = "sometable"."id")
36
37 This class is primarily used in Query.alias_map. All entries in alias_map
38 must be Join compatible by providing the following attributes and methods:
39 - table_name (string)
40 - table_alias (possible alias for the table, can be None)
41 - join_type (can be None for those entries that aren't joined from
42 anything)
43 - parent_alias (which table is this join's parent, can be None similarly
44 to join_type)
45 - as_sql()
46 - relabeled_clone()
47 """
48

88 def as_sql(self, compiler, connection):
89 """
90 Generate the full
91 LEFT OUTER JOIN sometable ON sometable.somecol = othertable.othercol, params
92 clause for this join.
93 """
94 join_conditions = []
95 params = []
96 qn = compiler.quote_name_unless_alias
97 qn2 = connection.ops.quote_name
98 # Add a join condition for each pair of joining columns.
99 # RemovedInDjango60Warning: when the depraction ends, replace with:
100 # for lhs, rhs in self.join_field:
101 join_fields = self.join_fields or self.join_cols
102 for lhs, rhs in join_fields:
103 if isinstance(lhs, str):
104 # RemovedInDjango60Warning: when the depraction ends, remove
105 # the branch for strings.
106 lhs_full_name = "%s.%s" % (qn(self.parent_alias), qn2(lhs))
107 rhs_full_name = "%s.%s" % (qn(self.table_alias), qn2(rhs))
108 else:
109 lhs, rhs = connection.ops.prepare_join_on_clause(
110 self.parent_alias, lhs, self.table_alias, rhs
111 )
112 lhs_sql, lhs_params = compiler.compile(lhs)
113 lhs_full_name = lhs_sql % lhs_params
114 rhs_sql, rhs_params = compiler.compile(rhs)
115 rhs_full_name = rhs_sql % rhs_params
116 join_conditions.append(f"{lhs_full_name} = {rhs_full_name}")
117
118 # Add a single condition inside parentheses for whatever
119 # get_extra_restriction() returns.
120 extra_cond = self.join_field.get_extra_restriction(
121 self.table_alias, self.parent_alias
122 )
123 if extra_cond:
124 extra_sql, extra_params = compiler.compile(extra_cond)
125 join_conditions.append("(%s)" % extra_sql)
126 params.extend(extra_params)
127 if self.filtered_relation:
128 try:
129 extra_sql, extra_params = compiler.compile(self.filtered_relation)
130 except FullResultSet:
131 pass
132 else:
133 join_conditions.append("(%s)" % extra_sql)
134 params.extend(extra_params)
135 if not join_conditions:
136 # This might be a rel on the other end of an actual declared field.
137 declared_field = getattr(self.join_field, "field", self.join_field)
138 raise ValueError(
139 "Join generated an empty ON clause. %s did not yield either "
140 "joining columns or extra restrictions." % declared_field.__class__
141 )
142 on_clause_sql = " AND ".join(join_conditions)
143 alias_str = (
144 "" if self.table_alias == self.table_name else (" %s" % self.table_alias)
145 )
146 sql = "%s %s%s ON (%s)" % (
147 self.join_type,
148 qn(self.table_name),
149 alias_str,
150 on_clause_sql,
151 )
152 return sql, params
