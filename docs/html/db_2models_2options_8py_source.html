57def normalize_together(option_together):
58 """
59 option_together can be either a tuple of tuples, or a single
60 tuple of two strings. Normalize it to a tuple of tuples, so that
61 calling code can uniformly expect that.
62 """
63 try:
64 if not option_together:
65 return ()
66 if not isinstance(option_together, (tuple, list)):
67 raise TypeError
68 first_element = option_together[0]
69 if not isinstance(first_element, (tuple, list)):
70 option_together = (option_together,)
71 # Normalize everything to tuples
72 return tuple(tuple(ot) for ot in option_together)
73 except TypeError:
74 # If the value of option_together isn't valid, return it
75 # verbatim; this will be picked up by the check framework later.
76 return option_together

326 def add_field(self, field, private=False):
327 # Insert the given field in the order in which it was created, using
328 # the "creation_counter" attribute of the field.
329 # Move many-to-many related fields from self.fields into
330 # self.many_to_many.
331 if private:
332 self.private_fields.append(field)
333 elif field.is_relation and field.many_to_many:
334 bisect.insort(self.local_many_to_many, field)
335 else:
336 bisect.insort(self.local_fields, field)
337 self.setup_pk(field)
338
339 # If the field being added is a relation to another known field,
340 # expire the cache on this field and the forward cache on the field
341 # being referenced, because there will be new relationships in the
342 # cache. Otherwise, expire the cache of references *to* this field.
343 # The mechanism for getting at the related model is slightly odd -
344 # ideally, we'd just ask for field.related_model. However, related_model
345 # is a cached property, and all the models haven't been loaded yet, so
346 # we need to make sure we don't cache a string reference.
347 if (
348 field.is_relation
349 and hasattr(field.remote_field, "model")
350 and field.remote_field.model
351 ):
352 try:
353 field.remote_field.model._meta._expire_cache(forward=False)
354 except AttributeError:
355 pass
356 self._expire_cache()
357 else:
358 self._expire_cache(reverse=False)

682 def get_base_chain(self, model):
683 """
684 Return a list of parent classes leading to `model` (ordered from

686 `model` is a grandparent or even more distant relation.

718 def get_ancestor_link(self, ancestor):
719 """
720 Return the field on the current model which points to the given
721 "ancestor". This is possible an indirect link (a pointer to a parent
722 model, which points, eventually, to the ancestor). Used when
723 constructing table joins for model inheritance.
724
725 Return None if the model isn't an ancestor of this one.
726 """

733 # In case of a proxied model, the first link

750 for int_model in self.get_base_chain(parent):
751 if int_model is proxied_model:
752 opts = int_model._meta
753 else:
754 final_field = opts.parents[int_model]
755 targets = (final_field.remote_field.get_related_field(),)
756 opts = int_model._meta
757 path.append(
758 PathInfo(
759 from_opts=final_field.model._meta,
760 to_opts=opts,
761 target_fields=targets,
762 join_field=final_field,
763 m2m=False,
764 direct=True,
765 filtered_relation=None,
766 )
767 )
768 return path

770 def get_path_from_parent(self, parent):
771 """
772 Return a list of PathInfos containing the path from the parent
773 model to the current model, or an empty list if parent is not a
774 parent of the current model.
775 """
776 if self.model is parent:
777 return []
778 model = self.concrete_model
779 # Get a reversed base chain including both the current and parent
780 # models.
781 chain = model._meta.get_base_chain(parent)
782 chain.reverse()
783 chain.append(model)
784 # Construct a list of the PathInfos between models in chain.
785 path = []
786 for i, ancestor in enumerate(chain[:-1]):
787 child = chain[i + 1]
788 link = child._meta.get_ancestor_link(ancestor)
789 path.extend(link.reverse_path_infos)
790 return path

792 def _populate_directed_relation_graph(self):
793 """
794 This method is used by each model to find its reverse objects. As this
795 method is very expensive and is accessed frequently (it looks up every
796 field in a model, in every app), it is computed on first access and then
797 is set as a property on every model.
798 """
799 related_objects_graph = defaultdict(list)
800
801 all_models = self.apps.get_models(include_auto_created=True)
802 for model in all_models:
803 opts = model._meta
804 # Abstract model's fields are copied to child models, hence we will
805 # see the fields from the child models.
806 if opts.abstract:
807 continue
808 fields_with_relations = (
809 f
810 for f in opts._get_fields(reverse=False, include_parents=False)
811 if f.is_relation and f.related_model is not None
812 )
813 for f in fields_with_relations:
814 if not isinstance(f.remote_field.model, str):
815 remote_label = f.remote_field.model._meta.concrete_model._meta.label
816 related_objects_graph[remote_label].append(f)
817
818 for model in all_models:
819 # Set the relation_tree using the internal __dict__. In this way
820 # we avoid calling the cached property. In attribute lookup,
821 # __dict__ takes precedence over a data descriptor (such as
822 # @cached_property). This means that the _meta._relation_tree is
823 # only called if related_objects is not in __dict__.
824 related_objects = related_objects_graph[
825 model._meta.concrete_model._meta.label
826 ]
827 model._meta.__dict__["_relation_tree"] = related_objects
828 # It seems it is possible that self is not in all_models, so guard
829 # against that with default for get().
830 return self.__dict__.get("_relation_tree", EMPTY_RELATION_TREE)
831
832 @cached_property
833 def _relation_tree(self):
834 return self._populate_directed_relation_graph()

851 Return a list of fields associated to the model. By default, include
852 forward and reverse fields, fields derived from inheritance, but not

865 def _get_fields(
866 self,
867 forward=True,
868 reverse=True,
869 include_parents=True,
870 include_hidden=False,
871 topmost_call=True,
872 ):
873 """
874 Internal helper function to return fields of the model.
875 * If forward=True, then fields defined on this model are returned.
876 * If reverse=True, then relations pointing to this model are returned.
877 * If include_hidden=True, then fields with is_hidden=True are returned.
878 * The include_parents argument toggles if fields from parent models
879 should be included. It has three values: True, False, and
880 PROXY_PARENTS. When set to PROXY_PARENTS, the call will return all
881 fields defined for the current model or any of its parents in the
882 parent chain to the model's concrete model.
883 """
884 if include_parents not in (True, False, PROXY_PARENTS):
885 raise TypeError(
886 "Invalid argument for include_parents: %s" % (include_parents,)
887 )
888 # This helper function is used to allow recursion in ``get_fields()``
889 # implementation and to provide a fast way for Django's internals to
890 # access specific subsets of fields.
891
892 # Creates a cache key composed of all arguments
893 cache_key = (forward, reverse, include_parents, include_hidden, topmost_call)
894
895 try:
896 # In order to avoid list manipulation. Always return a shallow copy
897 # of the results.
898 return self._get_fields_cache[cache_key]
899 except KeyError:
900 pass
901
902 fields = []
903 # Recursively call _get_fields() on each parent, with the same
904 # options provided in this call.
905 if include_parents is not False:
906 # In diamond inheritance it is possible that we see the same model
907 # from two different routes. In that case, avoid adding fields from
908 # the same parent again.
909 parent_fields = set()
910 for parent in self.parents:
911 if (
912 parent._meta.concrete_model != self.concrete_model
913 and include_parents == PROXY_PARENTS
914 ):
915 continue
916 for obj in parent._meta._get_fields(
917 forward=forward,
918 reverse=reverse,
919 include_parents=include_parents,
920 include_hidden=include_hidden,
921 topmost_call=False,
922 ):
923 if (
924 not getattr(obj, "parent_link", False)
925 or obj.model == self.concrete_model
926 ) and obj not in parent_fields:
927 fields.append(obj)
928 parent_fields.add(obj)
929
930 if reverse and not self.proxy:
931 # Tree is computed once and cached until the app cache is expired.
932 # It is composed of a list of fields pointing to the current model
933 # from other models.
934 all_fields = self._relation_tree
935 for field in all_fields:
936 # If hidden fields should be included or the relation is not
937 # intentionally hidden, add to the fields dict.
938 if include_hidden or not field.remote_field.hidden:
939 fields.append(field.remote_field)
940
941 if forward:
942 fields += self.local_fields
943 fields += self.local_many_to_many
944 # Private fields are recopied to each child model, and they get a
945 # different model as field.model in each child. Hence we have to
946 # add the private fields separately from the topmost call. If we
947 # did this recursively similar to local_fields, we would get field
948 # instances with field.model != self.model.
949 if topmost_call:
950 fields += self.private_fields
951
952 # In order to avoid list manipulation. Always
953 # return a shallow copy of the results
954 fields = make_immutable_fields_list("get_fields()", fields)
955
956 # Store result into cache for later access
957 self._get_fields_cache[cache_key] = fields
958 return fields
