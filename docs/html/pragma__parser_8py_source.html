11# Allow stopping after the first semicolon/hash encountered,
12# so that an option can be continued with the reasons
13# why it is active or disabled.
14OPTION_RGX = r"""
15 (?:^\s*\#.*|\s*| # Comment line, or whitespaces,
16 \s*\#.*(?=\#.*?\bpylint:)) # or a beginning of an inline comment
17 # followed by "pylint:" pragma
18 (\# # Beginning of comment
19 .*? # Anything (as little as possible)
20 \bpylint: # pylint word and column
21 \s* # Any number of whitespaces
22 ([^;#\n]+)) # Anything except semicolon or hash or
23 # newline (it is the second matched group)
24 # and end of the first matched group
25 [;#]{0,1} # From 0 to 1 repetition of semicolon or hash
26"""
27OPTION_PO = re.compile(OPTION_RGX, re.VERBOSE)

35ATOMIC_KEYWORDS = frozenset(("disable-all", "skip-file"))
36MESSAGE_KEYWORDS = frozenset(
37 ("disable-next", "disable-msg", "enable-msg", "disable", "enable")
38)
39# sorted is necessary because sets are unordered collections and ALL_KEYWORDS
40# string should not vary between executions
41# reverse is necessary in order to have the longest keywords first, so that, for example,
42# 'disable' string should not be matched instead of 'disable-all'
43ALL_KEYWORDS = "|".join(
44 sorted(ATOMIC_KEYWORDS | MESSAGE_KEYWORDS, key=len, reverse=True)
45)
46
47
48TOKEN_SPECIFICATION = [
49 ("KEYWORD", rf"\b({ALL_KEYWORDS:s})\b"),
50 ("MESSAGE_STRING", r"[0-9A-Za-z\-\_]{2,}"), # Identifiers
51 ("ASSIGN", r"="), # Assignment operator
52 ("MESSAGE_NUMBER", r"[CREIWF]{1}\d*"),
53]
54
55TOK_REGEX = "|".join(
56 f"(?P<{token_name:s}>{token_rgx:s})"
57 for token_name, token_rgx in TOKEN_SPECIFICATION
58)

61def emit_pragma_representer(action: str, messages: list[str]) -> PragmaRepresenter:
62 if not messages and action in MESSAGE_KEYWORDS:
63 raise InvalidPragmaError(
64 "The keyword is not followed by message identifier", action
65 )
66 return PragmaRepresenter(action, messages)
