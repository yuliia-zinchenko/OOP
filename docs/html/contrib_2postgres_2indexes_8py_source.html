48 def __init__(self, *expressions, length=None, columns=(), **kwargs):
49 super().__init__(*expressions, **kwargs)
50 if len(self.fields) > 32:
51 raise ValueError("Bloom indexes support a maximum of 32 fields.")
52 if not isinstance(columns, (list, tuple)):
53 raise ValueError("BloomIndex.columns must be a list or tuple.")
54 if len(columns) > len(self.fields):
55 raise ValueError("BloomIndex.columns cannot have more values than fields.")
56 if not all(0 < col <= 4095 for col in columns):
57 raise ValueError(
58 "BloomIndex.columns must contain integers from 1 to 4095.",
59 )
60 if length is not None and not 0 < length <= 4096:
61 raise ValueError(
62 "BloomIndex.length must be None or an integer from 1 to 4096.",
63 )
64 self.length = length
65 self.columns = columns
66
67 def deconstruct(self):
68 path, args, kwargs = super().deconstruct()
69 if self.length is not None:
70 kwargs["length"] = self.length
71 if self.columns:
72 kwargs["columns"] = self.columns
73 return path, args, kwargs
74
75 def get_with_params(self):
76 with_params = []
77 if self.length is not None:
78 with_params.append("length = %d" % self.length)
79 if self.columns:
80 with_params.extend(
81 "col%d = %d" % (i, v) for i, v in enumerate(self.columns, start=1)
