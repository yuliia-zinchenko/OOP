19class DunderChecker(BaseChecker):
20 """Checks related to dunder methods."""

22 name = "dunder"
23 msgs = {
24 "W3201": (
25 "Bad or misspelled dunder method name %s.",
26 "bad-dunder-name",
27 "Used when a dunder method is misspelled or defined with a name "

34 {

43 def open(self) -> None:
44 self._dunder_methods = (
45 EXTRA_DUNDER_METHODS
46 + DUNDER_PROPERTIES
47 + self.linter.config.good_dunder_names
48 )
49 for since_vers, dunder_methods in DUNDER_METHODS.items():
50 if since_vers <= self.linter.config.py_version:
51 self._dunder_methods.extend(list(dunder_methods.keys()))
52
53 def visit_functiondef(self, node: nodes.FunctionDef) -> None:
54 """Check if known dunder method is misspelled or dunder name is not one
55 of the pre-defined names.
56 """
57 # ignore module-level functions
58 if not node.is_method():
59 return
60
61 # Detect something that could be a bad dunder method
62 if (
63 node.name.startswith("_")
64 and node.name.endswith("_")
65 and node.name not in self._dunder_methods
66 ):
67 self.add_message(
68 "bad-dunder-name",
69 node=node,
70 args=(node.name),
71 confidence=HIGH,
72 )
