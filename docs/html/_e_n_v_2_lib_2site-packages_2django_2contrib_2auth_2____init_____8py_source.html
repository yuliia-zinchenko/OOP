44def _clean_credentials(credentials):
45 """
46 Clean a dictionary of credentials of potentially sensitive info before
47 sending to less secure functions.
48
49 Not comprehensive - intended for user_login_failed signal
50 """
51 SENSITIVE_CREDENTIALS = re.compile("api|token|key|secret|password|signature", re.I)
52 CLEANSED_SUBSTITUTE = "********************"
53 for key in credentials:
54 if SENSITIVE_CREDENTIALS.search(key):
55 credentials[key] = CLEANSED_SUBSTITUTE
56 return credentials

65@sensitive_variables("credentials")
66def authenticate(request=None, **credentials):
67 """
68 If the given credentials are valid, return a User object.
69 """
70 for backend, backend_path in _get_backends(return_tuples=True):
71 backend_signature = inspect.signature(backend.authenticate)
72 try:
73 backend_signature.bind(request, **credentials)
74 except TypeError:
75 # This backend doesn't accept these credentials as arguments. Try
76 # the next one.
77 continue
78 try:
79 user = backend.authenticate(request, **credentials)
80 except PermissionDenied:
81 # This backend says to stop in our tracks - this user should not be
82 # allowed in at all.
83 break
84 if user is None:
85 continue
86 # Annotate the user object with the path of the backend.
87 user.backend = backend_path
88 return user
89
90 # The credentials supplied are invalid to all backends, fire signal
91 user_login_failed.send(
92 sender=__name__, credentials=_clean_credentials(credentials), request=request
93 )

102def login(request, user, backend=None):
103 """
104 Persist a user id and a backend in the request. This way a user doesn't
105 have to reauthenticate on every request. Note that data set during
106 the anonymous session is retained when the user logs in.
107 """
108 session_auth_hash = ""
109 if user is None:
110 user = request.user
111 if hasattr(user, "get_session_auth_hash"):
112 session_auth_hash = user.get_session_auth_hash()
113
114 if SESSION_KEY in request.session:
115 if _get_user_session_key(request) != user.pk or (
116 session_auth_hash
117 and not constant_time_compare(
118 request.session.get(HASH_SESSION_KEY, ""), session_auth_hash
119 )
120 ):
121 # To avoid reusing another user's session, create a new, empty
122 # session if the existing session corresponds to a different
123 # authenticated user.
124 request.session.flush()
125 else:
126 request.session.cycle_key()
127
128 try:
129 backend = backend or user.backend
130 except AttributeError:
131 backends = _get_backends(return_tuples=True)
132 if len(backends) == 1:
133 _, backend = backends[0]
134 else:
135 raise ValueError(
136 "You have multiple authentication backends configured and "
137 "therefore must provide the `backend` argument or set the "
138 "`backend` attribute on the user."
139 )
140 else:
141 if not isinstance(backend, str):
142 raise TypeError(
143 "backend must be a dotted import path string (got %r)." % backend
144 )
145
146 request.session[SESSION_KEY] = user._meta.pk.value_to_string(user)
147 request.session[BACKEND_SESSION_KEY] = backend
148 request.session[HASH_SESSION_KEY] = session_auth_hash
149 if hasattr(request, "user"):
150 request.user = user
151 rotate_token(request)
152 user_logged_in.send(sender=user.__class__, request=request, user=user)

160def logout(request):
161 """
162 Remove the authenticated user's ID from the request and flush their session
163 data.
164 """
165 # Dispatch the signal before the user is logged out so the receivers have a
166 # chance to find out *who* logged out.
167 user = getattr(request, "user", None)
168 if not getattr(user, "is_authenticated", True):
169 user = None
170 user_logged_out.send(sender=user.__class__, request=request, user=user)
171 request.session.flush()
172 if hasattr(request, "user"):
173 from django.contrib.auth.models import AnonymousUser

175 request.user = AnonymousUser()

200def get_user(request):
201 """
202 Return the user model instance associated with the given request session.
203 If no user is retrieved, return an instance of `AnonymousUser`.
204 """
205 from .models import AnonymousUser
206
207 user = None
208 try:
209 user_id = _get_user_session_key(request)
210 backend_path = request.session[BACKEND_SESSION_KEY]
211 except KeyError:
212 pass
213 else:
214 if backend_path in settings.AUTHENTICATION_BACKENDS:
215 backend = load_backend(backend_path)
216 user = backend.get_user(user_id)
217 # Verify the session
218 if hasattr(user, "get_session_auth_hash"):
219 session_hash = request.session.get(HASH_SESSION_KEY)
220 if not session_hash:
221 session_hash_verified = False
222 else:
223 session_auth_hash = user.get_session_auth_hash()
224 session_hash_verified = constant_time_compare(
225 session_hash, session_auth_hash
226 )
227 if not session_hash_verified:
228 # If the current secret does not verify the session, try
229 # with the fallback secrets and stop when a matching one is
230 # found.
231 if session_hash and any(
232 constant_time_compare(session_hash, fallback_auth_hash)
233 for fallback_auth_hash in user.get_session_auth_fallback_hash()
234 ):
235 request.session.cycle_key()
236 request.session[HASH_SESSION_KEY] = session_auth_hash
237 else:
238 request.session.flush()
239 user = None
240
241 return user or AnonymousUser()
