73class TranslationCatalog:
74 """
75 Simulate a dict for DjangoTranslation._catalog so as multiple catalogs
76 with different plural equations are kept separate.
77 """
78
79 def __init__(self, trans=None):
80 self._catalogs = [trans._catalog.copy()] if trans else [{}]
81 self._plurals = [trans.plural] if trans else [lambda n: int(n != 1)]
82
83 def __getitem__(self, key):
84 for cat in self._catalogs:
85 try:
86 return cat[key]
87 except KeyError:
88 pass
89 raise KeyError(key)
90
91 def __setitem__(self, key, value):
92 self._catalogs[0][key] = value
93
94 def __contains__(self, key):
95 return any(key in cat for cat in self._catalogs)
96
97 def items(self):
98 for cat in self._catalogs:
99 yield from cat.items()
100
101 def keys(self):
102 for cat in self._catalogs:
103 yield from cat.keys()
104
105 def update(self, trans):
106 # Merge if plural function is the same, else prepend.
107 for cat, plural in zip(self._catalogs, self._plurals):
108 if trans.plural.__code__ == plural.__code__:
109 cat.update(trans._catalog)
110 break
111 else:
112 self._catalogs.insert(0, trans._catalog.copy())
113 self._plurals.insert(0, trans.plural)
114
115 def get(self, key, default=None):
116 missing = object()
117 for cat in self._catalogs:
118 result = cat.get(key, missing)
119 if result is not missing:
120 return result
121 return default
122
123 def plural(self, msgid, num):
124 for cat, plural in zip(self._catalogs, self._plurals):
125 tmsg = cat.get((msgid, plural(num)))
126 if tmsg is not None:
127 return tmsg
128 raise KeyError

370def gettext(message):
371 """
372 Translate the 'message' string. It uses the current thread to find the
373 translation object to use. If no current translation is activated, the
374 message will be run through the default translation object.
375 """
376 global _default
377
378 eol_message = message.replace("\r\n", "\n").replace("\r", "\n")
379
380 if eol_message:
381 _default = _default or translation(settings.LANGUAGE_CODE)
382 translation_object = getattr(_active, "value", _default)
383
384 result = translation_object.gettext(eol_message)
385 else:
386 # Return an empty value of the corresponding type if an empty message
387 # is given, instead of metadata, which is the default gettext behavior.
388 result = type(message)("")
389
390 if isinstance(message, SafeData):
391 return mark_safe(result)
392
393 return result

493@functools.lru_cache(maxsize=1000)
494def get_supported_language_variant(lang_code, strict=False):
495 """
496 Return the language code that's listed in supported languages, possibly
497 selecting a more generic variant. Raise LookupError if nothing is found.
498
499 If `strict` is False (the default), look for a country-specific variant
500 when neither the language code nor its generic variant is found.
501
502 The language code is truncated to a maximum length to avoid potential
503 denial of service attacks.
504
505 lru_cache should have a maxsize to prevent from memory exhaustion attacks,
506 as the provided language codes are taken from the HTTP request. See also
507 <https://www.djangoproject.com/weblog/2007/oct/26/security-fix/>.
508 """
509 if lang_code:
510 # Truncate the language code to a maximum length to avoid potential
511 # denial of service attacks.
512 if len(lang_code) > LANGUAGE_CODE_MAX_LENGTH:
513 if (
514 not strict
515 and (index := lang_code.rfind("-", 0, LANGUAGE_CODE_MAX_LENGTH)) > 0
516 ):
517 # There is a generic variant under the maximum length accepted length.
518 lang_code = lang_code[:index]
519 else:
520 raise LookupError(lang_code)
521 # If 'zh-hant-tw' is not supported, try special fallback or subsequent
522 # language codes i.e. 'zh-hant' and 'zh'.
523 possible_lang_codes = [lang_code]
524 try:
525 possible_lang_codes.extend(LANG_INFO[lang_code]["fallback"])
526 except KeyError:
527 pass
528 i = None
529 while (i := lang_code.rfind("-", 0, i)) > -1:
530 possible_lang_codes.append(lang_code[:i])
531 generic_lang_code = possible_lang_codes[-1]
532 supported_lang_codes = get_languages()
533
534 for code in possible_lang_codes:
535 if code.lower() in supported_lang_codes and check_for_language(code):
536 return code
537 if not strict:
538 # if fr-fr is not supported, try fr-ca.
539 for supported_code in supported_lang_codes:
540 if supported_code.startswith(generic_lang_code + "-"):
541 return supported_code
542 raise LookupError(lang_code)

562def get_language_from_request(request, check_path=False):
563 """
564 Analyze the request to find what language the user wants the system to
565 show. Only languages listed in settings.LANGUAGES are taken into account.
566 If the user requests a sublanguage where we have a main language, we send
567 out the main language.
568
569 If check_path is True, the URL path prefix will be checked for a language
570 code, otherwise this is skipped for backwards compatibility.
571 """
572 if check_path:
573 lang_code = get_language_from_path(request.path_info)
574 if lang_code is not None:
575 return lang_code
576
577 lang_code = request.COOKIES.get(settings.LANGUAGE_COOKIE_NAME)
578 if (
579 lang_code is not None
580 and lang_code in get_languages()
581 and check_for_language(lang_code)
582 ):
583 return lang_code
584
585 try:
586 return get_supported_language_variant(lang_code)
587 except LookupError:
588 pass
589
590 accept = request.META.get("HTTP_ACCEPT_LANGUAGE", "")
591 for accept_lang, unused in parse_accept_lang_header(accept):
592 if accept_lang == "*":
593 break
594
595 if not language_code_re.search(accept_lang):
596 continue
597
598 try:
599 return get_supported_language_variant(accept_lang)
600 except LookupError:
601 continue
602
603 try:
604 return get_supported_language_variant(settings.LANGUAGE_CODE)
605 except LookupError:
606 return settings.LANGUAGE_CODE

609@functools.lru_cache(maxsize=1000)
610def _parse_accept_lang_header(lang_string):
611 """
612 Parse the lang_string, which is the body of an HTTP Accept-Language
613 header, and return a tuple of (lang, q-value), ordered by 'q' values.
614
615 Return an empty tuple if there are any format errors in lang_string.
616 """
617 result = []
618 pieces = accept_language_re.split(lang_string.lower())
619 if pieces[-1]:
620 return ()
621 for i in range(0, len(pieces) - 1, 3):
622 first, lang, priority = pieces[i : i + 3]
623 if first:
624 return ()
625 if priority:
626 priority = float(priority)
627 else:
628 priority = 1.0
629 result.append((lang, priority))
630 result.sort(key=lambda k: k[1], reverse=True)
631 return tuple(result)
