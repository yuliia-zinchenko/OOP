55 def visit_asyncwith(self, node: nodes.AsyncWith) -> None:

57 inferred = checker_utils.safe_infer(ctx_mgr)
58 if inferred is None or isinstance(inferred, util.UninferableBase):

62 # Check if we are dealing with a function decorated

65 continue
66 elif isinstance(inferred, astroid.bases.AsyncGenerator):
67 # Check if we are dealing with a function decorated
68 # with contextlib.asynccontextmanager.
69 if decorated_with(inferred.parent, self._async_generators):
70 continue
71 else:
72 try:

74 inferred.getattr("__aexit__")
75 except astroid.exceptions.NotFoundError:

77 # If we do not know the bases of this class,

83 "not-async-context-manager"
