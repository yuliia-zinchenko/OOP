102class BaseArchive:
103 """
104 Base Archive class. Implementations should inherit this class.
105 """
106
107 @staticmethod
108 def _copy_permissions(mode, filename):
109 """
110 If the file in the archive has some permissions (this assumes a file
111 won't be writable/executable without being readable), apply those
112 permissions to the unarchived file.
113 """
114 if mode & stat.S_IROTH:
115 os.chmod(filename, mode)
116
117 def split_leading_dir(self, path):
118 path = str(path)
119 path = path.lstrip("/").lstrip("\\")
120 if "/" in path and (
121 ("\\" in path and path.find("/") < path.find("\\")) or "\\" not in path
122 ):
123 return path.split("/", 1)
124 elif "\\" in path:
125 return path.split("\\", 1)
126 else:
127 return path, ""
128
129 def has_leading_dir(self, paths):
130 """
131 Return True if all the paths have the same leading path name
132 (i.e., everything is in one subdirectory in an archive).
133 """
134 common_prefix = None
135 for path in paths:
136 prefix, rest = self.split_leading_dir(path)
137 if not prefix:
138 return False
139 elif common_prefix is None:
140 common_prefix = prefix
141 elif prefix != common_prefix:
142 return False
143 return True

145 def target_filename(self, to_path, name):
146 target_path = os.path.abspath(to_path)
147 filename = os.path.abspath(os.path.join(target_path, name))
148 if not filename.startswith(target_path):
149 raise SuspiciousOperation("Archive contains invalid path: '%s'" % name)
150 return filename

163class TarArchive(BaseArchive):
164 def __init__(self, file):
165 self._archive = tarfile.open(file)
166
167 def list(self, *args, **kwargs):
168 self._archive.list(*args, **kwargs)
169
170 def extract(self, to_path):
171 members = self._archive.getmembers()
172 leading = self.has_leading_dir(x.name for x in members)
173 for member in members:
174 name = member.name
175 if leading:
176 name = self.split_leading_dir(name)[1]
177 filename = self.target_filename(to_path, name)
178 if member.isdir():
179 if filename:
180 os.makedirs(filename, exist_ok=True)
181 else:
182 try:
183 extracted = self._archive.extractfile(member)
184 except (KeyError, AttributeError) as exc:
185 # Some corrupt tar files seem to produce this
186 # (specifically bad symlinks)
187 print(
188 "In the tar file %s the member %s is invalid: %s"
189 % (name, member.name, exc)
190 )
191 else:
192 dirname = os.path.dirname(filename)
193 if dirname:
194 os.makedirs(dirname, exist_ok=True)
195 with open(filename, "wb") as outfile:
196 shutil.copyfileobj(extracted, outfile)
197 self._copy_permissions(member.mode, filename)
198 finally:
199 if extracted:
200 extracted.close()
201
202 def close(self):
203 self._archive.close()

206class ZipArchive(BaseArchive):
207 def __init__(self, file):
208 self._archive = zipfile.ZipFile(file)
209
210 def list(self, *args, **kwargs):
211 self._archive.printdir(*args, **kwargs)

213 def extract(self, to_path):
214 namelist = self._archive.namelist()
215 leading = self.has_leading_dir(namelist)
216 for name in namelist:
217 data = self._archive.read(name)
218 info = self._archive.getinfo(name)
219 if leading:
220 name = self.split_leading_dir(name)[1]
221 if not name:
222 continue
223 filename = self.target_filename(to_path, name)
224 if name.endswith(("/", "\\")):
225 # A directory
226 os.makedirs(filename, exist_ok=True)
227 else:
228 dirname = os.path.dirname(filename)
229 if dirname:
230 os.makedirs(dirname, exist_ok=True)
231 with open(filename, "wb") as outfile:
232 outfile.write(data)
233 # Convert ZipInfo.external_attr to mode
234 mode = info.external_attr >> 16
235 self._copy_permissions(mode, filename)

238 self._archive.close()
