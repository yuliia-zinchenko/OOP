28def _builtin_exceptions() -> set[str]:
29 def predicate(obj: Any) -> bool:
30 return isinstance(obj, type) and issubclass(obj, BaseException)
31
32 members = inspect.getmembers(builtins, predicate)
33 return {exc.__name__ for (_, exc) in members}

36def _annotated_unpack_infer(
37 stmt: nodes.NodeNG, context: InferenceContext | None = None
38) -> Generator[tuple[nodes.NodeNG, SuccessfulInferenceResult]]:
39 """Recursively generate nodes inferred by the given statement.
40
41 If the inferred value is a list or a tuple, recurse on the elements.
42 Returns an iterator which yields tuples in the format
43 ('original node', 'inferred node').
44 """
45 if isinstance(stmt, (nodes.List, nodes.Tuple)):
46 for elt in stmt.elts:
47 inferred = utils.safe_infer(elt)
48 if inferred and not isinstance(inferred, util.UninferableBase):
49 yield elt, inferred
50 return
51 for inferred in stmt.infer(context):
52 if isinstance(inferred, util.UninferableBase):
53 continue
54 yield stmt, inferred

337 def _check_misplaced_bare_raise(self, node: nodes.Raise) -> None:
338 # Filter out if it's present in __exit__.
339 scope = node.scope()
340 if (
341 isinstance(scope, nodes.FunctionDef)
342 and scope.is_method()
343 and scope.name == "__exit__"
344 ):
345 return
346
347 current = node
348 # Stop when a new scope is generated or when the raise
349 # statement is found inside a Try.
350 ignores = (nodes.ExceptHandler, nodes.FunctionDef)
351 while current and not isinstance(current.parent, ignores):
352 current = current.parent
353
354 expected = (nodes.ExceptHandler,)
355 if not current or not isinstance(current.parent, expected):
356 self.add_message("misplaced-bare-raise", node=node, confidence=HIGH)

358 def _check_bad_exception_cause(self, node: nodes.Raise) -> None:
359 """Verify that the exception cause is properly set.
360
361 An exception cause can be only `None` or an exception.
362 """
363 cause = utils.safe_infer(node.cause)
364 if cause is None or isinstance(cause, util.UninferableBase):
365 return
366
367 if isinstance(cause, nodes.Const):
368 if cause.value is not None:
369 self.add_message("bad-exception-cause", node=node, confidence=INFERENCE)
370 elif not isinstance(cause, nodes.ClassDef) and not utils.inherit_from_std_ex(
371 cause
372 ):
373 self.add_message("bad-exception-cause", node=node, confidence=INFERENCE)

375 def _check_raise_missing_from(self, node: nodes.Raise) -> None:
376 if node.exc is None:
377 # This is a plain `raise`, raising the previously-caught exception. No need for a
378 # cause.
379 return
380 # We'd like to check whether we're inside an `except` clause:
381 containing_except_node = utils.find_except_wrapper_node_in_scope(node)
382 if not containing_except_node:
383 return
384 # We found a surrounding `except`! We're almost done proving there's a
385 # `raise-missing-from` here. The only thing we need to protect against is that maybe
386 # the `raise` is raising the exception that was caught, possibly with some shenanigans
387 # like `exc.with_traceback(whatever)`. We won't analyze these, we'll just assume
388 # there's a violation on two simple cases: `raise SomeException(whatever)` and `raise
389 # SomeException`.
390 if containing_except_node.name is None:
391 # The `except` doesn't have an `as exception:` part, meaning there's no way that
392 # the `raise` is raising the same exception.
393 class_of_old_error = "Exception"
394 if isinstance(containing_except_node.type, (nodes.Name, nodes.Tuple)):
395 # 'except ZeroDivisionError' or 'except (ZeroDivisionError, ValueError)'
396 class_of_old_error = containing_except_node.type.as_string()
397 self.add_message(
398 "raise-missing-from",
399 node=node,
400 args=(
401 f"'except {class_of_old_error} as exc' and ",
402 node.as_string(),
403 "exc",
404 ),
405 confidence=HIGH,
406 )
407 elif (
408 isinstance(node.exc, nodes.Call)
409 and isinstance(node.exc.func, nodes.Name)
410 or isinstance(node.exc, nodes.Name)
411 and node.exc.name != containing_except_node.name.name
412 ):
413 # We have a `raise SomeException(whatever)` or a `raise SomeException`
414 self.add_message(
415 "raise-missing-from",
416 node=node,
417 args=("", node.as_string(), containing_except_node.name.name),
418 confidence=HIGH,
419 )

421 def _check_catching_non_exception(
422 self,
423 handler: nodes.ExceptHandler,
424 exc: SuccessfulInferenceResult,
425 part: nodes.NodeNG,
426 ) -> None:
427 if isinstance(exc, nodes.Tuple):
428 # Check if it is a tuple of exceptions.
429 inferred = [utils.safe_infer(elt) for elt in exc.elts]
430 if any(isinstance(node, util.UninferableBase) for node in inferred):
431 # Don't emit if we don't know every component.
432 return
433 if all(
434 node
435 and (utils.inherit_from_std_ex(node) or not utils.has_known_bases(node))
436 for node in inferred
437 ):
438 return
439
440 if not isinstance(exc, nodes.ClassDef):
441 # Don't emit the warning if the inferred stmt
442 # is None, but the exception handler is something else,
443 # maybe it was redefined.
444 if isinstance(exc, nodes.Const) and exc.value is None:
445 if (
446 isinstance(handler.type, nodes.Const) and handler.type.value is None
447 ) or handler.type.parent_of(exc):
448 # If the exception handler catches None or
449 # the exception component, which is None, is
450 # defined by the entire exception handler, then
451 # emit a warning.
452 self.add_message(
453 "catching-non-exception",
454 node=handler.type,
455 args=(part.as_string(),),
456 )
457 else:
458 self.add_message(
459 "catching-non-exception",
460 node=handler.type,
461 args=(part.as_string(),),
462 )
463 return
464
465 if (
466 not utils.inherit_from_std_ex(exc)
467 and exc.name not in self._builtin_exceptions
468 ):
469 if utils.has_known_bases(exc):
470 self.add_message(
471 "catching-non-exception", node=handler.type, args=(exc.name,)

474 def _check_try_except_raise(self, node: nodes.Try) -> None:
475 def gather_exceptions_from_handler(
476 handler: nodes.ExceptHandler,
477 ) -> list[InferenceResult] | None:
478 exceptions: list[InferenceResult] = []
479 if handler.type:
480 exceptions_in_handler = utils.safe_infer(handler.type)
481 if isinstance(exceptions_in_handler, nodes.Tuple):
482 exceptions = list(
483 {
484 exception
485 for exception in exceptions_in_handler.elts
486 if isinstance(exception, (nodes.Name, nodes.Attribute))
487 }
488 )
489 elif exceptions_in_handler:
490 exceptions = [exceptions_in_handler]
491 else:
492 # Break when we cannot infer anything reliably.
493 return None
494 return exceptions
495
496 bare_raise = False
497 handler_having_bare_raise = None
498 exceptions_in_bare_handler: list[InferenceResult] | None = []
499 for handler in node.handlers:
500 if bare_raise:
501 # check that subsequent handler is not parent of handler which had bare raise.
502 # since utils.safe_infer can fail for bare except, check it before.
503 # also break early if bare except is followed by bare except.
504
505 excs_in_current_handler = gather_exceptions_from_handler(handler)
506 if not excs_in_current_handler:
507 break
508 if exceptions_in_bare_handler is None:
509 # It can be `None` when the inference failed
510 break
511 for exc_in_current_handler in excs_in_current_handler:
512 inferred_current = utils.safe_infer(exc_in_current_handler)
513 if any(
514 utils.is_subclass_of(utils.safe_infer(e), inferred_current)
515 for e in exceptions_in_bare_handler
516 ):
517 bare_raise = False
518 break
519
520 # `raise` as the first operator inside the except handler
521 if _is_raising([handler.body[0]]):
522 # flags when there is a bare raise
523 if handler.body[0].exc is None:
524 bare_raise = True
525 handler_having_bare_raise = handler
526 exceptions_in_bare_handler = gather_exceptions_from_handler(handler)
527 else:
528 if bare_raise:
529 self.add_message("try-except-raise", node=handler_having_bare_raise)

572 def visit_try(self, node: nodes.Try) -> None:
573 """Check for empty except."""
574 self._check_try_except_raise(node)
575 exceptions_classes: list[Any] = []
576 nb_handlers = len(node.handlers)
577 for index, handler in enumerate(node.handlers):
578 if handler.type is None:
579 if not _is_raising(handler.body):
580 self.add_message("bare-except", node=handler, confidence=HIGH)
581
582 # check if an "except:" is followed by some other
583 # except
584 if index < (nb_handlers - 1):
585 msg = "empty except clause should always appear last"
586 self.add_message(
587 "bad-except-order", node=node, args=msg, confidence=HIGH
588 )
589
590 elif isinstance(handler.type, nodes.BoolOp):
591 self.add_message(
592 "binary-op-exception",
593 node=handler,
594 args=handler.type.op,
595 confidence=HIGH,
596 )
597 else:
598 try:
599 exceptions = list(_annotated_unpack_infer(handler.type))
600 except astroid.InferenceError:
601 continue
602
603 for part, exception in exceptions:
604 if isinstance(
605 exception, astroid.Instance
606 ) and utils.inherit_from_std_ex(exception):
607 exception = exception._proxied
608
609 self._check_catching_non_exception(handler, exception, part)
610
611 if not isinstance(exception, nodes.ClassDef):
612 continue
613
614 exc_ancestors = [
615 anc
616 for anc in exception.ancestors()
617 if isinstance(anc, nodes.ClassDef)
618 ]
619
620 for previous_exc in exceptions_classes:
621 if previous_exc in exc_ancestors:
622 msg = f"{previous_exc.name} is an ancestor class of {exception.name}"
623 self.add_message(
624 "bad-except-order",
625 node=handler.type,
626 args=msg,
627 confidence=INFERENCE,
628 )
629 if self._is_overgeneral_exception(exception) and not _is_raising(
630 handler.body
631 ):
632 self.add_message(
633 "broad-exception-caught",
634 args=exception.name,
635 node=handler.type,
636 confidence=INFERENCE,
637 )
638

640 self.add_message(
641 "duplicate-except",
642 args=exception.name,
643 node=handler.type,
644 confidence=INFERENCE,
645 )

647 exceptions_classes += [exc for _, exc in exceptions]
