28def as_crispy_form(form, template_pack=TEMPLATE_PACK, label_class="", field_class=""):
29 """
30 The original and still very useful way to generate a div elegant form/formset::
31
32 {% load crispy_forms_tags %}
33
34 <form class="my-class" method="post">
35 {% csrf_token %}
36 {{ myform|crispy }}
37 </form>
38
39 or, if you want to explicitly set the template pack::
40
41 {{ myform|crispy:"bootstrap4" }}
42
43 In ``bootstrap3`` or ``bootstrap4`` for horizontal forms you can do::
44
45 {{ myform|label_class:"col-lg-2",field_class:"col-lg-8" }}
46 """
47 c = {
48 "field_class": field_class,
49 "field_template": "%s/field.html" % template_pack,
50 "form_show_errors": True,
51 "form_show_labels": True,
52 "label_class": label_class,
53 }
54 if isinstance(form, BaseFormSet):
55 template = uni_formset_template(template_pack)
56 c["formset"] = form
57 else:
58 template = uni_form_template(template_pack)
59 c["form"] = form
60
61 return template.render(c)

87def as_crispy_field(field, template_pack=TEMPLATE_PACK, label_class="", field_class=""):
88 """
89 Renders a form field like a django-crispy-forms field::
90
91 {% load crispy_forms_tags %}
92 {{ form.field|as_crispy_field }}
93
94 or::
95
96 {{ form.field|as_crispy_field:"bootstrap4" }}
97 """
98 if not isinstance(field, boundfield.BoundField) and settings.DEBUG:
99 raise CrispyError("|as_crispy_field got passed an invalid or inexistent field")
100
101 attributes = {
102 "field": field,
103 "form_show_errors": True,
104 "form_show_labels": True,
105 "label_class": label_class,
106 "field_class": field_class,
107 }
108 helper = getattr(field.form, "helper", None)
109
110 template_path = None
111 if helper is not None:
112 attributes.update(helper.get_attributes(template_pack))
113 template_path = helper.field_template
114 if not template_path:
115 template_path = "%s/field.html" % template_pack
116 template = get_template(template_path)
117
118 return template.render(attributes)

128 """
129 A template filter to help rendering of fields with optgroups.
130
131 Returns:
132 A tuple of label, option, index
133
134 label: Group label for grouped optgroups (`None` if inputs are not

136
137 option: A dict containing information to render the option::

140 "name": "checkbox_select_multiple",
141 "value": 1,
142 "label": 1,
143 "selected": False,
144 "index": "0",
145 "attrs": {"id": "id_checkbox_select_multiple_0"},

150

152
153 """
