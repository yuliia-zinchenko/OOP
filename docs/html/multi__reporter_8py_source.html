21class MultiReporter:
22 """Reports messages and layouts in plain text."""
23
24 name = "_internal_multi_reporter"
25 # Note: do not register this reporter with linter.register_reporter as it is
26 # not intended to be used directly like a regular reporter, but is
27 # instead used to implement the
28 # `--output-format=json:somefile.json,colorized`
29 # multiple output formats feature
30
31 extension = ""

33 def __init__(
34 self,
35 sub_reporters: list[BaseReporter],
36 close_output_files: Callable[[], None],
37 output: TextIO | None = None,
38 ):
39 self._sub_reporters = sub_reporters
40 self.close_output_files = close_output_files
41 self._path_strip_prefix = os.getcwd() + os.sep
42 self._linter: PyLinter | None = None
43 self.out = output
44 self.messages: list[Message] = []

46 @property
47 def out(self) -> TextIO | None:
48 return self.__out
49
50 @out.setter
51 def out(self, output: TextIO | None = None) -> None:
52 """MultiReporter doesn't have its own output.
53
54 This method is only provided for API parity with BaseReporter
55 and should not be called with non-None values for 'output'.
56 """
57 self.__out = None
58 if output is not None:
59 raise NotImplementedError("MultiReporter does not support direct output.")
60
61 def __del__(self) -> None:

78 def handle_message(self, msg: Message) -> None:
79 """Handle a new message triggered on the current file."""
80 for rep in self._sub_reporters:
81 # We provide a copy so reporters can't modify message for others.
82 rep.handle_message(copy(msg))
83
84 def writeln(self, string: str = "") -> None:
85 """Write a line in the output buffer."""
86 for rep in self._sub_reporters:
87 rep.writeln(string)
88
89 def display_reports(self, layout: Section) -> None:
90 """Display results encapsulated in the layout tree."""
91 for rep in self._sub_reporters:
92 rep.display_reports(layout)
93
94 def display_messages(self, layout: Section | None) -> None:
95 """Hook for displaying the messages of the reporter."""
96 for rep in self._sub_reporters:
97 rep.display_messages(layout)
98
99 def on_set_current_module(self, module: str, filepath: str | None) -> None:
100 """Hook called when a module starts to be analysed."""
101 for rep in self._sub_reporters:
102 rep.on_set_current_module(module, filepath)

108 ) -> None:
109 """Hook called when a module finished analyzing."""
