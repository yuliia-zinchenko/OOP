51 def render(self, context):
52 block_context = context.render_context.get(BLOCK_CONTEXT_KEY)
53 with context.push():
54 if block_context is None:
55 context["block"] = self
56 result = self.nodelist.render(context)
57 else:
58 push = block = block_context.pop(self.name)
59 if block is None:
60 block = self
61 # Create new block so we can store context without thread-safety issues.
62 block = type(self)(block.name, block.nodelist)
63 block.context = context
64 context["block"] = block
65 result = block.nodelist.render(context)
66 if push is not None:
67 block_context.push(self.name, push)
68 return result

134 compiled_parent = self.get_parent(context)
135
136 if BLOCK_CONTEXT_KEY not in context.render_context:

144 # and its block nodes also need to be added to the block context.

176 def render(self, context):
177 """
178 Render the specified template and context. Cache the template object
179 in render_context to avoid reparsing and loading when used in a for
180 loop.
181 """
182 template = self.template.resolve(context)
183 # Does this quack like a Template?
184 if not callable(getattr(template, "render", None)):
185 # If not, try the cache and select_template().
186 template_name = template or ()
187 if isinstance(template_name, str):
188 template_name = (
189 construct_relative_path(
190 self.origin.template_name,
191 template_name,
192 ),
193 )
194 else:
195 template_name = tuple(template_name)
196 cache = context.render_context.dicts[0].setdefault(self, {})
197 template = cache.get(template_name)
198 if template is None:
199 template = context.template.engine.select_template(template_name)
200 cache[template_name] = template
201 # Use the base.Template of a backends.django.Template.
202 elif hasattr(template, "template"):
203 template = template.template
204 values = {
205 name: var.resolve(context) for name, var in self.extra_context.items()
206 }
207 if self.isolated_context:
208 return template.render(context.new(values))
209 with context.push(**values):
210 return template.render(context)

213@register.tag("block")
214def do_block(parser, token):
215 """
216 Define a block that can be overridden by child templates.
217 """
218 # token.split_contents() isn't useful here because this tag doesn't accept
219 # variable as arguments.
220 bits = token.contents.split()
221 if len(bits) != 2:
222 raise TemplateSyntaxError("'%s' tag takes only one argument" % bits[0])
223 block_name = bits[1]
224 # Keep track of the names of BlockNodes found in this template, so we can
225 # check for duplication.
226 try:
227 if block_name in parser.__loaded_blocks:
228 raise TemplateSyntaxError(
229 "'%s' tag with name '%s' appears more than once" % (bits[0], block_name)
230 )
231 parser.__loaded_blocks.append(block_name)
232 except AttributeError: # parser.__loaded_blocks isn't a list yet
233 parser.__loaded_blocks = [block_name]
234 nodelist = parser.parse(("endblock",))
235
236 # This check is kept for backwards-compatibility. See #3100.
237 endblock = parser.next_token()
238 acceptable_endblocks = ("endblock", "endblock %s" % block_name)
239 if endblock.contents not in acceptable_endblocks:
240 parser.invalid_block_tag(endblock, "endblock", acceptable_endblocks)
241
242 return BlockNode(block_name, nodelist)

245def construct_relative_path(current_template_name, relative_name):
246 """
247 Convert a relative path (starting with './' or '../') to the full template
248 name based on the current_template_name.
249 """
250 new_name = relative_name.strip("'\"")
251 if not new_name.startswith(("./", "../")):
252 # relative_name is a variable or a literal that doesn't contain a
253 # relative path.
254 return relative_name
255
256 new_name = posixpath.normpath(
257 posixpath.join(
258 posixpath.dirname(current_template_name.lstrip("/")),
259 new_name,
260 )
261 )
262 if new_name.startswith("../"):
263 raise TemplateSyntaxError(
264 "The relative path '%s' points outside the file hierarchy that "
265 "template '%s' is in." % (relative_name, current_template_name)
266 )
267 if current_template_name.lstrip("/") == new_name:
268 raise TemplateSyntaxError(
269 "The relative path '%s' was translated to template name '%s', the "
270 "same template in which the tag appears."
271 % (relative_name, current_template_name)
272 )
273 has_quotes = (
274 relative_name.startswith(('"', "'")) and relative_name[0] == relative_name[-1]
275 )
276 return f'"{new_name}"' if has_quotes else new_name

279@register.tag("extends")
280def do_extends(parser, token):
281 """
282 Signal that this template extends a parent template.
283
284 This tag may be used in two ways: ``{% extends "base" %}`` (with quotes)
285 uses the literal value "base" as the name of the parent template to extend,
286 or ``{% extends variable %}`` uses the value of ``variable`` as either the
287 name of the parent template to extend (if it evaluates to a string) or as
288 the parent template itself (if it evaluates to a Template object).
289 """
290 bits = token.split_contents()
291 if len(bits) != 2:
292 raise TemplateSyntaxError("'%s' takes one argument" % bits[0])
293 bits[1] = construct_relative_path(parser.origin.template_name, bits[1])
294 parent_name = parser.compile_filter(bits[1])
295 nodelist = parser.parse()
296 if nodelist.get_nodes_by_type(ExtendsNode):
297 raise TemplateSyntaxError(
298 "'%s' cannot appear more than once in the same template" % bits[0]
299 )
300 return ExtendsNode(nodelist, parent_name)

303@register.tag("include")
304def do_include(parser, token):
305 """
306 Load a template and render it with the current context. You can pass
307 additional context using keyword arguments.
308
309 Example::
310
311 {% include "foo/some_include" %}
312 {% include "foo/some_include" with bar="BAZZ!" baz="BING!" %}
313
314 Use the ``only`` argument to exclude the current context when rendering
315 the included template::
316
317 {% include "foo/some_include" only %}
318 {% include "foo/some_include" with bar="1" only %}
319 """
320 bits = token.split_contents()
321 if len(bits) < 2:
322 raise TemplateSyntaxError(
323 "%r tag takes at least one argument: the name of the template to "
324 "be included." % bits[0]
325 )
326 options = {}
327 remaining_bits = bits[2:]
328 while remaining_bits:
329 option = remaining_bits.pop(0)
330 if option in options:
331 raise TemplateSyntaxError(
332 "The %r option was specified more than once." % option
333 )
334 if option == "with":
335 value = token_kwargs(remaining_bits, parser, support_legacy=False)
336 if not value:
337 raise TemplateSyntaxError(
338 '"with" in %r tag needs at least one keyword argument.' % bits[0]
339 )
340 elif option == "only":
341 value = True
342 else:
343 raise TemplateSyntaxError(
344 "Unknown argument for %r tag: %r." % (bits[0], option)
345 )
346 options[option] = value
347 isolated_context = options.get("only", False)
348 namemap = options.get("with", {})
349 bits[1] = construct_relative_path(parser.origin.template_name, bits[1])
350 return IncludeNode(
351 parser.compile_filter(bits[1]),
352 extra_context=namemap,
353 isolated_context=isolated_context,
354 )
