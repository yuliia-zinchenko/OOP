Detailed Description

Django uses "format" (e.g. '%s') style placeholders, but Oracle uses ":var"
style. This fixes it -- but note that if you want to use a literal "%s" in
a query, you'll need to use "%%s".

530 def _fix_for_params(self, query, params, unify_by_values=False):
531 # oracledb wants no trailing ';' for SQL statements. For PL/SQL, it
532 # it does want a trailing ';' but not a trailing '/'. However, these
533 # characters must be included in the original query in case the query
534 # is being passed to SQL*Plus.
535 if query.endswith(";") or query.endswith("/"):
536 query = query[:-1]
537 if params is None:
538 params = []
539 elif hasattr(params, "keys"):
540 # Handle params as dict
541 args = {k: ":%s" % k for k in params}
542 query %= args
543 elif unify_by_values and params:
544 # Handle params as a dict with unified query parameters by their
545 # values. It can be used only in single query execute() because
546 # executemany() shares the formatted query with each of the params
547 # list. e.g. for input params = [0.75, 2, 0.75, 'sth', 0.75]
548 # params_dict = {
549 # (float, 0.75): ':arg0',
550 # (int, 2): ':arg1',
551 # (str, 'sth'): ':arg2',
552 # }
553 # args = [':arg0', ':arg1', ':arg0', ':arg2', ':arg0']
554 # params = {':arg0': 0.75, ':arg1': 2, ':arg2': 'sth'}
555 # The type of parameters in param_types keys is necessary to avoid
556 # unifying 0/1 with False/True.
557 param_types = [(type(param), param) for param in params]
558 params_dict = {
559 param_type: ":arg%d" % i
560 for i, param_type in enumerate(dict.fromkeys(param_types))
561 }
562 args = [params_dict[param_type] for param_type in param_types]
563 params = {
564 placeholder: param for (_, param), placeholder in params_dict.items()
565 }
566 query %= tuple(args)
567 else:
568 # Handle params as sequence
569 args = [(":arg%d" % i) for i in range(len(params))]
570 query %= tuple(args)
571 return query, self._format_params(params)
