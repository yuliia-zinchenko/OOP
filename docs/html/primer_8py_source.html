115 def _minimum_python_supported(package_data: dict[str, str]) -> bool:
116 min_python_str = package_data.get("minimum_python", None)
117 if not min_python_str:
118 return True
119 min_python_tuple = tuple(int(n) for n in min_python_str.split("."))
120 return min_python_tuple <= sys.version_info[:2]
