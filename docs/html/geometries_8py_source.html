1"""
2 The OGRGeometry is a wrapper for using the OGR Geometry class
3 (see https://gdal.org/api/ogrgeometry_cpp.html#_CPPv411OGRGeometry).
4 OGRGeometry may be instantiated when reading geometries from OGR Data Sources
5 (e.g. SHP files), or when given OGC WKT (a string).
6
7 While the 'full' API is not present yet, the API is "pythonic" unlike
8 the traditional and "next-generation" OGR Python bindings. One major
9 advantage OGR Geometries have over their GEOS counterparts is support
10 for spatial reference systems and their transformation.
11
12 Example:
13 >>> from django.contrib.gis.gdal import OGRGeometry, OGRGeomType, SpatialReference
14 >>> wkt1, wkt2 = 'POINT(-90 30)', 'POLYGON((0 0, 5 0, 5 5, 0 5)'
15 >>> pnt = OGRGeometry(wkt1)
16 >>> print(pnt)
17 POINT (-90 30)
18 >>> mpnt = OGRGeometry(OGRGeomType('MultiPoint'), SpatialReference('WGS84'))
19 >>> mpnt.add(wkt1)
20 >>> mpnt.add(wkt1)
21 >>> print(mpnt)
22 MULTIPOINT (-90 30,-90 30)
23 >>> print(mpnt.srs.name)
24 WGS 84
25 >>> print(mpnt.srs.proj)
26 +proj=longlat +ellps=WGS84 +datum=WGS84 +no_defs
27 >>> mpnt.transform(SpatialReference('NAD27'))
28 >>> print(mpnt.proj)
29 +proj=longlat +ellps=clrk66 +datum=NAD27 +no_defs
30 >>> print(mpnt)
31 MULTIPOINT (-89.99993037860248 29.99979788655764,-89.99993037860248 29.99979788655764)
32
33 The OGRGeomType class is to make it easy to specify an OGR geometry type:
34 >>> from django.contrib.gis.gdal import OGRGeomType
35 >>> gt1 = OGRGeomType(3) # Using an integer for the type
36 >>> gt2 = OGRGeomType('Polygon') # Using a string
37 >>> gt3 = OGRGeomType('POLYGON') # It's case-insensitive
38 >>> print(gt1 == 3, gt1 == 'Polygon') # Equivalence works w/non-OGRGeomType objects
39 True True
40"""

68 def __init__(self, geom_input, srs=None):
69 """Initialize Geometry on either WKT or an OGR pointer as input."""
70 str_instance = isinstance(geom_input, str)
71
72 # If HEX, unpack input to a binary buffer.
73 if str_instance and hex_regex.match(geom_input):
74 geom_input = memoryview(bytes.fromhex(geom_input))
75 str_instance = False
76
77 # Constructing the geometry,
78 if str_instance:
79 wkt_m = wkt_regex.match(geom_input)
80 json_m = json_regex.match(geom_input)
81 if wkt_m:
82 if wkt_m["srid"]:
83 # If there's EWKT, set the SRS w/value of the SRID.
84 srs = int(wkt_m["srid"])
85 if wkt_m["type"].upper() == "LINEARRING":
86 # OGR_G_CreateFromWkt doesn't work with LINEARRING WKT.
87 # See https://trac.osgeo.org/gdal/ticket/1992.
88 g = capi.create_geom(OGRGeomType(wkt_m["type"]).num)
89 capi.import_wkt(g, byref(c_char_p(wkt_m["wkt"].encode())))
90 else:
91 g = capi.from_wkt(
92 byref(c_char_p(wkt_m["wkt"].encode())), None, byref(c_void_p())
93 )
94 elif json_m:
95 g = self._from_json(geom_input.encode())
96 else:
97 # Seeing if the input is a valid short-hand string
98 # (e.g., 'Point', 'POLYGON').
99 OGRGeomType(geom_input)
100 g = capi.create_geom(OGRGeomType(geom_input).num)
101 elif isinstance(geom_input, memoryview):
102 # WKB was passed in
103 g = self._from_wkb(geom_input)
104 elif isinstance(geom_input, OGRGeomType):
105 # OGRGeomType was passed in, an empty geometry will be created.
106 g = capi.create_geom(geom_input.num)
107 elif isinstance(geom_input, self.ptr_type):
108 # OGR pointer (c_void_p) was the input.
109 g = geom_input
110 else:
111 raise GDALException(
112 "Invalid input type for OGR Geometry construction: %s"
113 % type(geom_input)
114 )
115
116 # Now checking the Geometry pointer before finishing initialization
117 # by setting the pointer for the object.
118 if not g:
119 raise GDALException(
120 "Cannot create OGR Geometry from input: %s" % geom_input
121 )
122 self.ptr = g
123
124 # Assigning the SpatialReference object to the geometry, if valid.
125 if srs:
126 self.srs = srs
127
128 # Setting the class depending upon the OGR Geometry Type
129 if (geo_class := GEO_CLASSES.get(self.geom_type.num)) is None:
130 raise TypeError(f"Unsupported geometry type: {self.geom_type}")
131 self.__class__ = geo_class

320 # Do not have to clone the `SpatialReference` object pointer because

447 """
