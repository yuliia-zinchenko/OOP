46def get_tag_uri(url, date):
47 """
48 Create a TagURI.
49
50 See
51 https://web.archive.org/web/20110514113830/http://diveintomark.org/archives/2004/05/28/howto-atom-id
52 """
53 bits = urlparse(url)
54 d = ""
55 if date is not None:
56 d = ",%s" % date.strftime("%Y-%m-%d")
57 return "tag:%s%s:%s/%s" % (bits.hostname, d, bits.path, bits.fragment)

198 def latest_post_date(self):
199 """
200 Return the latest item's pubdate or updateddate. If no items
201 have either of these attributes this return the current UTC date/time.
202 """
203 latest_date = None
204 date_keys = ("updateddate", "pubdate")
205
206 for item in self.items:
207 for date_key in date_keys:
208 item_date = item.get(date_key)
209 if item_date:
210 if latest_date is None or item_date > latest_date:
211 latest_date = item_date
212
213 return latest_date or datetime.datetime.now(tz=datetime.timezone.utc)
