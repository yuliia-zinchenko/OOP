16class Job:
17 """
18 Contains the options given when scheduling callables and its current schedule and other state.
19 This class should never be instantiated by the user.
20
21 :var str id: the unique identifier of this job
22 :var str name: the description of this job
23 :var func: the callable to execute
24 :var tuple|list args: positional arguments to the callable
25 :var dict kwargs: keyword arguments to the callable
26 :var bool coalesce: whether to only run the job once when several run times are due
27 :var trigger: the trigger object that controls the schedule of this job
28 :var str executor: the name of the executor that will run this job
29 :var int misfire_grace_time: the time (in seconds) how much this job's execution is allowed to
30 be late (``None`` means "allow the job to run no matter how late it is")
31 :var int max_instances: the maximum number of concurrently executing instances allowed for this
32 job
33 :var datetime.datetime next_run_time: the next scheduled run time of this job
34
35 .. note::
36 The ``misfire_grace_time`` has some non-obvious effects on job execution. See the
37 :ref:`missed-job-executions` section in the documentation for an in-depth explanation.
38 """

138 def _get_run_times(self, now):
139 """
140 Computes the scheduled run times between ``next_run_time`` and ``now`` (inclusive).
141
142 :type now: datetime.datetime
143 :rtype: list[datetime.datetime]
144
145 """
146 run_times = []
147 next_run_time = self.next_run_time
148 while next_run_time and next_run_time <= now:
149 run_times.append(next_run_time)
150 next_run_time = self.trigger.get_next_fire_time(next_run_time, now)
151
152 return run_times

154 def _modify(self, **changes):
155 """
156 Validates the changes to the Job and makes the modifications if and only if all of them
157 validate.
158
159 """
160 approved = {}
161
162 if "id" in changes:
163 value = changes.pop("id")
164 if not isinstance(value, str):
165 raise TypeError("id must be a nonempty string")
166 if hasattr(self, "id"):
167 raise ValueError("The job ID may not be changed")
168 approved["id"] = value
169
170 if "func" in changes or "args" in changes or "kwargs" in changes:
171 func = changes.pop("func") if "func" in changes else self.func
172 args = changes.pop("args") if "args" in changes else self.args
173 kwargs = changes.pop("kwargs") if "kwargs" in changes else self.kwargs
174
175 if isinstance(func, str):
176 func_ref = func
177 func = ref_to_obj(func)
178 elif callable(func):
179 try:
180 func_ref = obj_to_ref(func)
181 except ValueError:
182 # If this happens, this Job won't be serializable
183 func_ref = None
184 else:
185 raise TypeError("func must be a callable or a textual reference to one")
186
187 if not hasattr(self, "name") and changes.get("name", None) is None:
188 changes["name"] = get_callable_name(func)
189
190 if isinstance(args, str) or not isinstance(args, Iterable):
191 raise TypeError("args must be a non-string iterable")
192 if isinstance(kwargs, str) or not isinstance(kwargs, Mapping):
193 raise TypeError("kwargs must be a dict-like object")
194
195 check_callable_args(func, args, kwargs)
196
197 approved["func"] = func
198 approved["func_ref"] = func_ref
199 approved["args"] = args
200 approved["kwargs"] = kwargs
201
202 if "name" in changes:
203 value = changes.pop("name")
204 if not value or not isinstance(value, str):
205 raise TypeError("name must be a nonempty string")
206 approved["name"] = value
207
208 if "misfire_grace_time" in changes:
209 value = changes.pop("misfire_grace_time")
210 if value is not None and (not isinstance(value, int) or value <= 0):
211 raise TypeError(
212 "misfire_grace_time must be either None or a positive integer"
213 )
214 approved["misfire_grace_time"] = value
215
216 if "coalesce" in changes:
217 value = bool(changes.pop("coalesce"))
218 approved["coalesce"] = value
219
220 if "max_instances" in changes:
221 value = changes.pop("max_instances")
222 if not isinstance(value, int) or value <= 0:
223 raise TypeError("max_instances must be a positive integer")
224 approved["max_instances"] = value
