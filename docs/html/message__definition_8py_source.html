75 def may_be_emitted(self, py_version: tuple[int, ...] | sys._version_info) -> bool:
76 """May the message be emitted using the configured py_version?"""
77 if self.minversion is not None and self.minversion > py_version:

83 def format_help(self, checkerref: bool = False) -> str:
84 """Return the help string for the given message id."""
85 desc = self.description
86 if checkerref:
87 desc += f" This message belongs to the {self.checker_name} checker."
88 title = self.msg
89 if self.minversion or self.maxversion:
90 restr = []
91 if self.minversion:
92 restr.append(f"< {'.'.join(str(n) for n in self.minversion)}")
93 if self.maxversion:
94 restr.append(f">= {'.'.join(str(n) for n in self.maxversion)}")
95 restriction = " or ".join(restr)
96 if checkerref:
97 desc += f" It can't be emitted when using Python {restriction}."
98 else:
99 desc += (
100 f" This message can't be emitted when using Python {restriction}."
101 )
102 msg_help = normalize_text(" ".join(desc.split()), indent=" ")
103 message_id = f"{self.symbol} ({self.msgid})"
104 if title != "%s":
105 title = title.splitlines()[0]
106 return f":{message_id}: *{title.rstrip(' ')}*\n{msg_help}"
107 return f":{message_id}:\n{msg_help}"
