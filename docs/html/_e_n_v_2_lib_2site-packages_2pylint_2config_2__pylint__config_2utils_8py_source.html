35def should_retry_after_invalid_input(
36 func: Callable[_P, _ReturnValueT]
37) -> Callable[_P, _ReturnValueT]:
38 """Decorator that handles InvalidUserInput exceptions and retries."""
39
40 def inner_function(*args: _P.args, **kwargs: _P.kwargs) -> _ReturnValueT:
41 called_once = False
42 while True:
43 try:
44 return func(*args, **kwargs)
45 except InvalidUserInput as exc:
46 if called_once and exc.input == "exit()":
47 print("Stopping 'pylint-config'.")
48 sys.exit()
49 print(f"Answer should be one of {exc.valid}.")
50 print("Type 'exit()' if you want to exit the program.")
51 called_once = True
52
53 return inner_function

56@should_retry_after_invalid_input
57def get_and_validate_format() -> Literal["toml", "ini"]:
58 """Make sure that the output format is either .toml or .ini."""
59 # pylint: disable-next=bad-builtin
60 format_type = input(
61 "Please choose the format of configuration, (T)oml or (I)ni (.cfg): "
62 ).lower()
63
64 if format_type not in SUPPORTED_FORMATS:
65 raise InvalidUserInput(", ".join(sorted(SUPPORTED_FORMATS)), format_type)
66
67 if format_type.startswith("t"):
68 return "toml"
69 return "ini"

72@should_retry_after_invalid_input
73def validate_yes_no(question: str, default: Literal["yes", "no"] | None) -> bool:
74 """Validate that a yes or no answer is correct."""
75 question = f"{question} (y)es or (n)o "
76 if default:
77 question += f" (default={default}) "
78 # pylint: disable-next=bad-builtin
79 answer = input(question).lower()
80
81 if not answer and default:
82 answer = default
83
84 if answer not in YES_NO_ANSWERS:
85 raise InvalidUserInput(", ".join(sorted(YES_NO_ANSWERS)), answer)
86
87 return answer.startswith("y")
