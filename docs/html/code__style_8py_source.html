25class CodeStyleChecker(BaseChecker):
26 """Checkers that can improve code consistency.
27
28 As such they don't necessarily provide a performance benefit and
29 are often times opinionated.
30
31 Before adding another checker here, consider this:
32 1. Does the checker provide a clear benefit,
33 i.e. detect a common issue or improve performance
34 => it should probably be part of the core checker classes
35 2. Is it something that would improve code consistency,
36 maybe because it's slightly better with regard to performance
37 and therefore preferred => this is the right place
38 3. Everything else should go into another extension
39 """

138 def _check_dict_consider_namedtuple_dataclass(self, node: nodes.Dict) -> None:
139 """Check if dictionary values can be replaced by Namedtuple or Dataclass."""
140 if not (
141 isinstance(node.parent, (nodes.Assign, nodes.AnnAssign))
142 and isinstance(node.parent.parent, nodes.Module)
143 or isinstance(node.parent, nodes.AnnAssign)
144 and isinstance(node.parent.target, nodes.AssignName)
145 and utils.is_assign_name_annotated_with(node.parent.target, "Final")
146 ):
147 # If dict is not part of an 'Assign' or 'AnnAssign' node in
148 # a module context OR 'AnnAssign' with 'Final' annotation, skip check.
149 return
150
151 # All dict_values are itself dict nodes
152 if len(node.items) > 1 and all(
153 isinstance(dict_value, nodes.Dict) for _, dict_value in node.items
154 ):
155 KeyTupleT = tuple[type[nodes.NodeNG], str]
156
157 # Makes sure all keys are 'Const' string nodes
158 keys_checked: set[KeyTupleT] = set()
159 for _, dict_value in node.items:
160 dict_value = cast(nodes.Dict, dict_value)
161 for key, _ in dict_value.items:
162 key_tuple = (type(key), key.as_string())
163 if key_tuple in keys_checked:
164 continue
165 inferred = safe_infer(key)
166 if not (
167 isinstance(inferred, nodes.Const)
168 and inferred.pytype() == "builtins.str"
169 ):
170 return
171 keys_checked.add(key_tuple)
172
173 # Makes sure all subdicts have at least 1 common key
174 key_tuples: list[tuple[KeyTupleT, ...]] = []
175 for _, dict_value in node.items:
176 dict_value = cast(nodes.Dict, dict_value)
177 key_tuples.append(
178 tuple((type(key), key.as_string()) for key, _ in dict_value.items)
179 )
180 keys_intersection: set[KeyTupleT] = set(key_tuples[0])
181 for sub_key_tuples in key_tuples[1:]:
182 keys_intersection.intersection_update(sub_key_tuples)
183 if not keys_intersection:
184 return
185
186 self.add_message("consider-using-namedtuple-or-dataclass", node=node)
187 return
188
189 # All dict_values are itself either list or tuple nodes
190 if len(node.items) > 1 and all(
191 isinstance(dict_value, (nodes.List, nodes.Tuple))
192 for _, dict_value in node.items
193 ):
194 # Make sure all sublists have the same length > 0
195 list_length = len(node.items[0][1].elts)
196 if list_length == 0:
197 return
198 for _, dict_value in node.items[1:]:
199 if len(dict_value.elts) != list_length:
200 return
201
202 # Make sure at least one list entry isn't a dict
203 for _, dict_value in node.items:
204 if all(isinstance(entry, nodes.Dict) for entry in dict_value.elts):
205 return
206
207 self.add_message("consider-using-namedtuple-or-dataclass", node=node)
208 return

210 def _check_consider_using_assignment_expr(self, node: nodes.If) -> None:
211 """Check if an assignment expression (walrus operator) can be used.
212
213 For example if an assignment is directly followed by an if statement:
214 >>> x = 2
215 >>> if x:
216 >>> ...
217
218 Can be replaced by:
219 >>> if (x := 2):
220 >>> ...
221
222 Note: Assignment expressions were added in Python 3.8
223 """
224 # Check if `node.test` contains a `Name` node
225 node_name: nodes.Name | None = None
226 if isinstance(node.test, nodes.Name):
227 node_name = node.test
228 elif (
229 isinstance(node.test, nodes.UnaryOp)
230 and node.test.op == "not"
231 and isinstance(node.test.operand, nodes.Name)
232 ):
233 node_name = node.test.operand
234 elif (
235 isinstance(node.test, nodes.Compare)
236 and isinstance(node.test.left, nodes.Name)
237 and len(node.test.ops) == 1
238 ):
239 node_name = node.test.left
240 else:
241 return
242
243 # Make sure the previous node is an assignment to the same name
244 # used in `node.test`. Furthermore, ignore if assignment spans multiple lines.
245 prev_sibling = node.previous_sibling()
246 if CodeStyleChecker._check_prev_sibling_to_if_stmt(
247 prev_sibling, node_name.name
248 ):
249 # Check if match statement would be a better fit.
250 # I.e. multiple ifs that test the same name.
251 if CodeStyleChecker._check_ignore_assignment_expr_suggestion(
252 node, node_name.name
253 ):
254 return
255
256 # Build suggestion string. Check length of suggestion
257 # does not exceed max-line-length-suggestions
258 test_str = node.test.as_string().replace(
259 node_name.name,
260 f"({node_name.name} := {prev_sibling.value.as_string()})",
261 1,
262 )
263 suggestion = f"if {test_str}:"
264 if (
265 node.col_offset is not None
266 and len(suggestion) + node.col_offset > self._max_length
267 or len(suggestion) > self._max_length
268 ):
269 return
270
271 self.add_message(
272 "consider-using-assignment-expr",
273 node=node_name,
274 args=(suggestion,),
275 )

277 @staticmethod
278 def _check_prev_sibling_to_if_stmt(
279 prev_sibling: nodes.NodeNG | None, name: str | None
280 ) -> TypeGuard[nodes.Assign | nodes.AnnAssign]:
281 """Check if previous sibling is an assignment with the same name.
282
283 Ignore statements which span multiple lines.
284 """
285 if prev_sibling is None or prev_sibling.tolineno - prev_sibling.fromlineno != 0:
286 return False
287
288 if (
289 isinstance(prev_sibling, nodes.Assign)
290 and len(prev_sibling.targets) == 1
291 and isinstance(prev_sibling.targets[0], nodes.AssignName)
292 and prev_sibling.targets[0].name == name
293 ):
294 return True
295 if (
296 isinstance(prev_sibling, nodes.AnnAssign)
297 and isinstance(prev_sibling.target, nodes.AssignName)
298 and prev_sibling.target.name == name
299 ):
300 return True
301 return False

303 @staticmethod
304 def _check_ignore_assignment_expr_suggestion(
305 node: nodes.If, name: str | None
306 ) -> bool:
307 """Return True if suggestion for assignment expr should be ignored.
308
309 E.g., in cases where a match statement would be a better fit
310 (multiple conditions).
311 """
312 if isinstance(node.test, nodes.Compare):
313 next_if_node: nodes.If | None = None
314 next_sibling = node.next_sibling()
315 if len(node.orelse) == 1 and isinstance(node.orelse[0], nodes.If):
316 # elif block
317 next_if_node = node.orelse[0]
318 elif isinstance(next_sibling, nodes.If):
319 # separate if block
320 next_if_node = next_sibling
321
322 if ( # pylint: disable=too-many-boolean-expressions
323 next_if_node is not None
324 and (
325 isinstance(next_if_node.test, nodes.Compare)
326 and isinstance(next_if_node.test.left, nodes.Name)
327 and next_if_node.test.left.name == name
328 or isinstance(next_if_node.test, nodes.Name)
329 and next_if_node.test.name == name
330 )
331 ):
332 return True
333 return False
