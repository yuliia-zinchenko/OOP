93 def register(self, model_or_iterable, admin_class=None, **options):
94 """
95 Register the given model(s) with the given admin class.
96
97 The model(s) should be Model classes, not instances.
98
99 If an admin class isn't given, use ModelAdmin (the default admin
100 options). If keyword arguments are given -- e.g., list_display --
101 apply them as options to the admin class.
102
103 If a model is already registered, raise AlreadyRegistered.
104
105 If a model is abstract, raise ImproperlyConfigured.
106 """
107 admin_class = admin_class or ModelAdmin
108 if isinstance(model_or_iterable, ModelBase):
109 model_or_iterable = [model_or_iterable]
110 for model in model_or_iterable:
111 if model._meta.abstract:
112 raise ImproperlyConfigured(
113 "The model %s is abstract, so it cannot be registered with admin."
114 % model.__name__
115 )
116
117 if self.is_registered(model):
118 registered_admin = str(self.get_model_admin(model))
119 msg = "The model %s is already registered " % model.__name__
120 if registered_admin.endswith(".ModelAdmin"):
121 # Most likely registered without a ModelAdmin subclass.
122 msg += "in app %r." % registered_admin.removesuffix(".ModelAdmin")
123 else:
124 msg += "with %r." % registered_admin
125 raise AlreadyRegistered(msg)
126
127 # Ignore the registration if the model has been
128 # swapped out.
129 if not model._meta.swapped:
130 # If we got **options then dynamically construct a subclass of
131 # admin_class with those **options.
132 if options:
133 # For reasons I don't quite understand, without a __module__
134 # the created class appears to "live" in the wrong place,
135 # which causes issues later on.
136 options["__module__"] = __name__
137 admin_class = type(
138 "%sAdmin" % model.__name__, (admin_class,), options
139 )
140
141 # Instantiate the admin class to save in the registry
142 self._registry[model] = admin_class(model, self)
143
144 def unregister(self, model_or_iterable):
145 """
146 Unregister the given model(s).
147
148 If a model isn't already registered, raise NotRegistered.
149 """
150 if isinstance(model_or_iterable, ModelBase):
151 model_or_iterable = [model_or_iterable]

155 del self._registry[model]

204 def admin_view(self, view, cacheable=False):
205 """
206 Decorator to create an admin view attached to this ``AdminSite``. This
207 wraps the view and provides permission checking by calling
208 ``self.has_permission``.
209
210 You'll want to use this from within ``AdminSite.get_urls()``:
211
212 class MyAdminSite(AdminSite):

214 def get_urls(self):
215 from django.urls import path
216
217 urls = super().get_urls()
218 urls += [
219 path('my_view/', self.admin_view(some_view))
220 ]
221 return urls
222
223 By default, admin_views are marked non-cacheable using the
224 ``never_cache`` decorator. If the view can be safely cached, set
225 cacheable=True.
226 """

228 def inner(request, *args, **kwargs):
229 if not self.has_permission(request):
230 if request.path == reverse("admin:logout", current_app=self.name):
231 index_path = reverse("admin:index", current_app=self.name)
232 return HttpResponseRedirect(index_path)
233 # Inner import to prevent django.contrib.admin (app) from
234 # importing django.contrib.auth.models.User (unrelated model).
235 from django.contrib.auth.views import redirect_to_login
236
237 return redirect_to_login(
238 request.get_full_path(),
239 reverse("admin:login", current_app=self.name),
240 )
241 return view(request, *args, **kwargs)
242
243 if not cacheable:
244 inner = never_cache(inner)
245 # We add csrf_protect here so this function can be used as a utility
246 # function for any view, without having to repeat 'csrf_protect'.
247 if not getattr(view, "csrf_exempt", False):
248 inner = csrf_protect(inner)
249 return update_wrapper(inner, view)

464 def _build_app_dict(self, request, label=None):
465 """
466 Build the app dictionary. The optional `label` parameter filters models
467 of a specific app.
468 """
469 app_dict = {}
470
471 if label:
472 models = {
473 m: m_a
474 for m, m_a in self._registry.items()
475 if m._meta.app_label == label
476 }
477 else:
478 models = self._registry
479
480 for model, model_admin in models.items():
481 app_label = model._meta.app_label
482
483 has_module_perms = model_admin.has_module_permission(request)
484 if not has_module_perms:
485 continue
486
487 perms = model_admin.get_model_perms(request)
488
489 # Check whether user has any perm for this module.
490 # If so, add the module to the model_list.
491 if True not in perms.values():
492 continue
493
494 info = (app_label, model._meta.model_name)
495 model_dict = {
496 "model": model,
497 "name": capfirst(model._meta.verbose_name_plural),
498 "object_name": model._meta.object_name,
499 "perms": perms,
500 "admin_url": None,
501 "add_url": None,
502 }
503 if perms.get("change") or perms.get("view"):
504 model_dict["view_only"] = not perms.get("change")
505 try:
506 model_dict["admin_url"] = reverse(
507 "admin:%s_%s_changelist" % info, current_app=self.name
508 )
509 except NoReverseMatch:
510 pass
511 if perms.get("add"):
512 try:
513 model_dict["add_url"] = reverse(
514 "admin:%s_%s_add" % info, current_app=self.name
515 )
516 except NoReverseMatch:
517 pass
518
519 if app_label in app_dict:
520 app_dict[app_label]["models"].append(model_dict)
521 else:
522 app_dict[app_label] = {
523 "name": apps.get_app_config(app_label).verbose_name,
524 "app_label": app_label,
525 "app_url": reverse(
526 "admin:app_list",
527 kwargs={"app_label": app_label},
528 current_app=self.name,
529 ),
530 "has_module_perms": has_module_perms,
531 "models": [model_dict],
532 }
533
534 return app_dict
