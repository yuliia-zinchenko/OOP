173class RenderContext(BaseContext):
174 """
175 A stack container for storing Template state.
176
177 RenderContext simplifies the implementation of template Nodes by providing a
178 safe place to store state between invocations of a node's `render` method.
179
180 The RenderContext also provides scoping rules that are more sensible for
181 'template local' variables. The render context stack is pushed before each
182 template is rendered, creating a fresh scope with nothing in it. Name
183 resolution fails if a variable is not found at the top of the RequestContext
184 stack. Thus, variables are local to a specific template and don't affect the
185 rendering of other templates as they would if they were stored in the normal
186 template context.
187 """

217class RequestContext(Context):
218 """
219 This subclass of template.Context automatically populates itself using
220 the processors defined in the engine's configuration.
221 Additional processors can be specified as a list of callables
222 using the "processors" keyword argument.
223 """
224
225 def __init__(
226 self,
227 request,
228 dict_=None,
229 processors=None,
230 use_l10n=None,
231 use_tz=None,
232 autoescape=True,
233 ):
234 super().__init__(dict_, use_l10n=use_l10n, use_tz=use_tz, autoescape=autoescape)
235 self.request = request
236 self._processors = () if processors is None else tuple(processors)
237 self._processors_index = len(self.dicts)
238
239 # placeholder for context processors output
240 self.update({})
241
242 # empty dict for any new modifications
243 # (so that context processors don't overwrite them)
244 self.update({})
245
246 @contextmanager
247 def bind_template(self, template):
248 if self.template is not None:
249 raise RuntimeError("Context is already bound to a template")
250
251 self.template = template
252 # Set context processors according to the template engine's settings.
253 processors = template.engine.template_context_processors + self._processors
254 updates = {}
255 for processor in processors:
256 context = processor(self.request)
257 try:
258 updates.update(context)
259 except TypeError as e:
260 raise TypeError(
261 f"Context processor {processor.__qualname__} didn't return a "
262 "dictionary."
263 ) from e
264
265 self.dicts[self._processors_index] = updates
266
267 try:
268 yield
269 finally:
270 self.template = None
271 # Unset context processors.
272 self.dicts[self._processors_index] = {}

277 # Context.new don't include values from context processors.

284 """
285 Create a suitable Context from a plain dict and optionally an HttpRequest.
