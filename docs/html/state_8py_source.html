33def _get_related_models(m):
34 """Return all models that have a direct relationship to the given model."""
35 related_models = [
36 subclass
37 for subclass in m.__subclasses__()
38 if issubclass(subclass, models.Model)
39 ]
40 related_fields_models = set()
41 for f in m._meta.get_fields(include_parents=True, include_hidden=True):
42 if (
43 f.is_relation
44 and f.related_model is not None
45 and not isinstance(f.related_model, str)
46 ):
47 related_fields_models.add(f.model)
48 related_models.append(f.related_model)
49 # Reverse accessors of foreign keys to proxy models are attached to their
50 # concrete proxied model.
51 opts = m._meta
52 if opts.proxy and m in related_fields_models:
53 related_models.append(opts.concrete_model)
54 return related_models

68def get_related_models_recursive(model):
69 """
70 Return all models that have a direct or indirect relationship
71 to the given model.
72
73 Relationships are either defined by explicit relational fields, like
74 ForeignKey, ManyToManyField or OneToOneField, or by inheriting from another
75 model (a superclass is related to its subclasses, but not vice versa). Note,
76 however, that a model inheriting from a concrete model is also related to
77 its superclass through the implicit *_ptr OneToOneField on the subclass.
78 """
79 seen = set()
80 queue = _get_related_models(model)
81 for rel_mod in queue:
82 rel_app_label, rel_model_name = (
83 rel_mod._meta.app_label,
84 rel_mod._meta.model_name,
85 )
86 if (rel_app_label, rel_model_name) in seen:
87 continue
88 seen.add((rel_app_label, rel_model_name))
89 queue.extend(_get_related_models(rel_mod))
90 return seen - {(model._meta.app_label, model._meta.model_name)}

346 def _find_reload_model(self, app_label, model_name, delay=False):
347 if delay:
348 self.is_delayed = True
349
350 related_models = set()
351
352 try:
353 old_model = self.apps.get_model(app_label, model_name)
354 except LookupError:
355 pass
356 else:
357 # Get all relations to and from the old model before reloading,
358 # as _meta.apps may change
359 if delay:
360 related_models = get_related_models_tuples(old_model)
361 else:
362 related_models = get_related_models_recursive(old_model)
363
364 # Get all outgoing references from the model to be rendered
365 model_state = self.models[(app_label, model_name)]
366 # Directly related models are the models pointed to by ForeignKeys,
367 # OneToOneFields, and ManyToManyFields.
368 direct_related_models = set()
369 for field in model_state.fields.values():
370 if field.is_relation:
371 if field.remote_field.model == RECURSIVE_RELATIONSHIP_CONSTANT:
372 continue
373 rel_app_label, rel_model_name = _get_app_label_and_model_name(
374 field.related_model, app_label
375 )
376 direct_related_models.add((rel_app_label, rel_model_name.lower()))
377
378 # For all direct related models recursively get all related models.
379 related_models.update(direct_related_models)
380 for rel_app_label, rel_model_name in direct_related_models:
381 try:
382 rel_model = self.apps.get_model(rel_app_label, rel_model_name)
383 except LookupError:
384 pass
385 else:
386 if delay:
387 related_models.update(get_related_models_tuples(rel_model))
388 else:
389 related_models.update(get_related_models_recursive(rel_model))
390
391 # Include the model itself
392 related_models.add((app_label, model_name))
393
394 return related_models

651 # We keep trying to render the models in a loop, ignoring invalid
652 # base errors, until the size of the unrendered models doesn't
653 # decrease by at least one, meaning there's a base dependency loop/
654 # missing base.
655 if not model_states:
656 return
657 # Prevent that all model caches are expired for each render.
658 with self.bulk_update():
659 unrendered_models = model_states
660 while unrendered_models:
661 new_unrendered_models = []
662 for model in unrendered_models:
663 try:
664 model.render(self)
665 except InvalidBasesError:
666 new_unrendered_models.append(model)
667 if len(new_unrendered_models) == len(unrendered_models):
668 raise InvalidBasesError(
669 "Cannot resolve bases for %r\nThis can happen if you are "
670 "inheriting models from an app with migrations (e.g. "
671 "contrib.auth)\n in an app with no migrations; see "
672 "https://docs.djangoproject.com/en/%s/topics/migrations/"
673 "#dependencies for more"
674 % (new_unrendered_models, get_docs_version())
675 )
676 unrendered_models = new_unrendered_models

710class ModelState:
711 """
712 Represent a Django Model. Don't use the actual Model class as it's not
713 designed to have its options changed - instead, mutate this one and then
714 render it into a Model as required.
715
716 Note that while you are allowed to mutate .fields, you are not allowed
717 to mutate the Field instances inside there themselves - you must instead
718 assign new ones, as these are not detached during a clone.
719 """
720
721 def __init__(
722 self, app_label, name, fields, options=None, bases=None, managers=None
723 ):
724 self.app_label = app_label
725 self.name = name
726 self.fields = dict(fields)
727 self.options = options or {}
728 self.options.setdefault("indexes", [])
729 self.options.setdefault("constraints", [])
730 self.bases = bases or (models.Model,)
731 self.managers = managers or []
732 for name, field in self.fields.items():
733 # Sanity-check that fields are NOT already bound to a model.
734 if hasattr(field, "model"):
735 raise ValueError(
736 'ModelState.fields cannot be bound to a model - "%s" is.' % name
737 )
738 # Sanity-check that relation fields are NOT referring to a model class.
739 if field.is_relation and hasattr(field.related_model, "_meta"):
740 raise ValueError(
741 'Model fields in "ModelState.fields" cannot refer to a model class '
742 f'- "{self.app_label}.{self.name}.{name}.to" does. Use a string '
743 "reference instead."
744 )
745 if field.many_to_many and hasattr(field.remote_field.through, "_meta"):
746 raise ValueError(
747 'Model fields in "ModelState.fields" cannot refer to a model class '
748 f'- "{self.app_label}.{self.name}.{name}.through" does. Use a '
749 "string reference instead."
750 )
751 # Sanity-check that indexes have their name set.
752 for index in self.options["indexes"]:
753 if not index.name:
754 raise ValueError(
755 "Indexes passed to ModelState require a name attribute. "
756 "%r doesn't have one." % index
757 )

759 @cached_property
760 def name_lower(self):
761 return self.name.lower()

764 if field_name == "_order":
765 field_name = self.options.get("order_with_respect_to", field_name)

768 @classmethod
769 def from_model(cls, model, exclude_rels=False):
770 """Given a model, return a ModelState representing it."""
771 # Deconstruct the fields
772 fields = []
773 for field in model._meta.local_fields:
774 if getattr(field, "remote_field", None) and exclude_rels:
775 continue
776 if isinstance(field, models.OrderWrt):
777 continue
778 name = field.name
779 try:
780 fields.append((name, field.clone()))
781 except TypeError as e:
782 raise TypeError(
783 "Couldn't reconstruct field %s on %s: %s"
784 % (
785 name,
786 model._meta.label,
787 e,
788 )
789 )
790 if not exclude_rels:
791 for field in model._meta.local_many_to_many:
792 name = field.name
793 try:
794 fields.append((name, field.clone()))
795 except TypeError as e:
796 raise TypeError(
797 "Couldn't reconstruct m2m field %s on %s: %s"
798 % (
799 name,
800 model._meta.object_name,
801 e,
802 )
803 )
804 # Extract the options
805 options = {}
806 for name in DEFAULT_NAMES:
807 # Ignore some special options
808 if name in ["apps", "app_label"]:
809 continue
810 elif name in model._meta.original_attrs:
811 if name == "unique_together":
812 ut = model._meta.original_attrs["unique_together"]
813 options[name] = set(normalize_together(ut))
814 elif name == "index_together":
815 it = model._meta.original_attrs["index_together"]
816 options[name] = set(normalize_together(it))
817 elif name == "indexes":
818 indexes = [idx.clone() for idx in model._meta.indexes]
819 for index in indexes:
820 if not index.name:
821 index.set_name_with_model(model)
822 options["indexes"] = indexes
823 elif name == "constraints":
824 options["constraints"] = [
825 con.clone() for con in model._meta.constraints
826 ]
827 else:
828 options[name] = model._meta.original_attrs[name]
829 # If we're ignoring relationships, remove all field-listing model
830 # options (that option basically just means "make a stub model")
831 if exclude_rels:
832 for key in ["unique_together", "index_together", "order_with_respect_to"]:
833 if key in options:
834 del options[key]
835 # Private fields are ignored, so remove options that refer to them.
836 elif options.get("order_with_respect_to") in {
837 field.name for field in model._meta.private_fields
838 }:
839 del options["order_with_respect_to"]

841 def flatten_bases(model):
842 bases = []
843 for base in model.__bases__:
844 if hasattr(base, "_meta") and base._meta.abstract:
845 bases.extend(flatten_bases(base))
846 else:
847 bases.append(base)
848 return bases
849
850 # We can't rely on __mro__ directly because we only want to flatten
851 # abstract models and not the whole tree. However by recursing on
852 # __bases__ we may end up with duplicates and ordering issues, we
853 # therefore discard any duplicates and reorder the bases according
854 # to their index in the MRO.
855 flattened_bases = sorted(
856 set(flatten_bases(model)), key=lambda x: model.__mro__.index(x)
857 )
858
859 # Make our record
860 bases = tuple(
861 (base._meta.label_lower if hasattr(base, "_meta") else base)
862 for base in flattened_bases
863 )
864 # Ensure at least one base inherits from models.Model
865 if not any(
866 (isinstance(base, str) or issubclass(base, models.Model)) for base in bases
867 ):
868 bases = (models.Model,)
869
870 managers = []
871 manager_names = set()
872 default_manager_shim = None
873 for manager in model._meta.managers:
874 if manager.name in manager_names:
875 # Skip overridden managers.
876 continue
877 elif manager.use_in_migrations:
878 # Copy managers usable in migrations.
879 new_manager = copy.copy(manager)
880 new_manager._set_creation_counter()
881 elif manager is model._base_manager or manager is model._default_manager:
882 # Shim custom managers used as default and base managers.
883 new_manager = models.Manager()
884 new_manager.model = manager.model
885 new_manager.name = manager.name
886 if manager is model._default_manager:
887 default_manager_shim = new_manager
888 else:
889 continue
890 manager_names.add(manager.name)
891 managers.append((manager.name, new_manager))
892
893 # Ignore a shimmed default manager called objects if it's the only one.
894 if managers == [("objects", default_manager_shim)]:
895 managers = []
896
897 # Construct the new ModelState
898 return cls(
899 model._meta.app_label,
900 model._meta.object_name,
901 fields,
902 options,
903 bases,
904 managers,
905 )

907 def construct_managers(self):
908 """Deep-clone the managers using deconstruction."""
909 # Sort all managers by their creation counter
910 sorted_managers = sorted(self.managers, key=lambda v: v[1].creation_counter)
911 for mgr_name, manager in sorted_managers:
912 as_manager, manager_path, qs_path, args, kwargs = manager.deconstruct()
913 if as_manager:
914 qs_class = import_string(qs_path)
915 yield mgr_name, qs_class.as_manager()
916 else:
917 manager_class = import_string(manager_path)
918 yield mgr_name, manager_class(*args, **kwargs)

920 def clone(self):
921 """Return an exact copy of this ModelState."""
922 return self.__class__(
923 app_label=self.app_label,
924 name=self.name,
925 fields=dict(self.fields),
926 # Since options are shallow-copied here, operations such as
927 # AddIndex must replace their option (e.g 'indexes') rather
928 # than mutating it.
929 options=dict(self.options),
930 bases=self.bases,
931 managers=list(self.managers),
932 )

934 def render(self, apps):
935 """Create a Model object from our current state into the given apps."""
936 # First, make a Meta object
937 meta_contents = {"app_label": self.app_label, "apps": apps, **self.options}
938 meta = type("Meta", (), meta_contents)
939 # Then, work out our bases
940 try:
941 bases = tuple(
942 (apps.get_model(base) if isinstance(base, str) else base)
943 for base in self.bases
944 )
945 except LookupError:
946 raise InvalidBasesError(
947 "Cannot resolve one or more bases from %r" % (self.bases,)
948 )
949 # Clone fields for the body, add other bits.
950 body = {name: field.clone() for name, field in self.fields.items()}
951 body["Meta"] = meta
952 body["__module__"] = "__fake__"
953
954 # Restore managers
955 body.update(self.construct_managers())
956 # Then, make a Model object (apps.register_model is called in __new__)
957 return type(self.name, bases, body)
