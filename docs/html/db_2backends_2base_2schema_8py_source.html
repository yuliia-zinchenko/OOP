24def _is_relevant_relation(relation, altered_field):
25 """
26 When altering the given field, must constraints on its model from the given
27 relation be temporarily dropped?
28 """
29 field = relation.field
30 if field.many_to_many:
31 # M2M reverse field
32 return False
33 if altered_field.primary_key and field.to_fields == [None]:
34 # Foreign key constraint on the primary key, which is being altered.
35 return True
36 # Is the constraint targeting the field being altered?
37 return altered_field.name in field.to_fields

173 def execute(self, sql, params=()):
174 """Execute the given SQL statement, with optional parameters."""
175 # Don't perform the transactional DDL check if SQL is being collected
176 # as it's not going to be executed anyway.
177 if (
178 not self.collect_sql
179 and self.connection.in_atomic_block
180 and not self.connection.features.can_rollback_ddl
181 ):
182 raise TransactionManagementError(
183 "Executing DDL statements while in a transaction on databases "
184 "that can't perform a rollback is prohibited."
185 )
186 # Account for non-string statement objects.
187 sql = str(sql)
188 # Log the command we're running, then run it
189 logger.debug(
190 "%s; (params %r)", sql, params, extra={"params": params, "sql": sql}
191 )
192 if self.collect_sql:
193 ending = "" if sql.rstrip().endswith(";") else ";"
194 if params is not None:
195 self.collected_sql.append(
196 (sql % tuple(map(self.quote_value, params))) + ending
197 )
198 else:
199 self.collected_sql.append(sql + ending)
200 else:
201 with self.connection.cursor() as cursor:
202 cursor.execute(sql, params)

207 def table_sql(self, model):
208 """Take a model and return its table definition."""
209 # Add any unique_togethers (always deferred, as some fields might be
210 # created afterward, like geometry fields with some backends).
211 for field_names in model._meta.unique_together:
212 fields = [model._meta.get_field(field) for field in field_names]
213 self.deferred_sql.append(self._create_unique_sql(model, fields))
214 # Create column SQL, add FK deferreds if needed.
215 column_sqls = []
216 params = []
217 for field in model._meta.local_fields:
218 # SQL.
219 definition, extra_params = self.column_sql(model, field)
220 if definition is None:
221 continue
222 # Check constraints can go on the column SQL here.
223 db_params = field.db_parameters(connection=self.connection)
224 if db_params["check"]:
225 definition += " " + self.sql_check_constraint % db_params
226 # Autoincrement SQL (for backends with inline variant).
227 col_type_suffix = field.db_type_suffix(connection=self.connection)
228 if col_type_suffix:
229 definition += " %s" % col_type_suffix
230 params.extend(extra_params)
231 # FK.
232 if field.remote_field and field.db_constraint:
233 to_table = field.remote_field.model._meta.db_table
234 to_column = field.remote_field.model._meta.get_field(
235 field.remote_field.field_name
236 ).column
237 if self.sql_create_inline_fk:
238 definition += " " + self.sql_create_inline_fk % {
239 "to_table": self.quote_name(to_table),
240 "to_column": self.quote_name(to_column),
241 }
242 elif self.connection.features.supports_foreign_keys:
243 self.deferred_sql.append(
244 self._create_fk_sql(
245 model, field, "_fk_%(to_table)s_%(to_column)s"
246 )
247 )
248 # Add the SQL to our big list.
249 column_sqls.append(
250 "%s %s"
251 % (
252 self.quote_name(field.column),
253 definition,
254 )
255 )
256 # Autoincrement SQL (for backends with post table definition
257 # variant).
258 if field.get_internal_type() in (
259 "AutoField",
260 "BigAutoField",
261 "SmallAutoField",
262 ):
263 autoinc_sql = self.connection.ops.autoinc_sql(
264 model._meta.db_table, field.column
265 )
266 if autoinc_sql:
267 self.deferred_sql.extend(autoinc_sql)
268 # The BaseConstraint DDL creation methods such as constraint_sql(),
269 # create_sql(), and delete_sql(), were not designed in a way that
270 # separate SQL from parameters which make their generated SQL unfit to
271 # be used in a context where parametrization is delegated to the
272 # backend.
273 constraint_sqls = []
274 if params:
275 # If parameters are present (e.g. a DEFAULT clause on backends that
276 # allow parametrization) defer constraint creation so they are not
277 # mixed with SQL meant to be parametrized.
278 for constraint in model._meta.constraints:
279 self.deferred_sql.append(constraint.create_sql(model, self))
280 else:
281 constraint_sqls.extend(
282 constraint.constraint_sql(model, self)
283 for constraint in model._meta.constraints
284 )
285 sql = self.sql_create_table % {
286 "table": self.quote_name(model._meta.db_table),
287 "definition": ", ".join(
288 str(statement)
289 for statement in (*column_sqls, *constraint_sqls)
290 if statement
291 ),
292 }
293 if model._meta.db_tablespace:
294 tablespace_sql = self.connection.ops.tablespace_sql(
295 model._meta.db_tablespace
296 )
297 if tablespace_sql:
298 sql += " " + tablespace_sql
299 return sql, params

303 def _iter_column_sql(
304 self, column_db_type, params, model, field, field_db_params, include_default
305 ):
306 yield column_db_type
307 if collation := field_db_params.get("collation"):
308 yield self._collate_sql(collation)
309 if self.connection.features.supports_comments_inline and field.db_comment:
310 yield self._comment_sql(field.db_comment)
311 # Work out nullability.
312 null = field.null
313 # Add database default.
314 if field.db_default is not NOT_PROVIDED:
315 default_sql, default_params = self.db_default_sql(field)
316 yield f"DEFAULT {default_sql}"
317 params.extend(default_params)
318 include_default = False
319 # Include a default value, if requested.
320 include_default = (
321 include_default
322 and not self.skip_default(field)
323 and
324 # Don't include a default value if it's a nullable field and the
325 # default cannot be dropped in the ALTER COLUMN statement (e.g.
326 # MySQL longtext and longblob).
327 not (null and self.skip_default_on_alter(field))
328 )
329 if include_default:
330 default_value = self.effective_default(field)
331 if default_value is not None:
332 column_default = "DEFAULT " + self._column_default_sql(field)
333 if self.connection.features.requires_literal_defaults:
334 # Some databases can't take defaults as a parameter
335 # (Oracle, SQLite). If this is the case, the individual
336 # schema backend should implement prepare_default().
337 yield column_default % self.prepare_default(default_value)
338 else:
339 yield column_default
340 params.append(default_value)
341 # Oracle treats the empty string ('') as null, so coerce the null
342 # option whenever '' is a possible value.
343 if (
344 field.empty_strings_allowed
345 and not field.primary_key
346 and self.connection.features.interprets_empty_strings_as_nulls
347 ):
348 null = True
349 if field.generated:
350 generated_sql, generated_params = self._column_generated_sql(field)
351 params.extend(generated_params)
352 yield generated_sql
353 elif not null:
354 yield "NOT NULL"
355 elif not self.connection.features.implied_column_null:
356 yield "NULL"
357 if field.primary_key:
358 yield "PRIMARY KEY"
359 elif field.unique:
360 yield "UNIQUE"
361 # Optionally add the tablespace if it's an implicitly indexed column.
362 tablespace = field.db_tablespace or model._meta.db_tablespace
363 if (
364 tablespace
365 and self.connection.features.supports_tablespaces
366 and field.unique
367 ):
368 yield self.connection.ops.tablespace_sql(tablespace, inline=True)

713 def add_field(self, model, field):
714 """
715 Create a field on a model. Usually involves adding a column, but may
716 involve adding a table instead (for M2M fields).
717 """
718 # Special-case implicit M2M tables
719 if field.many_to_many and field.remote_field.through._meta.auto_created:
720 return self.create_model(field.remote_field.through)
721 # Get the column's definition
722 definition, params = self.column_sql(model, field, include_default=True)
723 # It might not actually have a column behind it
724 if definition is None:
725 return
726 if col_type_suffix := field.db_type_suffix(connection=self.connection):
727 definition += f" {col_type_suffix}"
728 # Check constraints can go on the column SQL here
729 db_params = field.db_parameters(connection=self.connection)
730 if db_params["check"]:
731 definition += " " + self.sql_check_constraint % db_params
732 if (
733 field.remote_field
734 and self.connection.features.supports_foreign_keys
735 and field.db_constraint
736 ):
737 constraint_suffix = "_fk_%(to_table)s_%(to_column)s"
738 # Add FK constraint inline, if supported.
739 if self.sql_create_column_inline_fk:
740 to_table = field.remote_field.model._meta.db_table
741 to_column = field.remote_field.model._meta.get_field(
742 field.remote_field.field_name
743 ).column
744 namespace, _ = split_identifier(model._meta.db_table)
745 definition += " " + self.sql_create_column_inline_fk % {
746 "name": self._fk_constraint_name(model, field, constraint_suffix),
747 "namespace": (
748 "%s." % self.quote_name(namespace) if namespace else ""
749 ),
750 "column": self.quote_name(field.column),
751 "to_table": self.quote_name(to_table),
752 "to_column": self.quote_name(to_column),
753 "deferrable": self.connection.ops.deferrable_sql(),
754 }
755 # Otherwise, add FK constraints later.
756 else:
757 self.deferred_sql.append(
758 self._create_fk_sql(model, field, constraint_suffix)
759 )
760 # Build the SQL and run it
761 sql = self.sql_create_column % {
762 "table": self.quote_name(model._meta.db_table),
763 "column": self.quote_name(field.column),
764 "definition": definition,
765 }
766 # Prevent using [] as params, in the case a literal '%' is used in the
767 # definition on backends that don't support parametrized DDL.
768 self.execute(sql, params or None)
769 # Drop the default if we need to
770 if (
771 field.db_default is NOT_PROVIDED
772 and not self.skip_default_on_alter(field)
773 and self.effective_default(field) is not None
774 ):
775 changes_sql, params = self._alter_column_default_sql(
776 model, None, field, drop=True
777 )
778 sql = self.sql_alter_column % {
779 "table": self.quote_name(model._meta.db_table),
780 "changes": changes_sql,
781 }
782 self.execute(sql, params)
783 # Add field comment, if required.
784 if (
785 field.db_comment
786 and self.connection.features.supports_comments
787 and not self.connection.features.supports_comments_inline
788 ):
789 field_type = db_params["type"]
790 self.execute(
791 *self._alter_column_comment_sql(
792 model, field, field_type, field.db_comment
793 )
794 )
795 # Add an index, if required
796 self.deferred_sql.extend(self._field_indexes_sql(model, field))
797 # Reset connection if required
798 if self.connection.features.connection_persists_old_columns:
799 self.connection.close()

801 def remove_field(self, model, field):
802 """
803 Remove a field from a model. Usually involves deleting a column,
804 but for M2Ms may involve deleting a table.
805 """
806 # Special-case implicit M2M tables
807 if field.many_to_many and field.remote_field.through._meta.auto_created:
808 return self.delete_model(field.remote_field.through)
809 # It might not actually have a column behind it
810 if field.db_parameters(connection=self.connection)["type"] is None:
811 return
812 # Drop any FK constraints, MySQL requires explicit deletion
813 if field.remote_field:
814 fk_names = self._constraint_names(model, [field.column], foreign_key=True)
815 for fk_name in fk_names:
816 self.execute(self._delete_fk_sql(model, fk_name))
817 # Delete the column
818 sql = self.sql_delete_column % {
819 "table": self.quote_name(model._meta.db_table),
820 "column": self.quote_name(field.column),
821 }
822 self.execute(sql)
823 # Reset connection if required
824 if self.connection.features.connection_persists_old_columns:
825 self.connection.close()
826 # Remove all deferred statements referencing the deleted column.
827 for sql in list(self.deferred_sql):
828 if isinstance(sql, Statement) and sql.references_column(
829 model._meta.db_table, field.column
830 ):
831 self.deferred_sql.remove(sql)

833 def alter_field(self, model, old_field, new_field, strict=False):
834 """
835 Allow a field's type, uniqueness, nullability, default, column,
836 constraints, etc. to be modified.
837 `old_field` is required to compute the necessary changes.
838 If `strict` is True, raise errors if the old column does not match
839 `old_field` precisely.
840 """
841 if not self._field_should_be_altered(old_field, new_field):
842 return
843 # Ensure this field is even column-based
844 old_db_params = old_field.db_parameters(connection=self.connection)
845 old_type = old_db_params["type"]
846 new_db_params = new_field.db_parameters(connection=self.connection)
847 new_type = new_db_params["type"]
848 modifying_generated_field = False
849 if (old_type is None and old_field.remote_field is None) or (
850 new_type is None and new_field.remote_field is None
851 ):
852 raise ValueError(
853 "Cannot alter field %s into %s - they do not properly define "
854 "db_type (are you using a badly-written custom field?)"
855 % (old_field, new_field),
856 )
857 elif (
858 old_type is None
859 and new_type is None
860 and (
861 old_field.remote_field.through
862 and new_field.remote_field.through
863 and old_field.remote_field.through._meta.auto_created
864 and new_field.remote_field.through._meta.auto_created
865 )
866 ):
867 return self._alter_many_to_many(model, old_field, new_field, strict)
868 elif (
869 old_type is None
870 and new_type is None
871 and (
872 old_field.remote_field.through
873 and new_field.remote_field.through
874 and not old_field.remote_field.through._meta.auto_created
875 and not new_field.remote_field.through._meta.auto_created
876 )
877 ):
878 # Both sides have through models; this is a no-op.
879 return
880 elif old_type is None or new_type is None:
881 raise ValueError(
882 "Cannot alter field %s into %s - they are not compatible types "
883 "(you cannot alter to or from M2M fields, or add or remove "
884 "through= on M2M fields)" % (old_field, new_field)
885 )
886 elif old_field.generated != new_field.generated or (
887 new_field.generated and old_field.db_persist != new_field.db_persist
888 ):
889 modifying_generated_field = True
890 elif new_field.generated:
891 try:
892 old_field_sql = old_field.generated_sql(self.connection)
893 except FieldError:
894 # Field used in a generated field was renamed.
895 modifying_generated_field = True
896 else:
897 new_field_sql = new_field.generated_sql(self.connection)
898 modifying_generated_field = old_field_sql != new_field_sql
899 if modifying_generated_field:
900 raise ValueError(
901 f"Modifying GeneratedFields is not supported - the field {new_field} "
902 "must be removed and re-added with the new definition."
903 )
904
905 self._alter_field(
906 model,
907 old_field,
908 new_field,
909 old_type,
910 new_type,
911 old_db_params,
912 new_db_params,
913 strict,
914 )

920 data = field.db_type_parameters(self.connection)

927 def _alter_field(
928 self,
929 model,
930 old_field,
931 new_field,
932 old_type,
933 new_type,
934 old_db_params,
935 new_db_params,
936 strict=False,
937 ):
938 """Perform a "physical" (non-ManyToMany) field update."""
939 # Drop any FK constraints, we'll remake them later
940 fks_dropped = set()
941 if (
942 self.connection.features.supports_foreign_keys
943 and old_field.remote_field
944 and old_field.db_constraint
945 and self._field_should_be_altered(
946 old_field,
947 new_field,
948 ignore={"db_comment"},
949 )
950 ):
951 fk_names = self._constraint_names(
952 model, [old_field.column], foreign_key=True
953 )
954 if strict and len(fk_names) != 1:
955 raise ValueError(
956 "Found wrong number (%s) of foreign key constraints for %s.%s"
957 % (
958 len(fk_names),
959 model._meta.db_table,
960 old_field.column,
961 )
962 )
963 for fk_name in fk_names:
964 fks_dropped.add((old_field.column,))
965 self.execute(self._delete_fk_sql(model, fk_name))
966 # Has unique been removed?
967 if old_field.unique and (
968 not new_field.unique or self._field_became_primary_key(old_field, new_field)
969 ):
970 # Find the unique constraint for this field
971 meta_constraint_names = {
972 constraint.name for constraint in model._meta.constraints
973 }
974 constraint_names = self._constraint_names(
975 model,
976 [old_field.column],
977 unique=True,
978 primary_key=False,
979 exclude=meta_constraint_names,
980 )
981 if strict and len(constraint_names) != 1:
982 raise ValueError(
983 "Found wrong number (%s) of unique constraints for %s.%s"
984 % (
985 len(constraint_names),
986 model._meta.db_table,
987 old_field.column,
988 )
989 )
990 for constraint_name in constraint_names:
991 self.execute(self._delete_unique_sql(model, constraint_name))
992 # Drop incoming FK constraints if the field is a primary key or unique,
993 # which might be a to_field target, and things are going to change.
994 old_collation = old_db_params.get("collation")
995 new_collation = new_db_params.get("collation")
996 drop_foreign_keys = (
997 self.connection.features.supports_foreign_keys
998 and (
999 (old_field.primary_key and new_field.primary_key)
1000 or (old_field.unique and new_field.unique)
1001 )
1002 and ((old_type != new_type) or (old_collation != new_collation))
1003 )
1004 if drop_foreign_keys:
1005 # '_meta.related_field' also contains M2M reverse fields, these
1006 # will be filtered out
1007 for _old_rel, new_rel in _related_non_m2m_objects(old_field, new_field):
1008 rel_fk_names = self._constraint_names(
1009 new_rel.related_model, [new_rel.field.column], foreign_key=True
1010 )
1011 for fk_name in rel_fk_names:
1012 self.execute(self._delete_fk_sql(new_rel.related_model, fk_name))
1013 # Removed an index? (no strict check, as multiple indexes are possible)
1014 # Remove indexes if db_index switched to False or a unique constraint
1015 # will now be used in lieu of an index. The following lines from the
1016 # truth table show all True cases; the rest are False:
1017 #
1018 # old_field.db_index | old_field.unique | new_field.db_index | new_field.unique
1019 # ------------------------------------------------------------------------------
1020 # True | False | False | False
1021 # True | False | False | True
1022 # True | False | True | True
1023 if (
1024 old_field.db_index
1025 and not old_field.unique
1026 and (not new_field.db_index or new_field.unique)
1027 ):
1028 # Find the index for this field
1029 meta_index_names = {index.name for index in model._meta.indexes}
1030 # Retrieve only BTREE indexes since this is what's created with
1031 # db_index=True.
1032 index_names = self._constraint_names(
1033 model,
1034 [old_field.column],
1035 index=True,
1036 type_=Index.suffix,
1037 exclude=meta_index_names,
1038 )
1039 for index_name in index_names:
1040 # The only way to check if an index was created with
1041 # db_index=True or with Index(['field'], name='foo')
1042 # is to look at its name (refs #28053).
1043 self.execute(self._delete_index_sql(model, index_name))
1044 # Change check constraints?
1045 old_db_check = self._field_db_check(old_field, old_db_params)
1046 new_db_check = self._field_db_check(new_field, new_db_params)
1047 if old_db_check != new_db_check and old_db_check:
1048 meta_constraint_names = {
1049 constraint.name for constraint in model._meta.constraints
1050 }
1051 constraint_names = self._constraint_names(
1052 model,
1053 [old_field.column],
1054 check=True,
1055 exclude=meta_constraint_names,
1056 )
1057 if strict and len(constraint_names) != 1:
1058 raise ValueError(
1059 "Found wrong number (%s) of check constraints for %s.%s"
1060 % (
1061 len(constraint_names),
1062 model._meta.db_table,
1063 old_field.column,
1064 )
1065 )
1066 for constraint_name in constraint_names:
1067 self.execute(self._delete_check_sql(model, constraint_name))
1068 # Have they renamed the column?
1069 if old_field.column != new_field.column:
1070 self.execute(
1071 self._rename_field_sql(
1072 model._meta.db_table, old_field, new_field, new_type
1073 )
1074 )
1075 # Rename all references to the renamed column.
1076 for sql in self.deferred_sql:
1077 if isinstance(sql, Statement):
1078 sql.rename_column_references(
1079 model._meta.db_table, old_field.column, new_field.column
1080 )
1081 # Next, start accumulating actions to do
1082 actions = []
1083 null_actions = []
1084 post_actions = []
1085 # Type suffix change? (e.g. auto increment).
1086 old_type_suffix = old_field.db_type_suffix(connection=self.connection)
1087 new_type_suffix = new_field.db_type_suffix(connection=self.connection)
1088 # Type, collation, or comment change?
1089 if (
1090 old_type != new_type
1091 or old_type_suffix != new_type_suffix
1092 or old_collation != new_collation
1093 or (
1094 self.connection.features.supports_comments
1095 and old_field.db_comment != new_field.db_comment
1096 )
1097 ):
1098 fragment, other_actions = self._alter_column_type_sql(
1099 model, old_field, new_field, new_type, old_collation, new_collation
1100 )
1101 actions.append(fragment)
1102 post_actions.extend(other_actions)
1103
1104 if new_field.db_default is not NOT_PROVIDED:
1105 if (
1106 old_field.db_default is NOT_PROVIDED
1107 or new_field.db_default != old_field.db_default
1108 ):
1109 actions.append(
1110 self._alter_column_database_default_sql(model, old_field, new_field)
1111 )
1112 elif old_field.db_default is not NOT_PROVIDED:
1113 actions.append(
1114 self._alter_column_database_default_sql(
1115 model, old_field, new_field, drop=True
1116 )
1117 )
1118 # When changing a column NULL constraint to NOT NULL with a given
1119 # default value, we need to perform 4 steps:
1120 # 1. Add a default for new incoming writes
1121 # 2. Update existing NULL rows with new default
1122 # 3. Replace NULL constraint with NOT NULL
1123 # 4. Drop the default again.
1124 # Default change?
1125 needs_database_default = False
1126 if (
1127 old_field.null
1128 and not new_field.null
1129 and new_field.db_default is NOT_PROVIDED
1130 ):
1131 old_default = self.effective_default(old_field)
1132 new_default = self.effective_default(new_field)
1133 if (
1134 not self.skip_default_on_alter(new_field)
1135 and old_default != new_default
1136 and new_default is not None
1137 ):
1138 needs_database_default = True
1139 actions.append(
1140 self._alter_column_default_sql(model, old_field, new_field)
1141 )
1142 # Nullability change?
1143 if old_field.null != new_field.null:
1144 fragment = self._alter_column_null_sql(model, old_field, new_field)
1145 if fragment:
1146 null_actions.append(fragment)
1147 # Only if we have a default and there is a change from NULL to NOT NULL
1148 four_way_default_alteration = (
1149 new_field.has_default() or new_field.db_default is not NOT_PROVIDED
1150 ) and (old_field.null and not new_field.null)
1151 if actions or null_actions:
1152 if not four_way_default_alteration:
1153 # If we don't have to do a 4-way default alteration we can
1154 # directly run a (NOT) NULL alteration
1155 actions += null_actions
1156 # Combine actions together if we can (e.g. postgres)
1157 if self.connection.features.supports_combined_alters and actions:
1158 sql, params = tuple(zip(*actions))
1159 actions = [(", ".join(sql), sum(params, []))]
1160 # Apply those actions
1161 for sql, params in actions:
1162 self.execute(
1163 self.sql_alter_column
1164 % {
1165 "table": self.quote_name(model._meta.db_table),
1166 "changes": sql,
1167 },
1168 params,
1169 )
1170 if four_way_default_alteration:
1171 if new_field.db_default is NOT_PROVIDED:
1172 default_sql = "%s"
1173 params = [new_default]
1174 else:
1175 default_sql, params = self.db_default_sql(new_field)
1176 # Update existing rows with default value
1177 self.execute(
1178 self.sql_update_with_default
1179 % {
1180 "table": self.quote_name(model._meta.db_table),
1181 "column": self.quote_name(new_field.column),
1182 "default": default_sql,
1183 },
1184 params,
1185 )
1186 # Since we didn't run a NOT NULL change before we need to do it
1187 # now
1188 for sql, params in null_actions:
1189 self.execute(
1190 self.sql_alter_column
1191 % {
1192 "table": self.quote_name(model._meta.db_table),
1193 "changes": sql,
1194 },
1195 params,
1196 )
1197 if post_actions:
1198 for sql, params in post_actions:
1199 self.execute(sql, params)
1200 # If primary_key changed to False, delete the primary key constraint.
1201 if old_field.primary_key and not new_field.primary_key:
1202 self._delete_primary_key(model, strict)
1203 # Added a unique?
1204 if self._unique_should_be_added(old_field, new_field):
1205 self.execute(self._create_unique_sql(model, [new_field]))
1206 # Added an index? Add an index if db_index switched to True or a unique
1207 # constraint will no longer be used in lieu of an index. The following
1208 # lines from the truth table show all True cases; the rest are False:
1209 #
1210 # old_field.db_index | old_field.unique | new_field.db_index | new_field.unique
1211 # ------------------------------------------------------------------------------
1212 # False | False | True | False
1213 # False | True | True | False
1214 # True | True | True | False
1215 if (
1216 (not old_field.db_index or old_field.unique)
1217 and new_field.db_index
1218 and not new_field.unique
1219 ):
1220 self.execute(self._create_index_sql(model, fields=[new_field]))
1221 # Type alteration on primary key? Then we need to alter the column
1222 # referring to us.
1223 rels_to_update = []
1224 if drop_foreign_keys:
1225 rels_to_update.extend(_related_non_m2m_objects(old_field, new_field))
1226 # Changed to become primary key?
1227 if self._field_became_primary_key(old_field, new_field):
1228 # Make the new one
1229 self.execute(self._create_primary_key_sql(model, new_field))
1230 # Update all referencing columns
1231 rels_to_update.extend(_related_non_m2m_objects(old_field, new_field))
1232 # Handle our type alters on the other end of rels from the PK stuff above
1233 for old_rel, new_rel in rels_to_update:
1234 rel_db_params = new_rel.field.db_parameters(connection=self.connection)
1235 rel_type = rel_db_params["type"]
1236 rel_collation = rel_db_params.get("collation")
1237 old_rel_db_params = old_rel.field.db_parameters(connection=self.connection)
1238 old_rel_collation = old_rel_db_params.get("collation")
1239 fragment, other_actions = self._alter_column_type_sql(
1240 new_rel.related_model,
1241 old_rel.field,
1242 new_rel.field,
1243 rel_type,
1244 old_rel_collation,
1245 rel_collation,
1246 )
1247 self.execute(
1248 self.sql_alter_column
1249 % {
1250 "table": self.quote_name(new_rel.related_model._meta.db_table),
1251 "changes": fragment[0],
1252 },
1253 fragment[1],
1254 )
1255 for sql, params in other_actions:
1256 self.execute(sql, params)
1257 # Does it have a foreign key?
1258 if (
1259 self.connection.features.supports_foreign_keys
1260 and new_field.remote_field
1261 and (
1262 fks_dropped or not old_field.remote_field or not old_field.db_constraint
1263 )
1264 and new_field.db_constraint
1265 ):
1266 self.execute(
1267 self._create_fk_sql(model, new_field, "_fk_%(to_table)s_%(to_column)s")
1268 )
1269 # Rebuild FKs that pointed to us if we previously had to drop them
1270 if drop_foreign_keys:
1271 for _, rel in rels_to_update:
1272 if rel.field.db_constraint:
1273 self.execute(
1274 self._create_fk_sql(rel.related_model, rel.field, "_fk")
1275 )
1276 # Does it have check constraints we need to add?
1277 if old_db_check != new_db_check and new_db_check:
1278 constraint_name = self._create_index_name(
1279 model._meta.db_table, [new_field.column], suffix="_check"
1280 )
1281 self.execute(
1282 self._create_check_sql(model, constraint_name, new_db_params["check"])
1283 )
1284 # Drop the default if we need to
1285 # (Django usually does not use in-database defaults)
1286 if needs_database_default:
1287 changes_sql, params = self._alter_column_default_sql(
1288 model, old_field, new_field, drop=True
1289 )
1290 sql = self.sql_alter_column % {
1291 "table": self.quote_name(model._meta.db_table),
1292 "changes": changes_sql,
1293 }
1294 self.execute(sql, params)
1295 # Reset connection if required
1296 if self.connection.features.connection_persists_old_columns:
1297 self.connection.close()

1304 as required by new_field, or None if no changes are required.
1305 """

1307 self.connection.features.interprets_empty_strings_as_nulls
1308 and new_field.empty_strings_allowed
1309 ):
1310 # The field is nullable in the database anyway, leave it alone.
1311 return
1312 else:
1313 new_db_params = new_field.db_parameters(connection=self.connection)
1314 sql = (
1315 self.sql_alter_column_null
1316 if new_field.null
1317 else self.sql_alter_column_not_null
1318 )

1320 sql

1486 def _create_index_name(self, table_name, column_names, suffix=""):
1487 """
1488 Generate a unique name for an index/unique constraint.
1489
1490 The name is divided into 3 parts: the table name, the column names,
1491 and a unique digest and suffix.
1492 """
1493 _, table_name = split_identifier(table_name)
1494 hash_suffix_part = "%s%s" % (
1495 names_digest(table_name, *column_names, length=8),
1496 suffix,
1497 )
1498 max_length = self.connection.ops.max_name_length() or 200
1499 # If everything fits into max_length, use that name.
1500 index_name = "%s_%s_%s" % (table_name, "_".join(column_names), hash_suffix_part)
1501 if len(index_name) <= max_length:
1502 return index_name
1503 # Shorten a long suffix.
1504 if len(hash_suffix_part) > max_length / 3:
1505 hash_suffix_part = hash_suffix_part[: max_length // 3]
1506 other_length = (max_length - len(hash_suffix_part)) // 2 - 1
1507 index_name = "%s_%s_%s" % (
1508 table_name[:other_length],
1509 "_".join(column_names)[:other_length],
1510 hash_suffix_part,
1511 )
1512 # Prepend D if needed to prevent the name from starting with an
1513 # underscore or a number (not permitted on Oracle).
1514 if index_name[0] == "_" or index_name[0].isdigit():
1515 index_name = "D%s" % index_name[:-1]
1516 return index_name
