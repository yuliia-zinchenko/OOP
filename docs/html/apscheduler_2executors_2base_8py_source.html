106def run_job(job, jobstore_alias, run_times, logger_name):
107 """
108 Called by executors to run the job. Returns a list of scheduler events to be dispatched by the
109 scheduler.
110
111 """
112 events = []
113 logger = logging.getLogger(logger_name)
114 for run_time in run_times:
115 # See if the job missed its run time window, and handle
116 # possible misfires accordingly
117 if job.misfire_grace_time is not None:
118 difference = datetime.now(timezone.utc) - run_time
119 grace_time = timedelta(seconds=job.misfire_grace_time)
120 if difference > grace_time:
121 events.append(
122 JobExecutionEvent(
123 EVENT_JOB_MISSED, job.id, jobstore_alias, run_time
124 )
125 )
126 logger.warning('Run time of job "%s" was missed by %s', job, difference)
127 continue
128
129 logger.info('Running job "%s" (scheduled at %s)', job, run_time)
130 try:
131 retval = job.func(*job.args, **job.kwargs)
132 except BaseException:
133 exc, tb = sys.exc_info()[1:]
134 formatted_tb = "".join(format_tb(tb))
135 events.append(
136 JobExecutionEvent(
137 EVENT_JOB_ERROR,
138 job.id,
139 jobstore_alias,
140 run_time,
141 exception=exc,
142 traceback=formatted_tb,
143 )
144 )
145 logger.exception('Job "%s" raised an exception', job)
146
147 # This is to prevent cyclic references that would lead to memory leaks
148 traceback.clear_frames(tb)
149 del tb
150 else:
151 events.append(
152 JobExecutionEvent(
153 EVENT_JOB_EXECUTED, job.id, jobstore_alias, run_time, retval=retval
154 )
155 )
156 logger.info('Job "%s" executed successfully', job)
157
158 return events
