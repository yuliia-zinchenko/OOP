70 A wrapper around BytesIO that restricts what can be read since data from

278def encode_multipart(boundary, data):
279 """
280 Encode multipart POST data from a dictionary of form values.
281
282 The key will be used as the form data name; the value will be transmitted
283 as content. If the value is a file, the contents of the file will be sent
284 as an application/octet-stream; otherwise, str(value) will be sent.
285 """
286 lines = []
287
288 def to_bytes(s):
289 return force_bytes(s, settings.DEFAULT_CHARSET)
290
291 # Not by any means perfect, but good enough for our purposes.
292 def is_file(thing):
293 return hasattr(thing, "read") and callable(thing.read)
294
295 # Each bit of the multipart form data could be either a form value or a
296 # file, or a *list* of form values and/or files. Remember that HTTP field
297 # names can be duplicated!
298 for key, value in data.items():
299 if value is None:
300 raise TypeError(
301 "Cannot encode None for key '%s' as POST data. Did you mean "
302 "to pass an empty string or omit the value?" % key
303 )
304 elif is_file(value):
305 lines.extend(encode_file(boundary, key, value))
306 elif not isinstance(value, str) and isinstance(value, Iterable):
307 for item in value:
308 if is_file(item):
309 lines.extend(encode_file(boundary, key, item))
310 else:
311 lines.extend(
312 to_bytes(val)
313 for val in [
314 "--%s" % boundary,
315 'Content-Disposition: form-data; name="%s"' % key,
316 "",
317 item,
318 ]
319 )
320 else:
321 lines.extend(
322 to_bytes(val)
323 for val in [
324 "--%s" % boundary,
325 'Content-Disposition: form-data; name="%s"' % key,
326 "",
327 value,
328 ]
329 )
330
331 lines.extend(
332 [
333 to_bytes("--%s--" % boundary),
334 b"",
335 ]
336 )
337 return b"\r\n".join(lines)

960 def _follow_redirect(
961 self,
962 response,
963 *,
964 data="",
965 content_type="",
966 headers=None,
967 query_params=None,
968 **extra,
969 ):
970 """Follow a single redirect contained in response using GET."""
971 response_url = response.url
972 redirect_chain = response.redirect_chain
973 redirect_chain.append((response_url, response.status_code))
974
975 url = urlsplit(response_url)
976 if url.scheme:
977 extra["wsgi.url_scheme"] = url.scheme
978 if url.hostname:
979 extra["SERVER_NAME"] = url.hostname
980 extra["HTTP_HOST"] = url.hostname
981 if url.port:
982 extra["SERVER_PORT"] = str(url.port)
983
984 path = url.path
985 # RFC 3986 Section 6.2.3: Empty path should be normalized to "/".
986 if not path and url.netloc:
987 path = "/"
988 # Prepend the request path to handle relative path redirects
989 if not path.startswith("/"):
990 path = urljoin(response.request["PATH_INFO"], path)
991
992 if response.status_code in (
993 HTTPStatus.TEMPORARY_REDIRECT,
994 HTTPStatus.PERMANENT_REDIRECT,
995 ):
996 # Preserve request method and query string (if needed)
997 # post-redirect for 307/308 responses.
998 request_method = response.request["REQUEST_METHOD"].lower()
999 if request_method not in ("get", "head"):
1000 extra["QUERY_STRING"] = url.query
1001 request_method = getattr(self, request_method)
1002 else:
1003 request_method = self.get
1004 data = QueryDict(url.query)
1005 content_type = None
1006
1007 return request_method(
1008 path,
1009 data=data,
1010 content_type=content_type,
1011 follow=False,
1012 headers=headers,
1013 query_params=query_params,
1014 **extra,
1015 )
1016
1017 def _ensure_redirects_not_cyclic(self, response):
1018 """
1019 Raise a RedirectCycleError if response contains too many redirects.
1020 """
1021 redirect_chain = response.redirect_chain
1022 if redirect_chain[-1] in redirect_chain[:-1]:
1023 # Check that we're not redirecting to somewhere we've already been
1024 # to, to prevent loops.
1025 raise RedirectCycleError("Redirect loop detected.", last_response=response)
1026 if len(redirect_chain) > 20:
1027 # Such a lengthy chain likely also means a loop, but one with a
1028 # growing path, changing view, or changing query argument. 20 is
1029 # the value of "network.http.redirection-limit" from Firefox.
1030 raise RedirectCycleError("Too many redirects.", last_response=response)

1033class Client(ClientMixin, RequestFactory):
1034 """
1035 A class that can act as a client for testing purposes.
1036
1037 It allows the user to compose GET and POST requests, and
1038 obtain the response that the server gave to those requests.
1039 The server Response objects are annotated with the details
1040 of the contexts and templates that were rendered during the
1041 process of serving the request.
1042
1043 Client objects are stateful - they will retain cookie (and
1044 thus session) details for the lifetime of the Client instance.
1045
1046 This is not intended as a replacement for Twill/Selenium or
1047 the like - it is here to allow testing against the
1048 contexts and templates produced by a view, rather than the
1049 HTML rendered to the end-user.
1050 """

1053 self,
1054 enforce_csrf_checks=False,
1055 raise_request_exception=True,
1056 *,
1057 headers=None,
1058 query_params=None,
1059 **defaults,
1060 ):
1061 super().__init__(headers=headers, query_params=query_params, **defaults)

1064 self.exc_info = None
1065 self.extra = None
1066 self.headers = None

1068 def request(self, **request):
1069 """
1070 Make a generic request. Compose the environment dictionary and pass
1071 to the handler, return the result of the handler. Assume defaults for
1072 the query environment, which can be overridden using the arguments to
1073 the request.
1074 """
1075 environ = self._base_environ(**request)
1076
1077 # Curry a data dictionary into an instance of the template renderer
1078 # callback function.
1079 data = {}
1080 on_template_render = partial(store_rendered_templates, data)
1081 signal_uid = "template-render-%s" % id(request)
1082 signals.template_rendered.connect(on_template_render, dispatch_uid=signal_uid)

1086 try:

1088 finally:
1089 signals.template_rendered.disconnect(dispatch_uid=signal_uid)
1090 got_request_exception.disconnect(dispatch_uid=exception_uid)
1091 # Check for signaled exceptions.
1092 self.check_exception(response)
1093 # Save the client and request that stimulated the response.
1094 response.client = self
1095 response.request = request
1096 # Add any rendered template detail to the response.
1097 response.templates = data.get("templates", [])
1098 response.context = data.get("context")

1100 # Attach the ResolverMatch instance to the response.
