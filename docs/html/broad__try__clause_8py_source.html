19class BroadTryClauseChecker(checkers.BaseChecker):
20 """Checks for try clauses with too many lines.
21
22 According to PEP 8, ``try`` clauses shall contain the absolute minimum
23 amount of code. This checker enforces a maximum number of statements within
24 ``try`` clauses.
25 """
26
27 # configuration section name

49 def _count_statements(
50 self, node: nodes.For | nodes.If | nodes.Try | nodes.While | nodes.With
51 ) -> int:
52 statement_count = len(node.body)
53
54 for body_node in node.body:
55 if isinstance(body_node, (nodes.For, nodes.If, nodes.While, nodes.With)):
56 statement_count += self._count_statements(body_node)
57
58 return statement_count
59
60 def visit_try(self, node: nodes.Try) -> None:
61 try_clause_statements = self._count_statements(node)
62 if try_clause_statements > self.linter.config.max_try_statements:
63 msg = (
64 f"try clause contains {try_clause_statements} statements, expected at"
65 f" most {self.linter.config.max_try_statements}"
66 )
67 self.add_message(
68 "too-many-try-statements", node.lineno, node=node, args=msg
69 )
