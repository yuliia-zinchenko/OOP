22def _get_filtered_node_statements(
23 base_node: nodes.NodeNG, stmt_nodes: list[nodes.NodeNG]
24) -> list[tuple[nodes.NodeNG, _base_nodes.Statement]]:
25 statements = [(node, node.statement()) for node in stmt_nodes]
26 # Next we check if we have ExceptHandlers that are parent
27 # of the underlying variable, in which case the last one survives
28 if len(statements) > 1 and all(
29 isinstance(stmt, nodes.ExceptHandler) for _, stmt in statements
30 ):
31 statements = [
32 (node, stmt) for node, stmt in statements if stmt.parent_of(base_node)
33 ]
34 return statements

37def _is_from_decorator(node) -> bool:
38 """Return whether the given node is the child of a decorator."""
39 return any(isinstance(parent, nodes.Decorators) for parent in node.node_ancestors())

42def _get_if_statement_ancestor(node: nodes.NodeNG) -> nodes.If | None:
43 """Return the first parent node that is an If node (or None)."""
44 for parent in node.node_ancestors():
45 if isinstance(parent, nodes.If):
46 return parent

50def _filter_stmts(
51 base_node: _base_nodes.LookupMixIn,
52 stmts: list[SuccessfulInferenceResult],
53 frame: nodes.LocalsDictNodeNG,
54 offset: int,
55) -> list[nodes.NodeNG]:
56 """Filter the given list of statements to remove ignorable statements.
57
58 If base_node is not a frame itself and the name is found in the inner
59 frame locals, statements will be filtered to remove ignorable
60 statements according to base_node's location.
61
62 :param stmts: The statements to filter.
63
64 :param frame: The frame that all of the given statements belong to.
65
66 :param offset: The line offset to filter statements up to.
67
68 :returns: The filtered statements.
69 """
70 # if offset == -1, my actual frame is not the inner frame but its parent
71 #
72 # class A(B): pass
73 #
74 # we need this to resolve B correctly
75 if offset == -1:
76 myframe = base_node.frame().parent.frame()
77 else:
78 myframe = base_node.frame()
79 # If the frame of this node is the same as the statement
80 # of this node, then the node is part of a class or
81 # a function definition and the frame of this node should be the
82 # the upper frame, not the frame of the definition.
83 # For more information why this is important,
84 # see Pylint issue #295.
85 # For example, for 'b', the statement is the same
86 # as the frame / scope:
87 #
88 # def test(b=1):
89 # ...
90 if base_node.parent and base_node.statement() is myframe and myframe.parent:
91 myframe = myframe.parent.frame()
92
93 mystmt: _base_nodes.Statement | None = None
94 if base_node.parent:
95 mystmt = base_node.statement()
96
97 # line filtering if we are in the same frame
98 #
99 # take care node may be missing lineno information (this is the case for
100 # nodes inserted for living objects)
101 if myframe is frame and mystmt and mystmt.fromlineno is not None:
102 assert mystmt.fromlineno is not None, mystmt
103 mylineno = mystmt.fromlineno + offset
104 else:
105 # disabling lineno filtering
106 mylineno = 0
107
108 _stmts: list[nodes.NodeNG] = []
109 _stmt_parents = []
110 statements = _get_filtered_node_statements(base_node, stmts)
111 for node, stmt in statements:
112 # line filtering is on and we have reached our location, break
113 if stmt.fromlineno and stmt.fromlineno > mylineno > 0:
114 break
115 # Ignore decorators with the same name as the
116 # decorated function
117 # Fixes issue #375
118 if mystmt is stmt and _is_from_decorator(base_node):
119 continue
120 if node.has_base(base_node):
121 break
122
123 if isinstance(node, nodes.EmptyNode):
124 # EmptyNode does not have assign_type(), so just add it and move on
125 _stmts.append(node)
126 continue
127
128 assign_type = node.assign_type()
129 _stmts, done = assign_type._get_filtered_stmts(base_node, node, _stmts, mystmt)
130 if done:
131 break
132
133 optional_assign = assign_type.optional_assign
134 if optional_assign and assign_type.parent_of(base_node):
135 # we are inside a loop, loop var assignment is hiding previous
136 # assignment
137 _stmts = [node]
138 _stmt_parents = [stmt.parent]
139 continue
140
141 if isinstance(assign_type, nodes.NamedExpr):
142 # If the NamedExpr is in an if statement we do some basic control flow inference
143 if_parent = _get_if_statement_ancestor(assign_type)
144 if if_parent:
145 # If the if statement is within another if statement we append the node
146 # to possible statements
147 if _get_if_statement_ancestor(if_parent):
148 optional_assign = False
149 _stmts.append(node)
150 _stmt_parents.append(stmt.parent)
151 # Else we assume that it will be evaluated
152 else:
153 _stmts = [node]
154 _stmt_parents = [stmt.parent]
155 else:
156 _stmts = [node]
157 _stmt_parents = [stmt.parent]
158
159 # XXX comment various branches below!!!
160 try:
161 pindex = _stmt_parents.index(stmt.parent)
162 except ValueError:
163 pass
164 else:
165 # we got a parent index, this means the currently visited node
166 # is at the same block level as a previously visited node
167 if _stmts[pindex].assign_type().parent_of(assign_type):
168 # both statements are not at the same block level
169 continue
170 # if currently visited node is following previously considered
171 # assignment and both are not exclusive, we can drop the
172 # previous one. For instance in the following code ::
173 #
174 # if a:
175 # x = 1
176 # else:
177 # x = 2
178 # print x
179 #
180 # we can't remove neither x = 1 nor x = 2 when looking for 'x'
181 # of 'print x'; while in the following ::
182 #
183 # x = 1
184 # x = 2
185 # print x
186 #
187 # we can remove x = 1 when we see x = 2
188 #
189 # moreover, on loop assignment types, assignment won't
190 # necessarily be done if the loop has no iteration, so we don't
191 # want to clear previous assignments if any (hence the test on
192 # optional_assign)
193 if not (optional_assign or nodes.are_exclusive(_stmts[pindex], node)):
194 del _stmt_parents[pindex]
195 del _stmts[pindex]
196
197 # If base_node and node are exclusive, then we can ignore node
198 if nodes.are_exclusive(base_node, node):
199 continue
200
201 # An AssignName node overrides previous assignments if:
202 # 1. node's statement always assigns
203 # 2. node and base_node are in the same block (i.e., has the same parent as base_node)
204 if isinstance(node, (nodes.NamedExpr, nodes.AssignName)):
205 if isinstance(stmt, nodes.ExceptHandler):
206 # If node's statement is an ExceptHandler, then it is the variable
207 # bound to the caught exception. If base_node is not contained within
208 # the exception handler block, node should override previous assignments;
209 # otherwise, node should be ignored, as an exception variable
210 # is local to the handler block.
211 if stmt.parent_of(base_node):
212 _stmts = []
213 _stmt_parents = []
214 else:
215 continue
216 elif not optional_assign and mystmt and stmt.parent is mystmt.parent:
217 _stmts = []
218 _stmt_parents = []
219 elif isinstance(node, nodes.DelName):
220 # Remove all previously stored assignments
221 _stmts = []
222 _stmt_parents = []
223 continue
224 # Add the new assignment
225 _stmts.append(node)
226 if isinstance(node, nodes.Arguments) or isinstance(
227 node.parent, nodes.Arguments
228 ):
229 # Special case for _stmt_parents when node is a function parameter;
230 # in this case, stmt is the enclosing FunctionDef, which is what we
231 # want to add to _stmt_parents, not stmt.parent. This case occurs when
232 # node is an Arguments node (representing varargs or kwargs parameter),
233 # and when node.parent is an Arguments node (other parameters).
234 # See issue #180.
235 _stmt_parents.append(stmt)
236 else:
237 _stmt_parents.append(stmt.parent)
238 return _stmts
