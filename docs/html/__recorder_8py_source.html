32def _remove_nones(d: "Any") -> "Any":
33 if isinstance(d, dict):
34 return {k: _remove_nones(v) for k, v in d.items() if v is not None}

36 return [_remove_nones(i) for i in d]

40def _remove_default_headers(data: "Any") -> "Any":
41 """
42 It would be too verbose to store these headers in the file generated by the
43 record functionality.
44 """
45 if isinstance(data, dict):
46 keys_to_remove = [
47 "Content-Length",
48 "Content-Type",
49 "Date",
50 "Server",
51 "Connection",
52 "Content-Encoding",
53 ]
54 for i, response in enumerate(data["responses"]):
55 for key in keys_to_remove:
56 if key in response["response"]["headers"]:
57 del data["responses"][i]["response"]["headers"][key]
58 if not response["response"]["headers"]:
59 del data["responses"][i]["response"]["headers"]
60 return data
