117def _last_token_on_line_is(tokens: TokenWrapper, line_end: int, token: str) -> bool:
118 return (
119 line_end > 0
120 and tokens.token(line_end - 1) == token
121 or line_end > 1
122 and tokens.token(line_end - 2) == token
123 and tokens.type(line_end - 1) == tokenize.COMMENT
124 )

273 # pylint: disable-next = too-many-return-statements, too-many-branches
274 def _check_keyword_parentheses(
275 self, tokens: list[tokenize.TokenInfo], start: int
276 ) -> None:
277 """Check that there are not unnecessary parentheses after a keyword.
278
279 Parens are unnecessary if there is exactly one balanced outer pair on a
280 line and contains no commas (i.e. is not a tuple).
281
282 Args:
283 tokens: The entire list of Tokens.
284 start: The position of the keyword in the token list.
285 """
286 # If the next token is not a paren, we're fine.
287 if tokens[start + 1].string != "(":
288 return
289 if (
290 tokens[start].string == "not"
291 and start > 0
292 and tokens[start - 1].string == "is"
293 ):
294 # If this is part of an `is not` expression, we have a binary operator
295 # so the parentheses are not necessarily redundant.
296 return
297 found_and_or = False
298 contains_walrus_operator = False
299 walrus_operator_depth = 0
300 contains_double_parens = 0
301 depth = 0
302 keyword_token = str(tokens[start].string)
303 line_num = tokens[start].start[0]
304 for i in range(start, len(tokens) - 1):
305 token = tokens[i]
306
307 # If we hit a newline, then assume any parens were for continuation.
308 if token.type == tokenize.NL:
309 return
310 # Since the walrus operator doesn't exist below python3.8, the tokenizer
311 # generates independent tokens
312 if (
313 token.string == ":=" # <-- python3.8+ path
314 or token.string + tokens[i + 1].string == ":="
315 ):
316 contains_walrus_operator = True
317 walrus_operator_depth = depth
318 if token.string == "(":
319 depth += 1
320 if tokens[i + 1].string == "(":
321 contains_double_parens = 1
322 elif token.string == ")":
323 depth -= 1
324 if depth:
325 if contains_double_parens and tokens[i + 1].string == ")":
326 # For walrus operators in `if (not)` conditions and comprehensions
327 if keyword_token in {"in", "if", "not"}:
328 continue
329 return
330 contains_double_parens -= 1
331 continue
332 # ')' can't happen after if (foo), since it would be a syntax error.
333 if tokens[i + 1].string in {":", ")", "]", "}", "in"} or tokens[
334 i + 1
335 ].type in {tokenize.NEWLINE, tokenize.ENDMARKER, tokenize.COMMENT}:
336 if contains_walrus_operator and walrus_operator_depth - 1 == depth:
337 return
338 # The empty tuple () is always accepted.
339 if i == start + 2:
340 return
341 if found_and_or:
342 return
343 if keyword_token == "in":
344 # This special case was added in https://github.com/pylint-dev/pylint/pull/4948
345 # but it could be removed in the future. Avoid churn for now.
346 return
347 self.add_message(
348 "superfluous-parens", line=line_num, args=keyword_token
349 )
350 return
351 elif depth == 1:
352 # This is a tuple, which is always acceptable.
353 if token[1] == ",":
354 return
355 # 'and' and 'or' are the only boolean operators with lower precedence
356 # than 'not', so parens are only required when they are found.
357 if token[1] in {"and", "or"}:
358 found_and_or = True
359 # A yield inside an expression must always be in parentheses,
360 # quit early without error.
361 elif token[1] == "yield":
362 return
363 # A generator expression always has a 'for' token in it, and
364 # the 'for' token is only legal inside parens when it is in a
365 # generator expression. The parens are necessary here, so bail
366 # without an error.
367 elif token[1] == "for":
368 return
369 # A generator expression can have an 'else' token in it.
370 # We check the rest of the tokens to see if any problems occur after
371 # the 'else'.
372 elif token[1] == "else":
373 if "(" in (i.string for i in tokens[i:]):
374 self._check_keyword_parentheses(tokens[i:], 0)
375 return

377 def process_tokens(self, tokens: list[tokenize.TokenInfo]) -> None:
378 """Process tokens and search for:
379
380 - too long lines (i.e. longer than <max_chars>)
381 - optionally bad construct (if given, bad_construct must be a compiled
382 regular expression).
383 """
384 indents = [0]
385 check_equal = False
386 line_num = 0
387 self._lines = {}
388 self._visited_lines = {}
389 self._last_line_ending: str | None = None
390 last_blank_line_num = 0
391 for idx, (tok_type, string, start, _, line) in enumerate(tokens):
392 if start[0] != line_num:
393 line_num = start[0]
394 # A tokenizer oddity: if an indented line contains a multi-line
395 # docstring, the line member of the INDENT token does not contain
396 # the full line; therefore we check the next token on the line.
397 if tok_type == tokenize.INDENT:
398 self.new_line(TokenWrapper(tokens), idx - 1, idx + 1)
399 else:
400 self.new_line(TokenWrapper(tokens), idx - 1, idx)
401
402 if tok_type == tokenize.NEWLINE:
403 # a program statement, or ENDMARKER, will eventually follow,
404 # after some (possibly empty) run of tokens of the form
405 # (NL | COMMENT)* (INDENT | DEDENT+)?
406 # If an INDENT appears, setting check_equal is wrong, and will
407 # be undone when we see the INDENT.
408 check_equal = True
409 self._check_line_ending(string, line_num)
410 elif tok_type == tokenize.INDENT:
411 check_equal = False
412 self.check_indent_level(string, indents[-1] + 1, line_num)
413 indents.append(indents[-1] + 1)
414 elif tok_type == tokenize.DEDENT:
415 # there's nothing we need to check here! what's important is
416 # that when the run of DEDENTs ends, the indentation of the
417 # program statement (or ENDMARKER) that triggered the run is
418 # equal to what's left at the top of the indents stack
419 check_equal = True
420 if len(indents) > 1:
421 del indents[-1]
422 elif tok_type == tokenize.NL:
423 if not line.strip("\r\n"):
424 last_blank_line_num = line_num
425 elif tok_type not in (tokenize.COMMENT, tokenize.ENCODING):
426 # This is the first concrete token following a NEWLINE, so it
427 # must be the first token of the next program statement, or an
428 # ENDMARKER; the "line" argument exposes the leading white-space
429 # for this statement; in the case of ENDMARKER, line is an empty
430 # string, so will properly match the empty string with which the
431 # "indents" stack was seeded
432 if check_equal:
433 check_equal = False
434 self.check_indent_level(line, indents[-1], line_num)
435
436 if tok_type == tokenize.NUMBER and string.endswith("l"):
437 self.add_message("lowercase-l-suffix", line=line_num)
438
439 if string in _KEYWORD_TOKENS:
440 self._check_keyword_parentheses(tokens, idx)
441
442 line_num -= 1 # to be ok with "wc -l"
443 if line_num > self.linter.config.max_module_lines:
444 # Get the line where the too-many-lines (or its message id)
445 # was disabled or default to 1.
446 message_definition = self.linter.msgs_store.get_message_definitions(
447 "too-many-lines"
448 )[0]
449 names = (message_definition.msgid, "too-many-lines")
450 lineno = next(
451 filter(None, (self.linter._pragma_lineno.get(name) for name in names)),
452 1,
453 )
454 self.add_message(
455 "too-many-lines",
456 args=(line_num, self.linter.config.max_module_lines),
457 line=lineno,
458 )
459
460 # See if there are any trailing lines. Do not complain about empty
461 # files like __init__.py markers.
462 if line_num == last_blank_line_num and line_num > 0:
463 self.add_message("trailing-newlines", line=line_num)
464
465 def _check_line_ending(self, line_ending: str, line_num: int) -> None:
466 # check if line endings are mixed
467 if self._last_line_ending is not None:
468 # line_ending == "" indicates a synthetic newline added at
469 # the end of a file that does not, in fact, end with a
470 # newline.
471 if line_ending and line_ending != self._last_line_ending:
472 self.add_message("mixed-line-endings", line=line_num)
473
474 self._last_line_ending = line_ending
475
476 # check if line ending is as expected
477 expected = self.linter.config.expected_line_ending_format
478 if expected:
479 # reduce multiple \n\n\n\n to one \n
480 line_ending = reduce(lambda x, y: x + y if x != y else x, line_ending, "")
481 line_ending = "LF" if line_ending == "\n" else "CRLF"
482 if line_ending != expected:
483 self.add_message(
484 "unexpected-line-ending-format",
485 args=(line_ending, expected),
486 line=line_num,
487 )

489 @only_required_for_messages("multiple-statements")
490 def visit_default(self, node: nodes.NodeNG) -> None:
491 """Check the node line number and check it if not yet done."""
492 if not node.is_statement:
493 return
494 if not node.root().pure_python:
495 return
496 prev_sibl = node.previous_sibling()
497 if prev_sibl is not None:
498 prev_line = prev_sibl.fromlineno
499 elif isinstance(
500 node.parent, nodes.Try
501 ) and self._is_first_node_in_else_finally_body(node, node.parent):
502 prev_line = self._infer_else_finally_line_number(node, node.parent)
503 elif isinstance(node.parent, nodes.Module):
504 prev_line = 0
505 else:
506 prev_line = node.parent.statement().fromlineno
507 line = node.fromlineno
508 assert line, node
509 if prev_line == line and self._visited_lines.get(line) != 2:
510 self._check_multi_statement_line(node, line)
511 return
512 if line in self._visited_lines:
513 return
514 try:
515 tolineno = node.blockstart_tolineno
516 except AttributeError:
517 tolineno = node.tolineno
518 assert tolineno, node
519 lines: list[str] = []
520 for line in range(line, tolineno + 1): # noqa: B020
521 self._visited_lines[line] = 1
522 try:
523 lines.append(self._lines[line].rstrip())
524 except KeyError:
525 lines.append("")

527 def _is_first_node_in_else_finally_body(
528 self, node: nodes.NodeNG, parent: nodes.Try
529 ) -> bool:
530 if parent.orelse and node == parent.orelse[0]:
531 return True
532 if parent.finalbody and node == parent.finalbody[0]:
533 return True
534 return False
535
536 def _infer_else_finally_line_number(
537 self, node: nodes.NodeNG, parent: nodes.Try
538 ) -> int:
539 last_line_of_prev_block = 0
540 if node in parent.finalbody and parent.orelse:
541 last_line_of_prev_block = parent.orelse[-1].tolineno
542 elif parent.handlers and parent.handlers[-1].body:
543 last_line_of_prev_block = parent.handlers[-1].body[-1].tolineno
544 elif parent.body:
545 last_line_of_prev_block = parent.body[-1].tolineno
546
547 return last_line_of_prev_block + 1 if last_line_of_prev_block else 0

549 def _check_multi_statement_line(self, node: nodes.NodeNG, line: int) -> None:
550 """Check for lines containing multiple statements."""
551 if isinstance(node, nodes.With):
552 # Do not warn about multiple nested context managers in with statements.
553 return
554 if (
555 isinstance(node.parent, nodes.If)
556 and not node.parent.orelse
557 and self.linter.config.single_line_if_stmt
558 ):
559 return
560 if (
561 isinstance(node.parent, nodes.ClassDef)
562 and len(node.parent.body) == 1
563 and self.linter.config.single_line_class_stmt
564 ):
565 return
566
567 # Functions stubs and class with ``Ellipsis`` as body are exempted.
568 if (
569 isinstance(node, nodes.Expr)
570 and isinstance(node.parent, (nodes.FunctionDef, nodes.ClassDef))
571 and isinstance(node.value, nodes.Const)
572 and node.value.value is Ellipsis
573 ):
574 return
575
576 self.add_message("multiple-statements", node=node, confidence=HIGH)
577 self._visited_lines[line] = 2

579 def check_trailing_whitespace_ending(self, line: str, i: int) -> None:
580 """Check that there is no trailing white-space."""
581 # exclude \f (formfeed) from the rstrip
582 stripped_line = line.rstrip("\t\n\r\v ")
583 if line[len(stripped_line) :] not in ("\n", "\r\n"):
584 self.add_message(
585 "trailing-whitespace",
586 line=i,
587 col_offset=len(stripped_line),
588 confidence=HIGH,
589 )

591 def check_line_length(self, line: str, i: int, checker_off: bool) -> None:
592 """Check that the line length is less than the authorized value."""
593 max_chars = self.linter.config.max_line_length
594 ignore_long_line = self.linter.config.ignore_long_lines
595 line = line.rstrip()
596 if len(line) > max_chars and not ignore_long_line.search(line):
597 if checker_off:
598 self.linter.add_ignored_message("line-too-long", i)
599 else:
600 self.add_message("line-too-long", line=i, args=(len(line), max_chars))

602 @staticmethod
603 def remove_pylint_option_from_lines(options_pattern_obj: Match[str]) -> str:
604 """Remove the `# pylint ...` pattern from lines."""
605 lines = options_pattern_obj.string
606 purged_lines = (
607 lines[: options_pattern_obj.start(1)].rstrip()
608 + lines[options_pattern_obj.end(1) :]
609 )
610 return purged_lines

612 @staticmethod
613 def is_line_length_check_activated(pylint_pattern_match_object: Match[str]) -> bool:
614 """Return True if the line length check is activated."""
615 try:
616 for pragma in parse_pragma(pylint_pattern_match_object.group(2)):
617 if pragma.action == "disable" and "line-too-long" in pragma.messages:
618 return False
619 except PragmaParserError:
620 # Printing useful information dealing with this error is done in the lint package
621 pass
622 return True

624 @staticmethod
625 def specific_splitlines(lines: str) -> list[str]:
626 """Split lines according to universal newlines except those in a specific
627 sets.
628 """
629 unsplit_ends = {
630 "\x0b", # synonym of \v
631 "\x0c", # synonym of \f
632 "\x1c",
633 "\x1d",
634 "\x1e",
635 "\x85",
636 "\u2028",
637 "\u2029",
638 }
639 res: list[str] = []
640 buffer = ""
641 for atomic_line in lines.splitlines(True):
642 if atomic_line[-1] not in unsplit_ends:
643 res.append(buffer + atomic_line)
644 buffer = ""
645 else:
646 buffer += atomic_line
647 return res

649 def check_lines(
650 self, tokens: TokenWrapper, line_start: int, lines: str, lineno: int
651 ) -> None:
652 """Check given lines for potential messages.
653
654 Check if lines have:
655 - a final newline
656 - no trailing white-space
657 - less than a maximum number of characters
658 """
659 # we're first going to do a rough check whether any lines in this set
660 # go over the line limit. If none of them do, then we don't need to
661 # parse out the pylint options later on and can just assume that these
662 # lines are clean
663
664 # we'll also handle the line ending check here to avoid double-iteration
665 # unless the line lengths are suspect
666
667 max_chars = self.linter.config.max_line_length
668
669 split_lines = self.specific_splitlines(lines)
670
671 for offset, line in enumerate(split_lines):
672 if not line.endswith("\n"):
673 self.add_message("missing-final-newline", line=lineno + offset)
674 continue
675 # We don't test for trailing whitespaces in strings
676 # See https://github.com/pylint-dev/pylint/issues/6936
677 # and https://github.com/pylint-dev/pylint/issues/3822
678 if tokens.type(line_start) != tokenize.STRING:
679 self.check_trailing_whitespace_ending(line, lineno + offset)
680
681 # This check is purposefully simple and doesn't rstrip since this is running
682 # on every line you're checking it's advantageous to avoid doing a lot of work
683 potential_line_length_warning = any(
684 len(line) > max_chars for line in split_lines
685 )
686
687 # if there were no lines passing the max_chars config, we don't bother
688 # running the full line check (as we've met an even more strict condition)
689 if not potential_line_length_warning:
690 return
691
692 # Line length check may be deactivated through `pylint: disable` comment
693 mobj = OPTION_PO.search(lines)
694 checker_off = False
695 if mobj:
696 if not self.is_line_length_check_activated(mobj):
697 checker_off = True
698 # The 'pylint: disable whatever' should not be taken into account for line length count
699 lines = self.remove_pylint_option_from_lines(mobj)
700
701 # here we re-run specific_splitlines since we have filtered out pylint options above
702 for offset, line in enumerate(self.specific_splitlines(lines)):
703 self.check_line_length(line, lineno + offset, checker_off)

705 def check_indent_level(self, string: str, expected: int, line_num: int) -> None:
