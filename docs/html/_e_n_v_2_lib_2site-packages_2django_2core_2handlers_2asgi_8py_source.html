48 def __init__(self, scope, body_file):

50 self._post_parse_error = False
51 self._read_started = False
52 self.resolver_match = None
53 self.path = scope["path"]
54 self.script_name = get_script_prefix(scope)
55 if self.script_name:
56 # TODO: Better is-prefix checking, slash handling?
57 self.path_info = scope["path"].removeprefix(self.script_name)
58 else:
59 self.path_info = scope["path"]
60 # HTTP basics.
61 self.method = self.scope["method"].upper()
62 # Ensure query string is encoded correctly.
63 query_string = self.scope.get("query_string", "")
64 if isinstance(query_string, bytes):
65 query_string = query_string.decode()
66 self.META = {
67 "REQUEST_METHOD": self.method,
68 "QUERY_STRING": query_string,
69 "SCRIPT_NAME": self.script_name,
70 "PATH_INFO": self.path_info,
71 # WSGI-expecting code will need these for a while
72 "wsgi.multithread": True,
73 "wsgi.multiprocess": True,
74 }
75 if self.scope.get("client"):
76 self.META["REMOTE_ADDR"] = self.scope["client"][0]
77 self.META["REMOTE_HOST"] = self.META["REMOTE_ADDR"]
78 self.META["REMOTE_PORT"] = self.scope["client"][1]
79 if self.scope.get("server"):
80 self.META["SERVER_NAME"] = self.scope["server"][0]
81 self.META["SERVER_PORT"] = str(self.scope["server"][1])
82 else:
83 self.META["SERVER_NAME"] = "unknown"
84 self.META["SERVER_PORT"] = "0"
85 # Headers go into META.
86 for name, value in self.scope.get("headers", []):
87 name = name.decode("latin1")
88 if name == "content-length":
89 corrected_name = "CONTENT_LENGTH"
90 elif name == "content-type":
91 corrected_name = "CONTENT_TYPE"
92 else:
93 corrected_name = "HTTP_%s" % name.upper().replace("-", "_")
94 # HTTP/2 say only ASCII chars are allowed in headers, but decode
95 # latin1 just in case.
96 value = value.decode("latin1")
97 if corrected_name in self.META:
98 value = self.META[corrected_name] + "," + value
99 self.META[corrected_name] = value
100 # Pull out request encoding, if provided.
101 self._set_content_type_params(self.META)
102 # Directly assign the body file to be our stream.
103 self._stream = body_file
104 # Other bits.
105 self.resolver_match = None

164 async def handle(self, scope, receive, send):
165 """
166 Handles the ASGI request. Called via the __call__ method.
167 """
168 # Receive the HTTP request body as a stream object.
169 try:
170 body_file = await self.read_body(receive)
171 except RequestAborted:
172 return
173 # Request is complete and can be served.
174 set_script_prefix(get_script_prefix(scope))
175 await signals.request_started.asend(sender=self.__class__, scope=scope)
176 # Get the request and check for basic issues.
177 request, error_response = self.create_request(scope, body_file)
178 if request is None:
179 body_file.close()
180 await self.send_response(error_response, send)
181 await sync_to_async(error_response.close)()
182 return

184 async def process_request(request, send):
185 response = await self.run_get_response(request)
186 try:
187 await self.send_response(response, send)
188 except asyncio.CancelledError:
189 # Client disconnected during send_response (ignore exception).
190 pass
191
192 return response
193
194 # Try to catch a disconnect while getting response.
195 tasks = [
196 # Check the status of these tasks and (optionally) terminate them
197 # in this order. The listen_for_disconnect() task goes first
198 # because it should not raise unexpected errors that would prevent
199 # us from cancelling process_request().
200 asyncio.create_task(self.listen_for_disconnect(receive)),
201 asyncio.create_task(process_request(request, send)),
202 ]
203 await asyncio.wait(tasks, return_when=asyncio.FIRST_COMPLETED)
204 # Now wait on both tasks (they may have both finished by now).
205 for task in tasks:
206 if task.done():
207 try:
208 task.result()
209 except RequestAborted:
210 # Ignore client disconnects.
211 pass
212 except AssertionError:
213 body_file.close()
214 raise
215 else:
216 # Allow views to handle cancellation.
217 task.cancel()
218 try:
219 await task
220 except asyncio.CancelledError:
221 # Task re-raised the CancelledError as expected.
222 pass
223
224 try:
225 response = tasks[1].result()
226 except asyncio.CancelledError:
227 await signals.request_finished.asend(sender=self.__class__)
228 else:
229 await sync_to_async(response.close)()

231 body_file.close()

252 async def read_body(self, receive):
253 """Reads an HTTP body from an ASGI connection."""
254 # Use the tempfile that auto rolls-over to a disk file as it fills up.
255 body_file = tempfile.SpooledTemporaryFile(
256 max_size=settings.FILE_UPLOAD_MAX_MEMORY_SIZE, mode="w+b"
257 )
258 while True:
259 message = await receive()
260 if message["type"] == "http.disconnect":
261 body_file.close()
262 # Early client disconnect.
263 raise RequestAborted()
264 # Add a body chunk from the message, if provided.
265 if "body" in message:
266 body_file.write(message["body"])
267 # Quit out if that's the end.
268 if not message.get("more_body", False):
269 break
270 body_file.seek(0)
271 return body_file

302 async def send_response(self, response, send):
303 """Encode and send a response out over ASGI."""
304 # Collect cookies into headers. Have to preserve header case as there
305 # are some non-RFC compliant clients that require e.g. Content-Type.
306 response_headers = []
307 for header, value in response.items():
308 if isinstance(header, str):
309 header = header.encode("ascii")
310 if isinstance(value, str):
311 value = value.encode("latin1")
312 response_headers.append((bytes(header), bytes(value)))
313 for c in response.cookies.values():
314 response_headers.append(
315 (b"Set-Cookie", c.output(header="").encode("ascii").strip())
316 )
317 # Initial response message.
318 await send(
319 {
320 "type": "http.response.start",
321 "status": response.status_code,
322 "headers": response_headers,
323 }
324 )
325 # Streaming responses need to be pinned to their iterator.
326 if response.streaming:
327 # - Consume via `__aiter__` and not `streaming_content` directly, to
328 # allow mapping of a sync iterator.
329 # - Use aclosing() when consuming aiter.
330 # See https://github.com/python/cpython/commit/6e8dcda
331 async with aclosing(aiter(response)) as content:
332 async for part in content:
333 for chunk, _ in self.chunk_bytes(part):
334 await send(
335 {
336 "type": "http.response.body",
337 "body": chunk,
338 # Ignore "more" as there may be more parts; instead,
339 # use an empty final closing message with False.
340 "more_body": True,
341 }
342 )
343 # Final closing message.
344 await send({"type": "http.response.body"})
345 # Other responses just need chunking.
346 else:
347 # Yield chunks of response.
348 for chunk, last in self.chunk_bytes(response.content):
349 await send(
350 {
351 "type": "http.response.body",
352 "body": chunk,
353 "more_body": not last,
354 }
355 )

357 @classmethod
358 def chunk_bytes(cls, data):
359 """
360 Chunks some data up so it can be sent in reasonable size messages.
361 Yields (chunk, last_chunk) tuples.
362 """
363 position = 0
364 if not data:
365 yield data, True
366 return
367 while position < len(data):
368 yield (
369 data[position : position + cls.chunk_size],
370 (position + cls.chunk_size) >= len(data),
371 )
