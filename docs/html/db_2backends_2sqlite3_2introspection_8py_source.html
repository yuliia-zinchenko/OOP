171 def _parse_column_or_constraint_definition(self, tokens, columns):
172 token = None
173 is_constraint_definition = None
174 field_name = None
175 constraint_name = None
176 unique = False
177 unique_columns = []
178 check = False
179 check_columns = []
180 braces_deep = 0
181 for token in tokens:
182 if token.match(sqlparse.tokens.Punctuation, "("):
183 braces_deep += 1
184 elif token.match(sqlparse.tokens.Punctuation, ")"):
185 braces_deep -= 1
186 if braces_deep < 0:
187 # End of columns and constraints for table definition.
188 break
189 elif braces_deep == 0 and token.match(sqlparse.tokens.Punctuation, ","):
190 # End of current column or constraint definition.
191 break
192 # Detect column or constraint definition by first token.
193 if is_constraint_definition is None:
194 is_constraint_definition = token.match(
195 sqlparse.tokens.Keyword, "CONSTRAINT"
196 )
197 if is_constraint_definition:
198 continue
199 if is_constraint_definition:
200 # Detect constraint name by second token.
201 if constraint_name is None:
202 if token.ttype in (sqlparse.tokens.Name, sqlparse.tokens.Keyword):
203 constraint_name = token.value
204 elif token.ttype == sqlparse.tokens.Literal.String.Symbol:
205 constraint_name = token.value[1:-1]
206 # Start constraint columns parsing after UNIQUE keyword.
207 if token.match(sqlparse.tokens.Keyword, "UNIQUE"):
208 unique = True
209 unique_braces_deep = braces_deep
210 elif unique:
211 if unique_braces_deep == braces_deep:
212 if unique_columns:
213 # Stop constraint parsing.
214 unique = False
215 continue
216 if token.ttype in (sqlparse.tokens.Name, sqlparse.tokens.Keyword):
217 unique_columns.append(token.value)
218 elif token.ttype == sqlparse.tokens.Literal.String.Symbol:
219 unique_columns.append(token.value[1:-1])
220 else:
221 # Detect field name by first token.
222 if field_name is None:
223 if token.ttype in (sqlparse.tokens.Name, sqlparse.tokens.Keyword):
224 field_name = token.value
225 elif token.ttype == sqlparse.tokens.Literal.String.Symbol:
226 field_name = token.value[1:-1]
227 if token.match(sqlparse.tokens.Keyword, "UNIQUE"):
228 unique_columns = [field_name]
229 # Start constraint columns parsing after CHECK keyword.
230 if token.match(sqlparse.tokens.Keyword, "CHECK"):
231 check = True
232 check_braces_deep = braces_deep
233 elif check:
234 if check_braces_deep == braces_deep:
235 if check_columns:
236 # Stop constraint parsing.
237 check = False
238 continue
239 if token.ttype in (sqlparse.tokens.Name, sqlparse.tokens.Keyword):
240 if token.value in columns:
241 check_columns.append(token.value)
242 elif token.ttype == sqlparse.tokens.Literal.String.Symbol:
243 if token.value[1:-1] in columns:
244 check_columns.append(token.value[1:-1])
245 unique_constraint = (
246 {
247 "unique": True,
248 "columns": unique_columns,
249 "primary_key": False,
250 "foreign_key": None,
251 "check": False,
252 "index": False,
253 }
254 if unique_columns
255 else None
256 )
257 check_constraint = (
258 {
259 "check": True,
260 "columns": check_columns,
261 "primary_key": False,
262 "unique": False,
263 "foreign_key": None,
264 "index": False,
265 }
266 if check_columns
267 else None
268 )
269 return constraint_name, unique_constraint, check_constraint, token

271 def _parse_table_constraints(self, sql, columns):
272 # Check constraint parsing is based of SQLite syntax diagram.
273 # https://www.sqlite.org/syntaxdiagrams.html#table-constraint
274 statement = sqlparse.parse(sql)[0]
275 constraints = {}
276 unnamed_constrains_index = 0
277 tokens = (token for token in statement.flatten() if not token.is_whitespace)
278 # Go to columns and constraint definition
279 for token in tokens:
280 if token.match(sqlparse.tokens.Punctuation, "("):
281 break
282 # Parse columns and constraint definition
283 while True:
284 (
285 constraint_name,
286 unique,
287 check,
288 end_token,
289 ) = self._parse_column_or_constraint_definition(tokens, columns)
290 if unique:
291 if constraint_name:
292 constraints[constraint_name] = unique
293 else:
294 unnamed_constrains_index += 1
295 constraints[
296 "__unnamed_constraint_%s__" % unnamed_constrains_index
297 ] = unique
298 if check:
299 if constraint_name:
300 constraints[constraint_name] = check
301 else:
302 unnamed_constrains_index += 1
303 constraints[
304 "__unnamed_constraint_%s__" % unnamed_constrains_index
305 ] = check
306 if end_token.match(sqlparse.tokens.Punctuation, ")"):
307 break
308 return constraints

310 def get_constraints(self, cursor, table_name):
311 """
312 Retrieve any constraints or keys (unique, pk, fk, check, index) across
313 one or more columns.
314 """
315 constraints = {}
316 # Find inline check constraints.
317 try:
318 table_schema = cursor.execute(
319 "SELECT sql FROM sqlite_master WHERE type='table' and name=%s"
320 % (self.connection.ops.quote_name(table_name),)
321 ).fetchone()[0]
322 except TypeError:
323 # table_name is a view.
324 pass
325 else:
326 columns = {
327 info.name for info in self.get_table_description(cursor, table_name)
328 }
329 constraints.update(self._parse_table_constraints(table_schema, columns))
330
331 # Get the index info
332 cursor.execute(
333 "PRAGMA index_list(%s)" % self.connection.ops.quote_name(table_name)
334 )
335 for row in cursor.fetchall():
336 # SQLite 3.8.9+ has 5 columns, however older versions only give 3
337 # columns. Discard last 2 columns if there.
338 number, index, unique = row[:3]
339 cursor.execute(
340 "SELECT sql FROM sqlite_master "
341 "WHERE type='index' AND name=%s" % self.connection.ops.quote_name(index)
342 )
343 # There's at most one row.
344 (sql,) = cursor.fetchone() or (None,)
345 # Inline constraints are already detected in
346 # _parse_table_constraints(). The reasons to avoid fetching inline
347 # constraints from `PRAGMA index_list` are:
348 # - Inline constraints can have a different name and information
349 # than what `PRAGMA index_list` gives.
350 # - Not all inline constraints may appear in `PRAGMA index_list`.
351 if not sql:
352 # An inline constraint
353 continue
354 # Get the index info for that index
355 cursor.execute(
356 "PRAGMA index_info(%s)" % self.connection.ops.quote_name(index)
357 )
358 for index_rank, column_rank, column in cursor.fetchall():
359 if index not in constraints:
360 constraints[index] = {
361 "columns": [],
362 "primary_key": False,
363 "unique": bool(unique),
364 "foreign_key": None,
365 "check": False,
366 "index": True,
367 }
368 constraints[index]["columns"].append(column)
369 # Add type and column orders for indexes
370 if constraints[index]["index"]:
371 # SQLite doesn't support any index type other than b-tree
372 constraints[index]["type"] = Index.suffix
373 orders = self._get_index_columns_orders(sql)
374 if orders is not None:
375 constraints[index]["orders"] = orders
376 # Get the PK
377 pk_columns = self.get_primary_key_columns(cursor, table_name)
378 if pk_columns:
379 # SQLite doesn't actually give a name to the PK constraint,
380 # so we invent one. This is fine, as the SQLite backend never
381 # deletes PK constraints by name, as you can't delete constraints
382 # in SQLite; we remake the table with a new PK instead.
383 constraints["__primary__"] = {
384 "columns": pk_columns,
385 "primary_key": True,
386 "unique": False, # It's not actually a unique constraint.
387 "foreign_key": None,
388 "check": False,
389 "index": False,
390 }
391 relations = enumerate(self.get_relations(cursor, table_name).items())
392 constraints.update(
393 {
394 f"fk_{index}": {
395 "columns": [column_name],
396 "primary_key": False,
397 "unique": False,
398 "foreign_key": (ref_table_name, ref_column_name),
399 "check": False,
400 "index": False,
401 }
402 for index, (column_name, (ref_column_name, ref_table_name)) in relations
403 }
404 )
405 return constraints

415 def _get_column_collations(self, cursor, table_name):
416 row = cursor.execute(
417 """
418 SELECT sql
419 FROM sqlite_master
420 WHERE type = 'table' AND name = %s
421 """,
422 [table_name],
423 ).fetchone()
424 if not row:
425 return {}
426
427 sql = row[0]
428 columns = str(sqlparse.parse(sql)[0][-1]).strip("()").split(", ")
429 collations = {}
430 for column in columns:
431 tokens = column[1:].split()
432 column_name = tokens[0].strip('"')
433 for index, token in enumerate(tokens):
434 if token == "COLLATE":
435 collation = tokens[index + 1]
436 break
437 else:
438 collation = None
439 collations[column_name] = collation
440 return collations
