20class DocStringStyleChecker(checkers.BaseChecker):
21 """Checks format of docstrings based on PEP 0257."""
22
23 name = "docstyle"
24
25 msgs = {
26 "C0198": (
27 'Bad docstring quotes in %s, expected """, given %s',
28 "bad-docstring-quotes",
29 "Used when a docstring does not have triple double quotes.",
30 ),
31 "C0199": (
32 "First line empty in %s docstring",
33 "docstring-first-line-empty",
34 "Used when a blank line is found at the beginning of a docstring.",

38 @only_required_for_messages("docstring-first-line-empty", "bad-docstring-quotes")
39 def visit_module(self, node: nodes.Module) -> None:
40 self._check_docstring("module", node)
41
42 def visit_classdef(self, node: nodes.ClassDef) -> None:
43 self._check_docstring("class", node)
44
45 def visit_functiondef(self, node: nodes.FunctionDef) -> None:
46 ftype = "method" if node.is_method() else "function"
47 self._check_docstring(ftype, node)
48
49 visit_asyncfunctiondef = visit_functiondef

51 def _check_docstring(
52 self, node_type: str, node: nodes.Module | nodes.ClassDef | nodes.FunctionDef
53 ) -> None:
54 docstring = node.doc_node.value if node.doc_node else None
55 if docstring and docstring[0] == "\n":
56 self.add_message(
57 "docstring-first-line-empty",
58 node=node,
59 args=(node_type,),
60 confidence=HIGH,
61 )
62
63 # Use "linecache", instead of node.as_string(), because the latter
64 # looses the original form of the docstrings.
65
66 if docstring:
67 lineno = node.fromlineno + 1
68 line = linecache.getline(node.root().file, lineno).lstrip()
69 if line and line.find('"""') == 0:
70 return
71 if line and "'''" in line:
72 quotes = "'''"
73 elif line and line[0] == '"':
74 quotes = '"'
75 elif line and line[0] == "'":
76 quotes = "'"
77 else:
78 quotes = ""
79 if quotes:
80 self.add_message(
81 "bad-docstring-quotes",
82 node=node,
83 args=(node_type, quotes),
84 confidence=HIGH,
85 )
