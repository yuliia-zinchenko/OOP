66 def _set_one_msg_status(
67 self, scope: str, msg: MessageDefinition, line: int | None, enable: bool
68 ) -> None:
69 """Set the status of an individual message."""
70 if scope in {"module", "line"}:
71 assert isinstance(line, int) # should always be int inside module scope
72
73 self.linter.file_state.set_msg_status(msg, line, enable, scope)
74 if not enable and msg.symbol != "locally-disabled":
75 self.linter.add_message(
76 "locally-disabled", line=line, args=(msg.symbol, msg.msgid)
77 )
78 else:
79 msgs = self._msgs_state
80 msgs[msg.msgid] = enable

82 def _get_messages_to_set(
83 self, msgid: str, enable: bool, ignore_unknown: bool = False
84 ) -> list[MessageDefinition]:
85 """Do some tests and find the actual messages of which the status should be set."""
86 message_definitions: list[MessageDefinition] = []
87 if msgid == "all":
88 for _msgid in MSG_TYPES:
89 message_definitions.extend(
90 self._get_messages_to_set(_msgid, enable, ignore_unknown)
91 )
92 if not enable:
93 # "all" should not disable pylint's own warnings
94 message_definitions = list(
95 filter(
96 lambda m: m.msgid not in self.default_enabled_messages,
97 message_definitions,
98 )
99 )
100 return message_definitions
101
102 # msgid is a category?
103 category_id = msgid.upper()
104 if category_id not in MSG_TYPES:
105 category_id_formatted = MSG_TYPES_LONG.get(category_id)
106 else:
107 category_id_formatted = category_id
108 if category_id_formatted is not None:
109 for _msgid in self.linter.msgs_store._msgs_by_category[
110 category_id_formatted
111 ]:
112 message_definitions.extend(
113 self._get_messages_to_set(_msgid, enable, ignore_unknown)
114 )
115 return message_definitions
116
117 # msgid is a checker name?
118 if msgid.lower() in self.linter._checkers:
119 for checker in self.linter._checkers[msgid.lower()]:
120 for _msgid in checker.msgs:
121 message_definitions.extend(
122 self._get_messages_to_set(_msgid, enable, ignore_unknown)
123 )
124 return message_definitions
125
126 # msgid is report id?
127 if msgid.lower().startswith("rp"):
128 if enable:
129 self.linter.enable_report(msgid)
130 else:
131 self.linter.disable_report(msgid)
132 return message_definitions
133
134 try:
135 # msgid is a symbolic or numeric msgid.
136 message_definitions = self.linter.msgs_store.get_message_definitions(msgid)
137 except exceptions.UnknownMessageError:
138 if not ignore_unknown:
139 raise
140 return message_definitions

142 def _set_msg_status(
143 self,
144 msgid: str,
145 enable: bool,
146 scope: str = "package",
147 line: int | None = None,
148 ignore_unknown: bool = False,
149 ) -> None:
150 """Do some tests and then iterate over message definitions to set state."""
151 assert scope in {"package", "module", "line"}
152
153 message_definitions = self._get_messages_to_set(msgid, enable, ignore_unknown)
154
155 for message_definition in message_definitions:
156 self._set_one_msg_status(scope, message_definition, line, enable)
157
158 # sync configuration object
159 self.linter.config.enable = []
160 self.linter.config.disable = []
161 for msgid_or_symbol, is_enabled in self._msgs_state.items():
162 symbols = [
163 m.symbol
164 for m in self.linter.msgs_store.get_message_definitions(msgid_or_symbol)
165 ]
166 if is_enabled:
167 self.linter.config.enable += symbols
168 else:
169 self.linter.config.disable += symbols

241 def list_messages_enabled(self) -> None:
242 emittable, non_emittable = self.linter.msgs_store.find_emittable_messages()
243 enabled: list[str] = []
244 disabled: list[str] = []
245 for message in emittable:
246 if self.is_message_enabled(message.msgid):
247 enabled.append(f" {message.symbol} ({message.msgid})")
248 else:
249 disabled.append(f" {message.symbol} ({message.msgid})")
250 print("Enabled messages:")
251 for msg in enabled:
252 print(msg)
253 print("\nDisabled messages:")
254 for msg in disabled:
255 print(msg)
256 print("\nNon-emittable messages with current interpreter:")
257 for msg_def in non_emittable:
258 print(f" {msg_def.symbol} ({msg_def.msgid})")

261 def _get_message_state_scope(
262 self,
263 msgid: str,
264 line: int | None = None,
265 confidence: interfaces.Confidence | None = None,
266 ) -> Literal[0, 1, 2] | None:
267 """Returns the scope at which a message was enabled/disabled."""
268 if confidence is None:
269 confidence = interfaces.UNDEFINED
270 if confidence.name not in self.linter.config.confidence:
271 return MSG_STATE_CONFIDENCE # type: ignore[return-value] # mypy does not infer Literal correctly
272 try:
273 if line in self.linter.file_state._module_msgs_state[msgid]:
274 return MSG_STATE_SCOPE_MODULE # type: ignore[return-value]
275 except (KeyError, TypeError):
276 return MSG_STATE_SCOPE_CONFIG # type: ignore[return-value]
277 return None

279 def _is_one_message_enabled(self, msgid: str, line: int | None) -> bool:
280 """Checks state of a single message for the current file.
281
282 This function can't be cached as it depends on self.file_state which can
283 change.
284 """
285 if line is None:
286 return self._msgs_state.get(msgid, True)
287 try:
288 return self.linter.file_state._module_msgs_state[msgid][line]
289 except KeyError:
290 # Check if the message's line is after the maximum line existing in ast tree.
291 # This line won't appear in the ast tree and won't be referred in
292 # self.file_state._module_msgs_state
293 # This happens for example with a commented line at the end of a module.
294 max_line_number = self.linter.file_state.get_effective_max_line_number()
295 if max_line_number and line > max_line_number:
296 fallback = True
297 lines = self.linter.file_state._raw_module_msgs_state.get(msgid, {})
298
299 # Doesn't consider scopes, as a 'disable' can be in a
300 # different scope than that of the current line.
301 closest_lines = reversed(
302 [
303 (message_line, enable)
304 for message_line, enable in lines.items()
305 if message_line <= line
306 ]
307 )
308 _, fallback_iter = next(closest_lines, (None, None))
309 if fallback_iter is not None:
310 fallback = fallback_iter
311
312 return self._msgs_state.get(msgid, fallback)
313 return self._msgs_state.get(msgid, True)

315 def is_message_enabled(
316 self,
317 msg_descr: str,
318 line: int | None = None,
319 confidence: interfaces.Confidence | None = None,
320 ) -> bool:
321 """Is this message enabled for the current file ?
322
323 Optionally, is it enabled for this line and confidence level ?
324
325 The current file is implicit and mandatory. As a result this function
326 can't be cached right now as the line is the line of the currently
327 analysed file (self.file_state), if it changes, then the result for
328 the same msg_descr/line might need to change.
329
330 :param msg_descr: Either the msgid or the symbol for a MessageDefinition
331 :param line: The line of the currently analysed file
332 :param confidence: The confidence of the message
333 """
334 if confidence and confidence.name not in self.linter.config.confidence:
335 return False
336 try:
337 msgids = self.linter.msgs_store.message_id_store.get_active_msgids(
338 msg_descr
339 )
340 except exceptions.UnknownMessageError:
341 # The linter checks for messages that are not registered
342 # due to version mismatch, just treat them as message IDs
343 # for now.
344 msgids = [msg_descr]
345 return any(self._is_one_message_enabled(msgid, line) for msgid in msgids)

347 def process_tokens(self, tokens: list[tokenize.TokenInfo]) -> None:
348 """Process tokens from the current module to search for module/block level
349 options.
350
351 See func_block_disable_msg.py test case for expected behaviour.
352 """
353 control_pragmas = {"disable", "disable-next", "enable"}
354 prev_line = None
355 saw_newline = True
356 seen_newline = True
357 for tok_type, content, start, _, _ in tokens:
358 if prev_line and prev_line != start[0]:
359 saw_newline = seen_newline
360 seen_newline = False
361
362 prev_line = start[0]
363 if tok_type in (tokenize.NL, tokenize.NEWLINE):
364 seen_newline = True
365
366 if tok_type != tokenize.COMMENT:
367 continue
368 match = OPTION_PO.search(content)
369 if match is None:
370 continue
371 try: # pylint: disable = too-many-try-statements
372 for pragma_repr in parse_pragma(match.group(2)):
373 if pragma_repr.action in {"disable-all", "skip-file"}:
374 if pragma_repr.action == "disable-all":
375 self.linter.add_message(
376 "deprecated-pragma",
377 line=start[0],
378 args=("disable-all", "skip-file"),
379 )
380 self.linter.add_message("file-ignored", line=start[0])
381 self._ignore_file = True
382 return
383 try:
384 meth = self._options_methods[pragma_repr.action]
385 except KeyError:
386 meth = self._bw_options_methods[pragma_repr.action]
387 # found a "(dis|en)able-msg" pragma deprecated suppression
388 self.linter.add_message(
389 "deprecated-pragma",
390 line=start[0],
391 args=(
392 pragma_repr.action,
393 pragma_repr.action.replace("-msg", ""),
394 ),
395 )
396 for msgid in pragma_repr.messages:
397 # Add the line where a control pragma was encountered.
398 if pragma_repr.action in control_pragmas:
399 self._pragma_lineno[msgid] = start[0]
400
401 if (pragma_repr.action, msgid) == ("disable", "all"):
402 self.linter.add_message(
403 "deprecated-pragma",
404 line=start[0],
405 args=("disable=all", "skip-file"),
406 )
407 self.linter.add_message("file-ignored", line=start[0])
408 self._ignore_file = True
409 return
410 # If we did not see a newline between the previous line and now,
411 # we saw a backslash so treat the two lines as one.
412 l_start = start[0]
413 if not saw_newline:
414 l_start -= 1
415 try:
416 meth(msgid, "module", l_start)
417 except (
418 exceptions.DeletedMessageError,
419 exceptions.MessageBecameExtensionError,
420 ) as e:
421 self.linter.add_message(
422 "useless-option-value",
423 args=(pragma_repr.action, e),
424 line=start[0],
425 confidence=HIGH,
426 )
427 except exceptions.UnknownMessageError:
428 self.linter.add_message(
429 "unknown-option-value",
430 args=(pragma_repr.action, msgid),
431 line=start[0],
432 confidence=HIGH,
433 )
434
435 except UnRecognizedOptionError as err:
436 self.linter.add_message(
437 "unrecognized-inline-option", args=err.token, line=start[0]
438 )
439 continue
440 except InvalidPragmaError as err:
441 self.linter.add_message(
442 "bad-inline-option", args=err.token, line=start[0]
443 )
444 continue
