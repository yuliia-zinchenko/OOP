103def get_module_and_frameid(node: nodes.NodeNG) -> tuple[str, str]:
104 """Return the module name and the frame id in the module."""
105 frame = node.frame()
106 module, obj = "", []
107 while frame:
108 if isinstance(frame, Module):
109 module = frame.name
110 else:
111 obj.append(getattr(frame, "name", "<lambda>"))
112 try:
113 frame = frame.parent.frame()
114 except AttributeError:
115 break
116 obj.reverse()
117 return module, ".".join(obj)

127def get_rst_section(
128 section: str | None,
129 options: list[tuple[str, OptionDict, Any]],
130 doc: str | None = None,
131) -> str:
132 """Format an option's section using as a ReStructuredText formatted output."""
133 result = ""
134 if section:
135 result += get_rst_title(section, "'")
136 if doc:
137 formatted_doc = normalize_text(doc)
138 result += f"{formatted_doc}\n\n"
139 for optname, optdict, value in options:
140 help_opt = optdict.get("help")
141 result += f":{optname}:\n"
142 if help_opt:
143 assert isinstance(help_opt, str)
144 formatted_help = normalize_text(help_opt, indent=" ")
145 result += f"{formatted_help}\n"
146 if value and optname != "py-version":
147 value = str(_format_option_value(optdict, value))
148 result += f"\n Default: ``{value.replace('`` ', '```` ``')}``\n"
149 return result

170def register_plugins(linter: PyLinter, directory: str) -> None:
171 """Load all module and package in the given directory, looking for a
172 'register' function in each one, used to register pylint checkers.
173 """
174 imported = {}
175 for filename in os.listdir(directory):
176 base, extension = os.path.splitext(filename)
177 if base in imported or base == "__pycache__":
178 continue
179 if (
180 extension in PY_EXTS
181 and base != "__init__"
182 or (
183 not extension
184 and os.path.isdir(os.path.join(directory, base))
185 and not filename.startswith(".")
186 )
187 ):
188 try:
189 module = modutils.load_module_from_file(
190 os.path.join(directory, filename)
191 )
192 except ValueError:
193 # empty module name (usually Emacs auto-save files)
194 continue
195 except ImportError as exc:
196 print(f"Problem importing module {filename}: {exc}", file=sys.stderr)
197 else:
198 if hasattr(module, "register"):
199 module.register(linter)
200 imported[base] = 1

203def _splitstrip(string: str, sep: str = ",") -> list[str]:
204 r"""Return a list of stripped string by splitting the string given as
205 argument on `sep` (',' by default), empty strings are discarded.
206
207 >>> _splitstrip('a, b, c , 4,,')
208 ['a', 'b', 'c', '4']
209 >>> _splitstrip('a')
210 ['a']
211 >>> _splitstrip('a,\nb,\nc,')
212 ['a', 'b', 'c']
213
214 :type string: str or unicode
215 :param string: a csv line
216
217 :type sep: str or unicode
218 :param sep: field separator, default to the comma (',')
219
220 :rtype: str or unicode
221 :return: the unquoted string (or the input string if it wasn't quoted)
222 """
223 return [word.strip() for word in string.split(sep) if word.strip()]

247def _check_regexp_csv(value: list[str] | tuple[str] | str) -> Iterable[str]:
248 r"""Split a comma-separated list of regexps, taking care to avoid splitting
249 a regex employing a comma as quantifier, as in `\d{1,2}`.
250 """
251 if isinstance(value, (list, tuple)):
252 yield from value
253 else:
254 # None is a sentinel value here
255 regexps: deque[deque[str] | None] = deque([None])
256 open_braces = False
257 for char in value:
258 if char == "{":
259 open_braces = True
260 elif char == "}" and open_braces:
261 open_braces = False
262
263 if char == "," and not open_braces:
264 regexps.append(None)
265 elif regexps[-1] is None:
266 regexps.pop()
267 regexps.append(deque([char]))
268 else:
269 regexps[-1].append(char)
270 yield from ("".join(regexp).strip() for regexp in regexps if regexp is not None)

273def _comment(string: str) -> str:
274 """Return string as a comment."""
275 lines = [line.strip() for line in string.splitlines()]
276 sep = "\n"
277 return "# " + f"{sep}# ".join(lines)

280def _format_option_value(optdict: OptionDict, value: Any) -> str:
281 """Return the user input's value from a 'compiled' value.
282
283 TODO: Refactor the code to not use this deprecated function
284 """
285 if optdict.get("type", None) == "py_version":
286 value = ".".join(str(item) for item in value)
287 elif isinstance(value, (list, tuple)):
288 value = ",".join(_format_option_value(optdict, item) for item in value)
289 elif isinstance(value, dict):
290 value = ",".join(f"{k}:{v}" for k, v in value.items())
291 elif hasattr(value, "match"): # optdict.get('type') == 'regexp'
292 # compiled regexp
293 value = value.pattern
294 elif optdict.get("type") == "yn":
295 value = "yes" if value else "no"
296 elif isinstance(value, str) and value.isspace():
297 value = f"'{value}'"
298 return str(value)
