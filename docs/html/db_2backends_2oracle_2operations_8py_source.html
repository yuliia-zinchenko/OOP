83 def date_extract_sql(self, lookup_type, sql, params):
84 extract_sql = f"TO_CHAR({sql}, %s)"
85 extract_param = None
86 if lookup_type == "week_day":
87 # TO_CHAR(field, 'D') returns an integer from 1-7, where 1=Sunday.
88 extract_param = "D"

91 extract_param = "D"

324 def last_executed_query(self, cursor, sql, params):
325 # https://python-oracledb.readthedocs.io/en/latest/api_manual/cursor.html#Cursor.statement
326 # The DB API definition does not define this attribute.
327 statement = cursor.statement
328 # Unlike Psycopg's `query` and MySQLdb`'s `_executed`, oracledb's
329 # `statement` doesn't contain the query parameters. Substitute

332 if isinstance(params, (tuple, list)):
333 params = {

339 statement = statement.replace(
340 key, force_str(params[key], errors="replace")
341 )

376 def quote_name(self, name):
377 # SQL92 requires delimited (quoted) names to be case-sensitive. When
378 # not quoted, Oracle has case-insensitive behavior for identifiers, but
379 # always defaults to uppercase.
380 # We simplify things by making Oracle identifiers always uppercase.
381 if not name.startswith('"') and not name.endswith('"'):
382 name = '"%s"' % truncate_name(name, self.max_name_length())
383 # Oracle puts the query text into a (query % args) construct, so % signs
384 # in names need to be escaped. The '%%' will be collapsed back to '%' at
385 # that stage so we aren't really making the name longer here.
386 name = name.replace("%", "%%")
387 return name.upper()

416 with self.connection.cursor() as cursor:
417 if recursive:
418 cursor.execute(
419 """
420 SELECT
421 user_tables.table_name, rcons.constraint_name

427 AND cons.constraint_type = ANY('P', 'U'))

463 def sql_flush(self, style, tables, *, reset_sequences=False, allow_cascade=False):
464 if not tables:
465 return []
466
467 truncated_tables = {table.upper() for table in tables}
468 constraints = set()
469 # Oracle's TRUNCATE CASCADE only works with ON DELETE CASCADE foreign
470 # keys which Django doesn't define. Emulate the PostgreSQL behavior
471 # which truncates all dependent tables by manually retrieving all
472 # foreign key constraints and resolving dependencies.
473 for table in tables:
474 for foreign_table, constraint in self._foreign_key_constraints(
475 table, recursive=allow_cascade
476 ):
477 if allow_cascade:
478 truncated_tables.add(foreign_table)
479 constraints.add((foreign_table, constraint))
480 sql = (
481 [
482 "%s %s %s %s %s %s %s %s;"
483 % (
484 style.SQL_KEYWORD("ALTER"),
485 style.SQL_KEYWORD("TABLE"),
486 style.SQL_FIELD(self.quote_name(table)),
487 style.SQL_KEYWORD("DISABLE"),
488 style.SQL_KEYWORD("CONSTRAINT"),
489 style.SQL_FIELD(self.quote_name(constraint)),
490 style.SQL_KEYWORD("KEEP"),
491 style.SQL_KEYWORD("INDEX"),
492 )
493 for table, constraint in constraints
494 ]
495 + [
496 "%s %s %s;"
497 % (
498 style.SQL_KEYWORD("TRUNCATE"),
499 style.SQL_KEYWORD("TABLE"),
500 style.SQL_FIELD(self.quote_name(table)),
501 )
502 for table in truncated_tables
503 ]
504 + [
505 "%s %s %s %s %s %s;"
506 % (
507 style.SQL_KEYWORD("ALTER"),
508 style.SQL_KEYWORD("TABLE"),
509 style.SQL_FIELD(self.quote_name(table)),
510 style.SQL_KEYWORD("ENABLE"),
511 style.SQL_KEYWORD("CONSTRAINT"),
512 style.SQL_FIELD(self.quote_name(constraint)),
513 )
514 for table, constraint in constraints
515 ]
516 )
517 if reset_sequences:
518 sequences = [
519 sequence
520 for sequence in self.connection.introspection.sequence_list()
521 if sequence["table"].upper() in truncated_tables
522 ]
523 # Since we've just deleted all the rows, running our sequence ALTER
524 # code will reset the sequence to 0.
525 sql.extend(self.sequence_reset_by_name_sql(style, sequences))
526 return sql

635 def combine_expression(self, connector, sub_expressions):
636 lhs, rhs = sub_expressions
637 if connector == "%%":
638 return "MOD(%s)" % ",".join(sub_expressions)
639 elif connector == "&":
640 return "BITAND(%s)" % ",".join(sub_expressions)
641 elif connector == "|":
642 return "BITAND(-%(lhs)s-1,%(rhs)s)+%(lhs)s" % {"lhs": lhs, "rhs": rhs}
643 elif connector == "<<":
644 return "(%(lhs)s * POWER(2, %(rhs)s))" % {"lhs": lhs, "rhs": rhs}
645 elif connector == ">>":
646 return "FLOOR(%(lhs)s / POWER(2, %(rhs)s))" % {"lhs": lhs, "rhs": rhs}
647 elif connector == "^":
648 return "POWER(%s)" % ",".join(sub_expressions)
649 elif connector == "#":
650 raise NotSupportedError("Bitwise XOR is not supported in Oracle.")
651 return super().combine_expression(connector, sub_expressions)

673 def bulk_insert_sql(self, fields, placeholder_rows):
674 field_placeholders = [
675 BulkInsertMapper.types.get(
676 getattr(field, "target_field", field).get_internal_type(), "%s"
677 )
678 for field in fields
679 if field
680 ]
681 query = []
682 for row in placeholder_rows:
683 select = []
684 for i, placeholder in enumerate(row):
685 # A model without any fields has fields=[None].
686 if fields[i]:
687 placeholder = field_placeholders[i] % placeholder
688 # Add columns aliases to the first select to avoid "ORA-00918:
689 # column ambiguously defined" when two or more columns in the
690 # first select have the same value.
691 if not query:
692 placeholder = "%s col_%s" % (placeholder, i)
693 select.append(placeholder)
694 suffix = self.connection.features.bare_select_suffix
695 query.append(f"SELECT %s{suffix}" % ", ".join(select))
696 # Bulk insert to tables with Oracle identity columns causes Oracle to
697 # add sequence.nextval to it. Sequence.nextval cannot be used with the
698 # UNION operator. To prevent incorrect SQL, move UNION to a subquery.
699 return "SELECT * FROM (%s)" % " UNION ALL ".join(query)
