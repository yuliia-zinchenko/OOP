305 def get_new_connection(self, conn_params):
306 # self.isolation_level must be set:
307 # - after connecting to the database in order to obtain the database's
308 # default when no value is explicitly specified in options.
309 # - before calling _set_autocommit() because if autocommit is on, that
310 # will set connection.isolation_level to ISOLATION_LEVEL_AUTOCOMMIT.
311 options = self.settings_dict["OPTIONS"]
312 set_isolation_level = False

368 def _configure_connection(self, connection):
369 # This function is called from init_connection_state and from the

447 # Get the current async task
448 # Note that right now this is behind @async_unsafe, so this is
449 # unreachable, but in future we'll start loosening this restriction.

557 def callproc(self, name, args=None):
558 if not isinstance(name, sql.Identifier):
559 name = sql.Identifier(name)
560
561 qparts = [sql.SQL("SELECT * FROM "), name, sql.SQL("(")]
562 if args:
563 for item in args:
564 qparts.append(sql.Literal(item))
565 qparts.append(sql.SQL(","))
566 del qparts[-1]
567
568 qparts.append(sql.SQL(")"))
569 stmt = sql.Composed(qparts)
570 self.execute(stmt)
571 return args
