44 def date_extract_sql(self, lookup_type, sql, params):
45 # https://dev.mysql.com/doc/mysql/en/date-and-time-functions.html
46 if lookup_type == "week_day":
47 # DAYOFWEEK() returns an integer, 1-7, Sunday=1.
48 return f"DAYOFWEEK({sql})", params
49 elif lookup_type == "iso_week_day":
50 # WEEKDAY() returns an integer, 0-6, Monday=0.
51 return f"WEEKDAY({sql}) + 1", params
52 elif lookup_type == "week":
53 # Override the value of default_week_format for consistency with
54 # other database backends.
55 # Mode 3: Monday, 1-53, with 4 or more days this year.
56 return f"WEEK({sql}, 3)", params
57 elif lookup_type == "iso_year":
58 # Get the year part from the YEARWEEK function, which returns a
59 # number as year * 100 + week.
60 return f"TRUNCATE(YEARWEEK({sql}, 3), -2) / 100", params
61 else:
62 # EXTRACT returns 1-53 based on ISO-8601 for the week number.
63 lookup_type = lookup_type.upper()
64 if not self._extract_format_re.fullmatch(lookup_type):
65 raise ValueError(f"Invalid loookup type: {lookup_type!r}")
66 return f"EXTRACT({lookup_type} FROM {sql})", params

202 def sql_flush(self, style, tables, *, reset_sequences=False, allow_cascade=False):
203 if not tables:
204 return []
205
206 sql = ["SET FOREIGN_KEY_CHECKS = 0;"]
207 if reset_sequences:
208 # It's faster to TRUNCATE tables that require a sequence reset
209 # since ALTER TABLE AUTO_INCREMENT is slower than TRUNCATE.
210 sql.extend(
211 "%s %s;"
212 % (
213 style.SQL_KEYWORD("TRUNCATE"),
214 style.SQL_FIELD(self.quote_name(table_name)),
215 )
216 for table_name in tables
217 )
218 else:
219 # Otherwise issue a simple DELETE since it's faster than TRUNCATE
220 # and preserves sequences.
221 sql.extend(
222 "%s %s %s;"
223 % (
224 style.SQL_KEYWORD("DELETE"),
225 style.SQL_KEYWORD("FROM"),
226 style.SQL_FIELD(self.quote_name(table_name)),
227 )
228 for table_name in tables
229 )
230 sql.append("SET FOREIGN_KEY_CHECKS = 1;")
231 return sql

293 def combine_expression(self, connector, sub_expressions):
294 if connector == "^":
295 return "POW(%s)" % ",".join(sub_expressions)
296 # Convert the result to a signed integer since MySQL's binary operators
297 # return an unsigned integer.
298 elif connector in ("&", "|", "<<", "#"):
299 connector = "^" if connector == "#" else connector
300 return "CONVERT(%s, SIGNED)" % connector.join(sub_expressions)

302 lhs, rhs = sub_expressions

364 def explain_query_prefix(self, format=None, **options):
365 # Alias MySQL's TRADITIONAL to TEXT for consistency with other backends.
366 if format and format.upper() == "TEXT":
367 format = "TRADITIONAL"
368 elif (
