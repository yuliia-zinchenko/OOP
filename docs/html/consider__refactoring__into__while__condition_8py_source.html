48 def _check_breaking_after_while_true(self, node: nodes.While) -> None:
49 """Check that any loop with an ``if`` clause has a break statement."""
50 if not isinstance(node.test, nodes.Const) or not node.test.bool_value():
51 return
52 pri_candidates: list[nodes.If] = []
53 for n in node.body:
54 if not isinstance(n, nodes.If):
55 break
56 pri_candidates.append(n)
57 candidates = []
58 tainted = False
59 for c in pri_candidates:
60 if tainted or not isinstance(c.body[0], nodes.Break):
61 break
62 candidates.append(c)
63 orelse = c.orelse
64 while orelse:
65 orelse_node = orelse[0]
66 if not isinstance(orelse_node, nodes.If):
67 tainted = True
68 else:
69 candidates.append(orelse_node)
70 if not isinstance(orelse_node, nodes.If):
71 break
72 orelse = orelse_node.orelse
73
74 candidates = [n for n in candidates if isinstance(n.body[0], nodes.Break)]
75 msg = " and ".join(
76 [f"({utils.not_condition_as_string(c.test)})" for c in candidates]
77 )
78 if len(candidates) == 1:
79 msg = utils.not_condition_as_string(candidates[0].test)

81 return
