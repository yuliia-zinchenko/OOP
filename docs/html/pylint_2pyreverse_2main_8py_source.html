269class Run(_ArgumentsManager, _ArgumentsProvider):
270 """Base class providing common behaviour for pyreverse commands."""
271
272 options = OPTIONS
273 name = "pyreverse"
274
275 def __init__(self, args: Sequence[str]) -> NoReturn:
276 # Immediately exit if user asks for version
277 if "--version" in args:
278 print("pyreverse is included in pylint:")
279 print(constants.full_version)
280 sys.exit(0)
281
282 _ArgumentsManager.__init__(self, prog="pyreverse", description=__doc__)
283 _ArgumentsProvider.__init__(self, self)
284
285 # Parse options
286 insert_default_options()
287 args = self._parse_command_line_configuration(args)
288
289 if self.config.output_format not in DIRECTLY_SUPPORTED_FORMATS:
290 check_graphviz_availability()
291 print(
292 f"Format {self.config.output_format} is not supported natively."
293 " Pyreverse will try to generate it using Graphviz..."
294 )
295 check_if_graphviz_supports_format(self.config.output_format)
296
297 sys.exit(self.run(args))
298
299 def run(self, args: list[str]) -> int:
300 """Checking arguments and run project."""
301 if not args:
302 print(self.help())
303 return 1
304 extra_packages_paths = list(
305 {discover_package_path(arg, self.config.source_roots) for arg in args}
306 )
307 with augmented_sys_path(extra_packages_paths):
308 project = project_from_files(
309 args,
310 project_name=self.config.project,
311 black_list=self.config.ignore_list,
312 verbose=self.config.verbose,
313 )
314 linker = Linker(project, tag=True)
315 handler = DiadefsHandler(self.config)
316 diadefs = handler.get_diadefs(project, linker)
317 writer.DiagramWriter(self.config).write(diadefs)
318 return 0
