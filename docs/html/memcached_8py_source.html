42 def get_backend_timeout(self, timeout=DEFAULT_TIMEOUT):
43 """
44 Memcached deals with long (> 30 days) timeouts in a special
45 way. Call this function to obtain a safe value for your timeout.
46 """
47 if timeout == DEFAULT_TIMEOUT:
48 timeout = self.default_timeout
49
50 if timeout is None:
51 # Using 0 in memcache sets a non-expiring timeout.
52 return 0
53 elif int(timeout) == 0:
54 # Other cache backends treat 0 as set-and-expire. To achieve this
55 # in memcache backends, a negative timeout must be passed.
56 timeout = -1
57
58 if timeout > 2592000: # 60*60*24*30, 30 days
59 # See https://github.com/memcached/memcached/wiki/Programming#expiration
60 # "Expiration times can be set from 0, meaning "never expire", to
61 # 30 days. Any time higher than 30 days is interpreted as a Unix
62 # timestamp date. If you want to expire an object on January 1st of
63 # next year, this is how you do that."
64 #
65 # This means that we have to switch to absolute timestamps.
66 timeout += int(time.time())
67 return int(timeout)

103 def incr(self, key, delta=1, version=None):
104 key = self.make_and_validate_key(key, version=version)
105 try:
106 # Memcached doesn't support negative delta.
107 if delta < 0:
108 val = self._cache.decr(key, -delta)
109 else:
110 val = self._cache.incr(key, delta)
111 # Normalize an exception raised by the underlying client library to
112 # ValueError in the event of a nonexistent key when calling
113 # incr()/decr().
114 except self.LibraryValueNotFoundException:
115 val = None
116 if val is None:
117 raise ValueError("Key '%s' not found" % key)
118 return val
