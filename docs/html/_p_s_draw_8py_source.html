34 def __init__(self, fp: IO[bytes] | None = None) -> None:

39 def begin_document(self, id: str | None = None) -> None:
40 """Set up printing of a document. (Write PostScript DSC header.)"""
41 # FIXME: incomplete
42 self.fp.write(
43 b"%!PS-Adobe-3.0\n"
44 b"save\n"
45 b"/showpage { } def\n"
46 b"%%EndComments\n"

48 )
49 # self.fp.write(ERROR_PS) # debugging!
50 self.fp.write(EDROFF_PS)
51 self.fp.write(VDI_PS)
52 self.fp.write(b"%%EndProlog\n")
53 self.isofont: dict[bytes, int] = {}
54
55 def end_document(self) -> None:
56 """Ends printing. (Write PostScript DSC footer.)"""
57 self.fp.write(b"%%EndDocument\nrestore showpage\n%%End\n")
58 if hasattr(self.fp, "flush"):
59 self.fp.flush()

61 def setfont(self, font: str, size: int) -> None:
62 """
63 Selects which font to use.
64
65 :param font: A PostScript font name
66 :param size: Size in points.
67 """
68 font_bytes = bytes(font, "UTF-8")
69 if font_bytes not in self.isofont:
70 # reencode font
71 self.fp.write(
72 b"/PSDraw-%s ISOLatin1Encoding /%s E\n" % (font_bytes, font_bytes)
73 )
74 self.isofont[font_bytes] = 1
75 # rough
76 self.fp.write(b"/F0 %d /PSDraw-%s F\n" % (size, font_bytes))

78 def line(self, xy0: tuple[int, int], xy1: tuple[int, int]) -> None:
79 """
80 Draws a line between the two points. Coordinates are given in
81 PostScript point coordinates (72 points per inch, (0, 0) is the lower
82 left corner of the page).
83 """
84 self.fp.write(b"%d %d %d %d Vl\n" % (*xy0, *xy1))
85
86 def rectangle(self, box: tuple[int, int, int, int]) -> None:
87 """
88 Draws a rectangle.
89
90 :param box: A tuple of four integers, specifying left, bottom, width and
91 height.
92 """
93 self.fp.write(b"%d %d M 0 %d %d Vr\n" % box)

95 def text(self, xy: tuple[int, int], text: str) -> None:
96 """
97 Draws text at the given position. You must use
98 :py:meth:`~PIL.PSDraw.PSDraw.setfont` before calling this method.
99 """
100 text_bytes = bytes(text, "UTF-8")
101 text_bytes = b"\\(".join(text_bytes.split(b"("))
102 text_bytes = b"\\)".join(text_bytes.split(b")"))
103 self.fp.write(b"%d %d M (%s) S\n" % (xy + (text_bytes,)))

108 def image(
109 self, box: tuple[int, int, int, int], im: Image.Image, dpi: int | None = None
110 ) -> None:
111 """Draw a PIL image, centered in the given box."""
112 # default resolution depends on mode
113 if not dpi:
114 if im.mode == "1":
115 dpi = 200 # fax
116 else:
117 dpi = 100 # grayscale
118 # image size (on paper)
119 x = im.size[0] * 72 / dpi
120 y = im.size[1] * 72 / dpi
121 # max allowed size
122 xmax = float(box[2] - box[0])
123 ymax = float(box[3] - box[1])
124 if x > xmax:
125 y = y * xmax / x
126 x = xmax
127 if y > ymax:
128 x = x * ymax / y
129 y = ymax
130 dx = (xmax - x) / 2 + box[0]
131 dy = (ymax - y) / 2 + box[1]
132 self.fp.write(b"gsave\n%f %f translate\n" % (dx, dy))
133 if (x, y) != im.size:
134 # EpsImagePlugin._save prints the image at (0,0,xsize,ysize)
135 sx = x / im.size[0]
136 sy = y / im.size[1]
137 self.fp.write(b"%f %f scale\n" % (sx, sy))
138 EpsImagePlugin._save(im, self.fp, "", 0)
139 self.fp.write(b"\ngrestore\n")
