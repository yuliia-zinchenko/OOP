56 def get_table_list(self, cursor):
57 """Return a list of table and view names in the current database."""
58 cursor.execute(
59 """
60 SELECT
61 c.relname,
62 CASE
63 WHEN c.relispartition THEN 'p'
64 WHEN c.relkind IN ('m', 'v') THEN 'v'
65 ELSE 't'
66 END,
67 obj_description(c.oid, 'pg_class')
68 FROM pg_catalog.pg_class c
69 LEFT JOIN pg_catalog.pg_namespace n ON n.oid = c.relnamespace
70 WHERE c.relkind IN ('f', 'm', 'p', 'r', 'v')
71 AND n.nspname NOT IN ('pg_catalog', 'pg_toast')
72 AND pg_catalog.pg_table_is_visible(c.oid)
73 """
74 )
75 return [
76 TableInfo(*row)
77 for row in cursor.fetchall()
78 if row[0] not in self.ignored_tables
79 ]

81 def get_table_description(self, cursor, table_name):
82 """
83 Return a description of the table with the DB-API cursor.description
84 interface.
85 """
86 # Query the pg_catalog tables as cursor.description does not reliably
87 # return the nullable property and information_schema.columns does not
88 # contain details of materialized views.
89 cursor.execute(
90 """
91 SELECT
92 a.attname AS column_name,
93 NOT (a.attnotnull OR (t.typtype = 'd' AND t.typnotnull)) AS is_nullable,
94 pg_get_expr(ad.adbin, ad.adrelid) AS column_default,
95 CASE WHEN collname = 'default' THEN NULL ELSE collname END AS collation,
96 a.attidentity != '' AS is_autofield,
97 col_description(a.attrelid, a.attnum) AS column_comment
98 FROM pg_attribute a
99 LEFT JOIN pg_attrdef ad ON a.attrelid = ad.adrelid AND a.attnum = ad.adnum
100 LEFT JOIN pg_collation co ON a.attcollation = co.oid
101 JOIN pg_type t ON a.atttypid = t.oid
102 JOIN pg_class c ON a.attrelid = c.oid
103 JOIN pg_namespace n ON c.relnamespace = n.oid
104 WHERE c.relkind IN ('f', 'm', 'p', 'r', 'v')
105 AND c.relname = %s
106 AND n.nspname NOT IN ('pg_catalog', 'pg_toast')
107 AND pg_catalog.pg_table_is_visible(c.oid)
108 """,
109 [table_name],
110 )
111 field_map = {line[0]: line[1:] for line in cursor.fetchall()}
112 cursor.execute(
113 "SELECT * FROM %s LIMIT 1" % self.connection.ops.quote_name(table_name)
114 )
115 return [
116 FieldInfo(
117 line.name,
118 line.type_code,
119 # display_size is always None on psycopg2.
120 line.internal_size if line.display_size is None else line.display_size,
121 line.internal_size,
122 line.precision,
123 line.scale,
124 *field_map[line.name],
125 )
126 for line in cursor.description
127 ]

156 """
157 Return a dictionary of {field_name: (field_name_other_table, other_table)}

159 """
160 cursor.execute(

165 LEFT JOIN pg_class c2 ON con.confrelid = c2.oid

167 pg_attribute a1 ON c1.oid = a1.attrelid AND a1.attnum = con.conkey[1]
168 LEFT JOIN
169 pg_attribute a2 ON c2.oid = a2.attrelid AND a2.attnum = con.confkey[1]

171 c1.relname = %s AND
172 con.contype = 'f' AND
173 c1.relnamespace = c2.relnamespace AND
174 pg_catalog.pg_table_is_visible(c1.oid)
175 """,
176 [table_name],

180 def get_constraints(self, cursor, table_name):
181 """
182 Retrieve any constraints or keys (unique, pk, fk, check, index) across
183 one or more columns. Also retrieve the definition of expression-based
184 indexes.
185 """
186 constraints = {}
187 # Loop over the key table, collecting things as constraints. The column
188 # array must return column names in the same order in which they were
189 # created.
190 cursor.execute(
191 """
192 SELECT
193 c.conname,
194 array(
195 SELECT attname
196 FROM unnest(c.conkey) WITH ORDINALITY cols(colid, arridx)
197 JOIN pg_attribute AS ca ON cols.colid = ca.attnum
198 WHERE ca.attrelid = c.conrelid
199 ORDER BY cols.arridx
200 ),
201 c.contype,
202 (SELECT fkc.relname || '.' || fka.attname
203 FROM pg_attribute AS fka
204 JOIN pg_class AS fkc ON fka.attrelid = fkc.oid
205 WHERE fka.attrelid = c.confrelid AND fka.attnum = c.confkey[1]),
206 cl.reloptions
207 FROM pg_constraint AS c
208 JOIN pg_class AS cl ON c.conrelid = cl.oid
209 WHERE cl.relname = %s AND pg_catalog.pg_table_is_visible(cl.oid)
210 """,
211 [table_name],
212 )
213 for constraint, columns, kind, used_cols, options in cursor.fetchall():
214 constraints[constraint] = {
215 "columns": columns,
216 "primary_key": kind == "p",
217 "unique": kind in ["p", "u"],
218 "foreign_key": tuple(used_cols.split(".", 1)) if kind == "f" else None,
219 "check": kind == "c",
220 "index": False,
221 "definition": None,
222 "options": options,
223 }
224 # Now get indexes
225 cursor.execute(
226 """
227 SELECT
228 indexname,
229 array_agg(attname ORDER BY arridx),
230 indisunique,
231 indisprimary,
232 array_agg(ordering ORDER BY arridx),
233 amname,
234 exprdef,
235 s2.attoptions
236 FROM (
237 SELECT
238 c2.relname as indexname, idx.*, attr.attname, am.amname,
239 CASE
240 WHEN idx.indexprs IS NOT NULL THEN
241 pg_get_indexdef(idx.indexrelid)
242 END AS exprdef,
243 CASE am.amname
244 WHEN %s THEN
245 CASE (option & 1)
246 WHEN 1 THEN 'DESC' ELSE 'ASC'
247 END
248 END as ordering,
249 c2.reloptions as attoptions
250 FROM (
251 SELECT *
252 FROM
253 pg_index i,
254 unnest(i.indkey, i.indoption)
255 WITH ORDINALITY koi(key, option, arridx)
256 ) idx
257 LEFT JOIN pg_class c ON idx.indrelid = c.oid
258 LEFT JOIN pg_class c2 ON idx.indexrelid = c2.oid
259 LEFT JOIN pg_am am ON c2.relam = am.oid
260 LEFT JOIN
261 pg_attribute attr ON attr.attrelid = c.oid AND attr.attnum = idx.key
262 WHERE c.relname = %s AND pg_catalog.pg_table_is_visible(c.oid)
263 ) s2
264 GROUP BY indexname, indisunique, indisprimary, amname, exprdef, attoptions;
265 """,
266 [self.index_default_access_method, table_name],
267 )
268 for (
269 index,
270 columns,
271 unique,
272 primary,
273 orders,
274 type_,
275 definition,
276 options,
277 ) in cursor.fetchall():
278 if index not in constraints:
279 basic_index = (
280 type_ == self.index_default_access_method
281 and
282 # '_btree' references
283 # django.contrib.postgres.indexes.BTreeIndex.suffix.
284 not index.endswith("_btree")
285 and options is None
286 )
287 constraints[index] = {
288 "columns": columns if columns != [None] else [],
289 "orders": orders if orders != [None] else [],
290 "primary_key": primary,
291 "unique": unique,
292 "foreign_key": None,
293 "check": False,
294 "index": True,
295 "type": Index.suffix if basic_index else type_,
296 "definition": definition,
297 "options": options,
298 }
299 return constraints
