5"""A similarities / code duplication command line tool and pylint checker.
6
7The algorithm is based on comparing the hash value of n successive lines of a file.
8First the files are read and any line that doesn't fulfill requirement are removed
9(comments, docstrings...)
10
11Those stripped lines are stored in the LineSet class which gives access to them.
12Then each index of the stripped lines collection is associated with the hash of n
13successive entries of the stripped lines starting at the current index (n is the
14minimum common lines option).
15
16The common hashes between both linesets are then looked for. If there are matches, then
17the match indices in both linesets are stored and associated with the corresponding
18couples (start line number/end line number) in both files.
19
20This association is then post-processed to handle the case of successive matches. For
21example if the minimum common lines setting is set to four, then the hashes are
22computed with four lines. If one of match indices couple (12, 34) is the
23successor of another one (11, 33) then it means that there are in fact five lines which
24are common.
25
26Once post-processed the values of association table are the result looked for, i.e.
27start and end lines numbers of common lines in both files.
28"""
29

207def hash_lineset(
208 lineset: LineSet, min_common_lines: int = DEFAULT_MIN_SIMILARITY_LINE
209) -> tuple[HashToIndex_T, IndexToLines_T]:
210 """Return two dicts.
211
212 The first associates the hash of successive stripped lines of a lineset
213 to the indices of the starting lines.
214 The second dict, associates the index of the starting line in the lineset's stripped lines to the
215 couple [start, end] lines number in the corresponding file.
216
217 :param lineset: lineset object (i.e the lines in a file)
218 :param min_common_lines: number of successive lines that are used to compute the hash
219 :return: a dict linking hashes to corresponding start index and a dict that links this
220 index to the start and end lines in the file
221 """
222 hash2index = defaultdict(list)
223 index2lines = {}
224 # Comments, docstring and other specific patterns maybe excluded -> call to stripped_lines
225 # to get only what is desired
226 lines = tuple(x.text for x in lineset.stripped_lines)
227 # Need different iterators on same lines but each one is shifted 1 from the precedent
228 shifted_lines = [iter(lines[i:]) for i in range(min_common_lines)]
229
230 for i, *succ_lines in enumerate(zip(*shifted_lines)):
231 start_linenumber = LineNumber(lineset.stripped_lines[i].line_number)
232 try:
233 end_linenumber = lineset.stripped_lines[i + min_common_lines].line_number
234 except IndexError:
235 end_linenumber = LineNumber(lineset.stripped_lines[-1].line_number + 1)
236
237 index = Index(i)
238 index2lines[index] = SuccessiveLinesLimits(
239 start=start_linenumber, end=end_linenumber
240 )
241
242 l_c = LinesChunk(lineset.name, index, *succ_lines)
243 hash2index[l_c].append(index)
244
245 return hash2index, index2lines

248def remove_successive(all_couples: CplIndexToCplLines_T) -> None:
249 """Removes all successive entries in the dictionary in argument.
250
251 :param all_couples: collection that has to be cleaned up from successive entries.
252 The keys are couples of indices that mark the beginning of common entries
253 in both linesets. The values have two parts. The first one is the couple
254 of starting and ending line numbers of common successive lines in the first file.
255 The second part is the same for the second file.
256
257 For example consider the following dict:
258
259 >>> all_couples
260 {(11, 34): ([5, 9], [27, 31]),
261 (23, 79): ([15, 19], [45, 49]),
262 (12, 35): ([6, 10], [28, 32])}
263
264 There are two successive keys (11, 34) and (12, 35).
265 It means there are two consecutive similar chunks of lines in both files.
266 Thus remove last entry and update the last line numbers in the first entry
267
268 >>> remove_successive(all_couples)
269 >>> all_couples
270 {(11, 34): ([5, 10], [27, 32]),
271 (23, 79): ([15, 19], [45, 49])}
272 """
273 couple: LineSetStartCouple
274 for couple in tuple(all_couples.keys()):
275 to_remove = []
276 test = couple.increment(Index(1))
277 while test in all_couples:
278 all_couples[couple].first_file.end = all_couples[test].first_file.end
279 all_couples[couple].second_file.end = all_couples[test].second_file.end
280 all_couples[couple].effective_cmn_lines_nb += 1
281 to_remove.append(test)
282 test = test.increment(Index(1))
283
284 for target in to_remove:
285 try:
286 all_couples.pop(target)
287 except KeyError:
288 pass

291def filter_noncode_lines(
292 ls_1: LineSet,
293 stindex_1: Index,
294 ls_2: LineSet,
295 stindex_2: Index,
296 common_lines_nb: int,
297) -> int:
298 """Return the effective number of common lines between lineset1
299 and lineset2 filtered from non code lines.
300
301 That is to say the number of common successive stripped
302 lines except those that do not contain code (for example
303 a line with only an ending parenthesis)
304
305 :param ls_1: first lineset
306 :param stindex_1: first lineset starting index
307 :param ls_2: second lineset
308 :param stindex_2: second lineset starting index
309 :param common_lines_nb: number of common successive stripped lines before being filtered from non code lines
310 :return: the number of common successive stripped lines that contain code
311 """
312 stripped_l1 = [
313 lspecif.text
314 for lspecif in ls_1.stripped_lines[stindex_1 : stindex_1 + common_lines_nb]
315 if REGEX_FOR_LINES_WITH_CONTENT.match(lspecif.text)
316 ]
317 stripped_l2 = [
318 lspecif.text
319 for lspecif in ls_2.stripped_lines[stindex_2 : stindex_2 + common_lines_nb]
320 if REGEX_FOR_LINES_WITH_CONTENT.match(lspecif.text)
321 ]
322 return sum(sline_1 == sline_2 for sline_1, sline_2 in zip(stripped_l1, stripped_l2))

398 def _compute_sims(self) -> list[tuple[int, set[LinesChunkLimits_T]]]:
399 """Compute similarities in appended files."""
400 no_duplicates: dict[int, list[set[LinesChunkLimits_T]]] = defaultdict(list)
401
402 for commonality in self._iter_sims():
403 num = commonality.cmn_lines_nb
404 lineset1 = commonality.fst_lset
405 start_line_1 = commonality.fst_file_start
406 end_line_1 = commonality.fst_file_end
407 lineset2 = commonality.snd_lset
408 start_line_2 = commonality.snd_file_start
409 end_line_2 = commonality.snd_file_end
410
411 duplicate = no_duplicates[num]
412 couples: set[LinesChunkLimits_T]
413 for couples in duplicate:
414 if (lineset1, start_line_1, end_line_1) in couples or (
415 lineset2,
416 start_line_2,
417 end_line_2,
418 ) in couples:
419 break
420 else:
421 duplicate.append(
422 {
423 (lineset1, start_line_1, end_line_1),
424 (lineset2, start_line_2, end_line_2),
425 }
426 )
427 sims: list[tuple[int, set[LinesChunkLimits_T]]] = []
428 ensembles: list[set[LinesChunkLimits_T]]
429 for num, ensembles in no_duplicates.items():
430 cpls: set[LinesChunkLimits_T]
431 for cpls in ensembles:
432 sims.append((num, cpls))
433 sims.sort()
434 sims.reverse()
435 return sims

444 def _get_similarity_report(
445 self, similarities: list[tuple[int, set[LinesChunkLimits_T]]]
446 ) -> str:
447 """Create a report from similarities."""
448 report: str = ""
449 duplicated_line_number: int = 0
450 for number, couples in similarities:
451 report += f"\n{number} similar lines in {len(couples)} files\n"
452 couples_l = sorted(couples)
453 line_set = start_line = end_line = None
454 for line_set, start_line, end_line in couples_l:
455 report += f"=={line_set.name}:[{start_line}:{end_line}]\n"
456 if line_set:
457 for line in line_set._real_lines[start_line:end_line]:
458 report += f" {line.rstrip()}\n" if line.rstrip() else "\n"
459 duplicated_line_number += number * (len(couples_l) - 1)
460 total_line_number: int = sum(len(lineset) for lineset in self.linesets)
461 report += (
462 f"TOTAL lines={total_line_number} "
463 f"duplicates={duplicated_line_number} "
464 f"percent={duplicated_line_number * 100.0 / total_line_number:.2f}\n"
465 )
466 return report

469 def _find_common(
470 self, lineset1: LineSet, lineset2: LineSet
471 ) -> Generator[Commonality]:
472 """Find similarities in the two given linesets.
473
474 This the core of the algorithm. The idea is to compute the hashes of a
475 minimal number of successive lines of each lineset and then compare the
476 hashes. Every match of such comparison is stored in a dict that links the
477 couple of starting indices in both linesets to the couple of corresponding
478 starting and ending lines in both files.
479
480 Last regroups all successive couples in a bigger one. It allows to take into
481 account common chunk of lines that have more than the minimal number of
482 successive lines required.
483 """
484 hash_to_index_1: HashToIndex_T
485 hash_to_index_2: HashToIndex_T
486 index_to_lines_1: IndexToLines_T
487 index_to_lines_2: IndexToLines_T
488 hash_to_index_1, index_to_lines_1 = hash_lineset(
489 lineset1, self.namespace.min_similarity_lines
490 )
491 hash_to_index_2, index_to_lines_2 = hash_lineset(
492 lineset2, self.namespace.min_similarity_lines
493 )
494
495 hash_1: frozenset[LinesChunk] = frozenset(hash_to_index_1.keys())
496 hash_2: frozenset[LinesChunk] = frozenset(hash_to_index_2.keys())
497
498 common_hashes: Iterable[LinesChunk] = sorted(
499 hash_1 & hash_2, key=lambda m: hash_to_index_1[m][0]
500 )
501
502 # all_couples is a dict that links the couple of indices in both linesets that mark the beginning of
503 # successive common lines, to the corresponding starting and ending number lines in both files
504 all_couples: CplIndexToCplLines_T = {}
505
506 for c_hash in sorted(common_hashes, key=operator.attrgetter("_index")):
507 for indices_in_linesets in itertools.product(
508 hash_to_index_1[c_hash], hash_to_index_2[c_hash]
509 ):
510 index_1 = indices_in_linesets[0]
511 index_2 = indices_in_linesets[1]
512 all_couples[LineSetStartCouple(index_1, index_2)] = (
513 CplSuccessiveLinesLimits(
514 copy.copy(index_to_lines_1[index_1]),
515 copy.copy(index_to_lines_2[index_2]),
516 effective_cmn_lines_nb=self.namespace.min_similarity_lines,
517 )
518 )
519
520 remove_successive(all_couples)
521
522 for cml_stripped_l, cmn_l in all_couples.items():
523 start_index_1 = cml_stripped_l.fst_lineset_index
524 start_index_2 = cml_stripped_l.snd_lineset_index
525 nb_common_lines = cmn_l.effective_cmn_lines_nb
526
527 com = Commonality(
528 cmn_lines_nb=nb_common_lines,
529 fst_lset=lineset1,
530 fst_file_start=cmn_l.first_file.start,
531 fst_file_end=cmn_l.first_file.end,
532 snd_lset=lineset2,
533 snd_file_start=cmn_l.second_file.start,
534 snd_file_end=cmn_l.second_file.end,
535 )
536
537 eff_cmn_nb = filter_noncode_lines(
538 lineset1, start_index_1, lineset2, start_index_2, nb_common_lines
539 )
540
541 if eff_cmn_nb > self.namespace.min_similarity_lines:
542 yield com

544 def _iter_sims(self) -> Generator[Commonality]:
545 """Iterate on similarities among all files, by making a Cartesian
546 product.
547 """
548 for idx, lineset in enumerate(self.linesets[:-1]):
549 for lineset2 in self.linesets[idx + 1 :]:
550 yield from self._find_common(lineset, lineset2)

593 chain.from_iterable(
594 range(node.lineno, (node.end_lineno or node.lineno) + 1)
595 for node in tree.nodes_of_class((nodes.Import, nodes.ImportFrom))

600 def _get_functions(
601 functions: list[nodes.NodeNG], tree: nodes.NodeNG
602 ) -> list[nodes.NodeNG]:
603 """Recursively get all functions including nested in the classes from
604 the.
605
606 tree.
607 """
608 for node in tree.body:
609 if isinstance(node, (nodes.FunctionDef, nodes.AsyncFunctionDef)):
610 functions.append(node)
611
612 if isinstance(
613 node,
614 (nodes.ClassDef, nodes.FunctionDef, nodes.AsyncFunctionDef),
615 ):
616 _get_functions(functions, node)
617
618 return functions
619
620 functions = _get_functions([], tree)
621 ignore_lines.update(
622 chain.from_iterable(
623 range(
624 func.lineno,
625 func.body[0].lineno if func.body else func.tolineno + 1,
626 )
627 for func in functions
628 )
629 )
630
631 strippedlines = []
632 docstring = None
633 for lineno, line in enumerate(lines, start=1):
634 if line_enabled_callback is not None and not line_enabled_callback(
635 "R0801", lineno
636 ):
637 continue
638 line = line.strip()
639 if ignore_docstrings:
640 if not docstring:
641 if line.startswith(('"""', "'''")):
642 docstring = line[:3]
643 line = line[3:]
644 elif line.startswith(('r"""', "r'''")):
645 docstring = line[1:4]
646 line = line[4:]
647 if docstring:
648 if line.endswith(docstring):
649 docstring = None
650 line = ""
651 if ignore_comments:
652 line = line.split("#", 1)[0].strip()
653 if lineno in ignore_lines:
654 line = ""
655 if line:
656 strippedlines.append(
657 LineSpecifs(text=line, line_number=LineNumber(lineno - 1))
658 )

843 def close(self) -> None:
844 """Compute and display similarities on closing (i.e. end of parsing)."""
845 total = sum(len(lineset) for lineset in self.linesets)
846 duplicated = 0
847 stats = self.linter.stats
848 for num, couples in self._compute_sims():
849 msg = []
850 lineset = start_line = end_line = None
851 for lineset, start_line, end_line in couples:
852 msg.append(f"=={lineset.name}:[{start_line}:{end_line}]")
853 msg.sort()
854
855 if lineset:
856 for line in lineset.real_lines[start_line:end_line]:
857 msg.append(line.rstrip())
858
859 self.add_message("R0801", args=(len(couples), "\n".join(msg)))
860 duplicated += num * (len(couples) - 1)
861 stats.nb_duplicated_lines += int(duplicated)
862 stats.percent_duplicated_lines += float(total and duplicated * 100.0 / total)
