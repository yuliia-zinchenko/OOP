123 def _add_view(self, request, form_url="", extra_context=None):
124 # It's an error for a user to have add permission but NOT change
125 # permission for users. If we allowed such users to add users, they
126 # could create superusers, which would mean they would essentially have
127 # the permission to change users. To avoid the problem entirely, we
128 # disallow users from adding users if they don't have change
129 # permission.
130 if not self.has_change_permission(request):
131 if self.has_add_permission(request) and settings.DEBUG:
132 # Raise Http404 in debug mode so that the user gets a helpful
133 # error message.
134 raise Http404(
135 'Your user does not have the "Change user" permission. In '
136 "order to add users, Django requires that your user "
137 'account have both the "Add user" and "Change user" '
138 "permissions set."
139 )
140 raise PermissionDenied
141 if extra_context is None:
142 extra_context = {}
143 username_field = self.opts.get_field(self.model.USERNAME_FIELD)
144 defaults = {
145 "auto_populated_fields": (),
146 "username_help_text": username_field.help_text,
147 }
148 extra_context.update(defaults)
149 return super().add_view(request, form_url, extra_context)

152 def user_change_password(self, request, id, form_url=""):
153 user = self.get_object(request, unquote(id))
154 if not self.has_change_permission(request, user):
155 raise PermissionDenied
156 if user is None:
157 raise Http404(
158 _("%(name)s object with primary key %(key)r does not exist.")
159 % {
160 "name": self.opts.verbose_name,
161 "key": escape(id),
162 }
163 )
164 if request.method == "POST":
165 form = self.change_password_form(user, request.POST)
166 if form.is_valid():
167 # If disabling password-based authentication was requested
168 # (via the form field `usable_password`), the submit action
169 # must be "unset-password". This check is most relevant when
170 # the admin user has two submit buttons available (for example
171 # when Javascript is disabled).
172 valid_submission = (
173 form.cleaned_data["set_usable_password"]
174 or "unset-password" in request.POST
175 )
176 if not valid_submission:
177 msg = gettext("Conflicting form data submitted. Please try again.")
178 messages.error(request, msg)
179 return HttpResponseRedirect(request.get_full_path())
180
181 user = form.save()
182 change_message = self.construct_change_message(request, form, None)
183 self.log_change(request, user, change_message)
184 if user.has_usable_password():
185 msg = gettext("Password changed successfully.")
186 else:
187 msg = gettext("Password-based authentication was disabled.")
188 messages.success(request, msg)
189 update_session_auth_hash(request, form.user)
190 return HttpResponseRedirect(
191 reverse(
192 "%s:%s_%s_change"
193 % (
194 self.admin_site.name,
195 user._meta.app_label,
196 user._meta.model_name,
197 ),
198 args=(user.pk,),
199 )
200 )
201 else:
202 form = self.change_password_form(user)
203
204 fieldsets = [(None, {"fields": list(form.base_fields)})]
205 admin_form = admin.helpers.AdminForm(form, fieldsets, {})
206
207 if user.has_usable_password():
208 title = _("Change password: %s")
209 else:
210 title = _("Set password: %s")
211 context = {
212 "title": title % escape(user.get_username()),
213 "adminForm": admin_form,
214 "form_url": form_url,
215 "form": form,
216 "is_popup": (IS_POPUP_VAR in request.POST or IS_POPUP_VAR in request.GET),
217 "is_popup_var": IS_POPUP_VAR,
218 "add": True,
219 "change": False,
220 "has_delete_permission": False,
221 "has_change_permission": True,
222 "has_absolute_url": False,
223 "opts": self.opts,
224 "original": user,
225 "save_as": False,
226 "show_save": True,
227 **self.admin_site.each_context(request),
228 }
229
