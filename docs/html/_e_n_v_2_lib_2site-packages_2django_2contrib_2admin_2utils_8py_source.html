34def lookup_spawns_duplicates(opts, lookup_path):
35 """
36 Return True if the given lookup path spawns duplicates.
37 """
38 lookup_fields = lookup_path.split(LOOKUP_SEP)
39 # Go through the fields (following all relations) and look for an m2m.
40 for field_name in lookup_fields:
41 if field_name == "pk":
42 field_name = opts.pk.name
43 try:
44 field = opts.get_field(field_name)
45 except FieldDoesNotExist:
46 # Ignore query lookups.
47 continue
48 else:
49 if hasattr(field, "path_infos"):
50 # This field is a relation; update opts to follow the relation.
51 path_info = field.path_infos
52 opts = path_info[-1].to_opts
53 if any(path.m2m for path in path_info):
54 # This field is a m2m relation so duplicates must be
55 # handled.
56 return True
57 return False

65def prepare_lookup_value(key, value, separator=","):
66 """
67 Return a lookup value prepared to be used in queryset filtering.
68 """
69 if isinstance(value, list):
70 return [prepare_lookup_value(key, v, separator=separator) for v in value]
71 # if key ends with __in, split parameter into separate values
72 if key.endswith("__in"):
73 value = value.split(separator)
74 # if key ends with __isnull, special case '' and the string literals 'false' and '0'
75 elif key.endswith("__isnull"):
76 value = value.lower() not in ("", "false", "0")
77 return value

87def quote(s):
88 """
89 Ensure that primary key values do not confuse the admin URLs by escaping
90 any '/', '_' and ':' and similarly problematic characters.
91 Similar to urllib.parse.quote(), except that the quoting is slightly
92 different so that it doesn't get automatically unquoted by the web browser.
93 """
94 return s.translate(QUOTE_MAP) if isinstance(s, str) else s

123def get_deleted_objects(objs, request, admin_site):
124 """
125 Find all objects related to ``objs`` that should also be deleted. ``objs``
126 must be a homogeneous iterable of objects (e.g. a QuerySet).
127
128 Return a nested list of strings suitable for display in the
129 template with the ``unordered_list`` filter.
130 """

132 obj = objs[0]
133 except IndexError:
134 return [], {}, set(), []
135 else:
136 using = router.db_for_write(obj._meta.model)
137 collector = NestedObjects(using=using, origin=objs)
138 collector.collect(objs)
139 perms_needed = set()

141 def format_callback(obj):
142 model = obj.__class__
143 opts = obj._meta
144
145 no_edit_link = "%s: %s" % (capfirst(opts.verbose_name), obj)
146
147 if admin_site.is_registered(model):

149 request, obj

151 perms_needed.add(opts.verbose_name)

153 admin_url = reverse(
154 "%s:%s_%s_change"

156 None,

161 return no_edit_link

169 # admin or is edited inline.

175 model_count = {
176 model._meta.verbose_name_plural: len(objs)
177 for model, objs in collector.model_objs.items()
178 }

183class NestedObjects(Collector):
184 def __init__(self, *args, **kwargs):
185 super().__init__(*args, **kwargs)
186 self.edges = {} # {from_instance: [to_instances]}
187 self.protected = set()
188 self.model_objs = defaultdict(set)
189
190 def add_edge(self, source, target):
191 self.edges.setdefault(source, []).append(target)
192
193 def collect(self, objs, source=None, source_attr=None, **kwargs):
194 for obj in objs:
195 if source_attr and not source_attr.endswith("+"):
196 related_name = source_attr % {
197 "class": source._meta.model_name,
198 "app_label": source._meta.app_label,
199 }
200 self.add_edge(getattr(obj, related_name), obj)
201 else:
202 self.add_edge(None, obj)
203 self.model_objs[obj._meta.model].add(obj)
204 try:
205 return super().collect(objs, source_attr=source_attr, **kwargs)
206 except models.ProtectedError as e:
207 self.protected.update(e.protected_objects)
208 except models.RestrictedError as e:
209 self.protected.update(e.restricted_objects)

211 def related_objects(self, related_model, related_fields, objs):
212 qs = super().related_objects(related_model, related_fields, objs)
213 return qs.select_related(
214 *[related_field.name for related_field in related_fields]
215 )

217 def _nested(self, obj, seen, format_callback):
218 if obj in seen:
219 return []
220 seen.add(obj)
221 children = []
222 for child in self.edges.get(obj, ()):
223 children.extend(self._nested(child, seen, format_callback))
224 if format_callback:
225 ret = [format_callback(obj)]
226 else:
227 ret = [obj]
228 if children:
229 ret.append(children)
230 return ret

232 def nested(self, format_callback=None):
233 """
234 Return the graph as a nested list.
235 """
236 seen = set()
237 roots = []
238 for root in self.edges.get(None, ()):
239 roots.extend(self._nested(root, seen, format_callback))
240 return roots

322def _get_non_gfk_field(opts, name):
323 """
324 For historical reasons, the admin app relies on GenericForeignKeys as being
325 "not found" by get_field(). This could likely be cleaned up.
326
327 Reverse relations should also be excluded as these aren't attributes of the
328 model (rather something like `foo_set`).
329 """
330 field = opts.get_field(name)
331 if (
332 field.is_relation
333 and
334 # Generic foreign keys OR reverse relations
335 ((field.many_to_one and not field.related_model) or field.one_to_many)
336 ):
337 raise FieldDoesNotExist()
338
339 # Avoid coercing <FK>_id fields to FK
340 if (
341 field.is_relation
342 and not field.many_to_many
343 and hasattr(field, "attname")
344 and field.attname == name
345 ):
346 raise FieldIsAForeignKeyColumnName()
347
348 return field

498def reverse_field_path(model, path):
499 """Create a reversed field path.
500
501 E.g. Given (Order, "user__groups"),
502 return (Group, "user__order").
503
504 Final field must be a related model, not a data field.
505 """
506 reversed_path = []
507 parent = model
508 pieces = path.split(LOOKUP_SEP)
509 for piece in pieces:
510 field = parent._meta.get_field(piece)
511 # skip trailing data field if extant:
512 if len(reversed_path) == len(pieces) - 1: # final iteration
513 try:
514 get_model_from_relation(field)
515 except NotRelationField:
516 break
517
518 # Field should point to another model
519 if field.is_relation and not (field.auto_created and not field.concrete):
520 related_name = field.related_query_name()
521 parent = field.remote_field.model
522 else:
523 related_name = field.field.name
524 parent = field.related_model
525 reversed_path.insert(0, related_name)
526 return (parent, LOOKUP_SEP.join(reversed_path))

549def construct_change_message(form, formsets, add):
550 """
551 Construct a JSON structure describing changes from a changed object.
552 Translations are deactivated so that strings are stored untranslated.
553 Translation happens later on LogEntry access.
554 """
555 # Evaluating `form.changed_data` prior to disabling translations is required
556 # to avoid fields affected by localization from being included incorrectly,
557 # e.g. where date formats differ such as MM/DD/YYYY vs DD/MM/YYYY.
558 changed_data = form.changed_data
559 with translation_override(None):
560 # Deactivate translations while fetching verbose_name for form
561 # field labels and using `field_name`, if verbose_name is not provided.
562 # Translations will happen later on LogEntry access.
563 changed_field_labels = _get_changed_field_labels_from_form(form, changed_data)
564
565 change_message = []
566 if add:
567 change_message.append({"added": {}})
568 elif form.changed_data:
569 change_message.append({"changed": {"fields": changed_field_labels}})
570 if formsets:
571 with translation_override(None):
572 for formset in formsets:
573 for added_object in formset.new_objects:
574 change_message.append(
575 {
576 "added": {
577 "name": str(added_object._meta.verbose_name),
578 "object": str(added_object),
579 }
580 }
581 )
582 for changed_object, changed_fields in formset.changed_objects:
583 change_message.append(
584 {
585 "changed": {
586 "name": str(changed_object._meta.verbose_name),
587 "object": str(changed_object),
588 "fields": _get_changed_field_labels_from_form(
589 formset.forms[0], changed_fields
590 ),
591 }
592 }
593 )
594 for deleted_object in formset.deleted_objects:
595 change_message.append(
596 {
597 "deleted": {
598 "name": str(deleted_object._meta.verbose_name),
599 "object": str(deleted_object),
600 }
601 }
602 )
603 return change_message
