116 def check(self, against, using=DEFAULT_DB_ALIAS):
117 """
118 Do a database query to check if the expressions of the Q instance
119 matches against the expressions.
120 """
121 # Avoid circular imports.
122 from django.db.models import BooleanField, Value
123 from django.db.models.functions import Coalesce
124 from django.db.models.sql import Query
125 from django.db.models.sql.constants import SINGLE
126
127 query = Query(None)
128 for name, value in against.items():
129 if not hasattr(value, "resolve_expression"):
130 value = Value(value)
131 query.add_annotation(value, name, select=False)
132 query.add_annotation(Value(1), "_check")
133 # This will raise a FieldError if a field is missing in "against".
134 if connections[using].features.supports_comparing_boolean_expr:
135 query.add_q(Q(Coalesce(self, True, output_field=BooleanField())))
136 else:
137 query.add_q(self)
138 compiler = query.get_compiler(using=using)
139 try:
140 return compiler.execute_sql(SINGLE) is not None
141 except DatabaseError as e:
142 logger.warning("Got a database error calling check() on %r: %s", self, e)
143 return True

157 @cached_property
158 def identity(self):
159 path, args, kwargs = self.deconstruct()
160 identity = [path, *kwargs.items()]
161 for child in args:
162 if isinstance(child, tuple):

164 value = make_hashable(value)
165 identity.append((arg, value))
166 else:

192class DeferredAttribute:
193 """
194 A wrapper for a deferred-loading field. When the value is read from this
195 object the first time, the query is executed.
196 """
197
198 def __init__(self, field):
199 self.field = field
200
201 def __get__(self, instance, cls=None):
202 """
203 Retrieve and caches the value from the datastore on the first lookup.
204 Return the cached value.
205 """
206 if instance is None:
207 return self
208 data = instance.__dict__
209 field_name = self.field.attname
210 if field_name not in data:
211 # Let's see if the field is part of the parent chain. If so we
212 # might be able to reuse the already loaded value. Refs #18343.
213 val = self._check_parent_chain(instance)
214 if val is None:
215 if instance.pk is None and self.field.generated:
216 raise AttributeError(
217 "Cannot read a generated field from an unsaved model."
218 )
219 instance.refresh_from_db(fields=[field_name])
220 else:
221 data[field_name] = val
222 return data[field_name]
223
224 def _check_parent_chain(self, instance):
225 """
226 Check if the field value can be fetched from a parent field already
227 loaded in the instance. This can be done if the to-be fetched
228 field is a primary key field.
229 """
230 opts = instance._meta
231 link_field = opts.get_ancestor_link(self.field.model)
232 if self.field.primary_key and self.field != link_field:
233 return getattr(instance, link_field.attname)
234 return None

356def select_related_descend(field, restricted, requested, select_mask):
357 """
358 Return whether `field` should be used to descend deeper for
359 `select_related()` purposes.
360
361 Arguments:
362 * `field` - the field to be checked. Can be either a `Field` or
363 `ForeignObjectRel` instance.
364 * `restricted` - a boolean field, indicating if the field list has been
365 manually restricted using a select_related() clause.
366 * `requested` - the select_related() dictionary.
367 * `select_mask` - the dictionary of selected fields.
368 """
369 # Only relationships can be descended.
370 if not field.remote_field:
371 return False
372 # Forward MTI parent links should not be explicitly descended as they are
373 # always JOIN'ed against (unless excluded by `select_mask`).
374 if getattr(field.remote_field, "parent_link", False):
375 return False
376 # When `select_related()` is used without a `*requested` mask all
377 # relationships are descended unless they are nullable.
378 if not restricted:
379 return not field.null
380 # When `select_related(*requested)` is used only fields that are part of
381 # `requested` should be descended.
382 if field.name not in requested:
383 return False
384 # Prevent invalid usages of `select_related()` and `only()`/`defer()` such
385 # as `select_related("a").only("b")` and `select_related("a").defer("a")`.
386 if select_mask and field not in select_mask:
387 raise FieldError(
388 f"Field {field.model._meta.object_name}.{field.name} cannot be both "
389 "deferred and traversed using select_related at the same time."
390 )
391 return True

394def refs_expression(lookup_parts, annotations):
395 """
396 Check if the lookup_parts contains references to the given annotations set.
397 Because the LOOKUP_SEP is contained in the default annotation names, check
398 each prefix of the lookup_parts for a match.
399 """
400 for n in range(1, len(lookup_parts) + 1):
401 level_n_lookup = LOOKUP_SEP.join(lookup_parts[0:n])
402 if annotations.get(level_n_lookup):
403 return level_n_lookup, lookup_parts[n:]
404 return None, ()

407def check_rel_lookup_compatibility(model, target_opts, field):
408 """
409 Check that self.model is compatible with target_opts. Compatibility
410 is OK if:
411 1) model and opts match (where proxy inheritance is removed)
412 2) model is parent of opts' model or the other way around
413 """
414
415 def check(opts):
416 return (
417 model._meta.concrete_model == opts.concrete_model
418 or opts.concrete_model in model._meta.all_parents
419 or model in opts.all_parents
420 )
421
422 # If the field is a primary key, then doing a query against the field's
423 # model is ok, too. Consider the case:
424 # class Restaurant(models.Model):
425 # place = OneToOneField(Place, primary_key=True):
426 # Restaurant.objects.filter(pk__in=Restaurant.objects.all()).
427 # If we didn't have the primary key check, then pk__in (== place__in) would
428 # give Place's opts as the target opts, but Restaurant isn't compatible
429 # with that. This logic applies only to primary keys, as when doing __in=qs,
430 # we are going to turn this into __in=qs.values('pk') later on.
431 return check(target_opts) or (
432 getattr(field, "primary_key", False) and check(field.model._meta)
433 )

439 def __init__(self, relation_name, *, condition=Q()):
