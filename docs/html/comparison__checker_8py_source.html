20def _is_one_arg_pos_call(call: nodes.NodeNG) -> bool:
21 """Is this a call with exactly 1 positional argument ?"""
22 return isinstance(call, nodes.Call) and len(call.args) == 1 and not call.keywords

84 def _check_singleton_comparison(
85 self,
86 left_value: nodes.NodeNG,
87 right_value: nodes.NodeNG,
88 root_node: nodes.Compare,
89 checking_for_absence: bool = False,
90 ) -> None:
91 """Check if == or != is being used to compare a singleton value."""
92 if utils.is_singleton_const(left_value):
93 singleton, other_value = left_value.value, right_value
94 elif utils.is_singleton_const(right_value):
95 singleton, other_value = right_value.value, left_value
96 else:
97 return
98
99 singleton_comparison_example = {False: "'{} is {}'", True: "'{} is not {}'"}
100
101 # True/False singletons have a special-cased message in case the user is
102 # mistakenly using == or != to check for truthiness
103 if singleton in {True, False}:
104 suggestion_template = (
105 "{} if checking for the singleton value {}, or {} if testing for {}"
106 )
107 truthiness_example = {False: "not {}", True: "{}"}
108 truthiness_phrase = {True: "truthiness", False: "falsiness"}
109
110 # Looks for comparisons like x == True or x != False
111 checking_truthiness = singleton is not checking_for_absence
112
113 suggestion = suggestion_template.format(
114 singleton_comparison_example[checking_for_absence].format(
115 left_value.as_string(), right_value.as_string()
116 ),
117 singleton,
118 (
119 "'bool({})'"
120 if not utils.is_test_condition(root_node) and checking_truthiness
121 else "'{}'"
122 ).format(
123 truthiness_example[checking_truthiness].format(
124 other_value.as_string()
125 )
126 ),
127 truthiness_phrase[checking_truthiness],
128 )
129 else:
130 suggestion = singleton_comparison_example[checking_for_absence].format(
131 left_value.as_string(), right_value.as_string()
132 )
133 self.add_message(
134 "singleton-comparison",
135 node=root_node,
136 args=(f"'{root_node.as_string()}'", suggestion),
137 )

139 def _check_nan_comparison(
140 self,
141 left_value: nodes.NodeNG,
142 right_value: nodes.NodeNG,
143 root_node: nodes.Compare,
144 checking_for_absence: bool = False,
145 ) -> None:
146 def _is_float_nan(node: nodes.NodeNG) -> bool:
147 try:
148 if isinstance(node, nodes.Call) and len(node.args) == 1:
149 if (
150 node.args[0].value.lower() == "nan"
151 and node.inferred()[0].pytype() == "builtins.float"
152 ):
153 return True
154 return False
155 except AttributeError:
156 return False
157
158 def _is_numpy_nan(node: nodes.NodeNG) -> bool:
159 if isinstance(node, nodes.Attribute) and node.attrname == "NaN":
160 if isinstance(node.expr, nodes.Name):
161 return node.expr.name in {"numpy", "nmp", "np"}
162 return False
163
164 def _is_nan(node: nodes.NodeNG) -> bool:
165 return _is_float_nan(node) or _is_numpy_nan(node)
166
167 nan_left = _is_nan(left_value)
168 if not nan_left and not _is_nan(right_value):
169 return
170
171 absence_text = ""
172 if checking_for_absence:
173 absence_text = "not "
174 if nan_left:
175 suggestion = f"'{absence_text}math.isnan({right_value.as_string()})'"
176 else:
177 suggestion = f"'{absence_text}math.isnan({left_value.as_string()})'"
178 self.add_message(
179 "nan-comparison",
180 node=root_node,
181 args=(f"'{root_node.as_string()}'", suggestion),
182 )

184 def _check_literal_comparison(
185 self, literal: nodes.NodeNG, node: nodes.Compare
186 ) -> None:
187 """Check if we compare to a literal, which is usually what we do not want to do."""
188 is_other_literal = isinstance(literal, (nodes.List, nodes.Dict, nodes.Set))
189 is_const = False
190 if isinstance(literal, nodes.Const):
191 if isinstance(literal.value, bool) or literal.value is None:
192 # Not interested in these values.
193 return
194 is_const = isinstance(literal.value, (bytes, str, int, float))
195
196 if is_const or is_other_literal:
197 incorrect_node_str = node.as_string()
198 if "is not" in incorrect_node_str:
199 equal_or_not_equal = "!="
200 is_or_is_not = "is not"
201 else:
202 equal_or_not_equal = "=="
203 is_or_is_not = "is"
204 fixed_node_str = incorrect_node_str.replace(
205 is_or_is_not, equal_or_not_equal
206 )
207 self.add_message(
208 "literal-comparison",
209 args=(
210 incorrect_node_str,
211 equal_or_not_equal,
212 is_or_is_not,
213 fixed_node_str,
214 ),
215 node=node,
216 confidence=HIGH,
217 )

219 def _check_logical_tautology(self, node: nodes.Compare) -> None:
220 """Check if identifier is compared against itself.
221
222 :param node: Compare node
223 :Example:
224 val = 786
225 if val == val: # [comparison-with-itself]
226 pass
227 """
228 left_operand = node.left
229 right_operand = node.ops[0][1]
230 operator = node.ops[0][0]
231 if isinstance(left_operand, nodes.Const) and isinstance(
232 right_operand, nodes.Const
233 ):
234 left_operand = left_operand.value
235 right_operand = right_operand.value
236 elif isinstance(left_operand, nodes.Name) and isinstance(
237 right_operand, nodes.Name
238 ):
239 left_operand = left_operand.name
240 right_operand = right_operand.name
241
242 if left_operand == right_operand:
243 suggestion = f"{left_operand} {operator} {right_operand}"
244 self.add_message("comparison-with-itself", node=node, args=(suggestion,))

246 def _check_constants_comparison(self, node: nodes.Compare) -> None:
247 """When two constants are being compared it is always a logical tautology."""
248 left_operand = node.left
249 if not isinstance(left_operand, nodes.Const):
250 return
251
252 right_operand = node.ops[0][1]
253 if not isinstance(right_operand, nodes.Const):
254 return
255
256 operator = node.ops[0][0]
257 self.add_message(
258 "comparison-of-constants",
259 node=node,
260 args=(left_operand.value, operator, right_operand.value),
261 confidence=HIGH,

264 def _check_callable_comparison(self, node: nodes.Compare) -> None:
265 operator = node.ops[0][0]
266 if operator not in COMPARISON_OPERATORS:
267 return
268
269 bare_callables = (nodes.FunctionDef, astroid.BoundMethod)
270 left_operand, right_operand = node.left, node.ops[0][1]
271 # this message should be emitted only when there is comparison of bare callable
272 # with non bare callable.
273 number_of_bare_callables = 0
274 for operand in left_operand, right_operand:
275 inferred = utils.safe_infer(operand)
276 # Ignore callables that raise, as well as typing constants
277 # implemented as functions (that raise via their decorator)
278 if (
279 isinstance(inferred, bare_callables)
280 and "typing._SpecialForm" not in inferred.decoratornames()
281 and not any(isinstance(x, nodes.Raise) for x in inferred.body)
282 ):
283 number_of_bare_callables += 1
284 if number_of_bare_callables == 1:
285 self.add_message("comparison-with-callable", node=node)

296 def visit_compare(self, node: nodes.Compare) -> None:
297 self._check_callable_comparison(node)
298 self._check_logical_tautology(node)
299 self._check_unidiomatic_typecheck(node)
300 self._check_constants_comparison(node)
301 # NOTE: this checker only works with binary comparisons like 'x == 42'
302 # but not 'x == y == 42'
303 if len(node.ops) != 1:
304 return
305
306 left = node.left
307 operator, right = node.ops[0]
308
309 if operator in {"==", "!="}:
310 self._check_singleton_comparison(
311 left, right, node, checking_for_absence=operator == "!="
312 )
313
314 if operator in {"==", "!=", "is", "is not"}:
315 self._check_nan_comparison(
316 left, right, node, checking_for_absence=operator in {"!=", "is not"}
317 )
318 if operator in {"is", "is not"}:
319 self._check_literal_comparison(right, node)

321 def _check_unidiomatic_typecheck(self, node: nodes.Compare) -> None:
322 operator, right = node.ops[0]

324 left = node.left

329 self,

331 left: nodes.NodeNG,
332 operator: str,
333 right: nodes.NodeNG,
334 ) -> None:
335 """Check for expressions like type(x) == Y."""
336 left_func = utils.safe_infer(left.func)
337 if not (
338 isinstance(left_func, nodes.ClassDef) and left_func.qname() == TYPE_QNAME
339 ):

341

343 right_func = utils.safe_infer(right.func)

346 and right_func.qname() == TYPE_QNAME

348 # type(x) == type(a)
349 right_arg = utils.safe_infer(right.args[0])
350 if not isinstance(right_arg, LITERAL_NODE_TYPES):
351 # not e.g. type(x) == type([])
352 return
