1"""
2"Rel objects" for related fields.
3
4"Rel objects" (for lack of a better name) carry information about the relation
5modeled by a related field and provide some utility functions. They're stored
6in the ``remote_field`` attribute of the field.
7
8They also act as reverse fields for the purposes of the Meta API because
9they're the closest concept currently available.

176 self,

182 """

223 def get_accessor_name(self, model=None):
224 # This method encapsulates the logic that decides what name to give an
225 # accessor descriptor that retrieves related many-to-one or
226 # many-to-many objects. It uses the lowercased object_name + "_set",
227 # but this can be overridden with the "related_name" option. Due to
228 # backwards compatibility ModelForms need to be able to provide an
229 # alternate model. See BaseInlineFormSet.get_default_prefix().
230 opts = model._meta if model else self.related_model._meta
231 model = model or self.related_model
232 if self.multiple:
233 # If this is a symmetrical m2m relation on self, there is no
234 # reverse accessor.
235 if self.symmetrical and model == self.model:
236 return None
237 if self.related_name:
238 return self.related_name
239 return opts.model_name + ("_set" if self.multiple else "")

261 """
262 Used by the ForeignKey field to store information about the relation.

264 ``_meta.get_fields()`` returns this class to provide access to the field
265 flags for the reverse relation.

268 fields we get the funny situation where
269 ``ManyToOneRel.many_to_one == False`` and

402 def get_related_field(self):
403 """
404 Return the field in the 'to' object to which this relationship is tied.
405 Provided for symmetry with ManyToOneRel.
406 """
407 opts = self.through._meta
408 if self.through_fields:
409 field = opts.get_field(self.through_fields[0])
410 else:
411 for field in opts.fields:
412 rel = getattr(field, "remote_field", None)
413 if rel and rel.model == self.model:
414 break
415 return field.foreign_related_fields[0]
