21def target_info_from_filename(filename: str) -> tuple[str, str, str]:
22 """Transforms /some/path/foo.png into ('/some/path', 'foo.png', 'png')."""
23 basename = os.path.basename(filename)
24 storedir = os.path.dirname(os.path.abspath(filename))
25 target = os.path.splitext(filename)[-1][1:]
26 return storedir, basename, target

65 def get_source(self) -> str:
66 """Returns self._source."""
67 if self._source is None:
68 self.emit("}\n")

75 def generate(
76 self, outputfile: str | None = None, mapfile: str | None = None
77 ) -> str:
78 """Generates a graph file.
79
80 :param str outputfile: filename and path [defaults to graphname.png]
81 :param str mapfile: filename and path
82
83 :rtype: str
84 :return: a path to the generated file
85 :raises RuntimeError: if the executable for rendering was not found
86 """
87 # pylint: disable=duplicate-code
88 graphviz_extensions = ("dot", "gv")
89 name = self.graphname
90 if outputfile is None:
91 target = "png"
92 pdot, dot_sourcepath = tempfile.mkstemp(".gv", name)
93 ppng, outputfile = tempfile.mkstemp(".png", name)
94 os.close(pdot)
95 os.close(ppng)
96 else:
97 _, _, target = target_info_from_filename(outputfile)
98 if not target:
99 target = "png"
100 outputfile = outputfile + "." + target
101 if target not in graphviz_extensions:
102 pdot, dot_sourcepath = tempfile.mkstemp(".gv", name)
103 os.close(pdot)
104 else:
105 dot_sourcepath = outputfile
106 with codecs.open(dot_sourcepath, "w", encoding="utf8") as file:
107 file.write(self.source)
108 if target not in graphviz_extensions:
109 if shutil.which(self.renderer) is None:
110 raise RuntimeError(
111 f"Cannot generate `{outputfile}` because '{self.renderer}' "
112 "executable not found. Install graphviz, or specify a `.gv` "
113 "outputfile to produce the DOT source code."
114 )
115 if mapfile:
116 subprocess.run(
117 [
118 self.renderer,
119 "-Tcmapx",
120 "-o",
121 mapfile,
122 "-T",
123 target,
124 dot_sourcepath,
125 "-o",
126 outputfile,
127 ],
128 check=True,
129 )
130 else:
131 subprocess.run(
132 [self.renderer, "-T", target, dot_sourcepath, "-o", outputfile],
133 check=True,
134 )
135 os.unlink(dot_sourcepath)
136 return outputfile

165def get_cycles(
166 graph_dict: dict[str, set[str]], vertices: list[str] | None = None
167) -> Sequence[list[str]]:
168 """Return a list of detected cycles based on an ordered graph (i.e. keys are
169 vertices and values are lists of destination vertices representing edges).
170 """
171 if not graph_dict:
172 return ()
173 result: list[list[str]] = []
174 if vertices is None:
175 vertices = list(graph_dict.keys())
176 for vertice in vertices:
177 _get_cycles(graph_dict, [], set(), result, vertice)
178 return result

181def _get_cycles(
182 graph_dict: dict[str, set[str]],
183 path: list[str],
184 visited: set[str],
185 result: list[list[str]],
186 vertice: str,
187) -> None:
188 """Recursive function doing the real work for get_cycles."""
189 if vertice in path:
190 cycle = [vertice]
191 for node in path[::-1]:
192 if node == vertice:
193 break
194 cycle.insert(0, node)
195 # make a canonical representation
196 start_from = min(cycle)
197 index = cycle.index(start_from)
198 cycle = cycle[index:] + cycle[0:index]
199 # append it to result if not already in
200 if cycle not in result:
201 result.append(cycle)
202 return
203 path.append(vertice)
204 try:
205 for node in graph_dict[vertice]:
206 # don't check already visited nodes again
207 if node not in visited:
208 _get_cycles(graph_dict, path, visited, result, node)
209 visited.add(node)
210 except KeyError:
211 pass
212 path.pop()
