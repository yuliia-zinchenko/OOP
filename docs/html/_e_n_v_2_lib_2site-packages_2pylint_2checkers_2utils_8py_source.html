298def defnode_in_scope(
299 var_node: nodes.NodeNG,
300 varname: str,
301 scope: nodes.NodeNG,
302) -> nodes.NodeNG | None:
303 if isinstance(scope, nodes.If):
304 for node in scope.body:
305 if isinstance(node, nodes.Nonlocal) and varname in node.names:
306 return node
307 if isinstance(node, nodes.Assign):
308 for target in node.targets:
309 if isinstance(target, nodes.AssignName) and target.name == varname:
310 return target
311 elif isinstance(scope, (COMP_NODE_TYPES, nodes.For)):
312 for ass_node in scope.nodes_of_class(nodes.AssignName):
313 if ass_node.name == varname:
314 return ass_node
315 elif isinstance(scope, nodes.With):
316 for expr, ids in scope.items:
317 if expr.parent_of(var_node):
318 break
319 if ids and isinstance(ids, nodes.AssignName) and ids.name == varname:
320 return ids
321 elif isinstance(scope, (nodes.Lambda, nodes.FunctionDef)):
322 if scope.args.is_argument(varname):
323 # If the name is found inside a default value
324 # of a function, then let the search continue
325 # in the parent's tree.
326 if scope.args.parent_of(var_node):
327 try:
328 scope.args.default_value(varname)
329 scope = scope.parent
330 defnode = defnode_in_scope(var_node, varname, scope)
331 except astroid.NoDefault:
332 pass
333 else:
334 return defnode
335 return scope
336 if getattr(scope, "name", None) == varname:
337 return scope
338 elif isinstance(scope, nodes.ExceptHandler):
339 if isinstance(scope.name, nodes.AssignName):
340 ass_node = scope.name
341 if ass_node.name == varname:
342 return ass_node
343 return None

346def is_defined_before(var_node: nodes.Name) -> bool:
347 """Check if the given variable node is defined before.
348
349 Verify that the variable node is defined by a parent node
350 (e.g. if or with) earlier than `var_node`, or is defined by a
351 (list, set, dict, or generator comprehension, lambda)
352 or in a previous sibling node on the same line
353 (statement_defining ; statement_using).
354 """
355 varname = var_node.name
356 for parent in var_node.node_ancestors():
357 defnode = defnode_in_scope(var_node, varname, parent)
358 if defnode is None:
359 continue
360 defnode_scope = defnode.scope()
361 if isinstance(
362 defnode_scope, (*COMP_NODE_TYPES, nodes.Lambda, nodes.FunctionDef)
363 ):
364 # Avoid the case where var_node_scope is a nested function
365 if isinstance(defnode_scope, nodes.FunctionDef):
366 var_node_scope = var_node.scope()
367 if var_node_scope is not defnode_scope and isinstance(
368 var_node_scope, nodes.FunctionDef
369 ):
370 return False
371 return True
372 if defnode.lineno < var_node.lineno:
373 return True
374 # `defnode` and `var_node` on the same line
375 for defnode_anc in defnode.node_ancestors():
376 if defnode_anc.lineno != var_node.lineno:
377 continue
378 if isinstance(
379 defnode_anc,
380 (
381 nodes.For,
382 nodes.While,
383 nodes.With,
384 nodes.Try,
385 nodes.ExceptHandler,
386 ),
387 ):
388 return True
389 # possibly multiple statements on the same line using semicolon separator
390 stmt = var_node.statement()
391 _node = stmt.previous_sibling()
392 lineno = stmt.fromlineno
393 while _node and _node.fromlineno == lineno:
394 for assign_node in _node.nodes_of_class(nodes.AssignName):
395 if assign_node.name == varname:
396 return True
397 for imp_node in _node.nodes_of_class((nodes.ImportFrom, nodes.Import)):
398 if varname in [name[1] or name[0] for name in imp_node.names]:
399 return True
400 _node = _node.previous_sibling()
401 return False

511def parse_format_string(
512 format_string: str,
513) -> tuple[set[str], int, dict[str, str], list[str]]:
514 """Parses a format string, returning a tuple (keys, num_args).
515
516 Where 'keys' is the set of mapping keys in the format string, and 'num_args' is the number
517 of arguments required by the format string. Raises IncompleteFormatString or
518 UnsupportedFormatCharacter if a parse error occurs.
519 """
520 keys = set()
521 key_types = {}
522 pos_types = []
523 num_args = 0
524
525 def next_char(i: int) -> tuple[int, str]:
526 i += 1
527 if i == len(format_string):
528 raise IncompleteFormatString
529 return (i, format_string[i])
530
531 i = 0
532 while i < len(format_string):
533 char = format_string[i]
534 if char == "%":
535 i, char = next_char(i)
536 # Parse the mapping key (optional).
537 key = None
538 if char == "(":
539 depth = 1
540 i, char = next_char(i)
541 key_start = i
542 while depth != 0:
543 if char == "(":
544 depth += 1
545 elif char == ")":
546 depth -= 1
547 i, char = next_char(i)
548 key_end = i - 1
549 key = format_string[key_start:key_end]
550
551 # Parse the conversion flags (optional).
552 while char in "#0- +":
553 i, char = next_char(i)
554 # Parse the minimum field width (optional).
555 if char == "*":
556 num_args += 1
557 i, char = next_char(i)
558 else:
559 while char in string.digits:
560 i, char = next_char(i)
561 # Parse the precision (optional).
562 if char == ".":
563 i, char = next_char(i)
564 if char == "*":
565 num_args += 1
566 i, char = next_char(i)
567 else:
568 while char in string.digits:
569 i, char = next_char(i)
570 # Parse the length modifier (optional).
571 if char in "hlL":
572 i, char = next_char(i)
573 # Parse the conversion type (mandatory).
574 flags = "diouxXeEfFgGcrs%a"
575 if char not in flags:
576 raise UnsupportedFormatCharacter(i)
577 if key:
578 keys.add(key)
579 key_types[key] = char
580 elif char != "%":
581 num_args += 1
582 pos_types.append(char)
583 i += 1
584 return keys, num_args, key_types, pos_types

596def collect_string_fields(format_string: str) -> Iterable[str | None]:
597 """Given a format string, return an iterator
598 of all the valid format fields.
599
600 It handles nested fields as well.
601 """
602 formatter = string.Formatter()
603 # pylint: disable = too-many-try-statements
604 try:
605 parseiterator = formatter.parse(format_string)
606 for result in parseiterator:
607 if all(item is None for item in result[1:]):
608 # not a replacement format
609 continue
610 name = result[1]
611 nested = result[2]
612 yield name
613 if nested:
614 yield from collect_string_fields(nested)
615 except ValueError as exc:
616 # Probably the format string is invalid.
617 if exc.args[0].startswith("cannot switch from manual"):
618 # On Jython, parsing a string with both manual
619 # and automatic positions will fail with a ValueError,
620 # while on CPython it will simply return the fields,
621 # the validation being done in the interpreter (?).
622 # We're just returning two mixed fields in order
623 # to trigger the format-combined-specification check.
624 yield ""
625 yield "1"
626 return
627 raise IncompleteFormatString(format_string) from exc

630def parse_format_method_string(
631 format_string: str,
632) -> tuple[list[tuple[str, list[tuple[bool, str]]]], int, int]:
633 """Parses a PEP 3101 format string, returning a tuple of
634 (keyword_arguments, implicit_pos_args_cnt, explicit_pos_args).
635
636 keyword_arguments is the set of mapping keys in the format string, implicit_pos_args_cnt
637 is the number of arguments required by the format string and
638 explicit_pos_args is the number of arguments passed with the position.
639 """
640 keyword_arguments = []
641 implicit_pos_args_cnt = 0
642 explicit_pos_args = set()
643 for name in collect_string_fields(format_string):
644 if name and str(name).isdigit():
645 explicit_pos_args.add(str(name))
646 elif name:
647 keyname, fielditerator = split_format_field_names(name)
648 if isinstance(keyname, numbers.Number):
649 explicit_pos_args.add(str(keyname))
650 try:
651 keyword_arguments.append((keyname, list(fielditerator)))
652 except ValueError as e:
653 raise IncompleteFormatString() from e
654 else:
655 implicit_pos_args_cnt += 1
656 return keyword_arguments, implicit_pos_args_cnt, len(explicit_pos_args)

893 with an unsupported Python version; the decorator cannot be inferred when

896 decorators = []
897 for decorator in getattr(node, "nodes", []):

926 is_import = ("typing" in import_names) and getattr(
927 decorator, "attrname", None

932 if inferred is None or isinstance(inferred, util.UninferableBase):

1343@lru_cache(maxsize=1024)
1344def safe_infer(
1345 node: nodes.NodeNG,
1346 context: InferenceContext | None = None,
1347 *,
1348 compare_constants: bool = False,
1349 compare_constructors: bool = False,
1350) -> InferenceResult | None:
1351 """Return the inferred value for the given node.
1352
1353 Return None if inference failed or if there is some ambiguity (more than
1354 one node has been inferred of different types).
1355
1356 If compare_constants is True and if multiple constants are inferred,
1357 unequal inferred values are also considered ambiguous and return None.
1358
1359 If compare_constructors is True and if multiple classes are inferred,
1360 constructors with different signatures are held ambiguous and return None.
1361 """
1362 inferred_types: set[str | None] = set()
1363 try:
1364 infer_gen = node.infer(context=context)
1365 value = next(infer_gen)
1366 except astroid.InferenceError:
1367 return None
1368 except Exception as e: # pragma: no cover
1369 raise AstroidError from e
1370
1371 if not isinstance(value, util.UninferableBase):
1372 inferred_types.add(_get_python_type_of_node(value))
1373
1374 # pylint: disable = too-many-try-statements
1375 try:
1376 for inferred in infer_gen:
1377 inferred_type = _get_python_type_of_node(inferred)
1378 if inferred_type not in inferred_types:
1379 return None # If there is ambiguity on the inferred node.
1380 if (
1381 compare_constants
1382 and isinstance(inferred, nodes.Const)
1383 and isinstance(value, nodes.Const)
1384 and inferred.value != value.value
1385 ):
1386 return None
1387 if (
1388 isinstance(inferred, nodes.FunctionDef)
1389 and isinstance(value, nodes.FunctionDef)
1390 and function_arguments_are_ambiguous(inferred, value)
1391 ):
1392 return None
1393 if (
1394 compare_constructors
1395 and isinstance(inferred, nodes.ClassDef)
1396 and isinstance(value, nodes.ClassDef)
1397 and class_constructors_are_ambiguous(inferred, value)
1398 ):
1399 return None
1400 except astroid.InferenceError:
1401 return None # There is some kind of ambiguity
1402 except StopIteration:
1403 return value
1404 except Exception as e: # pragma: no cover
1405 raise AstroidError from e
1406 return value if len(inferred_types) <= 1 else None
