349class _XableAction(_AccessLinterObjectAction):
350 """Callback action for enabling or disabling a message."""
351
352 def _call(
353 self,
354 xabling_function: Callable[[str], None],
355 values: str | Sequence[Any] | None,
356 option_string: str | None,
357 ) -> None:
358 assert isinstance(values, (tuple, list))
359 for msgid in utils._check_csv(values[0]):
360 try:
361 xabling_function(msgid)
362 except (
363 exceptions.DeletedMessageError,
364 exceptions.MessageBecameExtensionError,
365 ) as e:
366 self.linter._stashed_messages[
367 (self.linter.current_name, "useless-option-value")
368 ].append((option_string, str(e)))
369 except exceptions.UnknownMessageError:
370 self.linter._stashed_messages[
371 (self.linter.current_name, "unknown-option-value")
372 ].append((option_string, msgid))
