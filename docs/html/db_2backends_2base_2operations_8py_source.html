600 def year_lookup_bounds_for_date_field(self, value, iso_year=False):
601 """
602 Return a two-elements list with the lower and upper bound to be used
603 with a BETWEEN operator to query a DateField value using a year
604 lookup.
605
606 `value` is an int, containing the looked-up year.
607 If `iso_year` is True, return bounds for ISO-8601 week-numbering years.
608 """
609 if iso_year:

618 second = self.adapt_datefield_value(second)
619 return [first, second]

730 def window_frame_rows_start_end(self, start=None, end=None):
731 """
732 Return SQL for start and end points in an OVER clause window frame.
733 """
734 if isinstance(start, int) and isinstance(end, int) and start > end:
735 raise ValueError("start cannot be greater than end.")
736 if start is not None and not isinstance(start, int):
737 raise ValueError(
738 f"start argument must be an integer, zero, or None, but got '{start}'."
739 )
740 if end is not None and not isinstance(end, int):
741 raise ValueError(
742 f"end argument must be an integer, zero, or None, but got '{end}'."
743 )
744 start_ = self.window_frame_value(start) or self.UNBOUNDED_PRECEDING
745 end_ = self.window_frame_value(end) or self.UNBOUNDED_FOLLOWING
746 return start_, end_
