27def _multi_decorate(decorators, method):
28 """
29 Decorate `method` with one or more function decorators. `decorators` can be
30 a single decorator or an iterable of decorators.
31 """
32 if hasattr(decorators, "__iter__"):
33 # Apply a list/tuple of decorators if 'decorators' is one. Decorator
34 # functions are applied so that the call order is the same as the
35 # order in which they appear in the iterable.
36 decorators = decorators[::-1]
37 else:
38 decorators = [decorators]
39
40 def _wrapper(self, *args, **kwargs):
41 # bound_method has the signature that 'decorator' expects i.e. no
42 # 'self' argument, but it's a closure over self so it can call
43 # 'func'. Also, wrap method.__get__() in a function because new
44 # attributes can't be set on bound method objects, only on functions.
45 bound_method = wraps(method)(partial(method.__get__(self, type(self))))
46 for dec in decorators:
47 bound_method = dec(bound_method)
48 return bound_method(*args, **kwargs)
49
50 # Copy any attributes that a decorator adds to the function it decorates.
51 for dec in decorators:
52 _update_method_wrapper(_wrapper, dec)
53 # Preserve any existing attributes of 'method', including the name.
54 update_wrapper(_wrapper, method)
55 return _wrapper

58def method_decorator(decorator, name=""):
59 """
60 Convert a function decorator into a method decorator
61 """
62
63 # 'obj' can be a class or a function. If 'obj' is a function at the time it
64 # is passed to _dec, it will eventually be a method of the class it is
65 # defined on. If 'obj' is a class, the 'name' is required to be the name
66 # of the method that will be decorated.
67 def _dec(obj):
68 if not isinstance(obj, type):
69 return _multi_decorate(decorator, obj)
70 if not (name and hasattr(obj, name)):
71 raise ValueError(
72 "The keyword argument `name` must be the name of a method "
73 "of the decorated class: %s. Got '%s' instead." % (obj, name)
74 )
75 method = getattr(obj, name)
76 if not callable(method):
77 raise TypeError(
78 "Cannot decorate '%s' as it isn't a callable attribute of "
79 "%s (%s)." % (name, obj, method)
80 )
81 _wrapper = _multi_decorate(decorator, method)
82 setattr(obj, name, _wrapper)
83 return obj
84
85 # Don't worry about making _dec look similar to a list/tuple as it's rather
86 # meaningless.
87 if not hasattr(decorator, "__iter__"):
88 update_wrapper(_dec, decorator)
89 # Change the name to aid debugging.
90 obj = decorator if hasattr(decorator, "__name__") else decorator.__class__
91 _dec.__name__ = "method_decorator(%s)" % obj.__name__
92 return _dec
