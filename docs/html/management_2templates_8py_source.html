100 raise CommandError("'%s' already exists" % top_dir)
101 except OSError as e:
102 raise CommandError(e)
103 else:

105 if app_or_project == "app":

118 extra_files = []
119 excluded_directories = [".git", "__pycache__"]
120 for file in options["files"]:
121 extra_files.extend(map(lambda x: x.strip(), file.split(",")))
122 if exclude := options.get("exclude"):

124 excluded_directories.append(directory.strip())
125 if self.verbosity >= 2:

135 base_subdir = "%s_template" % app_or_project
136 base_directory = "%s_directory" % app_or_project

138 camel_case_value = "".join(x for x in name.title() if x != "_")

143 base_name: name,
144 base_directory: top_dir,

147 "django_version": django.__version__,

172 dirs.remove(dirname)

176 # Ignore some files as they cause various breakages.
177 continue

181 )

195 )

234 def handle_template(self, template, subdir):
235 """
236 Determine where the app or project templates are.
237 Use django.__path__[0] as the default because the Django install
238 directory isn't known.
239 """
240 if template is None:
241 return os.path.join(django.__path__[0], "conf", subdir)
242 else:
243 template = template.removeprefix("file://")
244 expanded_template = os.path.expanduser(template)
245 expanded_template = os.path.normpath(expanded_template)
246 if os.path.isdir(expanded_template):
247 return expanded_template
248 if self.is_url(template):
249 # downloads the file and returns the path
250 absolute_path = self.download(template)
251 else:
252 absolute_path = os.path.abspath(expanded_template)
253 if os.path.exists(absolute_path):
254 return self.extract(absolute_path)
255
256 raise CommandError(
257 "couldn't handle %s template %s." % (self.app_or_project, template)

291 def download(self, url):
292 """
293 Download the given URL and return the file name.
294 """

296 def cleanup_url(url):
297 tmp = url.rstrip("/")
298 filename = tmp.split("/")[-1]
299 if url.endswith("/"):
300 display_url = tmp + "/"
301 else:
302 display_url = url
303 return filename, display_url
304
305 prefix = "django_%s_template_" % self.app_or_project
306 tempdir = tempfile.mkdtemp(prefix=prefix, suffix="_download")
307 self.paths_to_remove.append(tempdir)
308 filename, display_url = cleanup_url(url)
309
310 if self.verbosity >= 2:
311 self.stdout.write("Downloading %s" % display_url)
312
313 the_path = os.path.join(tempdir, filename)
314 opener = build_opener()
315 opener.addheaders = [("User-Agent", f"Django/{django.__version__}")]
316 try:
317 with opener.open(url) as source, open(the_path, "wb") as target:
318 headers = source.info()
319 target.write(source.read())
320 except OSError as e:
321 raise CommandError(
322 "couldn't download URL %s to %s: %s" % (url, filename, e)
323 )
324
325 used_name = the_path.split("/")[-1]
326
327 # Trying to get better name from response headers
328 content_disposition = headers["content-disposition"]
329 if content_disposition:
330 _, params = parse_header_parameters(content_disposition)
331 guessed_filename = params.get("filename") or used_name
332 else:
333 guessed_filename = used_name
334
335 # Falling back to content type guessing
336 ext = self.splitext(guessed_filename)[1]
337 content_type = headers["content-type"]
338 if not ext and content_type:
339 ext = mimetypes.guess_extension(content_type)
340 if ext:
341 guessed_filename += ext
342
343 # Move the temporary file to a filename that has better
344 # chances of being recognized by the archive utils
345 if used_name != guessed_filename:
346 guessed_path = os.path.join(tempdir, guessed_filename)
347 shutil.move(the_path, guessed_path)
348 return guessed_path
349
350 # Giving up
351 return the_path
