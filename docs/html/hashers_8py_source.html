36def verify_password(password, encoded, preferred="default"):
37 """
38 Return two booleans. The first is whether the raw password matches the
39 three part encoded digest, and the second whether to regenerate the
40 password.
41 """
42 fake_runtime = password is None or not is_password_usable(encoded)
43
44 preferred = get_hasher(preferred)
45 try:
46 hasher = identify_hasher(encoded)
47 except ValueError:
48 # encoded is gibberish or uses a hasher that's no longer installed.
49 fake_runtime = True
50
51 if fake_runtime:
52 # Run the default password hasher once to reduce the timing difference
53 # between an existing user with an unusable password and a nonexistent
54 # user or missing hasher (similar to #20760).
55 make_password(get_random_string(UNUSABLE_PASSWORD_SUFFIX_LENGTH))
56 return False, False
57
58 hasher_changed = hasher.algorithm != preferred.algorithm
59 must_update = hasher_changed or preferred.must_update(encoded)
60 is_correct = hasher.verify(password, encoded)
61
62 # If the hasher didn't change (we don't protect against enumeration if it
63 # does) and the password should get updated, try to close the timing gap
64 # between the work factor of the current encoded password and the default
65 # work factor.
66 if not is_correct and not hasher_changed and must_update:
67 hasher.harden_runtime(password, encoded)
68
69 return is_correct, must_update

94def make_password(password, salt=None, hasher="default"):
95 """
96 Turn a plain-text password into a hash for database storage
97
98 Same as encode() but generate a new random salt. If password is None then
99 return a concatenation of UNUSABLE_PASSWORD_PREFIX and a random string,
100 which disallows logins. Additional random string reduces chances of gaining
101 access to staff or superuser accounts. See ticket #20079 for more info.
102 """
103 if password is None:
104 return UNUSABLE_PASSWORD_PREFIX + get_random_string(
105 UNUSABLE_PASSWORD_SUFFIX_LENGTH
106 )
107 if not isinstance(password, (bytes, str)):
108 raise TypeError(
109 "Password must be a string or bytes, got %s." % type(password).__qualname__
110 )
111 hasher = get_hasher(hasher)
112 salt = salt or hasher.salt()
113 return hasher.encode(password, salt)

167def identify_hasher(encoded):
168 """
169 Return an instance of a loaded password hasher.
170
171 Identify hasher algorithm by examining encoded hash, and call
172 get_hasher() to return hasher. Raise ValueError if
173 algorithm cannot be identified, or if hasher is not loaded.
174 """
175 # Ancient versions of Django created plain MD5 passwords and accepted
176 # MD5 passwords with an empty salt.
177 if (len(encoded) == 32 and "$" not in encoded) or (
178 len(encoded) == 37 and encoded.startswith("md5$$")
179 ):
180 algorithm = "unsalted_md5"
181 # Ancient versions of Django accepted SHA1 passwords with an empty salt.
182 elif len(encoded) == 46 and encoded.startswith("sha1$$"):
183 algorithm = "unsalted_sha1"
184 else:
185 algorithm = encoded.split("$", 1)[0]
186 return get_hasher(algorithm)

189def mask_hash(hash, show=6, char="*"):
190 """
191 Return the given hash, with only the first ``show`` number shown. The
192 rest are masked with ``char`` for security reasons.
193 """
194 masked = hash[:show]
195 masked += char * len(hash[show:])
196 return masked

204class BasePasswordHasher:
205 """
206 Abstract base class for password hashers
207
208 When creating your own hasher, you need to override algorithm,
209 verify(), encode() and safe_summary().
210
211 PasswordHasher objects are immutable.
212 """
213
214 algorithm = None
215 library = None
216 salt_entropy = 128
217
218 def _load_library(self):
219 if self.library is not None:
220 if isinstance(self.library, (tuple, list)):
221 name, mod_path = self.library
222 else:
223 mod_path = self.library
224 try:
225 module = importlib.import_module(mod_path)
226 except ImportError as e:
227 raise ValueError(
228 "Couldn't load %r algorithm library: %s"
229 % (self.__class__.__name__, e)
230 )
231 return module
232 raise ValueError(
233 "Hasher %r doesn't specify a library attribute" % self.__class__.__name__
234 )

236 def salt(self):
237 """
238 Generate a cryptographically secure nonce salt in ASCII with an entropy
239 of at least `salt_entropy` bits.
240 """
241 # Each character in the salt provides
242 # log_2(len(alphabet)) bits of entropy.
243 char_count = math.ceil(self.salt_entropy / math.log2(len(RANDOM_STRING_CHARS)))
244 return get_random_string(char_count, allowed_chars=RANDOM_STRING_CHARS)
245
246 def verify(self, password, encoded):

248 raise NotImplementedError(
249 "subclasses of BasePasswordHasher must provide a verify() method"
250 )

295 def harden_runtime(self, password, encoded):
296 """
297 Bridge the runtime gap between the work factor supplied in `encoded`
298 and the work factor suggested by this hasher.
299
300 Taking PBKDF2 as an example, if `encoded` contains 20000 iterations and
301 `self.iterations` is 30000, this method should run password through
302 another 10000 iterations of PBKDF2. Similar approaches should exist
303 for any hasher that has a work factor. If not, this method should be
304 defined as a no-op to silence the warning.
305 """
306 warnings.warn(
307 "subclasses of BasePasswordHasher should provide a harden_runtime() method"
308 )

311class PBKDF2PasswordHasher(BasePasswordHasher):
312 """
313 Secure password hashing using the PBKDF2 algorithm (recommended)
314
315 Configured to use PBKDF2 + HMAC + SHA256.
316 The result is a 64 byte binary string. Iterations may be changed
317 safely but you must rename the algorithm if you change SHA256.
318 """
319
320 algorithm = "pbkdf2_sha256"
321 iterations = 870000
322 digest = hashlib.sha256
323
324 def encode(self, password, salt, iterations=None):
325 self._check_encode_args(password, salt)
326 iterations = iterations or self.iterations
327 hash = pbkdf2(password, salt, iterations, digest=self.digest)
328 hash = base64.b64encode(hash).decode("ascii").strip()
329 return "%s$%d$%s$%s" % (self.algorithm, iterations, salt, hash)
330
331 def decode(self, encoded):
332 algorithm, iterations, salt, hash = encoded.split("$", 3)
333 assert algorithm == self.algorithm
334 return {
335 "algorithm": algorithm,
336 "hash": hash,
337 "iterations": int(iterations),
338 "salt": salt,
339 }
340
341 def verify(self, password, encoded):
342 decoded = self.decode(encoded)
343 encoded_2 = self.encode(password, decoded["salt"], decoded["iterations"])
344 return constant_time_compare(encoded, encoded_2)
345
346 def safe_summary(self, encoded):
347 decoded = self.decode(encoded)
348 return {
349 _("algorithm"): decoded["algorithm"],
350 _("iterations"): decoded["iterations"],
351 _("salt"): mask_hash(decoded["salt"]),
352 _("hash"): mask_hash(decoded["hash"]),
353 }
354
355 def must_update(self, encoded):
356 decoded = self.decode(encoded)
357 update_salt = must_update_salt(decoded["salt"], self.salt_entropy)
358 return (decoded["iterations"] != self.iterations) or update_salt
359
360 def harden_runtime(self, password, encoded):
361 decoded = self.decode(encoded)
362 extra_iterations = self.iterations - decoded["iterations"]
363 if extra_iterations > 0:
364 self.encode(password, decoded["salt"], extra_iterations)

379class Argon2PasswordHasher(BasePasswordHasher):
380 """
381 Secure password hashing using the argon2 algorithm.
382
383 This is the winner of the Password Hashing Competition 2013-2015
384 (https://password-hashing.net). It requires the argon2-cffi library which
385 depends on native C code and might cause portability issues.
386 """
387
388 algorithm = "argon2"
389 library = "argon2"
390
391 time_cost = 2
392 memory_cost = 102400
393 parallelism = 8
394
395 def encode(self, password, salt):
396 argon2 = self._load_library()
397 params = self.params()
398 data = argon2.low_level.hash_secret(
399 password.encode(),
400 salt.encode(),
401 time_cost=params.time_cost,
402 memory_cost=params.memory_cost,
403 parallelism=params.parallelism,
404 hash_len=params.hash_len,
405 type=params.type,
406 )
407 return self.algorithm + data.decode("ascii")
408
409 def decode(self, encoded):
410 argon2 = self._load_library()
411 algorithm, rest = encoded.split("$", 1)
412 assert algorithm == self.algorithm
413 params = argon2.extract_parameters("$" + rest)
414 variety, *_, b64salt, hash = rest.split("$")
415 # Add padding.
416 b64salt += "=" * (-len(b64salt) % 4)
417 salt = base64.b64decode(b64salt).decode("latin1")
418 return {
419 "algorithm": algorithm,
420 "hash": hash,
421 "memory_cost": params.memory_cost,
422 "parallelism": params.parallelism,
423 "salt": salt,
424 "time_cost": params.time_cost,
425 "variety": variety,
426 "version": params.version,
427 "params": params,
428 }
429
430 def verify(self, password, encoded):
431 argon2 = self._load_library()
432 algorithm, rest = encoded.split("$", 1)
433 assert algorithm == self.algorithm
434 try:
435 return argon2.PasswordHasher().verify("$" + rest, password)
436 except argon2.exceptions.VerificationError:
437 return False
438
439 def safe_summary(self, encoded):
440 decoded = self.decode(encoded)
441 return {
442 _("algorithm"): decoded["algorithm"],
443 _("variety"): decoded["variety"],
444 _("version"): decoded["version"],
445 _("memory cost"): decoded["memory_cost"],
446 _("time cost"): decoded["time_cost"],
447 _("parallelism"): decoded["parallelism"],
448 _("salt"): mask_hash(decoded["salt"]),
449 _("hash"): mask_hash(decoded["hash"]),
450 }
451
452 def must_update(self, encoded):
453 decoded = self.decode(encoded)
454 current_params = decoded["params"]
455 new_params = self.params()
456 # Set salt_len to the salt_len of the current parameters because salt
457 # is explicitly passed to argon2.
458 new_params.salt_len = current_params.salt_len
459 update_salt = must_update_salt(decoded["salt"], self.salt_entropy)
460 return (current_params != new_params) or update_salt

462 def harden_runtime(self, password, encoded):
463 # The runtime for Argon2 is too complicated to implement a sensible
464 # hardening algorithm.
465 pass

468 argon2 = self._load_library()
469 # salt_len is a noop, because we provide our own salt.

481class BCryptSHA256PasswordHasher(BasePasswordHasher):
482 """
483 Secure password hashing using the bcrypt algorithm (recommended)
484
485 This is considered by many to be the most secure algorithm but you
486 must first install the bcrypt library. Please be warned that
487 this library depends on native C code and might cause portability
488 issues.
489 """
490
491 algorithm = "bcrypt_sha256"
492 digest = hashlib.sha256
493 library = ("bcrypt", "bcrypt")
494 rounds = 12
495
496 def salt(self):
497 bcrypt = self._load_library()
498 return bcrypt.gensalt(self.rounds)
499
500 def encode(self, password, salt):
501 bcrypt = self._load_library()
502 password = password.encode()
503 # Hash the password prior to using bcrypt to prevent password
504 # truncation as described in #20138.
505 if self.digest is not None:
506 # Use binascii.hexlify() because a hex encoded bytestring is str.
507 password = binascii.hexlify(self.digest(password).digest())
508
509 data = bcrypt.hashpw(password, salt)
510 return "%s$%s" % (self.algorithm, data.decode("ascii"))

514 assert algorithm == self.algorithm

518 "checksum": data[22:],
519 "salt": data[:22],
520 "work_factor": int(work_factor),
521 }

525 assert algorithm == self.algorithm
526 encoded_2 = self.encode(password, data.encode("ascii"))

542 def harden_runtime(self, password, encoded):
543 _, data = encoded.split("$", 1)
544 salt = data[:29] # Length of the salt in bcrypt.
