20def generate_interactive_config(linter: PyLinter) -> None:
21 print("Starting interactive pylint configuration generation")
22
23 format_type = utils.get_and_validate_format()
24 minimal = format_type == "toml" and utils.get_minimal_setting()
25 to_file, output_file_name = utils.get_and_validate_output_file()
26
27 if format_type == "toml":
28 config_string = linter._generate_config_file(minimal=minimal)
29 else:
30 output_stream = StringIO()
31 linter._generate_config(stream=output_stream, skipsections=("Commands",))
32 config_string = output_stream.getvalue()
33
34 if to_file:
35 with open(output_file_name, "w", encoding="utf-8") as f:
36 print(config_string, file=f)
37 print(f"Wrote configuration file to {output_file_name.resolve()}")
38 else:
39 print(config_string)
