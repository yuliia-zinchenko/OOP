173 RuntimeWarning,

224 def _savepoint_allowed(self):
225 # When 'isolation_level' is not None, sqlite3 commits before each
226 # savepoint; it's a bug. When it is None, savepoints don't make sense
227 # because autocommit is enabled. The only exception is inside 'atomic'
228 # blocks. To work around that bug, on SQLite, 'atomic' starts a
229 # transaction explicitly rather than simply disable autocommit.
230 return self.in_atomic_block

257 def check_constraints(self, table_names=None):
258 """
259 Check each table name in `table_names` for rows with invalid foreign
260 key references. This method is intended to be used in conjunction with
261 `disable_constraint_checking()` and `enable_constraint_checking()`, to
262 determine if rows with invalid references were entered while constraint
263 checks were off.
264 """
265 with self.cursor() as cursor:
266 if table_names is None:
267 violations = cursor.execute("PRAGMA foreign_key_check").fetchall()
268 else:
269 violations = chain.from_iterable(
270 cursor.execute(
271 "PRAGMA foreign_key_check(%s)" % self.ops.quote_name(table_name)
272 ).fetchall()
273 for table_name in table_names
274 )
275 # See https://www.sqlite.org/pragma.html#pragma_foreign_key_check
276 for (
277 table_name,
278 rowid,
279 referenced_table_name,
280 foreign_key_index,
281 ) in violations:
282 foreign_key = cursor.execute(
283 "PRAGMA foreign_key_list(%s)" % self.ops.quote_name(table_name)
284 ).fetchall()[foreign_key_index]
285 column_name, referenced_column_name = foreign_key[3:5]
286 primary_key_column_name = self.introspection.get_primary_key_column(
287 cursor, table_name
288 )
289 primary_key_value, bad_value = cursor.execute(
290 "SELECT %s, %s FROM %s WHERE rowid = %%s"
291 % (
292 self.ops.quote_name(primary_key_column_name),
293 self.ops.quote_name(column_name),
294 self.ops.quote_name(table_name),
295 ),
296 (rowid,),
297 ).fetchone()
298 raise IntegrityError(
299 "The row in table '%s' with primary key '%s' has an "
300 "invalid foreign key: %s.%s contains a value '%s' that "
301 "does not have a corresponding value in %s.%s."
302 % (
303 table_name,
304 primary_key_value,
305 table_name,
306 column_name,
307 bad_value,
308 referenced_table_name,
309 referenced_column_name,
310 )

335class SQLiteCursorWrapper(Database.Cursor):
336 """
337 Django uses the "format" and "pyformat" styles, but Python's sqlite3 module
338 supports neither of these styles.
339
340 This wrapper performs the following conversions:
341
342 - "format" style to "qmark" style
343 - "pyformat" style to "named" style
344
345 In both cases, if you want to use a literal "%s", you'll need to use "%%s".
346 """
347
348 def execute(self, query, params=None):
349 if params is None:
350 return super().execute(query)
351 # Extract names if params is a mapping, i.e. "pyformat" style is used.
352 param_names = list(params) if isinstance(params, Mapping) else None
353 query = self.convert_query(query, param_names=param_names)
354 return super().execute(query, params)
355
356 def executemany(self, query, param_list):
357 # Extract names if params is a mapping, i.e. "pyformat" style is used.
358 # Peek carefully as a generator can be passed instead of a list/tuple.
359 peekable, param_list = tee(iter(param_list))
360 if (params := next(peekable, None)) and isinstance(params, Mapping):
361 param_names = list(params)
362 else:
363 param_names = None
364 query = self.convert_query(query, param_names=param_names)
365 return super().executemany(query, param_list)
366
367 def convert_query(self, query, *, param_names=None):
368 if param_names is None:
369 # Convert from "format" style to "qmark" style.
370 return FORMAT_QMARK_REGEX.sub("?", query).replace("%%", "%")
371 else:
372 # Convert from "pyformat" style to "named" style.
373 return query % {name: f":{name}" for name in param_names}
