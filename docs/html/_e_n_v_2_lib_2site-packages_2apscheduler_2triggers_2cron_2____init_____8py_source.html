145 def _increment_field_value(self, dateval, fieldnum):
146 """
147 Increments the designated field and resets all less significant fields to their minimum
148 values.
149
150 :type dateval: datetime
151 :type fieldnum: int
152 :return: a tuple containing the new date, and the number of the field that was actually
153 incremented
154 :rtype: tuple
155 """
156
157 values = {}
158 i = 0
159 while i < len(self.fields):
160 field = self.fields[i]
161 if not field.REAL:
162 if i == fieldnum:
163 fieldnum -= 1
164 i -= 1
165 else:
166 i += 1
167 continue
168
169 if i < fieldnum:
170 values[field.name] = field.get_value(dateval)
171 i += 1
172 elif i > fieldnum:
173 values[field.name] = field.get_min(dateval)
174 i += 1
175 else:
176 value = field.get_value(dateval)
177 maxval = field.get_max(dateval)
178 if value == maxval:
179 fieldnum -= 1
180 i -= 1
181 else:
182 values[field.name] = value + 1
183 i += 1
184
185 difference = datetime(**values) - dateval.replace(tzinfo=None)
186 dateval = datetime.fromtimestamp(
187 dateval.timestamp() + difference.total_seconds(), self.timezone
188 )
189 return dateval, fieldnum

204 def get_next_fire_time(self, previous_fire_time, now):
205 if previous_fire_time:

207 if start_date == previous_fire_time:

210 start_date = max(now, self.start_date) if self.start_date else now

212 fieldnum = 0
213 next_date = datetime_ceil(start_date).astimezone(self.timezone)
214 while 0 <= fieldnum < len(self.fields):
215 field = self.fields[fieldnum]
216 curr_value = field.get_value(next_date)
217 next_value = field.get_next_value(next_date)

219 if next_value is None:
220 # No valid value was found

222 next_date, fieldnum - 1
223 )
