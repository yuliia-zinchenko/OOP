62def get_visibility(name: str) -> str:
63 """Return the visibility from a name: public, protected, private or special."""
64 if SPECIAL.match(name):
65 visibility = "special"
66 elif PRIVATE.match(name):
67 visibility = "private"
68 elif PROTECTED.match(name):
69 visibility = "protected"
70
71 else:
72 visibility = "public"
73 return visibility

119class LocalsVisitor:
120 """Visit a project by traversing the locals dictionary.
121
122 * visit_<class name> on entering a node, where class name is the class of
123 the node in lower case
124
125 * leave_<class name> on leaving a node, where class name is the class of
126 the node in lower case
127 """
128
129 def __init__(self) -> None:
130 self._cache: dict[type[nodes.NodeNG], _CallbackTupleT] = {}
131 self._visited: set[nodes.NodeNG] = set()
132
133 def get_callbacks(self, node: nodes.NodeNG) -> _CallbackTupleT:
134 """Get callbacks from handler for the visited node."""
135 klass = node.__class__
136 methods = self._cache.get(klass)
137 if methods is None:
138 kid = klass.__name__.lower()
139 e_method = getattr(
140 self, f"visit_{kid}", getattr(self, "visit_default", None)
141 )
142 l_method = getattr(
143 self, f"leave_{kid}", getattr(self, "leave_default", None)
144 )
145 self._cache[klass] = (e_method, l_method)
146 else:
147 e_method, l_method = methods
148 return e_method, l_method
149
150 def visit(self, node: nodes.NodeNG) -> Any:
151 """Launch the visit starting from the given node."""
152 if node in self._visited:
153 return None
154
155 self._visited.add(node)
156 methods = self.get_callbacks(node)
157 if methods[0] is not None:
158 methods[0](node)
159 if hasattr(node, "locals"): # skip Instance and other proxy
160 for local_node in node.values():
161 self.visit(local_node)
162 if methods[1] is not None:
163 return methods[1](node)
164 return None

175def get_annotation(
176 node: nodes.AssignAttr | nodes.AssignName,
177) -> nodes.Name | nodes.Subscript | None:
178 """Return the annotation for `node`."""
179 ann = None
180 if isinstance(node.parent, nodes.AnnAssign):
181 ann = node.parent.annotation
182 elif isinstance(node, nodes.AssignAttr):
183 init_method = node.parent.parent
184 try:
185 annotations = dict(zip(init_method.locals, init_method.args.annotations))
186 ann = annotations.get(node.parent.value.name)
187 except AttributeError:
188 pass
189 else:
190 return ann
191
192 try:
193 default, *_ = node.infer()
194 except astroid.InferenceError:
195 default = ""
196
197 label = get_annotation_label(ann)
198
199 if (
200 ann
201 and getattr(default, "value", "value") is None
202 and not label.startswith("Optional")
203 and (
204 not isinstance(ann, nodes.BinOp)
205 or not any(
206 isinstance(child, nodes.Const) and child.value is None
207 for child in ann.get_children()
208 )
209 )
210 ):
211 label = rf"Optional[{label}]"
212
213 if label and ann:
214 ann.name = label
215 return ann

218def infer_node(node: nodes.AssignAttr | nodes.AssignName) -> set[InferenceResult]:
219 """Return a set containing the node annotation if it exists
220 otherwise return a set of the inferred types using the NodeNG.infer method.
221 """
222 ann = get_annotation(node)
223 try:
224 if ann:
225 if isinstance(ann, nodes.Subscript) or (
226 isinstance(ann, nodes.BinOp) and ann.op == "|"
227 ):
228 return {ann}
229 return set(ann.infer())
230 return set(node.infer())
231 except astroid.InferenceError:
232 return {ann} if ann else set()

246def check_if_graphviz_supports_format(output_format: str) -> None:
247 """Check if the ``dot`` command supports the requested output format.
248
249 This is needed if image output is desired and ``dot`` is used to convert
250 from *.gv into the final output format.
251 """
252 dot_output = subprocess.run(
253 ["dot", "-T?"], capture_output=True, check=False, encoding="utf-8"
254 )
255 match = re.match(
256 pattern=r".*Use one of: (?P<formats>(\S*\s?)+)",
257 string=dot_output.stderr.strip(),
258 )
259 if not match:
260 print(
261 "Unable to determine Graphviz supported output formats. "
262 "Pyreverse will continue, but subsequent error messages "
263 "regarding the output format may come from Graphviz directly."
264 )
265 return
266 supported_formats = match.group("formats")
267 if output_format not in supported_formats.split():
268 print(
269 f"Format {output_format} is not supported by Graphviz. It supports: {supported_formats}"
270 )
271 sys.exit(32)
