43 def check_relate_argument(self, arg):
44 masks = (
45 "TOUCH|OVERLAPBDYDISJOINT|OVERLAPBDYINTERSECT|EQUAL|INSIDE|COVEREDBY|"
46 "CONTAINS|COVERS|ANYINTERACT|ON"
47 )
48 mask_regex = re.compile(r"^(%s)(\+(%s))*$" % (masks, masks), re.I)
49 if not isinstance(arg, str) or not mask_regex.match(arg):
50 raise ValueError('Invalid SDO_RELATE mask: "%s"' % arg)

140 def convert_extent(self, clob):
141 if clob:
142 # Generally, Oracle returns a polygon for the extent -- however,
143 # it can return a single point if there's only one Point in the
144 # table.
145 ext_geom = GEOSGeometry(memoryview(clob.read()))
146 gtype = str(ext_geom.geom_type)
147 if gtype == "Polygon":
148 # Construct the 4-tuple from the coordinates in the polygon.
149 shell = ext_geom.shell
150 ll, ur = shell[0][:2], shell[2][:2]
151 elif gtype == "Point":
152 ll = ext_geom.coords[:2]
153 ur = ll
154 else:
155 raise Exception(
156 "Unexpected geometry type returned for extent: %s" % gtype
157 )
158 xmin, ymin = ll
159 xmax, ymax = ur
160 return (xmin, ymin, xmax, ymax)
161 else:
162 return None

172 def get_distance(self, f, value, lookup_type):
173 """
174 Return the distance parameters given the value and the lookup type.
175 On Oracle, geometry columns with a geodetic coordinate system behave
176 implicitly like a geography column, and thus meters will be used as
177 the distance parameter on them.
178 """
179 if not value:
180 return []
181 value = value[0]
182 if isinstance(value, Distance):
183 if f.geodetic(self.connection):
184 dist_param = value.m
185 else:
186 dist_param = getattr(
187 value, Distance.unit_attname(f.units_name(self.connection))
188 )
189 else:
190 dist_param = value
191
192 # dwithin lookups on Oracle require a special string parameter
193 # that starts with "distance=".
194 if lookup_type == "dwithin":
195 dist_param = "distance=%s" % dist_param
196
197 return [dist_param]
