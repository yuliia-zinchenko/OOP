24def ordinal(value):
25 """
26 Convert an integer to its ordinal as a string. 1 is '1st', 2 is '2nd',
27 3 is '3rd', etc. Works for any non-negative integer.
28 """
29 try:
30 value = int(value)
31 except (TypeError, ValueError):
32 return value
33 if value < 0:
34 return str(value)
35 if value % 100 in (11, 12, 13):
36 # Translators: Ordinal format for 11 (11th), 12 (12th), and 13 (13th).
37 value = pgettext("ordinal 11, 12, 13", "{}th").format(value)
38 else:
39 templates = (
40 # Translators: Ordinal format when value ends with 0, e.g. 80th.
41 pgettext("ordinal 0", "{}th"),
42 # Translators: Ordinal format when value ends with 1, e.g. 81st, except 11.
43 pgettext("ordinal 1", "{}st"),
44 # Translators: Ordinal format when value ends with 2, e.g. 82nd, except 12.
45 pgettext("ordinal 2", "{}nd"),

66@register.filter(is_safe=True)
67def intcomma(value, use_l10n=True):
68 """
69 Convert an integer to a string containing commas every three digits.
70 For example, 3000 becomes '3,000' and 45000 becomes '45,000'.
71 """
72 if use_l10n:
73 try:
74 if not isinstance(value, (float, Decimal)):
75 value = int(value)
76 except (TypeError, ValueError):
77 return intcomma(value, False)
78 else:
79 return number_format(value, use_l10n=True, force_grouping=True)
80 result = str(value)
81 match = re.match(r"-?\d+", result)
82 if match:
83 prefix = match[0]
84 prefix_with_commas = re.sub(r"\d{3}", r"\g<0>,", prefix[::-1])[::-1]
85 # Remove a leading comma, if needed.
86 prefix_with_commas = re.sub(r"^(-?),", r"\1", prefix_with_commas)
87 result = prefix_with_commas + result[len(prefix) :]
88 return result

123@register.filter(is_safe=False)
124def intword(value):
125 """
126 Convert a large integer to a friendly text representation. Works best
127 for numbers over 1 million. For example, 1000000 becomes '1.0 million',
128 1200000 becomes '1.2 million' and '1200000000' becomes '1.2 billion'.
129 """
130 try:
131 value = int(value)
132 except (TypeError, ValueError):
133 return value
134
135 abs_value = abs(value)
136 if abs_value < 1000000:
137 return value
138
139 for exponent, converter in intword_converters:
140 large_number = 10**exponent
141 if abs_value < large_number * 1000:
142 new_value = value / large_number
143 rounded_value = round_away_from_one(new_value)
144 return converter(abs(rounded_value)) % {
145 "value": defaultfilters.floatformat(new_value, 1),
146 }
147 return value

177@register.filter(expects_localtime=True)
178def naturalday(value, arg=None):
179 """
180 For date values that are tomorrow, today or yesterday compared to
181 present day return representing string. Otherwise, return a string
182 formatted according to settings.DATE_FORMAT.
183 """
184 tzinfo = getattr(value, "tzinfo", None)
185 try:
186 value = date(value.year, value.month, value.day)
187 except AttributeError:
188 # Passed value wasn't a date object
189 return value
190 today = datetime.now(tzinfo).date()
191 delta = value - today
192 if delta.days == 0:
193 return _("today")
194 elif delta.days == 1:
195 return _("tomorrow")
196 elif delta.days == -1:
197 return _("yesterday")
198 return defaultfilters.date(value, arg)

288 if not isinstance(value, date): # datetime is a subclass of date
289 return value
290
291 now = datetime.now(timezone.utc if is_aware(value) else None)
292 if value < now:
293 delta = now - value
294 if delta.days != 0:
295 return cls.time_strings["past-day"] % {
296 "delta": defaultfilters.timesince(
297 value, now, time_strings=cls.past_substrings
298 ),
299 }
300 elif delta.seconds == 0:
301 return cls.time_strings["now"]
302 elif delta.seconds < 60:
303 return cls.time_strings["past-second"] % {"count": delta.seconds}
304 elif delta.seconds // 60 < 60:

307 else:
