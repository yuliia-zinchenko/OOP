17class NameAliasMixin:
18 """Implements get_real_name and get_alias."""
19
20 def get_real_name(self):
21 """Returns the real name (object name) of this identifier."""
22 # a.b
23 dot_idx, _ = self.token_next_by(m=(T.Punctuation, '.'))
24 return self._get_first_name(dot_idx, real_name=True)
25
26 def get_alias(self):
27 """Returns the alias for this identifier or ``None``."""
28
29 # "name AS alias"
30 kw_idx, kw = self.token_next_by(m=(T.Keyword, 'AS'))
31 if kw is not None:
32 return self._get_first_name(kw_idx + 1, keywords=True)
33
34 # "name alias" or "complicated column expression alias"
35 _, ws = self.token_next_by(t=T.Whitespace)
36 if len(self.tokens) > 2 and ws is not None:
37 return self._get_first_name(reverse=True)

40class Token:
41 """Base class for all other classes in this module.
42
43 It represents a single token and has two instance attributes:
44 ``value`` is the unchanged value of the token and ``ttype`` is
45 the type of the token.
46 """
47
48 __slots__ = ('value', 'ttype', 'parent', 'normalized', 'is_keyword',
49 'is_group', 'is_whitespace', 'is_newline')
50
51 def __init__(self, ttype, value):
52 value = str(value)
53 self.value = value
54 self.ttype = ttype
55 self.parent = None
56 self.is_group = False
57 self.is_keyword = ttype in T.Keyword
58 self.is_whitespace = self.ttype in T.Whitespace
59 self.is_newline = self.ttype in T.Newline
60 self.normalized = value.upper() if self.is_keyword else value
61
62 def __str__(self):
63 return self.value
64
65 # Pending tokenlist __len__ bug fix
66 # def __len__(self):
67 # return len(self.value)
68
69 def __repr__(self):
70 cls = self._get_repr_name()
71 value = self._get_repr_value()
72
73 q = '"' if value.startswith("'") and value.endswith("'") else "'"
74 return "<{cls} {q}{value}{q} at 0x{id:2X}>".format(
75 id=id(self), **locals())
76
77 def _get_repr_name(self):
78 return str(self.ttype).split('.')[-1]
79
80 def _get_repr_value(self):
81 raw = str(self)
82 if len(raw) > 7:
83 raw = raw[:6] + '...'
84 return re.sub(r'\s+', ' ', raw)

86 def flatten(self):
87 """Resolve subgroups."""
88 yield self

90 def match(self, ttype, values, regex=False):
91 """Checks whether the token matches the given arguments.
92
93 *ttype* is a token type. If this token doesn't match the given token
94 type.
95 *values* is a list of possible values for this token. The values
96 are OR'ed together so if only one of the values matches ``True``
97 is returned. Except for keyword tokens the comparison is
98 case-sensitive. For convenience it's OK to pass in a single string.
99 If *regex* is ``True`` (default is ``False``) the given values are
100 treated as regular expressions.
101 """
102 type_matched = self.ttype is ttype
103 if not type_matched or values is None:
104 return type_matched
105
106 if isinstance(values, str):
107 values = (values,)
108
109 if regex:
110 # TODO: Add test for regex with is_keyboard = false
111 flag = re.IGNORECASE if self.is_keyword else 0
112 values = (re.compile(v, flag) for v in values)
113
114 for pattern in values:
115 if pattern.search(self.normalized):
116 return True
117 return False
118
119 if self.is_keyword:
120 values = (v.upper() for v in values)
121
122 return self.normalized in values

124 def within(self, group_cls):
125 """Returns ``True`` if this token is within *group_cls*.
126
127 Use this method for example to check if an identifier is within
128 a function: ``t.within(sql.Function)``.
129 """
130 parent = self.parent
131 while parent:
132 if isinstance(parent, group_cls):
133 return True
134 parent = parent.parent
135 return False
136
137 def is_child_of(self, other):
138 """Returns ``True`` if this token is a direct child of *other*."""
139 return self.parent == other
140
141 def has_ancestor(self, other):
142 """Returns ``True`` if *other* is in this tokens ancestry."""
143 parent = self.parent
144 while parent:
145 if parent == other:
146 return True
147 parent = parent.parent
148 return False

151class TokenList(Token):

160 def __init__(self, tokens=None):
161 self.tokens = tokens or []
162 [setattr(token, 'parent', self) for token in self.tokens]
163 super().__init__(None, str(self))

182 def _pprint_tree(self, max_depth=None, depth=0, f=None, _pre=''):
183 """Pretty-print the object tree."""
184 token_count = len(self.tokens)
185 for idx, token in enumerate(self.tokens):
186 cls = token._get_repr_name()
187 value = token._get_repr_value()
188
189 last = idx == (token_count - 1)
190 pre = '`- ' if last else '|- '
191
192 q = '"' if value.startswith("'") and value.endswith("'") else "'"
193 print("{_pre}{pre}{idx} {cls} {q}{value}{q}"
194 .format(**locals()), file=f)
195
196 if token.is_group and (max_depth is None or depth < max_depth):
197 parent_pre = ' ' if last else '| '
198 token._pprint_tree(max_depth, depth + 1, f, _pre + parent_pre)

200 def get_token_at_offset(self, offset):
201 """Returns the token that is on position offset."""
202 idx = 0
203 for token in self.flatten():
204 end = idx + len(token.value)
205 if idx <= offset < end:
206 return token
207 idx = end

209 def flatten(self):
210 """Generator yielding ungrouped tokens.
211
212 This method is recursively called for all child tokens.
213 """
214 try:
215 for token in self.tokens:
216 if token.is_group:
217 yield from token.flatten()
218 else:
219 yield token
220 except RecursionError as err:
221 raise SQLParseError('Maximum recursion depth exceeded') from err

232 def _token_matching(self, funcs, start=0, end=None, reverse=False):
233 """next token that match functions"""
234 if start is None:
235 return None
236
237 if not isinstance(funcs, (list, tuple)):
238 funcs = (funcs,)
239
240 if reverse:
241 assert end is None
242 indexes = range(start - 2, -1, -1)
243 else:
244 if end is None:
245 end = len(self.tokens)
246 indexes = range(start, end)
247 for idx in indexes:
248 token = self.tokens[idx]
249 for func in funcs:
250 if func(token):
251 return idx, token
252 return None, None

254 def token_first(self, skip_ws=True, skip_cm=False):
255 """Returns the first child token.
256
257 If *skip_ws* is ``True`` (the default), whitespace
258 tokens are ignored.
259
260 if *skip_cm* is ``True`` (default: ``False``), comments are
261 ignored too.
262 """
263 # this on is inconsistent, using Comment instead of T.Comment...
264 def matcher(tk):
265 return not ((skip_ws and tk.is_whitespace)
266 or (skip_cm and imt(tk, t=T.Comment, i=Comment)))
267 return self._token_matching(matcher)[1]

269 def token_next_by(self, i=None, m=None, t=None, idx=-1, end=None):
270 idx += 1
271 return self._token_matching(lambda tk: imt(tk, i, m, t), idx, end)

273 def token_not_matching(self, funcs, idx):
274 funcs = (funcs,) if not isinstance(funcs, (list, tuple)) else funcs
275 funcs = [lambda tk: not func(tk) for func in funcs]
276 return self._token_matching(funcs, idx)

278 def token_matching(self, funcs, idx):
279 return self._token_matching(funcs, idx)[1]

282 """Returns the previous token relative to *idx*.

285 If *skip_cm* is ``True`` comments are ignored.

308 """Return list index of token."""
309 start = start if isinstance(start, int) else self.token_index(start)

312 def group_tokens(self, grp_cls, start, end, include_end=True,
313 extend=False):
314 """Replace tokens by an instance of *grp_cls*."""
315 start_idx = start
316 start = self.tokens[start_idx]
317
318 end_idx = end + include_end
319
320 # will be needed later for new group_clauses
321 # while skip_ws and tokens and tokens[-1].is_whitespace:
322 # tokens = tokens[:-1]
323
324 if extend and isinstance(start, grp_cls):
325 subtokens = self.tokens[start_idx + 1:end_idx]
326
327 grp = start
328 grp.tokens.extend(subtokens)
329 del self.tokens[start_idx + 1:end_idx]
330 grp.value = str(start)
331 else:
332 subtokens = self.tokens[start_idx:end_idx]
333 grp = grp_cls(subtokens)
334 self.tokens[start_idx:end_idx] = [grp]
335 grp.parent = self
336
337 for token in subtokens:
338 token.parent = grp
339
340 return grp

381 def get_parent_name(self):
382 """Return name of the parent object if any.
383
384 A parent object is identified by the first occurring dot.

386 dot_idx, _ = self.token_next_by(m=(T.Punctuation, '.'))

390 def _get_first_name(self, idx=None, reverse=False, keywords=False,
391 real_name=False):
392 """Returns the name of the first token with a name"""
393
394 tokens = self.tokens[idx:] if idx else self.tokens
395 tokens = reversed(tokens) if reverse else tokens
396 types = [T.Name, T.Wildcard, T.String.Symbol]
397
398 if keywords:
399 types.append(T.Keyword)
400
401 for token in tokens:
402 if token.ttype in types:
403 return remove_quotes(token.value)
404 elif isinstance(token, (Identifier, Function)):
405 return token.get_real_name() if real_name else token.get_name()

408class Statement(TokenList):
409 """Represents a SQL statement."""
410
411 def get_type(self):
412 """Returns the type of a statement.
413
414 The returned value is a string holding an upper-cased reprint of
415 the first DML or DDL keyword. If the first token in this group
416 isn't a DML or DDL keyword "UNKNOWN" is returned.
417
418 Whitespaces and comments at the beginning of the statement
419 are ignored.
420 """
421 token = self.token_first(skip_cm=True)
422 if token is None:
423 # An "empty" statement that either has not tokens at all
424 # or only whitespace tokens.
425 return 'UNKNOWN'
426
427 elif token.ttype in (T.Keyword.DML, T.Keyword.DDL):
428 return token.normalized
429
430 elif token.ttype == T.Keyword.CTE:
431 # The WITH keyword should be followed by either an Identifier or
432 # an IdentifierList containing the CTE definitions; the actual
433 # DML keyword (e.g. SELECT, INSERT) will follow next.
434 tidx = self.token_index(token)
435 while tidx is not None:
436 tidx, token = self.token_next(tidx, skip_ws=True)
437 if isinstance(token, (Identifier, IdentifierList)):
438 tidx, token = self.token_next(tidx, skip_ws=True)
439
440 if token is not None \
441 and token.ttype == T.Keyword.DML:
442 return token.normalized
443
444 # Hmm, probably invalid syntax, so return unknown.
445 return 'UNKNOWN'

573class Case(TokenList):
574 """A CASE statement with one or more WHEN and possibly an ELSE part."""
575 M_OPEN = T.Keyword, 'CASE'
576 M_CLOSE = T.Keyword, 'END'
577
578 def get_cases(self, skip_ws=False):
579 """Returns a list of 2-tuples (condition, value).
580
581 If an ELSE exists condition is None.
582 """
583 CONDITION = 1
584 VALUE = 2
585
586 ret = []
587 mode = CONDITION
588
589 for token in self.tokens:
590 # Set mode from the current statement
591 if token.match(T.Keyword, 'CASE'):
592 continue
593
594 elif skip_ws and token.ttype in T.Whitespace:
595 continue
596
597 elif token.match(T.Keyword, 'WHEN'):
598 ret.append(([], []))
599 mode = CONDITION
600
601 elif token.match(T.Keyword, 'THEN'):
602 mode = VALUE
603
604 elif token.match(T.Keyword, 'ELSE'):
605 ret.append((None, []))
606 mode = VALUE
607
608 elif token.match(T.Keyword, 'END'):
609 mode = None
610
611 # First condition without preceding WHEN
612 if mode and not ret:
613 ret.append(([], []))
614
615 # Append token depending of the current mode
616 if mode == CONDITION:
617 ret[-1][0].append(token)
618
619 elif mode == VALUE:
620 ret[-1][1].append(token)
621
622 # Return cases list
623 return ret
