23class MessageIdStore:
24 """The MessageIdStore store MessageId and make sure that there is a 1-1 relation
25 between msgid and symbol.
26 """

28 def __init__(self) -> None:
29 self.__msgid_to_symbol: dict[str, str] = {}
30 self.__symbol_to_msgid: dict[str, str] = {}
31 self.__old_names: dict[str, list[str]] = {}
32 self.__active_msgids: dict[str, list[str]] = {}

34 def __len__(self) -> int:

37 def __repr__(self) -> str:

44 def get_symbol(self, msgid: str) -> str:
45 try:
46 return self.__msgid_to_symbol[msgid.upper()]
47 except KeyError as e:
48 msg = f"'{msgid}' is not stored in the message store."

51 def get_msgid(self, symbol: str) -> str:

58 def register_message_definition(
59 self, msgid: str, symbol: str, old_names: list[tuple[str, str]]
60 ) -> None:
61 self.check_msgid_and_symbol(msgid, symbol)
62 self.add_msgid_and_symbol(msgid, symbol)
63 for old_msgid, old_symbol in old_names:
64 self.check_msgid_and_symbol(old_msgid, old_symbol)
65 self.add_legacy_msgid_and_symbol(old_msgid, old_symbol, msgid)

67 def add_msgid_and_symbol(self, msgid: str, symbol: str) -> None:
68 """Add valid message id.
69
70 There is a little duplication with add_legacy_msgid_and_symbol to avoid a function call,
71 this is called a lot at initialization.
72 """
73 self.__msgid_to_symbol[msgid] = symbol
74 self.__symbol_to_msgid[symbol] = msgid
75
76 def add_legacy_msgid_and_symbol(
77 self, msgid: str, symbol: str, new_msgid: str
78 ) -> None:
79 """Add valid legacy message id.
80
81 There is a little duplication with add_msgid_and_symbol to avoid a function call,
82 this is called a lot at initialization.
83 """
84 self.__msgid_to_symbol[msgid] = symbol
85 self.__symbol_to_msgid[symbol] = msgid
86 existing_old_names = self.__old_names.get(msgid, [])
87 existing_old_names.append(new_msgid)
88 self.__old_names[msgid] = existing_old_names

90 def check_msgid_and_symbol(self, msgid: str, symbol: str) -> None:
91 existing_msgid: str | None = self.__symbol_to_msgid.get(symbol)
92 existing_symbol: str | None = self.__msgid_to_symbol.get(msgid)
93 if existing_symbol is None and existing_msgid is None:
94 return # both symbol and msgid are usable
95 if existing_msgid is not None:
96 if existing_msgid != msgid:
97 self._raise_duplicate_msgid(symbol, msgid, existing_msgid)
98 if existing_symbol and existing_symbol != symbol:
99 # See https://github.com/python/mypy/issues/10559
100 self._raise_duplicate_symbol(msgid, symbol, existing_symbol)
101
102 @staticmethod
103 def _raise_duplicate_symbol(msgid: str, symbol: str, other_symbol: str) -> NoReturn:
104 """Raise an error when a symbol is duplicated."""
105 symbols = [symbol, other_symbol]
106 symbols.sort()
107 error_message = f"Message id '{msgid}' cannot have both "
108 error_message += f"'{symbols[0]}' and '{symbols[1]}' as symbolic name."
109 raise InvalidMessageError(error_message)
110
111 @staticmethod
112 def _raise_duplicate_msgid(symbol: str, msgid: str, other_msgid: str) -> NoReturn:
113 """Raise an error when a msgid is duplicated."""
114 msgids = [msgid, other_msgid]
115 msgids.sort()
116 error_message = (
117 f"Message symbol '{symbol}' cannot be used for "
118 f"'{msgids[0]}' and '{msgids[1]}' at the same time."
119 f" If you're creating an 'old_names' use 'old-{symbol}' as the old symbol."
120 )
121 raise InvalidMessageError(error_message)

123 def get_active_msgids(self, msgid_or_symbol: str) -> list[str]:
124 """Return msgids but the input can be a symbol.
125
126 self.__active_msgids is used to implement a primitive cache for this function.
127 """
128 try:
129 return self.__active_msgids[msgid_or_symbol]
130 except KeyError:
131 pass
132
133 # If we don't have a cached value yet we compute it
134 msgid: str | None
135 deletion_reason = None
136 moved_reason = None
137 if msgid_or_symbol[1:].isdigit():
138 # Only msgid can have a digit as second letter
139 msgid = msgid_or_symbol.upper()
140 symbol = self.__msgid_to_symbol.get(msgid)
141 if not symbol:
142 deletion_reason = is_deleted_msgid(msgid)
143 if deletion_reason is None:
144 moved_reason = is_moved_msgid(msgid)
145 else:
146 symbol = msgid_or_symbol
147 msgid = self.__symbol_to_msgid.get(msgid_or_symbol)
148 if not msgid:
149 deletion_reason = is_deleted_symbol(symbol)
150 if deletion_reason is None:
151 moved_reason = is_moved_symbol(symbol)
152 if not msgid or not symbol:
153 if deletion_reason is not None:
154 raise DeletedMessageError(msgid_or_symbol, deletion_reason)
155 if moved_reason is not None:
156 raise MessageBecameExtensionError(msgid_or_symbol, moved_reason)
157 error_msg = f"No such message id or symbol '{msgid_or_symbol}'."
158 raise UnknownMessageError(error_msg)
159 ids = self.__old_names.get(msgid, [msgid])
160 # Add to cache
161 self.__active_msgids[msgid_or_symbol] = ids
162 return ids
