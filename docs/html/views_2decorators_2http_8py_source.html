83def condition(etag_func=None, last_modified_func=None):
84 """
85 Decorator to support conditional retrieval (or change) for a view
86 function.
87
88 The parameters are callables to compute the ETag and last modified time for
89 the requested resource, respectively. The callables are passed the same
90 parameters as the view itself. The ETag function should return a string (or
91 None if the resource doesn't exist), while the last_modified function
92 should return a datetime object (or None if the resource doesn't exist).
93
94 The ETag function should return a complete ETag, including quotes (e.g.
95 '"etag"'), since that's the only way to distinguish between weak and strong
96 ETags. If an unquoted ETag is returned (e.g. 'etag'), it will be converted
97 to a strong ETag by adding quotes.
98
99 This decorator will either pass control to the wrapped view function or
100 return an HTTP 304 response (unmodified) or 412 response (precondition
101 failed), depending upon the request method. In either case, the decorator
102 will add the generated ETag and Last-Modified headers to the response if
103 the headers aren't already set and if the request's method is safe.
104 """
105
106 def decorator(func):
107 def _pre_process_request(request, *args, **kwargs):
108 # Compute values (if any) for the requested resource.
109 res_last_modified = None
110 if last_modified_func:
111 if dt := last_modified_func(request, *args, **kwargs):
112 if not timezone.is_aware(dt):
113 dt = timezone.make_aware(dt, datetime.timezone.utc)
114 res_last_modified = int(dt.timestamp())
115 # The value from etag_func() could be quoted or unquoted.
116 res_etag = etag_func(request, *args, **kwargs) if etag_func else None
117 res_etag = quote_etag(res_etag) if res_etag is not None else None
118 response = get_conditional_response(
119 request,
120 etag=res_etag,
121 last_modified=res_last_modified,
122 )
123 return response, res_etag, res_last_modified

129 if res_last_modified and not response.has_header("Last-Modified"):
130 response.headers["Last-Modified"] = http_date(res_last_modified)
131 if res_etag:
132 response.headers.setdefault("ETag", res_etag)
