19class LambdaExpressionChecker(BaseChecker):
20 """Check for unnecessary usage of lambda expressions."""

40 def visit_assign(self, node: nodes.Assign) -> None:
41 """Check if lambda expression is assigned to a variable."""
42 if isinstance(node.targets[0], nodes.AssignName) and isinstance(
43 node.value, nodes.Lambda
44 ):
45 self.add_message(
46 "unnecessary-lambda-assignment",
47 node=node.value,
48 confidence=HIGH,
49 )
50 elif isinstance(node.targets[0], nodes.Tuple) and isinstance(
51 node.value, (nodes.Tuple, nodes.List)
52 ):
53 # Iterate over tuple unpacking assignment elements and
54 # see if any lambdas are assigned to a variable.
55 # N.B. We may encounter W0632 (unbalanced-tuple-unpacking)
56 # and still need to flag the lambdas that are being assigned.
57 for lhs_elem, rhs_elem in zip_longest(
58 node.targets[0].elts, node.value.elts
59 ):
60 if lhs_elem is None or rhs_elem is None:
61 # unbalanced tuple unpacking. stop checking.
62 break
63 if isinstance(lhs_elem, nodes.AssignName) and isinstance(
64 rhs_elem, nodes.Lambda
65 ):
66 self.add_message(
67 "unnecessary-lambda-assignment",
68 node=rhs_elem,
69 confidence=HIGH,
70 )
71
72 def visit_namedexpr(self, node: nodes.NamedExpr) -> None:
73 if isinstance(node.target, nodes.AssignName) and isinstance(
74 node.value, nodes.Lambda
75 ):
76 self.add_message(
77 "unnecessary-lambda-assignment",
78 node=node.value,
79 confidence=HIGH,
80 )
81
82 def visit_call(self, node: nodes.Call) -> None:
83 """Check if lambda expression is called directly."""
84 if isinstance(node.func, nodes.Lambda):
85 self.add_message(
86 "unnecessary-direct-lambda-call",
87 node=node,
88 confidence=HIGH,
89 )
