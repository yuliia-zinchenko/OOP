163class ForNode(Node):
164 child_nodelists = ("nodelist_loop", "nodelist_empty")
165
166 def __init__(
167 self, loopvars, sequence, is_reversed, nodelist_loop, nodelist_empty=None
168 ):
169 self.loopvars = loopvars
170 self.sequence = sequence
171 self.is_reversed = is_reversed
172 self.nodelist_loop = nodelist_loop
173 if nodelist_empty is None:
174 self.nodelist_empty = NodeList()
175 else:
176 self.nodelist_empty = nodelist_empty
177
178 def __repr__(self):
179 reversed_text = " reversed" if self.is_reversed else ""
180 return "<%s: for %s in %s, tail_len: %d%s>" % (
181 self.__class__.__name__,
182 ", ".join(self.loopvars),
183 self.sequence,
184 len(self.nodelist_loop),
185 reversed_text,
186 )
187
188 def render(self, context):
189 if "forloop" in context:
190 parentloop = context["forloop"]
191 else:
192 parentloop = {}
193 with context.push():
194 values = self.sequence.resolve(context, ignore_failures=True)
195 if values is None:
196 values = []
197 if not hasattr(values, "__len__"):
198 values = list(values)
199 len_values = len(values)
200 if len_values < 1:
201 return self.nodelist_empty.render(context)
202 nodelist = []
203 if self.is_reversed:
204 values = reversed(values)
205 num_loopvars = len(self.loopvars)
206 unpack = num_loopvars > 1
207 # Create a forloop value in the context. We'll update counters on each
208 # iteration just below.
209 loop_dict = context["forloop"] = {"parentloop": parentloop}
210 for i, item in enumerate(values):
211 # Shortcuts for current loop iteration number.
212 loop_dict["counter0"] = i
213 loop_dict["counter"] = i + 1
214 # Reverse counter iteration numbers.
215 loop_dict["revcounter"] = len_values - i
216 loop_dict["revcounter0"] = len_values - i - 1
217 # Boolean values designating first and last times through loop.
218 loop_dict["first"] = i == 0
219 loop_dict["last"] = i == len_values - 1
220
221 pop_context = False
222 if unpack:
223 # If there are multiple loop variables, unpack the item into
224 # them.
225 try:
226 len_item = len(item)
227 except TypeError: # not an iterable
228 len_item = 1
229 # Check loop variable count before unpacking
230 if num_loopvars != len_item:
231 raise ValueError(
232 "Need {} values to unpack in for loop; got {}. ".format(
233 num_loopvars, len_item
234 ),
235 )
236 unpacked_vars = dict(zip(self.loopvars, item))
237 pop_context = True
238 context.update(unpacked_vars)
239 else:
240 context[self.loopvars[0]] = item
241
242 for node in self.nodelist_loop:
243 nodelist.append(node.render_annotated(context))
244
245 if pop_context:
246 # Pop the loop variables pushed on to the context to avoid
247 # the context ending up in an inconsistent state when other
248 # tags (e.g., include and with) push data to context.
249 context.pop()
250 return mark_safe("".join(nodelist))

254 child_nodelists = ("nodelist_true", "nodelist_false")
255
256 def __init__(self, nodelist_true, nodelist_false, *varlist):
257 self.nodelist_true = nodelist_true
258 self.nodelist_false = nodelist_false
259 self._varlist = varlist
260
261 def render(self, context):
262 # Init state storage
263 state_frame = self._get_context_stack_frame(context)
264 state_frame.setdefault(self)
265
266 nodelist_true_output = None
267 if self._varlist:
268 # Consider multiple parameters. This behaves like an OR evaluation
269 # of the multiple variables.
270 compare_to = [
271 var.resolve(context, ignore_failures=True) for var in self._varlist
272 ]
273 else:
274 # The "{% ifchanged %}" syntax (without any variables) compares
275 # the rendered output.
276 compare_to = nodelist_true_output = self.nodelist_true.render(context)
277
278 if compare_to != state_frame[self]:
279 state_frame[self] = compare_to
280 # render true block if not already rendered
281 return nodelist_true_output or self.nodelist_true.render(context)
282 elif self.nodelist_false:
283 return self.nodelist_false.render(context)
284 return ""

286 def _get_context_stack_frame(self, context):
287 # The Context object behaves like a stack where each template tag can
288 # create a new scope. Find the place where to store the state to detect
289 # changes.
290 if "forloop" in context:
291 # Ifchanged is bound to the local for loop.
292 # When there is a loop-in-loop, the state is bound to the inner loop,
293 # so it resets when the outer loop continues.
294 return context["forloop"]
295 else:
296 # Using ifchanged outside loops. Effectively this is a no-op
297 # because the state is associated with 'self'.
298 return context.render_context

580@register.tag
581def cycle(parser, token):
582 """
583 Cycle among the given strings each time this tag is encountered.
584
585 Within a loop, cycles among the given strings each time through
586 the loop::
587
588 {% for o in some_list %}
589 <tr class="{% cycle 'row1' 'row2' %}">
590 ...
591 </tr>
592 {% endfor %}
593
594 Outside of a loop, give the values a unique name the first time you call
595 it, then use that name each successive time through::
596
597 <tr class="{% cycle 'row1' 'row2' 'row3' as rowcolors %}">...</tr>
598 <tr class="{% cycle rowcolors %}">...</tr>
599 <tr class="{% cycle rowcolors %}">...</tr>
600
601 You can use any number of values, separated by spaces. Commas can also
602 be used to separate values; if a comma is used, the cycle values are
603 interpreted as literal strings.
604
605 The optional flag "silent" can be used to prevent the cycle declaration
606 from returning any value::
607
608 {% for o in some_list %}
609 {% cycle 'row1' 'row2' as rowcolors silent %}
610 <tr class="{{ rowcolors }}">{% include "subtemplate.html " %}</tr>
611 {% endfor %}
612 """
613 # Note: This returns the exact same node on each {% cycle name %} call;
614 # that is, the node object returned from {% cycle a b c as name %} and the
615 # one returned from {% cycle name %} are the exact same object. This
616 # shouldn't cause problems (heh), but if it does, now you know.
617 #
618 # Ugly hack warning: This stuffs the named template dict into parser so
619 # that names are only unique within each template (as opposed to using
620 # a global variable, which would make cycle names have to be unique across
621 # *all* templates.
622 #
623 # It keeps the last node in the parser to be able to reset it with
624 # {% resetcycle %}.
625
626 args = token.split_contents()
627
628 if len(args) < 2:
629 raise TemplateSyntaxError("'cycle' tag requires at least two arguments")
630
631 if len(args) == 2:
632 # {% cycle foo %} case.
633 name = args[1]
634 if not hasattr(parser, "_named_cycle_nodes"):
635 raise TemplateSyntaxError(
636 "No named cycles in template. '%s' is not defined" % name
637 )
638 if name not in parser._named_cycle_nodes:
639 raise TemplateSyntaxError("Named cycle '%s' does not exist" % name)
640 return parser._named_cycle_nodes[name]
641
642 as_form = False
643
644 if len(args) > 4:
645 # {% cycle ... as foo [silent] %} case.
646 if args[-3] == "as":
647 if args[-1] != "silent":
648 raise TemplateSyntaxError(
649 "Only 'silent' flag is allowed after cycle's name, not '%s'."
650 % args[-1]
651 )
652 as_form = True
653 silent = True
654 args = args[:-1]
655 elif args[-2] == "as":
656 as_form = True
657 silent = False
658
659 if as_form:
660 name = args[-1]
661 values = [parser.compile_filter(arg) for arg in args[1:-2]]
662 node = CycleNode(values, name, silent=silent)
663 if not hasattr(parser, "_named_cycle_nodes"):
664 parser._named_cycle_nodes = {}
665 parser._named_cycle_nodes[name] = node
666 else:
667 values = [parser.compile_filter(arg) for arg in args[1:]]
668 node = CycleNode(values)
669 parser._last_cycle_node = node
670 return node

779 Loop over each item in an array.
780

782
783 <ul>
784 {% for athlete in athlete_list %}

791

794 {% for key,value in dict.items %}

796 {% endfor %}

799 be displayed if the given array is empty or could not be found::

801 <ul>
802 {% for athlete in athlete_list %}
803 <li>{{ athlete.name }}</li>
804 {% empty %}
805 <li>Sorry, no athletes in this list.</li>
806 {% endfor %}
807 <ul>
808
809 The above is equivalent to -- but shorter, cleaner, and possibly faster
810 than -- the following::
811
812 <ul>
813 {% if athlete_list %}
