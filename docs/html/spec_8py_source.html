124 [(s, ModuleType.C_EXTENSION) for s in importlib.machinery.EXTENSION_SUFFIXES]
125 + [(s, ModuleType.PY_SOURCE) for s in importlib.machinery.SOURCE_SUFFIXES]
126 + [(s, ModuleType.PY_COMPILED) for s in importlib.machinery.BYTECODE_SUFFIXES]

129 def find_module(
130 self,
131 modname: str,
132 module_parts: Sequence[str],
133 processed: list[str],
134 submodule_path: Sequence[str] | None,
135 ) -> ModuleSpec | None:
136 # Although we should be able to use `find_spec` this doesn't work on PyPy for builtins.
137 # Therefore, we use the `builtin_module_nams` heuristic for these.
138 if submodule_path is None and modname in sys.builtin_module_names:
139 return ModuleSpec(
140 name=modname,
141 location=None,
142 type=ModuleType.C_BUILTIN,
143 )
144
145 if submodule_path is not None:
146 search_paths = list(submodule_path)
147 else:
148 search_paths = sys.path
149
150 suffixes = (".py", ".pyi", importlib.machinery.BYTECODE_SUFFIXES[0])
151 for entry in search_paths:
152 package_directory = os.path.join(entry, modname)
153 for suffix in suffixes:
154 package_file_name = "__init__" + suffix
155 file_path = os.path.join(package_directory, package_file_name)
156 if os.path.isfile(file_path):
157 return ModuleSpec(
158 name=modname,
159 location=package_directory,
160 type=ModuleType.PKG_DIRECTORY,
161 )
162 for suffix, type_ in ImportlibFinder._SUFFIXES:
163 file_name = modname + suffix
164 file_path = os.path.join(entry, file_name)
165 if os.path.isfile(file_path):
166 return ModuleSpec(name=modname, location=file_path, type=type_)
167
168 # sys.stdlib_module_names was added in Python 3.10
169 if PY310_PLUS:
170 # If the module name matches a stdlib module name, check whether this is a frozen
171 # module. Note that `find_spec` actually imports parent modules, so we want to make
172 # sure we only run this code for stuff that can be expected to be frozen. For now
173 # this is only stdlib.
174 if modname in sys.stdlib_module_names or (
175 processed and processed[0] in sys.stdlib_module_names
176 ):
177 try:
178 with warnings.catch_warnings():
179 warnings.filterwarnings("ignore", category=Warning)
180 spec = importlib.util.find_spec(".".join((*processed, modname)))
181 except ValueError:
182 spec = None
183
184 if (
185 spec
186 and spec.loader # type: ignore[comparison-overlap] # noqa: E501
187 is importlib.machinery.FrozenImporter
188 ):
189 return ModuleSpec(
190 name=modname,
191 location=getattr(spec.loader_state, "filename", None),
192 type=ModuleType.PY_FROZEN,
193 )
194 else:
195 # NOTE: This is broken code. It doesn't work on Python 3.13+ where submodules can also
196 # be frozen. However, we don't want to worry about this and we don't want to break
197 # support for older versions of Python. This is just copy-pasted from the old non
198 # working version to at least have no functional behaviour change on <=3.10.
199 # It can be removed after 3.10 is no longer supported in favour of the logic above.
200 if submodule_path is None: # pylint: disable=else-if-used
201 try:
202 with warnings.catch_warnings():
203 warnings.filterwarnings("ignore", category=UserWarning)
204 spec = importlib.util.find_spec(modname)
205 if (
206 spec
207 and spec.loader # type: ignore[comparison-overlap] # noqa: E501
208 is importlib.machinery.FrozenImporter
209 ):
210 # No need for BuiltinImporter; builtins handled above
211 return ModuleSpec(
212 name=modname,
213 location=getattr(spec.loader_state, "filename", None),
214 type=ModuleType.PY_FROZEN,
215 )
216 except ValueError:
217 pass
218
219 return None

221 def contribute_to_path(
222 self, spec: ModuleSpec, processed: list[str]
223 ) -> Sequence[str] | None:
224 if spec.location is None:
225 # Builtin.
226 return None
227
228 if _is_setuptools_namespace(Path(spec.location)):
229 # extend_path is called, search sys.path for module/packages
230 # of this name see pkgutil.extend_path documentation
231 path = [
232 os.path.join(p, *processed)
233 for p in sys.path
234 if os.path.isdir(os.path.join(p, *processed))
235 ]
236 elif spec.name == "distutils" and not any(
237 spec.location.lower().startswith(ext_lib_dir.lower())
238 for ext_lib_dir in EXT_LIB_DIRS
239 ):
240 # virtualenv below 20.0 patches distutils in an unexpected way
241 # so we just find the location of distutils that will be
242 # imported to avoid spurious import-error messages
243 # https://github.com/pylint-dev/pylint/issues/5645
244 # A regression test to create this scenario exists in release-tests.yml
245 # and can be triggered manually from GitHub Actions
246 distutils_spec = importlib.util.find_spec("distutils")
247 if distutils_spec and distutils_spec.origin:
248 origin_path = Path(
249 distutils_spec.origin
250 ) # e.g. .../distutils/__init__.py
251 path = [str(origin_path.parent)] # e.g. .../distutils
252 else:
253 path = [spec.location]
254 else:
255 path = [spec.location]
256 return path

363 try:

365 data = stream.read(4096)

381def _search_zip(
382 modpath: Sequence[str],
383) -> tuple[Literal[ModuleType.PY_ZIPMODULE], str, str]:
384 for filepath, importer in _get_zipimporters():
385 if PY310_PLUS:
386 found: Any = importer.find_spec(modpath[0])
387 else:
388 found = importer.find_module(modpath[0])
389 if found:
390 if PY310_PLUS:
391 if not importer.find_spec(os.path.sep.join(modpath)):
392 raise ImportError(
393 "No module named {} in {}/{}".format(
394 ".".join(modpath[1:]), filepath, modpath
395 )
396 )
397 elif not importer.find_module(os.path.sep.join(modpath)):
398 raise ImportError(
399 "No module named {} in {}/{}".format(
400 ".".join(modpath[1:]), filepath, modpath
401 )
402 )
403 return (
404 ModuleType.PY_ZIPMODULE,
405 os.path.abspath(filepath) + os.path.sep + os.path.sep.join(modpath),
406 filepath,
407 )
408 raise ImportError(f"No module named {'.'.join(modpath)}")

411def _find_spec_with_path(
412 search_path: Sequence[str],
413 modname: str,
414 module_parts: list[str],
415 processed: list[str],
416 submodule_path: Sequence[str] | None,
417) -> tuple[Finder | _MetaPathFinder, ModuleSpec]:
418 for finder in _SPEC_FINDERS:
419 finder_instance = finder(search_path)
420 spec = finder_instance.find_module(
421 modname, module_parts, processed, submodule_path
422 )
423 if spec is None:
424 continue
425 return finder_instance, spec
426
427 # Support for custom finders
428 for meta_finder in sys.meta_path:
429 # See if we support the customer import hook of the meta_finder
430 meta_finder_name = meta_finder.__class__.__name__
431 if meta_finder_name not in _MetaPathFinderModuleTypes:
432 # Setuptools>62 creates its EditableFinders dynamically and have
433 # "type" as their __class__.__name__. We check __name__ as well
434 # to see if we can support the finder.
435 try:
436 meta_finder_name = meta_finder.__name__ # type: ignore[attr-defined]
437 except AttributeError:
438 continue
439 if meta_finder_name not in _MetaPathFinderModuleTypes:
440 continue
441
442 module_type = _MetaPathFinderModuleTypes[meta_finder_name]
443
444 # Meta path finders are supposed to have a find_spec method since
445 # Python 3.4. However, some third-party finders do not implement it.
446 # PEP302 does not refer to find_spec as well.
447 # See: https://github.com/pylint-dev/astroid/pull/1752/
448 if not hasattr(meta_finder, "find_spec"):
449 continue
450
451 spec = meta_finder.find_spec(modname, submodule_path)
452 if spec:
453 return (
454 meta_finder,
455 ModuleSpec(
456 spec.name,
457 module_type,
458 spec.origin,
459 spec.origin,
460 spec.submodule_search_locations,
461 ),
462 )
463
464 raise ImportError(f"No module named {'.'.join(module_parts)}")

467def find_spec(modpath: Iterable[str], path: Iterable[str] | None = None) -> ModuleSpec:
468 """Find a spec for the given module.
469
470 :type modpath: list or tuple
471 :param modpath:
472 split module's name (i.e name of a module or package split
473 on '.'), with leading empty strings for explicit relative import
474
475 :type path: list or None
476 :param path:
477 optional list of path where the module or package should be
478 searched (use sys.path if nothing or None is given)
479
480 :rtype: ModuleSpec
481 :return: A module spec, which describes how the module was
482 found and where.
483 """
484 return _find_spec(tuple(modpath), tuple(path) if path else None)
485
486
487@lru_cache(maxsize=1024)
488def _find_spec(module_path: tuple, path: tuple) -> ModuleSpec:
489 _path = path or sys.path
490
491 # Need a copy for not mutating the argument.
492 modpath = list(module_path)
493
494 submodule_path = None
495 module_parts = modpath[:]
496 processed: list[str] = []
497
498 while modpath:
499 modname = modpath.pop(0)
500 finder, spec = _find_spec_with_path(
501 _path, modname, module_parts, processed, submodule_path or path
502 )
503 processed.append(modname)
504 if modpath:
505 if isinstance(finder, Finder):
506 submodule_path = finder.contribute_to_path(spec, processed)
507 # If modname is a package from an editable install, update submodule_path
508 # so that the next module in the path will be found inside of it using importlib.
509 # Existence of __name__ is guaranteed by _find_spec_with_path.
510 elif finder.__name__ in _EditableFinderClasses: # type: ignore[attr-defined]
511 submodule_path = spec.submodule_search_locations
512
513 if spec.type == ModuleType.PKG_DIRECTORY:
514 spec = spec._replace(submodule_search_locations=submodule_path)
515
516 return spec
