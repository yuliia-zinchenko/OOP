5"""All alphanumeric unicode character are allowed in Python but due

30class NonAsciiNameChecker(base_checker.BaseChecker):
31 """A strict name checker only allowing ASCII.
32
33 Note: This check only checks Names, so it ignores the content of
34 docstrings and comments!
35 """
36
37 msgs = {
38 "C2401": (
39 '%s name "%s" contains a non-ASCII character, consider renaming it.',
40 "non-ascii-name",
41 NON_ASCII_HELP,
42 {"old_names": [("C0144", "old-non-ascii-name")]},
43 ),
44 # First %s will always be "file"
45 "W2402": (
46 '%s name "%s" contains a non-ASCII character.',
47 "non-ascii-file-name",
48 (
49 # Some = PyCharm at the time of writing didn't display the non_ascii_name_lo≈Ç

52 "Since Python 3.5, even though Python supports UTF-8 files, some editors or tools "
53 "don't."
54 ),

56 # First %s will always be "module"

66 def _check_name(self, node_type: str, name: str | None, node: nodes.NodeNG) -> None:
67 """Check whether a name is using non-ASCII characters."""
68 if name is None:
69 # For some nodes i.e. *kwargs from a dict, the name will be empty
70 return
71
72 if not str(name).isascii():
73 type_label = constants.HUMAN_READABLE_TYPES[node_type]
74 args = (type_label.capitalize(), name)
75
76 msg = "non-ascii-name"
77
78 # Some node types have customized messages
79 if node_type == "file":
80 msg = "non-ascii-file-name"
81 elif node_type == "module":
82 msg = "non-ascii-module-import"
83
84 self.add_message(msg, node=node, args=args, confidence=interfaces.HIGH)
85
86 @utils.only_required_for_messages("non-ascii-name", "non-ascii-file-name")
87 def visit_module(self, node: nodes.Module) -> None:
88 self._check_name("file", node.name.split(".")[-1], node)

90 @utils.only_required_for_messages("non-ascii-name")
91 def visit_functiondef(
92 self, node: nodes.FunctionDef | nodes.AsyncFunctionDef
93 ) -> None:
94 self._check_name("function", node.name, node)
95
96 # Check argument names
97 arguments = node.args
98
99 # Check position only arguments
100 if arguments.posonlyargs:
101 for pos_only_arg in arguments.posonlyargs:
102 self._check_name("argument", pos_only_arg.name, pos_only_arg)
103
104 # Check "normal" arguments
105 if arguments.args:
106 for arg in arguments.args:
107 self._check_name("argument", arg.name, arg)
108
109 # Check key word only arguments
110 if arguments.kwonlyargs:
111 for kwarg in arguments.kwonlyargs:
112 self._check_name("argument", kwarg.name, kwarg)
113
114 visit_asyncfunctiondef = visit_functiondef

116 @utils.only_required_for_messages("non-ascii-name")
117 def visit_global(self, node: nodes.Global) -> None:
118 for name in node.names:
119 self._check_name("const", name, node)

121 @utils.only_required_for_messages("non-ascii-name")
122 def visit_assignname(self, node: nodes.AssignName) -> None:
123 """Check module level assigned names."""
124 # The NameChecker from which this Checker originates knows a lot of different
125 # versions of variables, i.e. constants, inline variables etc.
126 # To simplify we use only `variable` here, as we don't need to apply different
127 # rules to different types of variables.
128 frame = node.frame()
129
130 if isinstance(frame, nodes.FunctionDef):
131 if node.parent in frame.body:
132 # Only perform the check if the assignment was done in within the body
133 # of the function (and not the function parameter definition
134 # (will be handled in visit_functiondef)
135 # or within a decorator (handled in visit_call)
136 self._check_name("variable", node.name, node)
137 elif isinstance(frame, nodes.ClassDef):
138 self._check_name("attr", node.name, node)
139 else:
140 # Possibilities here:
141 # - isinstance(node.assign_type(), nodes.Comprehension) == inlinevar
142 # - isinstance(frame, nodes.Module) == variable (constant?)
143 # - some other kind of assignment missed but still most likely a variable
144 self._check_name("variable", node.name, node)
145
146 @utils.only_required_for_messages("non-ascii-name")
147 def visit_classdef(self, node: nodes.ClassDef) -> None:
148 self._check_name("class", node.name, node)
149 for attr, anodes in node.instance_attrs.items():
150 if not any(node.instance_attr_ancestors(attr)):
151 self._check_name("attr", attr, anodes[0])
152
153 def _check_module_import(self, node: nodes.ImportFrom | nodes.Import) -> None:
154 for module_name, alias in node.names:
155 name = alias or module_name
156 self._check_name("module", name, node)
157
158 @utils.only_required_for_messages("non-ascii-name", "non-ascii-module-import")
159 def visit_import(self, node: nodes.Import) -> None:
160 self._check_module_import(node)
161
162 @utils.only_required_for_messages("non-ascii-name", "non-ascii-module-import")
163 def visit_importfrom(self, node: nodes.ImportFrom) -> None:
164 self._check_module_import(node)
165
166 @utils.only_required_for_messages("non-ascii-name")
167 def visit_call(self, node: nodes.Call) -> None:
168 """Check if the used keyword args are correct."""
169 for keyword in node.keywords:
170 self._check_name("argument", keyword.arg, keyword)
