6class SeparateDatabaseAndState(Operation):
7 """
8 Take two lists of operations - ones that will be used for the database,
9 and ones that will be used for the state change. This allows operations
10 that don't support state change to have it applied, or have operations

12 """

33 def database_forwards(self, app_label, schema_editor, from_state, to_state):
34 # We calculate state separately in here since our state functions aren't useful
35 for database_operation in self.database_operations:
36 to_state = from_state.clone()
37 database_operation.state_forwards(app_label, to_state)
38 database_operation.database_forwards(
39 app_label, schema_editor, from_state, to_state
40 )
41 from_state = to_state

43 def database_backwards(self, app_label, schema_editor, from_state, to_state):
44 # We calculate state separately in here since our state functions aren't useful
45 to_states = {}
46 for dbop in self.database_operations:
47 to_states[dbop] = to_state
48 to_state = to_state.clone()
49 dbop.state_forwards(app_label, to_state)
50 # to_state now has the states of all the database_operations applied
51 # which is the from_state for the backwards migration of the last
52 # operation.
53 for database_operation in reversed(self.database_operations):
54 from_state = to_state
55 to_state = to_states[database_operation]
56 database_operation.database_backwards(
57 app_label, schema_editor, from_state, to_state
58 )
