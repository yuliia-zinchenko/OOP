137 def _get_jobs(self, conditions):
138 jobs = []
139 failed_job_ids = []
140 for document in self.collection.find(
141 conditions, ["_id", "job_state"], sort=[("next_run_time", ASCENDING)]
142 ):
143 try:
144 jobs.append(self._reconstitute_job(document["job_state"]))
145 except BaseException:
146 self._logger.exception(
147 'Unable to restore job "%s" -- removing it', document["_id"]
148 )
149 failed_job_ids.append(document["_id"])
150
151 # Remove all the jobs we failed to restore
152 if failed_job_ids:
153 self.collection.delete_many({"_id": {"$in": failed_job_ids}})
154
155 return jobs
