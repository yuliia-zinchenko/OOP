37 def _check_contextmanager_generator_missing_cleanup(
38 self, node: nodes.FunctionDef
39 ) -> None:
40 """Check a FunctionDef to find if it is a generator
41 that uses a contextmanager internally.
42
43 If it is, check if the contextmanager is properly cleaned up. Otherwise, add message.
44
45 :param node: FunctionDef node to check
46 :type node: nodes.FunctionDef
47 """
48 # if function does not use a Yield statement, it can't be a generator
49 with_nodes = list(node.nodes_of_class(nodes.With))
50 if not with_nodes:
51 return
52 # check for Yield inside the With statement
53 yield_nodes = list(
54 chain.from_iterable(
55 with_node.nodes_of_class(nodes.Yield) for with_node in with_nodes
56 )
57 )
58 if not yield_nodes:
59 return
60
61 # infer the call that yields a value, and check if it is a contextmanager
62 for with_node in with_nodes:
63 for call, held in with_node.items:
64 if held is None:
65 # if we discard the value, then we can skip checking it
66 continue
67
68 # safe infer is a generator
69 inferred_node = getattr(utils.safe_infer(call), "parent", None)
70 if not isinstance(inferred_node, nodes.FunctionDef):
71 continue
72 if self._node_fails_contextmanager_cleanup(inferred_node, yield_nodes):
73 self.add_message(
74 "contextmanager-generator-missing-cleanup",
75 node=with_node,
76 args=(node.name,),
77 )
78
79 @staticmethod
80 def _node_fails_contextmanager_cleanup(
81 node: nodes.FunctionDef, yield_nodes: list[nodes.Yield]
82 ) -> bool:
83 """Check if a node fails contextmanager cleanup.
84
85 Current checks for a contextmanager:
86 - only if the context manager yields a non-constant value
87 - only if the context manager lacks a finally, or does not catch GeneratorExit
88 - only if some statement follows the yield, some manually cleanup happens
89
90 :param node: Node to check
91 :type node: nodes.FunctionDef
92 :return: True if fails, False otherwise
93 :param yield_nodes: List of Yield nodes in the function body
94 :type yield_nodes: list[nodes.Yield]
95 :rtype: bool
96 """
97
98 def check_handles_generator_exceptions(try_node: nodes.Try) -> bool:
99 # needs to handle either GeneratorExit, Exception, or bare except
100 for handler in try_node.handlers:
101 if handler.type is None:
102 # handles all exceptions (bare except)
103 return True
104 inferred = utils.safe_infer(handler.type)
105 if inferred and inferred.qname() in {
106 "builtins.GeneratorExit",
107 "builtins.Exception",
108 }:
109 return True
110 return False
111
112 # if context manager yields a non-constant value, then continue checking
113 if any(
114 yield_node.value is None or isinstance(yield_node.value, nodes.Const)
115 for yield_node in yield_nodes
116 ):
117 return False
118
119 # Check if yield expression is last statement
120 yield_nodes = list(node.nodes_of_class(nodes.Yield))
121 if len(yield_nodes) == 1:
122 n = yield_nodes[0].parent
123 while n is not node:
124 if n.next_sibling() is not None:
125 break
126 n = n.parent
127 else:
128 # No next statement found
129 return False
130
131 # if function body has multiple Try, filter down to the ones that have a yield node
132 try_with_yield_nodes = [
133 try_node
134 for try_node in node.nodes_of_class(nodes.Try)
135 if any(try_node.nodes_of_class(nodes.Yield))
136 ]
137 if not try_with_yield_nodes:
138 # no try blocks at all, so checks after this line do not apply
139 return True
140 # if the contextmanager has a finally block, then it is fine
141 if all(try_node.finalbody for try_node in try_with_yield_nodes):
142 return False
143 # if the contextmanager catches GeneratorExit, then it is fine
144 if all(
145 check_handles_generator_exceptions(try_node)
146 for try_node in try_with_yield_nodes
147 ):
148 return False
149 return True
