50 def get_field_type(self, data_type, description):
51 field_type = super().get_field_type(data_type, description)
52 if "auto_increment" in description.extra:
53 if field_type == "IntegerField":
54 return "AutoField"
55 elif field_type == "BigIntegerField":
56 return "BigAutoField"
57 elif field_type == "SmallIntegerField":
58 return "SmallAutoField"
59 if description.is_unsigned:
60 if field_type == "BigIntegerField":
61 return "PositiveBigIntegerField"
62 elif field_type == "IntegerField":
63 return "PositiveIntegerField"
64 elif field_type == "SmallIntegerField":
65 return "PositiveSmallIntegerField"
66 if description.data_type.upper() == "UUID":
67 return "UUIDField"
68 # JSON data type is an alias for LONGTEXT in MariaDB, use check
69 # constraints clauses to introspect JSONField.
70 if description.has_json_constraint:
71 return "JSONField"
72 return field_type

91 def get_table_description(self, cursor, table_name):
92 """
93 Return a description of the table with the DB-API cursor.description
94 interface."
95 """
96 json_constraints = {}
97 if (
98 self.connection.mysql_is_mariadb
99 and self.connection.features.can_introspect_json_field
100 ):
101 # JSON data type is an alias for LONGTEXT in MariaDB, select
102 # JSON_VALID() constraints to introspect JSONField.
103 cursor.execute(
104 """
105 SELECT c.constraint_name AS column_name
106 FROM information_schema.check_constraints AS c
107 WHERE
108 c.table_name = %s AND
109 LOWER(c.check_clause) =
110 'json_valid(`' + LOWER(c.constraint_name) + '`)' AND
111 c.constraint_schema = DATABASE()
112 """,
113 [table_name],
114 )
115 json_constraints = {row[0] for row in cursor.fetchall()}
116 # A default collation for the given table.
117 cursor.execute(
118 """
119 SELECT table_collation
120 FROM information_schema.tables
121 WHERE table_schema = DATABASE()
122 AND table_name = %s
123 """,
124 [table_name],
125 )
126 row = cursor.fetchone()
127 default_column_collation = row[0] if row else ""
128 # information_schema database gives more accurate results for some figures:
129 # - varchar length returned by cursor.description is an internal length,
130 # not visible length (#5725)
131 # - precision and scale (for decimal fields) (#5014)
132 # - auto_increment is not available in cursor.description
133 cursor.execute(
134 """
135 SELECT
136 column_name, data_type, character_maximum_length,
137 numeric_precision, numeric_scale, extra, column_default,
138 CASE
139 WHEN collation_name = %s THEN NULL
140 ELSE collation_name
141 END AS collation_name,
142 CASE
143 WHEN column_type LIKE '%% unsigned' THEN 1
144 ELSE 0
145 END AS is_unsigned,
146 column_comment
147 FROM information_schema.columns
148 WHERE table_name = %s AND table_schema = DATABASE()
149 """,
150 [default_column_collation, table_name],
151 )
152 field_info = {line[0]: InfoLine(*line) for line in cursor.fetchall()}
153
154 cursor.execute(
155 "SELECT * FROM %s LIMIT 1" % self.connection.ops.quote_name(table_name)
156 )
157
158 def to_int(i):
159 return int(i) if i is not None else i
160
161 fields = []
162 for line in cursor.description:
163 info = field_info[line[0]]
164 fields.append(
165 FieldInfo(
166 *line[:2],
167 to_int(info.max_len) or line[2],
168 to_int(info.max_len) or line[3],
169 to_int(info.num_prec) or line[4],
170 to_int(info.num_scale) or line[5],
171 line[6],
172 info.column_default,
173 info.collation,
174 info.extra,
175 info.is_unsigned,
176 line[0] in json_constraints,
177 info.comment,
178 info.data_type,
179 )
180 )
181 return fields

232 def _parse_constraint_columns(self, check_clause, columns):
233 check_columns = OrderedSet()
234 statement = sqlparse.parse(check_clause)[0]
235 tokens = (token for token in statement.flatten() if not token.is_whitespace)
236 for token in tokens:
237 if (
238 token.ttype == sqlparse.tokens.Name
239 and self.connection.ops.quote_name(token.value) == token.value
240 and token.value[1:-1] in columns
241 ):
242 check_columns.add(token.value[1:-1])
243 return check_columns

245 def get_constraints(self, cursor, table_name):
246 """
247 Retrieve any constraints or keys (unique, pk, fk, check, index) across
248 one or more columns.
249 """
250 constraints = {}
251 # Get the actual constraint names and columns
252 name_query = """
253 SELECT kc.`constraint_name`, kc.`column_name`,
254 kc.`referenced_table_name`, kc.`referenced_column_name`,
255 c.`constraint_type`
256 FROM
257 information_schema.key_column_usage AS kc,
258 information_schema.table_constraints AS c
259 WHERE
260 kc.table_schema = DATABASE() AND
261 (
262 kc.referenced_table_schema = DATABASE() OR
263 kc.referenced_table_schema IS NULL
264 ) AND
265 c.table_schema = kc.table_schema AND
266 c.constraint_name = kc.constraint_name AND
267 c.constraint_type != 'CHECK' AND
268 kc.table_name = %s
269 ORDER BY kc.`ordinal_position`
270 """
271 cursor.execute(name_query, [table_name])
272 for constraint, column, ref_table, ref_column, kind in cursor.fetchall():
273 if constraint not in constraints:
274 constraints[constraint] = {
275 "columns": OrderedSet(),
276 "primary_key": kind == "PRIMARY KEY",
277 "unique": kind in {"PRIMARY KEY", "UNIQUE"},
278 "index": False,
279 "check": False,
280 "foreign_key": (ref_table, ref_column) if ref_column else None,
281 }
282 if self.connection.features.supports_index_column_ordering:
283 constraints[constraint]["orders"] = []
284 constraints[constraint]["columns"].add(column)
285 # Add check constraints.
286 if self.connection.features.can_introspect_check_constraints:
287 unnamed_constraints_index = 0
288 columns = {
289 info.name for info in self.get_table_description(cursor, table_name)
290 }
291 if self.connection.mysql_is_mariadb:
292 type_query = """
293 SELECT c.constraint_name, c.check_clause
294 FROM information_schema.check_constraints AS c
295 WHERE
296 c.constraint_schema = DATABASE() AND
297 c.table_name = %s
298 """
299 else:
300 type_query = """
301 SELECT cc.constraint_name, cc.check_clause
302 FROM
303 information_schema.check_constraints AS cc,
304 information_schema.table_constraints AS tc
305 WHERE
306 cc.constraint_schema = DATABASE() AND
307 tc.table_schema = cc.constraint_schema AND
308 cc.constraint_name = tc.constraint_name AND
309 tc.constraint_type = 'CHECK' AND
310 tc.table_name = %s
311 """
312 cursor.execute(type_query, [table_name])
313 for constraint, check_clause in cursor.fetchall():
314 constraint_columns = self._parse_constraint_columns(
315 check_clause, columns
316 )
317 # Ensure uniqueness of unnamed constraints. Unnamed unique
318 # and check columns constraints have the same name as
319 # a column.
320 if set(constraint_columns) == {constraint}:
321 unnamed_constraints_index += 1
322 constraint = "__unnamed_constraint_%s__" % unnamed_constraints_index
323 constraints[constraint] = {
324 "columns": constraint_columns,
325 "primary_key": False,
326 "unique": False,
327 "index": False,
328 "check": True,
329 "foreign_key": None,
330 }
331 # Now add in the indexes
332 cursor.execute(
333 "SHOW INDEX FROM %s" % self.connection.ops.quote_name(table_name)
334 )
335 for table, non_unique, index, colseq, column, order, type_ in [
336 x[:6] + (x[10],) for x in cursor.fetchall()
337 ]:
338 if index not in constraints:
339 constraints[index] = {
340 "columns": OrderedSet(),
341 "primary_key": False,
342 "unique": not non_unique,
343 "check": False,
344 "foreign_key": None,
345 }
346 if self.connection.features.supports_index_column_ordering:
347 constraints[index]["orders"] = []
348 constraints[index]["index"] = True
349 constraints[index]["type"] = (
350 Index.suffix if type_ == "BTREE" else type_.lower()
351 )
352 constraints[index]["columns"].add(column)
353 if self.connection.features.supports_index_column_ordering:
354 constraints[index]["orders"].append("DESC" if order == "D" else "ASC")
355 # Convert the sorted sets to lists
356 for constraint in constraints.values():
357 constraint["columns"] = list(constraint["columns"])
358 return constraints
