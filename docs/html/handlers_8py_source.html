33 def _should_handle(self, path):
34 """
35 Check if the path should be handled. Ignore the path if:
36 * the host is provided as part of the base_url
37 * the request's path isn't under the media path (or equal)
38 """
39 return path.startswith(self.base_url[2]) and not self.base_url[1]
40
41 def file_path(self, url):
42 """
43 Return the relative path to the media file on disk for the given URL.
44 """
45 relative_url = url.removeprefix(self.base_url[2])
46 return url2pathname(relative_url)

48 def serve(self, request):
49 """Serve the request path."""
50 return serve(request, self.file_path(request.path), insecure=True)
