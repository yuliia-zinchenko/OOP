113 def serialize_db_to_string(self):
114 """
115 Serialize all data in the database into a JSON string.
116 Designed only for test runner usage; will not handle large
117 amounts of data.
118 """
119
120 # Iteratively return every object for all models to serialize.
121 def get_objects():
122 from django.db.migrations.loader import MigrationLoader
123
124 loader = MigrationLoader(self.connection)
125 for app_config in apps.get_app_configs():
126 if (
127 app_config.models_module is not None
128 and app_config.label in loader.migrated_apps
129 and app_config.name not in settings.TEST_NON_SERIALIZED_APPS
130 ):
131 for model in app_config.get_models():
132 if model._meta.can_migrate(
133 self.connection
134 ) and router.allow_migrate_model(self.connection.alias, model):
135 queryset = model._base_manager.using(
136 self.connection.alias,
137 ).order_by(model._meta.pk.name)
138 chunk_size = (
139 2000 if queryset._prefetch_related_lookups else None
140 )
141 yield from queryset.iterator(chunk_size=chunk_size)

143 # Serialize to a string
144 out = StringIO()

148 def deserialize_db_from_string(self, data):
149 """
150 Reload the database with data from a string generated by
151 the serialize_db_to_string() method.
152 """
153 data = StringIO(data)
154 table_names = set()
155 # Load data in a transaction to handle forward references and cycles.
156 with atomic(using=self.connection.alias):
157 # Disable constraint checks, because some databases (MySQL) doesn't
158 # support deferred checks.
159 with self.connection.constraint_checks_disabled():
160 for obj in serializers.deserialize(
161 "json", data, using=self.connection.alias
162 ):
163 obj.save()
164 table_names.add(obj.object.__class__._meta.db_table)
165 # Manually check for any invalid keys that might have been added,
166 # because constraint checks were disabled.
167 self.connection.check_constraints(table_names=table_names)

192 def _create_test_db(self, verbosity, autoclobber, keepdb=False):
193 """
194 Internal implementation - create the test db tables.
195 """
196 test_database_name = self._get_test_db_name()
197 test_db_params = {
198 "dbname": self.connection.ops.quote_name(test_database_name),
199 "suffix": self.sql_table_creation_suffix(),
200 }
201 # Create the test database and connect to it.
202 with self._nodb_cursor() as cursor:
203 try:
204 self._execute_create_test_db(cursor, test_db_params, keepdb)
205 except Exception as e:
206 # if we want to keep the db, then no need to do any of the below,
207 # just return and skip it all.
208 if keepdb:
209 return test_database_name
210
211 self.log("Got an error creating the test database: %s" % e)
212 if not autoclobber:
213 confirm = input(
214 "Type 'yes' if you would like to try deleting the test "
215 "database '%s', or 'no' to cancel: " % test_database_name
216 )
217 if autoclobber or confirm == "yes":
218 try:
219 if verbosity >= 1:
220 self.log(
221 "Destroying old test database for alias %s..."
222 % (
223 self._get_database_display_str(
224 verbosity, test_database_name
225 ),
226 )
227 )
228 cursor.execute("DROP DATABASE %(dbname)s" % test_db_params)
229 self._execute_create_test_db(cursor, test_db_params, keepdb)
230 except Exception as e:
231 self.log("Got an error recreating the test database: %s" % e)
232 sys.exit(2)
233 else:
234 self.log("Tests cancelled.")
235 sys.exit(1)
