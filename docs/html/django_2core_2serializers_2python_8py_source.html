99def Deserializer(
100 object_list, *, using=DEFAULT_DB_ALIAS, ignorenonexistent=False, **options
101):
102 """
103 Deserialize simple Python objects back into Django ORM instances.
104
105 It's expected that you pass the Python objects themselves (instead of a
106 stream or a string) to the constructor
107 """
108 handle_forward_references = options.pop("handle_forward_references", False)
109 field_names_cache = {} # Model: <list of field_names>
110
111 for d in object_list:
112 # Look up the model and starting build a dict of data for it.
113 try:
114 Model = _get_model(d["model"])
115 except base.DeserializationError:
116 if ignorenonexistent:
117 continue
118 else:
119 raise
120 data = {}
121 if "pk" in d:
122 try:
123 data[Model._meta.pk.attname] = Model._meta.pk.to_python(d.get("pk"))
124 except Exception as e:
125 raise base.DeserializationError.WithData(
126 e, d["model"], d.get("pk"), None
127 )
128 m2m_data = {}
129 deferred_fields = {}
130
131 if Model not in field_names_cache:
132 field_names_cache[Model] = {f.name for f in Model._meta.get_fields()}
133 field_names = field_names_cache[Model]
134
135 # Handle each field
136 for field_name, field_value in d["fields"].items():
137 if ignorenonexistent and field_name not in field_names:
138 # skip fields no longer on model
139 continue
140
141 field = Model._meta.get_field(field_name)
142
143 # Handle M2M relations
144 if field.remote_field and isinstance(
145 field.remote_field, models.ManyToManyRel
146 ):
147 try:
148 values = base.deserialize_m2m_values(
149 field, field_value, using, handle_forward_references
150 )
151 except base.M2MDeserializationError as e:
152 raise base.DeserializationError.WithData(
153 e.original_exc, d["model"], d.get("pk"), e.pk
154 )
155 if values == base.DEFER_FIELD:
156 deferred_fields[field] = field_value
157 else:
158 m2m_data[field.name] = values
159 # Handle FK fields
160 elif field.remote_field and isinstance(
161 field.remote_field, models.ManyToOneRel
162 ):
163 try:
164 value = base.deserialize_fk_value(
165 field, field_value, using, handle_forward_references
166 )
167 except Exception as e:
168 raise base.DeserializationError.WithData(
169 e, d["model"], d.get("pk"), field_value
170 )
171 if value == base.DEFER_FIELD:
172 deferred_fields[field] = field_value
173 else:
174 data[field.attname] = value
175 # Handle all other fields
176 else:
177 try:
178 data[field.name] = field.to_python(field_value)
179 except Exception as e:
180 raise base.DeserializationError.WithData(
181 e, d["model"], d.get("pk"), field_value
182 )
183
184 obj = base.build_instance(Model, data, using)
185 yield base.DeserializedObject(obj, m2m_data, deferred_fields)
