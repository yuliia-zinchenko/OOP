263 def write_to_last_migration_files(self, changes):
264 loader = MigrationLoader(connections[DEFAULT_DB_ALIAS])
265 new_changes = {}
266 update_previous_migration_paths = {}
267 for app_label, app_migrations in changes.items():
268 # Find last migration.
269 leaf_migration_nodes = loader.graph.leaf_nodes(app=app_label)
270 if len(leaf_migration_nodes) == 0:
271 raise CommandError(
272 f"App {app_label} has no migration, cannot update last migration."
273 )
274 leaf_migration_node = leaf_migration_nodes[0]
275 # Multiple leaf nodes have already been checked earlier in command.
276 leaf_migration = loader.graph.nodes[leaf_migration_node]
277 # Updated migration cannot be a squash migration, a dependency of
278 # another migration, and cannot be already applied.
279 if leaf_migration.replaces:
280 raise CommandError(
281 f"Cannot update squash migration '{leaf_migration}'."
282 )
283 if leaf_migration_node in loader.applied_migrations:
284 raise CommandError(
285 f"Cannot update applied migration '{leaf_migration}'."
286 )
287 depending_migrations = [
288 migration
289 for migration in loader.disk_migrations.values()
290 if leaf_migration_node in migration.dependencies
291 ]
292 if depending_migrations:
293 formatted_migrations = ", ".join(
294 [f"'{migration}'" for migration in depending_migrations]
295 )
296 raise CommandError(
297 f"Cannot update migration '{leaf_migration}' that migrations "
298 f"{formatted_migrations} depend on."
299 )
300 # Build new migration.
301 for migration in app_migrations:
302 leaf_migration.operations.extend(migration.operations)
303
304 for dependency in migration.dependencies:
305 if isinstance(dependency, SwappableTuple):
306 if settings.AUTH_USER_MODEL == dependency.setting:
307 leaf_migration.dependencies.append(
308 ("__setting__", "AUTH_USER_MODEL")
309 )
310 else:
311 leaf_migration.dependencies.append(dependency)
312 elif dependency[0] != migration.app_label:
313 leaf_migration.dependencies.append(dependency)
314 # Optimize migration.
315 optimizer = MigrationOptimizer()
316 leaf_migration.operations = optimizer.optimize(
317 leaf_migration.operations, app_label
318 )
319 # Update name.
320 previous_migration_path = MigrationWriter(leaf_migration).path
321 name_fragment = self.migration_name or leaf_migration.suggest_name()
322 suggested_name = leaf_migration.name[:4] + f"_{name_fragment}"
323 if leaf_migration.name == suggested_name:
324 new_name = leaf_migration.name + "_updated"
325 else:
326 new_name = suggested_name
327 leaf_migration.name = new_name
328 # Register overridden migration.
329 new_changes[app_label] = [leaf_migration]
330 update_previous_migration_paths[app_label] = previous_migration_path
331

406 def handle_merge(self, loader, conflicts):
407 """
408 Handles merging together conflicted migrations interactively,
409 if it's safe; otherwise, advises on how to fix it.
410 """
411 if self.interactive:
412 questioner = InteractiveMigrationQuestioner(prompt_output=self.log_output)
413 else:
414 questioner = MigrationQuestioner(defaults={"ask_merge": True})
415
416 for app_label, migration_names in conflicts.items():
417 # Grab out the migrations in question, and work out their
418 # common ancestor.
419 merge_migrations = []
420 for migration_name in migration_names:
421 migration = loader.get_migration(app_label, migration_name)
422 migration.ancestry = [
423 mig
424 for mig in loader.graph.forwards_plan((app_label, migration_name))
425 if mig[0] == migration.app_label
426 ]
427 merge_migrations.append(migration)
428
429 def all_items_equal(seq):
430 return all(item == seq[0] for item in seq[1:])
431
432 merge_migrations_generations = zip(*(m.ancestry for m in merge_migrations))
433 common_ancestor_count = sum(
434 1
435 for common_ancestor_generation in takewhile(
436 all_items_equal, merge_migrations_generations
437 )
438 )
439 if not common_ancestor_count:
440 raise ValueError(
441 "Could not find common ancestor of %s" % migration_names
442 )
443 # Now work out the operations along each divergent branch
444 for migration in merge_migrations:
445 migration.branch = migration.ancestry[common_ancestor_count:]
446 migrations_ops = (
447 loader.get_migration(node_app, node_name).operations
448 for node_app, node_name in migration.branch
449 )
450 migration.merged_operations = sum(migrations_ops, [])
451 # In future, this could use some of the Optimizer code
452 # (can_optimize_through) to automatically see if they're
453 # mergeable. For now, we always just prompt the user.
454 if self.verbosity > 0:
455 self.log(self.style.MIGRATE_HEADING("Merging %s" % app_label))
456 for migration in merge_migrations:
457 self.log(self.style.MIGRATE_LABEL(" Branch %s" % migration.name))
458 for operation in migration.merged_operations:
459 self.log(" %s" % operation.formatted_description())
460 if questioner.ask_merge(app_label):
461 # If they still want to merge it, then write out an empty
462 # file depending on the migrations needing merging.
463 numbers = [
464 MigrationAutodetector.parse_number(migration.name)
465 for migration in merge_migrations
466 ]
467 try:
468 biggest_number = max(x for x in numbers if x is not None)
469 except ValueError:
470 biggest_number = 1
471 subclass = type(
472 "Migration",
473 (Migration,),
474 {
475 "dependencies": [
476 (app_label, migration.name)
477 for migration in merge_migrations
478 ],
479 },
480 )
481 parts = ["%04i" % (biggest_number + 1)]
482 if self.migration_name:
483 parts.append(self.migration_name)
484 else:
485 parts.append("merge")
486 leaf_names = "_".join(
487 sorted(migration.name for migration in merge_migrations)
488 )
489 if len(leaf_names) > 47:
490 parts.append(get_migration_name_timestamp())
491 else:
492 parts.append(leaf_names)
493 migration_name = "_".join(parts)
494 new_migration = subclass(migration_name, app_label)
495 writer = MigrationWriter(new_migration, self.include_header)
496
497 if not self.dry_run:
