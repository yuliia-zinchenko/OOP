157 self._ensure_paths()

160 all_ids = self.client.get_children(self.path)
161 for node_name in all_ids:

163 node_path = self.path + "/" + node_name
164 content, _ = self.client.get(node_path)
165 doc = pickle.loads(content)
166 job_def = {
167 "job_id": node_name,
168 "next_run_time": doc["next_run_time"]

170 else None,
171 "job_state": doc["job_state"],
172 "job": self._reconstitute_job(doc["job_state"]),
173 "creation_time": _.ctime,

175 jobs.append(job_def)
176 except BaseException:

182 # Remove all the jobs we failed to restore
