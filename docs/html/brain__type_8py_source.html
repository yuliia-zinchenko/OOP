5"""
6Astroid hooks for type support.
7
8Starting from python3.9, type object behaves as it had __class_getitem__ method.
9However it was not possible to simply add this method inside type's body, otherwise
10all types would also have this method. In this case it would have been possible
11to write str[int].
12Guido Van Rossum proposed a hack to handle this in the interpreter:
13https://github.com/python/cpython/blob/67e394562d67cbcd0ac8114e5439494e7645b8f5/Objects/abstract.c#L181-L184
14
15This brain follows the same logic. It is no wise to add permanently the __class_getitem__ method
16to the type object. Instead we choose to add it only in the case of a subscript node
17which inside name node is type.
18Doing this type[int] is allowed whereas str[int] is not.
19
20Thanks to Lukasz Langa for fruitful discussion.
21"""
22

31def _looks_like_type_subscript(node) -> bool:
32 """
33 Try to figure out if a Name node is used inside a type related subscript.
34
35 :param node: node to check
36 :type node: astroid.nodes.node_classes.NodeNG
37 :return: whether the node is a Name node inside a type related subscript
38 """
39 if isinstance(node, nodes.Name) and isinstance(node.parent, nodes.Subscript):
40 return node.name == "type"
41 return False

44def infer_type_sub(node, context: InferenceContext | None = None):
45 """
46 Infer a type[...] subscript.
47
48 :param node: node to infer
49 :type node: astroid.nodes.node_classes.NodeNG
50 :return: the inferred node
51 :rtype: nodes.NodeNG
52 """
53 node_scope, _ = node.scope().lookup("type")
54 if not isinstance(node_scope, nodes.Module) or node_scope.qname() != "builtins":
55 raise UseInferenceDefault()
56 class_src = """
57 class type:
58 def __class_getitem__(cls, key):
59 return cls
60 """
61 node = extract_node(class_src)
62 return node.infer(context=context)

65def register(manager: AstroidManager) -> None:
