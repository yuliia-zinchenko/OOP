480 def transform(self, ct, clone=False):
481 """
482 Requires GDAL. Transform the geometry according to the given
483 transformation object, which may be an integer SRID, and WKT or
484 PROJ string. By default, transform the geometry in-place and return
485 nothing. However if the `clone` keyword is set, don't modify the
486 geometry and return a transformed clone instead.
487 """
488 srid = self.srid
489
490 if ct == srid:
491 # short-circuit where source & dest SRIDs match
492 if clone:
493 return self.clone()
494 else:
495 return
496
497 if isinstance(ct, gdal.CoordTransform):
498 # We don't care about SRID because CoordTransform presupposes
499 # source SRS.
500 srid = None
501 elif srid is None or srid < 0:
502 raise GEOSException("Calling transform() with no SRID set is not supported")
503
504 # Creating an OGR Geometry, which is then transformed.
505 g = gdal.OGRGeometry(self._ogr_ptr(), srid)
506 g.transform(ct)
507 # Getting a new GEOS pointer
508 ptr = g._geos_ptr()
509 if clone:
510 # User wants a cloned transformed geometry returned.
511 return GEOSGeometry(ptr, srid=g.srid)
512 if ptr:
513 # Reassigning pointer, and performing post-initialization setup
514 # again due to the reassignment.
515 capi.destroy_geom(self.ptr)
516 self.ptr = ptr
517 self._post_init()
518 self.srid = g.srid
519 else:
520 raise GEOSException("Transformed WKB was invalid.")

723 def __init__(self, geo_input, srid=None):
724 """
725 The base constructor for GEOS geometry objects. It may take the
726 following inputs:
727
728 * strings:
729 - WKT
730 - HEXEWKB (a PostGIS-specific canonical form)
731 - GeoJSON (requires GDAL)
732 * memoryview:
733 - WKB
734
735 The `srid` keyword specifies the Source Reference Identifier (SRID)
736 number for this Geometry. If not provided, it defaults to None.
737 """
738 input_srid = None
739 if isinstance(geo_input, bytes):
740 geo_input = force_str(geo_input)
741 if isinstance(geo_input, str):
742 wkt_m = wkt_regex.match(geo_input)
743 if wkt_m:
744 # Handle WKT input.
745 if wkt_m["srid"]:
746 input_srid = int(wkt_m["srid"])
747 g = self._from_wkt(force_bytes(wkt_m["wkt"]))
748 elif hex_regex.match(geo_input):
749 # Handle HEXEWKB input.
750 g = wkb_r().read(force_bytes(geo_input))

752 # Handle GeoJSON input.
