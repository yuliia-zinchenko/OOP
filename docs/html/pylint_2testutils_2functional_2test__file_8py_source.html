75 def _parse_options(self) -> None:
76 cp = configparser.ConfigParser()
77 cp.add_section("testoptions")
78 try:
79 cp.read(self.option_file)
80 except NoFileError:
81 pass
82
83 for name, value in cp.items("testoptions"):
84 conv = self._CONVERTERS.get(name, lambda v: v)
85
86 assert (
87 name in POSSIBLE_TEST_OPTIONS
88 ), f"[testoptions]' can only contains one of {POSSIBLE_TEST_OPTIONS} and had '{name}'"
89 self.options[name] = conv(value) # type: ignore[literal-required]
