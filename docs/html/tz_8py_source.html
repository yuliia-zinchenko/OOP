39@register.filter("timezone")
40def do_timezone(value, arg):
41 """
42 Convert a datetime to local time in a given time zone.
43
44 The argument must be an instance of a tzinfo subclass or a time zone name.
45
46 Naive datetimes are assumed to be in local time in the default time zone.
47 """
48 if not isinstance(value, datetime):
49 return ""
50
51 # Obtain a timezone-aware datetime
52 try:
53 if timezone.is_naive(value):
54 default_timezone = timezone.get_default_timezone()
55 value = timezone.make_aware(value, default_timezone)
56 # Filters must never raise exceptionsm, so catch everything.
57 except Exception:
58 return ""
59
60 # Obtain a tzinfo instance
61 if isinstance(arg, tzinfo):
62 tz = arg
63 elif isinstance(arg, str):
64 try:
65 tz = zoneinfo.ZoneInfo(arg)
66 except zoneinfo.ZoneInfoNotFoundError:
67 return ""
68 else:
69 return ""
70
71 result = timezone.localtime(value, tz)
72
73 # HACK: the convert_to_local_time flag will prevent
74 # automatic conversion of the value to local time.
75 result = datetimeobject(
76 result.year,
77 result.month,
78 result.day,
79 result.hour,
80 result.minute,
81 result.second,
82 result.microsecond,
83 result.tzinfo,
84 )
85 result.convert_to_local_time = False
86 return result

137@register.tag("localtime")
138def localtime_tag(parser, token):
139 """
140 Force or prevent conversion of datetime objects to local time,
141 regardless of the value of ``settings.USE_TZ``.
142
143 Sample usage::
144
145 {% localtime off %}{{ value_in_utc }}{% endlocaltime %}
146 """
147 bits = token.split_contents()
148 if len(bits) == 1:
149 use_tz = True
150 elif len(bits) > 2 or bits[1] not in ("on", "off"):
151 raise TemplateSyntaxError("%r argument should be 'on' or 'off'" % bits[0])
152 else:
153 use_tz = bits[1] == "on"
154 nodelist = parser.parse(("endlocaltime",))
155 parser.delete_first_token()
156 return LocalTimeNode(nodelist, use_tz)

159@register.tag("timezone")
160def timezone_tag(parser, token):
161 """
162 Enable a given time zone just for this block.
163
164 The ``timezone`` argument must be an instance of a ``tzinfo`` subclass, a
165 time zone name, or ``None``. If it is ``None``, the default time zone is
166 used within the block.
167
168 Sample usage::
169
170 {% timezone "Europe/Paris" %}
171 It is {{ now }} in Paris.
172 {% endtimezone %}
173 """
174 bits = token.split_contents()
175 if len(bits) != 2:
176 raise TemplateSyntaxError("'%s' takes one argument (timezone)" % bits[0])
177 tz = parser.compile_filter(bits[1])
178 nodelist = parser.parse(("endtimezone",))
179 parser.delete_first_token()
180 return TimezoneNode(nodelist, tz)
