22class OverlappingExceptionsChecker(checkers.BaseChecker):
23 """Checks for two or more exceptions in the same exception handler
24 clause that are identical or parts of the same inheritance hierarchy.
25
26 (i.e. overlapping).
27 """
28
29 name = "overlap-except"
30 msgs = {
31 "W0714": (
32 "Overlapping exceptions (%s)",
33 "overlapping-except",
34 "Used when exceptions in handler overlap or are identical",
35 )
36 }
37 options = ()

39 @utils.only_required_for_messages("overlapping-except")
40 def visit_try(self, node: nodes.Try) -> None:
41 """Check for empty except."""
42 for handler in node.handlers:
43 if handler.type is None:
44 continue
45 if isinstance(handler.type, astroid.BoolOp):
46 continue
47 try:
48 excs = list(_annotated_unpack_infer(handler.type))
49 except astroid.InferenceError:
50 continue
51
52 handled_in_clause: list[tuple[Any, Any]] = []
53 for part, exc in excs:
54 if isinstance(exc, util.UninferableBase):
55 continue
56 if isinstance(exc, astroid.Instance) and utils.inherit_from_std_ex(exc):
57 exc = exc._proxied
58
59 if not isinstance(exc, astroid.ClassDef):
60 continue
61
62 exc_ancestors = [
63 anc for anc in exc.ancestors() if isinstance(anc, astroid.ClassDef)
64 ]
65
66 for prev_part, prev_exc in handled_in_clause:
67 prev_exc_ancestors = [
68 anc
69 for anc in prev_exc.ancestors()
70 if isinstance(anc, astroid.ClassDef)
71 ]
72 if exc == prev_exc:
73 self.add_message(
74 "overlapping-except",
75 node=handler.type,
76 args=f"{prev_part.as_string()} and {part.as_string()} are the same",
77 )
78 elif prev_exc in exc_ancestors or exc in prev_exc_ancestors:
79 ancestor = part if exc in prev_exc_ancestors else prev_part
80 descendant = part if prev_exc in exc_ancestors else prev_part
81 self.add_message(
82 "overlapping-except",
83 node=handler.type,
84 args=f"{ancestor.as_string()} is an ancestor class of {descendant.as_string()}",
85 )
86 handled_in_clause += [(part, exc)]
