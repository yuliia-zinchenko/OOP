104 raise ImproperlyConfigured("GeoDjango supports SpatiaLite 4.3.0 and above.")

107 def convert_extent(self, box):
108 """
109 Convert the polygon data received from SpatiaLite to min/max values.
110 """
111 if box is None:
112 return None
113 shell = GEOSGeometry(box).shell
114 xmin, ymin = shell[0][:2]
115 xmax, ymax = shell[2][:2]
116 return (xmin, ymin, xmax, ymax)

125 def get_distance(self, f, value, lookup_type):
126 """
127 Return the distance parameters for the given geometry field,
128 lookup value, and lookup type.
129 """
130 if not value:
131 return []
132 value = value[0]
133 if isinstance(value, Distance):
134 if f.geodetic(self.connection):
135 if lookup_type == "dwithin":
136 raise ValueError(
137 "Only numeric values of degree units are allowed on "
138 "geographic DWithin queries."
139 )
140 dist_param = value.m
141 else:
142 dist_param = getattr(
143 value, Distance.unit_attname(f.units_name(self.connection))
144 )
145 else:
146 dist_param = value
147 return [dist_param]

149 def _get_spatialite_func(self, func):
150 """
151 Helper routine for calling SpatiaLite functions and returning
152 their result.
153 Any error occurring in this method should be handled by the caller.
154 """
155 cursor = self.connection._cursor()
156 try:
157 cursor.execute("SELECT %s" % func)
158 row = cursor.fetchone()
159 finally:
160 cursor.close()
161 return row[0]
