17def get_subparser_help(linter: PyLinter, command: str) -> str:
18 """Get the help message for one of the subcommands."""
19 # Make sure subparsers are initialized properly
20 assert linter._arg_parser._subparsers
21 subparser_action = linter._arg_parser._subparsers._group_actions[0]
22 assert isinstance(subparser_action, argparse._SubParsersAction)
23
24 for name, subparser in subparser_action.choices.items():
25 assert isinstance(subparser, argparse.ArgumentParser)
26 if name == command:
27 # Remove last character which is an extra new line
28 return subparser.format_help()[:-1]
29 return "" # pragma: no cover

32def get_help(parser: argparse.ArgumentParser) -> str:
33 """Get the help message for the main 'pylint-config' command.
34
35 Taken from argparse.ArgumentParser.format_help.
36 """
37 formatter = parser._get_formatter()
38
39 # usage
40 formatter.add_usage(
41 parser.usage, parser._actions, parser._mutually_exclusive_groups
42 )
43
44 # description
45 formatter.add_text(parser.description)
46
47 # positionals, optionals and user-defined groups
48 for action_group in parser._action_groups:
49 if action_group.title == "Subcommands":
50 formatter.start_section(action_group.title)
51 formatter.add_text(action_group.description)
52 formatter.add_arguments(action_group._group_actions)
53 formatter.end_section()
54
55 # epilog
56 formatter.add_text(parser.epilog)
57
58 # determine help from format above
59 return formatter.format_help()
