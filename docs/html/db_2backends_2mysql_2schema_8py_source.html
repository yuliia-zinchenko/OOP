166 def _create_missing_fk_index(
167 self,

170 fields,
171 expressions=None,
172 ):
173 """
174 MySQL can remove an implicit FK index on a field when that field is
175 covered by another index like a unique_together. "covered" here means
176 that the more complex index has the FK field as its first field (see
177 https://bugs.mysql.com/bug.php?id=37910).
178
179 Manually create an implicit FK index to make it possible to remove the
180 composed index.
181 """
182 first_field_name = None
183 if fields:
184 first_field_name = fields[0]
185 elif (
186 expressions
187 and self.connection.features.supports_expression_indexes
188 and isinstance(expressions[0], F)
189 and LOOKUP_SEP not in expressions[0].name
190 ):
191 first_field_name = expressions[0].name
192
193 if not first_field_name:
194 return
195
196 first_field = model._meta.get_field(first_field_name)
197 if first_field.get_internal_type() == "ForeignKey":
198 column = self.connection.introspection.identifier_converter(
199 first_field.column
200 )
201 with self.connection.cursor() as cursor:
202 constraint_names = [
203 name
204 for name, infodict in self.connection.introspection.get_constraints(
205 cursor, model._meta.db_table
206 ).items()
207 if infodict["index"] and infodict["columns"][0] == column
208 ]
209 # There are no other indexes that starts with the FK field, only
210 # the index that is expected to be deleted.
211 if len(constraint_names) == 1:
212 self.execute(
213 self._create_index_sql(model, fields=[first_field], suffix="")
214 )
