204 def build_absolute_uri(self, location=None):
205 """
206 Build an absolute URI from the location and the variables available in
207 this request. If no ``location`` is specified, build the absolute URI
208 using request.get_full_path(). If the location is absolute, convert it
209 to an RFC 3987 compliant URI and return it. If location is relative or
210 is scheme-relative (i.e., ``//example.com/``), urljoin() it to a base
211 URL constructed from the request variables.
212 """

214 # Make it an absolute url (but schemeless and domainless) for the
215 # edge case that the path starts with '//'.
216 location = "//%s" % self.get_full_path()
217 else:
218 # Coerce lazy locations.
219 location = str(location)
220 bits = urlsplit(location)
221 if not (bits.scheme and bits.netloc):
222 # Handle the simple, most common case. If the location is absolute
223 # and a scheme or host (netloc) isn't provided, skip an expensive
224 # urljoin() as long as no path segments are '.' or '..'.
225 if (
226 bits.path.startswith("/")
