52def _check_link_requires_python(
53 link: Link,
54 version_info: Tuple[int, int, int],
55 ignore_requires_python: bool = False,
56) -> bool:
57 """
58 Return whether the given Python version is compatible with a link's
59 "Requires-Python" value.
60
61 :param version_info: A 3-tuple of ints representing the Python
62 major-minor-micro version to check.
63 :param ignore_requires_python: Whether to ignore the "Requires-Python"
64 value if the given Python version isn't compatible.
65 """
66 try:
67 is_compatible = check_requires_python(
68 link.requires_python,
69 version_info=version_info,
70 )
71 except specifiers.InvalidSpecifier:
72 logger.debug(
73 "Ignoring invalid Requires-Python (%r) for link: %s",
74 link.requires_python,
75 link,
76 )
77 else:
78 if not is_compatible:
79 version = ".".join(map(str, version_info))
80 if not ignore_requires_python:
81 logger.verbose(
82 "Link requires a different Python (%s not in: %r): %s",
83 version,
84 link.requires_python,
85 link,
86 )
87 return False
88
89 logger.debug(
90 "Ignoring failed Requires-Python check (%s not in: %r) for link: %s",
91 version,
92 link.requires_python,
93 link,
94 )
95
96 return True

141 of PEP 592) are permitted to be candidates for install.
142 :param ignore_requires_python: Whether to ignore incompatible
143 PEP 503 "data-requires-python" values in HTML links. Defaults
144 to False.
145 """
146 if ignore_requires_python is None:
147 ignore_requires_python = False

149 self._allow_yanked = allow_yanked

151 self._ignore_requires_python = ignore_requires_python

157 def evaluate_link(self, link: Link) -> Tuple[LinkType, str]:
158 """
159 Determine whether a link is a candidate for installation.
160
161 :return: A tuple (result, detail), where *result* is an enum
162 representing whether the evaluation found a candidate, or the reason
163 why one is not found. If a candidate is found, *detail* will be the
164 candidate's version string; if one is not found, it contains the
165 reason the link fails to qualify.
166 """
167 version = None
168 if link.is_yanked and not self._allow_yanked:
169 reason = link.yanked_reason or "<none given>"
170 return (LinkType.yanked, f"yanked for reason: {reason}")
171
172 if link.egg_fragment:
173 egg_info = link.egg_fragment
174 ext = link.ext
175 else:
176 egg_info, ext = link.splitext()
177 if not ext:
178 return (LinkType.format_unsupported, "not a file")
179 if ext not in SUPPORTED_EXTENSIONS:
180 return (
181 LinkType.format_unsupported,
182 f"unsupported archive format: {ext}",
183 )
184 if "binary" not in self._formats and ext == WHEEL_EXTENSION:
185 reason = f"No binaries permitted for {self.project_name}"
186 return (LinkType.format_unsupported, reason)
187 if "macosx10" in link.path and ext == ".zip":
188 return (LinkType.format_unsupported, "macosx10 one")
189 if ext == WHEEL_EXTENSION:
190 try:
191 wheel = Wheel(link.filename)
192 except InvalidWheelFilename:
193 return (
194 LinkType.format_invalid,
195 "invalid wheel filename",
196 )
197 if canonicalize_name(wheel.name) != self._canonical_name:
198 reason = f"wrong project name (not {self.project_name})"
199 return (LinkType.different_project, reason)
200
201 supported_tags = self._target_python.get_unsorted_tags()
202 if not wheel.supported(supported_tags):
203 # Include the wheel's tags in the reason string to
204 # simplify troubleshooting compatibility issues.
205 file_tags = ", ".join(wheel.get_formatted_file_tags())
206 reason = (
207 f"none of the wheel's tags ({file_tags}) are compatible "
208 f"(run pip debug --verbose to show compatible tags)"
209 )
210 return (LinkType.platform_mismatch, reason)
211
212 version = wheel.version
213
214 # This should be up by the self.ok_binary check, but see issue 2700.
215 if "source" not in self._formats and ext != WHEEL_EXTENSION:
216 reason = f"No sources permitted for {self.project_name}"
217 return (LinkType.format_unsupported, reason)
218
219 if not version:
220 version = _extract_version_from_fragment(
221 egg_info,
222 self._canonical_name,
223 )
224 if not version:
225 reason = f"Missing project version for {self.project_name}"
226 return (LinkType.format_invalid, reason)
227
228 match = self._py_version_re.search(version)
229 if match:
230 version = version[: match.start()]
231 py_version = match.group(1)
232 if py_version != self._target_python.py_version:
233 return (
234 LinkType.platform_mismatch,
235 "Python version is incorrect",
236 )
237
238 supports_python = _check_link_requires_python(
239 link,
240 version_info=self._target_python.py_version_info,
241 ignore_requires_python=self._ignore_requires_python,
242 )
243 if not supports_python:
244 reason = f"{version} Requires-Python {link.requires_python}"
245 return (LinkType.requires_python_mismatch, reason)
246
247 logger.debug("Found link %s, version: %s", link, version)
248
249 return (LinkType.candidate, version)

252def filter_unallowed_hashes(
253 candidates: List[InstallationCandidate],
254 hashes: Optional[Hashes],
255 project_name: str,
256) -> List[InstallationCandidate]:
257 """
258 Filter out candidates whose hashes aren't allowed, and return a new
259 list of candidates.
260
261 If at least one candidate has an allowed hash, then all candidates with
262 either an allowed hash or no hash specified are returned. Otherwise,
263 the given candidates are returned.
264
265 Including the candidates with no hash specified when there is a match
266 allows a warning to be logged if there is a more preferred candidate
267 with no hash specified. Returning all candidates in the case of no
268 matches lets pip report the hash of the candidate that would otherwise
269 have been installed (e.g. permitting the user to more easily update
270 their requirements file with the desired hash).
271 """
272 if not hashes:
273 logger.debug(
274 "Given no hashes to check %s links for project %r: "
275 "discarding no candidates",
276 len(candidates),
277 project_name,
278 )
279 # Make sure we're not returning back the given value.
280 return list(candidates)
281
282 matches_or_no_digest = []
283 # Collect the non-matches for logging purposes.
284 non_matches = []
285 match_count = 0
286 for candidate in candidates:
287 link = candidate.link
288 if not link.has_hash:
289 pass
290 elif link.is_hash_allowed(hashes=hashes):
291 match_count += 1
292 else:
293 non_matches.append(candidate)
294 continue
295
296 matches_or_no_digest.append(candidate)
297
298 if match_count:
299 filtered = matches_or_no_digest
300 else:
301 # Make sure we're not returning back the given value.
302 filtered = list(candidates)
303
304 if len(filtered) == len(candidates):
305 discard_message = "discarding no candidates"
306 else:
307 discard_message = "discarding {} non-matches:\n {}".format(
308 len(non_matches),
309 "\n ".join(str(candidate.link) for candidate in non_matches),
310 )
311
312 logger.debug(
313 "Checked %s links for project %r against %s hashes "
314 "(%s matches, %s no digest): %s",
315 len(candidates),
316 project_name,
317 hashes.digest_count,
318 match_count,
319 len(matches_or_no_digest) - match_count,
320 discard_message,
321 )
322
323 return filtered

445 def get_applicable_candidates(
446 self,
447 candidates: List[InstallationCandidate],
448 ) -> List[InstallationCandidate]:
449 """
450 Return the applicable candidates from a list of candidates.
451 """
452 # Using None infers from the specifier instead.
453 allow_prereleases = self._allow_all_prereleases or None
454 specifier = self._specifier
455
456 # We turn the version object into a str here because otherwise
457 # when we're debundled but setuptools isn't, Python will see
458 # packaging.version.Version and
459 # pkg_resources._vendor.packaging.version.Version as different
460 # types. This way we'll use a str as a common data interchange
461 # format. If we stop using the pkg_resources provided specifier
462 # and start using our own, we can drop the cast to str().
463 candidates_and_versions = [(c, str(c.version)) for c in candidates]
464 versions = set(
465 specifier.filter(
466 (v for _, v in candidates_and_versions),
467 prereleases=allow_prereleases,
468 )
469 )
470
471 applicable_candidates = [c for c, v in candidates_and_versions if v in versions]
472 filtered_applicable_candidates = filter_unallowed_hashes(
473 candidates=applicable_candidates,
474 hashes=self._hashes,
475 project_name=self._project_name,
476 )
477
478 return sorted(filtered_applicable_candidates, key=self._sort_key)

480 def _sort_key(self, candidate: InstallationCandidate) -> CandidateSortingKey:
481 """
482 Function to pass as the `key` argument to a call to sorted() to sort
483 InstallationCandidates by preference.
484
485 Returns a tuple such that tuples sorting as greater using Python's
486 default comparison operator are more preferred.
487
488 The preference is as follows:
489
490 First and foremost, candidates with allowed (matching) hashes are
491 always preferred over candidates without matching hashes. This is
492 because e.g. if the only candidate with an allowed hash is yanked,
493 we still want to use that candidate.
494
495 Second, excepting hash considerations, candidates that have been
496 yanked (in the sense of PEP 592) are always less preferred than
497 candidates that haven't been yanked. Then:
498
499 If not finding wheels, they are sorted by version only.
500 If finding wheels, then the sort order is by version, then:
501 1. existing installs
502 2. wheels ordered via Wheel.support_index_min(self._supported_tags)
503 3. source archives
504 If prefer_binary was set, then all wheels are sorted above sources.
505
506 Note: it was considered to embed this logic into the Link
507 comparison operators, but then different sdist links
508 with the same version, would have to be considered equal
509 """
510 valid_tags = self._supported_tags
511 support_num = len(valid_tags)
512 build_tag: BuildTag = ()
513 binary_preference = 0
514 link = candidate.link
515 if link.is_wheel:
516 # can raise InvalidWheelFilename
517 wheel = Wheel(link.filename)
518 try:
519 pri = -(
520 wheel.find_most_preferred_tag(
521 valid_tags, self._wheel_tag_preferences
522 )
523 )
524 except ValueError:
525 raise UnsupportedWheel(
526 f"{wheel.filename} is not a supported wheel for this platform. It "
527 "can't be sorted."
528 )
529 if self._prefer_binary:
530 binary_preference = 1
531 if wheel.build_tag is not None:
532 match = re.match(r"^(\d+)(.*)$", wheel.build_tag)
533 assert match is not None, "guaranteed by filename validation"
534 build_tag_groups = match.groups()
535 build_tag = (int(build_tag_groups[0]), build_tag_groups[1])
536 else: # sdist
537 pri = -(support_num)
538 has_allowed_hash = int(link.is_hash_allowed(self._hashes))
539 yank_value = -1 * int(link.is_yanked) # -1 for yanked.
540 return (
541 has_allowed_hash,
542 yank_value,
543 binary_preference,
544 candidate.version,
545 pri,
546 build_tag,
547 )

718 def _sort_links(self, links: Iterable[Link]) -> List[Link]:
719 """
720 Returns elements of links in order, non-egg links first, egg links
721 second, while eliminating duplicates
722 """
723 eggs, no_eggs = [], []
724 seen: Set[Link] = set()
725 for link in links:
726 if link not in seen:
727 seen.add(link)
728 if link.egg_fragment:
729 eggs.append(link)
730 else:
731 no_eggs.append(link)
732 return no_eggs + eggs

738 # the link string is usually very long.

742 def get_install_candidate(
743 self, link_evaluator: LinkEvaluator, link: Link
744 ) -> Optional[InstallationCandidate]:

746 If the link is a candidate for install, convert it to an
747 InstallationCandidate and return it. Otherwise, return None.
748 """
749 result, detail = link_evaluator.evaluate_link(link)
750 if result != LinkType.candidate:
751 self._log_skipped_link(link, result, detail)
752 return None
753
754 try:
755 return InstallationCandidate(
756 name=link_evaluator.project_name,
757 link=link,
758 version=detail,
759 )

798 @functools.lru_cache(maxsize=None)
799 def find_all_candidates(self, project_name: str) -> List[InstallationCandidate]:
800 """Find all available InstallationCandidate for project_name
801
802 This checks index_urls and find_links.
803 All versions found are returned as an InstallationCandidate list.
804
805 See LinkEvaluator.evaluate_link() for details on which files
806 are accepted.
807 """
808 link_evaluator = self.make_link_evaluator(project_name)
809
810 collected_sources = self._link_collector.collect_sources(
811 project_name=project_name,
812 candidates_from_page=functools.partial(
813 self.process_project_url,
814 link_evaluator=link_evaluator,
815 ),
816 )
817
818 page_candidates_it = itertools.chain.from_iterable(
819 source.page_candidates()
820 for sources in collected_sources
821 for source in sources
822 if source is not None
823 )
824 page_candidates = list(page_candidates_it)
825
826 file_links_it = itertools.chain.from_iterable(
827 source.file_links()
828 for sources in collected_sources
829 for source in sources
830 if source is not None
831 )
832 file_candidates = self.evaluate_links(
833 link_evaluator,
834 sorted(file_links_it, reverse=True),
835 )
836
837 if logger.isEnabledFor(logging.DEBUG) and file_candidates:
838 paths = []
839 for candidate in file_candidates:
840 assert candidate.link.url # we need to have a URL
841 try:
842 paths.append(candidate.link.file_path)
843 except Exception:
844 paths.append(candidate.link.url) # it's not a local file
845
846 logger.debug("Local files found: %s", ", ".join(paths))
847
848 # This is an intentional priority ordering
849 return file_candidates + page_candidates

891 def find_requirement(
892 self, req: InstallRequirement, upgrade: bool
893 ) -> Optional[InstallationCandidate]:
894 """Try to find a Link matching req
895
896 Expects req, an InstallRequirement and upgrade, a boolean
897 Returns a InstallationCandidate if found,
898 Raises DistributionNotFound or BestVersionAlreadyInstalled otherwise
899 """
900 hashes = req.hashes(trust_internet=False)
901 best_candidate_result = self.find_best_candidate(
902 req.name,
903 specifier=req.specifier,
904 hashes=hashes,
905 )
906 best_candidate = best_candidate_result.best_candidate
907
908 installed_version: Optional[_BaseVersion] = None
909 if req.satisfied_by is not None:
910 installed_version = req.satisfied_by.version
911
912 def _format_versions(cand_iter: Iterable[InstallationCandidate]) -> str:
913 # This repeated parse_version and str() conversion is needed to
914 # handle different vendoring sources from pip and pkg_resources.
915 # If we stop using the pkg_resources provided specifier and start
916 # using our own, we can drop the cast to str().
917 return (
918 ", ".join(
919 sorted(
920 {str(c.version) for c in cand_iter},
921 key=parse_version,
922 )
923 )
924 or "none"
925 )
926
927 if installed_version is None and best_candidate is None:
928 logger.critical(
929 "Could not find a version that satisfies the requirement %s "
930 "(from versions: %s)",
931 req,
932 _format_versions(best_candidate_result.iter_all()),
933 )
934
935 raise DistributionNotFound(f"No matching distribution found for {req}")

937 def _should_install_candidate(
938 candidate: Optional[InstallationCandidate],
939 ) -> "TypeGuard[InstallationCandidate]":
940 if installed_version is None:
941 return True
942 if best_candidate is None:
943 return False
944 return best_candidate.version > installed_version
945
946 if not upgrade and installed_version is not None:
947 if _should_install_candidate(best_candidate):
948 logger.debug(
949 "Existing installed version (%s) satisfies requirement "
950 "(most up-to-date version is %s)",
951 installed_version,
952 best_candidate.version,
953 )
954 else:
955 logger.debug(
956 "Existing installed version (%s) is most up-to-date and "
957 "satisfies requirement",
958 installed_version,
959 )
960 return None
961
962 if _should_install_candidate(best_candidate):
963 logger.debug(
964 "Using version %s (newest of versions: %s)",
965 best_candidate.version,
966 _format_versions(best_candidate_result.iter_applicable()),
967 )
968 return best_candidate
969
970 # We have an existing version, and its the best version
971 logger.debug(
972 "Installed version (%s) is most up-to-date (past versions: %s)",
973 installed_version,
974 _format_versions(best_candidate_result.iter_applicable()),
975 )
976 raise BestVersionAlreadyInstalled

979def _find_name_version_sep(fragment: str, canonical_name: str) -> int:
980 """Find the separator's index based on the package's canonical name.
981
982 :param fragment: A <package>+<version> filename "fragment" (stem) or
983 egg fragment.
984 :param canonical_name: The package's canonical name.
985
986 This function is needed since the canonicalized name does not necessarily
987 have the same length as the egg info's name part. An example::
988
989 >>> fragment = 'foo__bar-1.0'
990 >>> canonical_name = 'foo-bar'
991 >>> _find_name_version_sep(fragment, canonical_name)
992 8
993 """
994 # Project name and version must be separated by one single dash. Find all
995 # occurrences of dashes; if the string in front of it matches the canonical
996 # name, this is the one separating the name and version parts.
997 for i, c in enumerate(fragment):
998 if c != "-":
999 continue
1000 if canonicalize_name(fragment[:i]) == canonical_name:
1001 return i
1002 raise ValueError(f"{fragment} does not match {canonical_name}")
1003

1005def _extract_version_from_fragment(fragment: str, canonical_name: str) -> Optional[str]:
1006 """Parse the version string from a <package>+<version> filename
1007 "fragment" (stem) or egg fragment.
1008
1009 :param fragment: The string to parse. E.g. foo-2.1
1010 :param canonical_name: The canonicalized name of the package this
1011 belongs to.
1012 """
1013 try:
1014 version_start = _find_name_version_sep(fragment, canonical_name) + 1
1015 except ValueError:
1016 return None
1017 version = fragment[version_start:]
1018 if not version:
1019 return None
1020 return version
