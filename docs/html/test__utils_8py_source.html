20def require_version(minver: str = "0.0.0", maxver: str = "4.0.0") -> Callable:
21 """Compare version of python interpreter to the given one and skips the test if older."""
22
23 def parse(python_version: str) -> tuple[int, ...]:
24 try:
25 return tuple(int(v) for v in python_version.split("."))
26 except ValueError as e:
27 msg = f"{python_version} is not a correct version : should be X.Y[.Z]."
28 raise ValueError(msg) from e
29
30 min_version = parse(minver)
31 max_version = parse(maxver)
32
33 def check_require_version(f):
34 current: tuple[int, int, int] = sys.version_info[:3]
35 if min_version < current <= max_version:
36 return f
37
38 version: str = ".".join(str(v) for v in sys.version_info)
39
40 @functools.wraps(f)
41 def new_f(*args, **kwargs):
42 if current <= min_version:
43 pytest.skip(f"Needs Python > {minver}. Current version is {version}.")
44 elif current > max_version:
45 pytest.skip(f"Needs Python <= {maxver}. Current version is {version}.")
46
47 return new_f
48
49 return check_require_version
