19BAD_FUNCTIONS = ["map", "filter"]
20# Some hints regarding the use of bad builtins.

22BUILTIN_HINTS = {"map": LIST_COMP_MSG, "filter": LIST_COMP_MSG}

51 @only_required_for_messages("bad-builtin")
52 def visit_call(self, node: nodes.Call) -> None:
53 if isinstance(node.func, nodes.Name):
54 name = node.func.name
55 # ignore the name if it's not a builtin (i.e. not defined in the
56 # locals nor globals scope)
57 if not (name in node.frame() or name in node.root()):
58 if name in self.linter.config.bad_functions:
59 hint = BUILTIN_HINTS.get(name)
60 args = f"{name!r}. {hint}" if hint else repr(name)
61 self.add_message("bad-builtin", node=node, args=args)
