140 def loaddata(self, fixture_labels):
141 connection = connections[self.using]
142
143 # Keep a count of the installed objects and fixtures
144 self.fixture_count = 0
145 self.loaded_object_count = 0
146 self.fixture_object_count = 0
147 self.models = set()
148
149 self.serialization_formats = serializers.get_public_serializer_formats()
150
151 # Django's test suite repeatedly tries to load initial_data fixtures
152 # from apps that don't have any fixtures. Because disabling constraint
153 # checks can be expensive on some database (especially MSSQL), bail
154 # out early if no fixtures are found.
155 for fixture_label in fixture_labels:
156 if self.find_fixtures(fixture_label):
157 break
158 else:
159 return
160
161 self.objs_with_deferred_fields = []
162 with connection.constraint_checks_disabled():
163 for fixture_label in fixture_labels:
164 self.load_label(fixture_label)
165 for obj in self.objs_with_deferred_fields:
166 obj.save_deferred_fields(using=self.using)
167
168 # Since we disabled constraint checks, we must manually check for
169 # any invalid keys that might have been added
170 table_names = [model._meta.db_table for model in self.models]
171 try:
172 connection.check_constraints(table_names=table_names)
173 except Exception as e:
174 e.args = ("Problem installing fixtures: %s" % e,)
175 raise
176
177 # If we found even one object in a fixture, we need to reset the
178 # database sequences.
179 if self.loaded_object_count > 0:
180 self.reset_sequences(connection, self.models)
181
182 if self.verbosity >= 1:
183 if self.fixture_object_count == self.loaded_object_count:
184 self.stdout.write(
185 "Installed %d object(s) from %d fixture(s)"
186 % (self.loaded_object_count, self.fixture_count)
187 )
188 else:
189 self.stdout.write(
190 "Installed %d object(s) (of %d) from %d fixture(s)"
191 % (
192 self.loaded_object_count,
193 self.fixture_object_count,
194 self.fixture_count,
195 )
196 )

198 def save_obj(self, obj):
199 """Save an object if permitted."""
200 if (
201 obj.object._meta.app_config in self.excluded_apps
202 or type(obj.object) in self.excluded_models
203 ):
204 return False
205 saved = False
206 if router.allow_migrate_model(self.using, obj.object.__class__):
207 saved = True
208 self.models.add(obj.object.__class__)
209 try:
210 obj.save(using=self.using)
211 # psycopg raises ValueError if data contains NUL chars.
212 except (DatabaseError, IntegrityError, ValueError) as e:
213 e.args = (
214 "Could not load %(object_label)s(pk=%(pk)s): %(error_msg)s"
215 % {
216 "object_label": obj.object._meta.label,
217 "pk": obj.object.pk,
218 "error_msg": e,
219 },
220 )
221 raise
222 if obj.deferred_fields:
223 self.objs_with_deferred_fields.append(obj)
224 return saved

226 def load_label(self, fixture_label):
227 """Load fixtures files for a given label."""
228 show_progress = self.verbosity >= 3
229 for fixture_file, fixture_dir, fixture_name in self.find_fixtures(
230 fixture_label
231 ):
232 _, ser_fmt, cmp_fmt = self.parse_name(os.path.basename(fixture_file))
233 open_method, mode = self.compression_formats[cmp_fmt]
234 fixture = open_method(fixture_file, mode)
235 self.fixture_count += 1
236 objects_in_fixture = 0
237 loaded_objects_in_fixture = 0
238 if self.verbosity >= 2:
239 self.stdout.write(
240 "Installing %s fixture '%s' from %s."
241 % (ser_fmt, fixture_name, humanize(fixture_dir))
242 )
243 try:
244 objects = serializers.deserialize(
245 ser_fmt,
246 fixture,
247 using=self.using,
248 ignorenonexistent=self.ignore,
249 handle_forward_references=True,
250 )
251
252 for obj in objects:
253 objects_in_fixture += 1
254 if self.save_obj(obj):
255 loaded_objects_in_fixture += 1
256 if show_progress:
257 self.stdout.write(
258 "\rProcessed %i object(s)." % loaded_objects_in_fixture,
259 ending="",
260 )
261 except Exception as e:
262 if not isinstance(e, CommandError):
263 e.args = (
264 "Problem installing fixture '%s': %s" % (fixture_file, e),
265 )
266 raise
267 finally:
268 fixture.close()
269
270 if objects_in_fixture and show_progress:
271 self.stdout.write() # Add a newline after progress indicator.
272 self.loaded_object_count += loaded_objects_in_fixture
273 self.fixture_object_count += objects_in_fixture
274 # Warn if the fixture we loaded contains 0 objects.
275 if objects_in_fixture == 0:
276 warnings.warn(
277 "No fixture data found for '%s'. (File format may be "
278 "invalid.)" % fixture_name,
279 RuntimeWarning,
280 )

316 def find_fixtures(self, fixture_label):
317 """Find fixture files for a given label."""
318 if fixture_label == READ_STDIN:
319 return [(READ_STDIN, None, READ_STDIN)]
320
321 fixture_name, ser_fmt, cmp_fmt = self.parse_name(fixture_label)
322 if self.verbosity >= 2:
323 self.stdout.write("Loading '%s' fixtures..." % fixture_name)
324
325 fixture_name, fixture_dirs = self.get_fixture_name_and_dirs(fixture_name)
326 targets = self.get_targets(fixture_name, ser_fmt, cmp_fmt)
327 fixture_files = []
328 for fixture_dir in fixture_dirs:
329 if self.verbosity >= 2:
330 self.stdout.write("Checking %s for fixtures..." % humanize(fixture_dir))
331 fixture_files_in_dir = self.find_fixture_files_in_dir(
332 fixture_dir,
333 fixture_name,
334 targets,
335 )
336 if self.verbosity >= 2 and not fixture_files_in_dir:
337 self.stdout.write(
338 "No fixture '%s' in %s." % (fixture_name, humanize(fixture_dir))
339 )
340
341 # Check kept for backwards-compatibility; it isn't clear why
342 # duplicates are only allowed in different directories.
343 if len(fixture_files_in_dir) > 1:
344 raise CommandError(
345 "Multiple fixtures named '%s' in %s. Aborting."
346 % (fixture_name, humanize(fixture_dir))
347 )
348 fixture_files.extend(fixture_files_in_dir)
349
350 if not fixture_files:
351 raise CommandError("No fixture named '%s' found." % fixture_name)
352
353 return fixture_files

355 @cached_property
356 def fixture_dirs(self):
357 """
358 Return a list of fixture directories.
359
360 The list contains the 'fixtures' subdirectory of each installed
361 application, if it exists, the directories in FIXTURE_DIRS, and the
362 current directory.
363 """
364 dirs = []
365 fixture_dirs = settings.FIXTURE_DIRS
366 if len(fixture_dirs) != len(set(fixture_dirs)):
367 raise ImproperlyConfigured("settings.FIXTURE_DIRS contains duplicates.")
368 for app_config in apps.get_app_configs():
369 app_label = app_config.label
370 app_dir = os.path.join(app_config.path, "fixtures")
371 if app_dir in [str(d) for d in fixture_dirs]:
372 raise ImproperlyConfigured(
373 "'%s' is a default fixture directory for the '%s' app "
374 "and cannot be listed in settings.FIXTURE_DIRS."
375 % (app_dir, app_label)
376 )
377
378 if self.app_label and app_label != self.app_label:
379 continue
380 if os.path.isdir(app_dir):
381 dirs.append(app_dir)
382 dirs.extend(fixture_dirs)
383 dirs.append("")
384 return [os.path.realpath(d) for d in dirs]

386 def parse_name(self, fixture_name):
387 """
388 Split fixture name in name, serialization format, compression format.
389 """
390 if fixture_name == READ_STDIN:
391 if not self.format:
392 raise CommandError(
393 "--format must be specified when reading from stdin."
394 )
395 return READ_STDIN, self.format, "stdin"
396
397 parts = fixture_name.rsplit(".", 2)
398
399 if len(parts) > 1 and parts[-1] in self.compression_formats:
400 cmp_fmt = parts[-1]
401 parts = parts[:-1]
402 else:
403 cmp_fmt = None
404
405 if len(parts) > 1:
406 if parts[-1] in self.serialization_formats:
407 ser_fmt = parts[-1]
408 parts = parts[:-1]
409 else:
410 raise CommandError(
411 "Problem installing fixture '%s': %s is not a known "
412 "serialization format." % (".".join(parts[:-1]), parts[-1])
413 )
414 else:
415 ser_fmt = None
416
417 name = ".".join(parts)
418
419 return name, ser_fmt, cmp_fmt
