200 def default_units(self, kwargs):
201 """
202 Return the unit value and the default units specified
203 from the given keyword arguments dictionary.
204 """
205 val = 0.0
206 default_unit = self.STANDARD_UNIT
207 for unit, value in kwargs.items():
208 if not isinstance(value, float):
209 value = float(value)
210 if unit in self.UNITS:
211 val += self.UNITS[unit] * value
212 default_unit = unit
213 elif unit in self.ALIAS:
214 u = self.ALIAS[unit]
215 val += self.UNITS[u] * value
216 default_unit = u
217 else:
218 lower = unit.lower()
219 if lower in self.UNITS:
220 val += self.UNITS[lower] * value
221 default_unit = lower
222 elif lower in self.LALIAS:
223 u = self.LALIAS[lower]
224 val += self.UNITS[u] * value
225 default_unit = u
226 else:
227 raise AttributeError("Unknown unit type: %s" % unit)
228 return val, default_unit

230 @classmethod
231 def unit_attname(cls, unit_str):
232 """
233 Retrieve the unit attribute name for the given unit string.
234 For example, if the given unit string is 'metre', return 'm'.
235 Raise an AttributeError if an attribute cannot be found.
236 """
237 lower = unit_str.lower()
238 if unit_str in cls.UNITS:
239 return unit_str
240 elif lower in cls.UNITS:
241 return lower
242 elif lower in cls.LALIAS:
243 return cls.LALIAS[lower]
244 else:
245 raise AttributeError(f"Unknown unit type: {unit_str}")
