12class NotChecker(checkers.BaseChecker):
13 """Checks for too many not in comparison expressions.
14
15 - "not not" should trigger a warning
16 - "not" followed by a comparison should trigger a warning
17 """
18
19 msgs = {
20 "C0117": (
21 'Consider changing "%s" to "%s"',
22 "unnecessary-negation",
23 "Used when a boolean expression contains an unneeded negation, "
24 "e.g. when two negation operators cancel each other out.",
25 {"old_names": [("C0113", "unneeded-not")]},
26 )
27 }
28 name = "refactoring"
29 reverse_op = {
30 "<": ">=",
31 "<=": ">",
32 ">": "<=",
33 ">=": "<",
34 "==": "!=",
35 "!=": "==",
36 "in": "not in",
37 "is": "is not",
38 }
39 # sets are not ordered, so for example "not set(LEFT_VALS) <= set(RIGHT_VALS)" is
40 # not equivalent to "set(LEFT_VALS) > set(RIGHT_VALS)"
41 skipped_nodes = (nodes.Set,)
42 # 'builtins' py3, '__builtin__' py2
43 skipped_classnames = [f"builtins.{qname}" for qname in ("set", "frozenset")]
44
45 @utils.only_required_for_messages("unnecessary-negation")
46 def visit_unaryop(self, node: nodes.UnaryOp) -> None:
47 if node.op != "not":
48 return
49 operand = node.operand
50
51 if isinstance(operand, nodes.UnaryOp) and operand.op == "not":
52 self.add_message(
53 "unnecessary-negation",
54 node=node,
55 args=(node.as_string(), operand.operand.as_string()),
56 )
57 elif isinstance(operand, nodes.Compare):
58 left = operand.left
59 # ignore multiple comparisons
60 if len(operand.ops) > 1:
61 return
62 operator, right = operand.ops[0]
63 if operator not in self.reverse_op:
64 return
65 # Ignore __ne__ as function of __eq__
66 frame = node.frame()
67 if frame.name == "__ne__" and operator == "==":
68 return
69 for _type in (utils.node_type(left), utils.node_type(right)):
70 if not _type:
71 return
72 if isinstance(_type, self.skipped_nodes):
73 return
74 if (
75 isinstance(_type, astroid.Instance)
76 and _type.qname() in self.skipped_classnames
77 ):
78 return
79 suggestion = (
80 f"{left.as_string()} {self.reverse_op[operator]} {right.as_string()}"
81 )
82 self.add_message(
83 "unnecessary-negation", node=node, args=(node.as_string(), suggestion)
84 )
