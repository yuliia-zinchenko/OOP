54class CursorWrapper:
55 """
56 A thin wrapper around MySQLdb's normal cursor class that catches particular
57 exception instances and reraises them with the correct types.
58
59 Implemented as a wrapper, rather than a subclass, so that it isn't stuck
60 to the particular underlying representation returned by Connection.cursor().
61 """

63 codes_for_integrityerror = (
64 1048, # Column cannot be null
65 1690, # BIGINT UNSIGNED value is out of range
66 3819, # CHECK constraint is violated
67 4025, # CHECK constraint failed
68 )

70 def __init__(self, cursor):
71 self.cursor = cursor

73 def execute(self, query, args=None):
74 try:
75 # args is None means no string interpolation
76 return self.cursor.execute(query, args)
77 except Database.OperationalError as e:
78 # Map some error codes to IntegrityError, since they seem to be
79 # misclassified and Django would prefer the more logical place.
80 if e.args[0] in self.codes_for_integrityerror:
81 raise IntegrityError(*tuple(e.args))
82 raise
83
84 def executemany(self, query, args):
85 try:
86 return self.cursor.executemany(query, args)

89 # misclassified and Django would prefer the more logical place.
90 if e.args[0] in self.codes_for_integrityerror:
91 raise IntegrityError(*tuple(e.args))

215 def get_connection_params(self):
216 kwargs = {
217 "conv": django_conversions,
218 "charset": "utf8",
219 }
220 settings_dict = self.settings_dict
221 if settings_dict["USER"]:
222 kwargs["user"] = settings_dict["USER"]
223 if settings_dict["NAME"]:
224 kwargs["database"] = settings_dict["NAME"]
225 if settings_dict["PASSWORD"]:
226 kwargs["password"] = settings_dict["PASSWORD"]
227 if settings_dict["HOST"].startswith("/"):
228 kwargs["unix_socket"] = settings_dict["HOST"]
229 elif settings_dict["HOST"]:
230 kwargs["host"] = settings_dict["HOST"]
231 if settings_dict["PORT"]:
232 kwargs["port"] = int(settings_dict["PORT"])
233 # We need the number of potentially affected rows after an
234 # "UPDATE", not the number of changed rows.
235 kwargs["client_flag"] = CLIENT.FOUND_ROWS
236 # Validate the transaction isolation level, if specified.
237 options = settings_dict["OPTIONS"].copy()
238 isolation_level = options.pop("isolation_level", "read committed")
239 if isolation_level:
240 isolation_level = isolation_level.lower()
241 if isolation_level not in self.isolation_levels:
242 raise ImproperlyConfigured(
243 "Invalid transaction isolation level '%s' specified.\n"
244 "Use one of %s, or None."
245 % (
246 isolation_level,
247 ", ".join("'%s'" % s for s in sorted(self.isolation_levels)),
248 )
249 )
250 self.isolation_level = isolation_level
251 kwargs.update(options)
252 return kwargs

265 def init_connection_state(self):
266 super().init_connection_state()

269 # SQL_AUTO_IS_NULL controls whether an AUTO_INCREMENT column on

300 def disable_constraint_checking(self):
301 """
302 Disable foreign key checks, primarily for use in adding rows with
303 forward references. Always return True to indicate constraint checks
304 need to be re-enabled.
305 """
306 with self.cursor() as cursor:
307 cursor.execute("SET foreign_key_checks=0")

312 Re-enable foreign key checks after they have been disabled.

315 # nested inside transaction.atomic.
316 self.needs_rollback, needs_rollback = False, self.needs_rollback

323 def check_constraints(self, table_names=None):
324 """
325 Check each table name in `table_names` for rows with invalid foreign
326 key references. This method is intended to be used in conjunction with
327 `disable_constraint_checking()` and `enable_constraint_checking()`, to
328 determine if rows with invalid references were entered while constraint
329 checks were off.
330 """
331 with self.cursor() as cursor:
332 if table_names is None:
333 table_names = self.introspection.table_names(cursor)
334 for table_name in table_names:
335 primary_key_column_name = self.introspection.get_primary_key_column(
336 cursor, table_name
337 )
338 if not primary_key_column_name:
339 continue
340 relations = self.introspection.get_relations(cursor, table_name)
341 for column_name, (
342 referenced_column_name,
343 referenced_table_name,
344 ) in relations.items():
345 cursor.execute(
346 """
347 SELECT REFERRING.`%s`, REFERRING.`%s` FROM `%s` as REFERRING
348 LEFT JOIN `%s` as REFERRED
349 ON (REFERRING.`%s` = REFERRED.`%s`)
350 WHERE REFERRING.`%s` IS NOT NULL AND REFERRED.`%s` IS NULL
351 """
352 % (
353 primary_key_column_name,
354 column_name,
355 table_name,
356 referenced_table_name,
357 column_name,
358 referenced_column_name,
359 column_name,
360 referenced_column_name,
361 )
362 )
363 for bad_row in cursor.fetchall():
364 raise IntegrityError(
365 "The row in table '%s' with primary key '%s' has an "
366 "invalid foreign key: %s.%s contains a value '%s' that "
367 "does not have a corresponding value in %s.%s."
368 % (
369 table_name,
370 bad_row[0],
371 table_name,
372 column_name,
373 bad_row[1],
374 referenced_table_name,
375 referenced_column_name,
376 )
377 )
