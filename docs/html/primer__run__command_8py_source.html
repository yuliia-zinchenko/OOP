58 @staticmethod
59 def _filter_fatal_errors(
60 messages: list[OldJsonExport],
61 ) -> list[Message]:
62 """Separate fatal errors so we can report them independently."""
63 fatal_msgs: list[Message] = []
64 for raw_message in messages:
65 message = JSONReporter.deserialize(raw_message)
66 if message.category == "fatal":
67 if GITHUB_CRASH_TEMPLATE_LOCATION in message.msg:
68 # Remove the crash template location if we're running on GitHub.
69 # We were falsely getting "new" errors when the timestamp changed.
70 message.msg = message.msg.rsplit(CRASH_TEMPLATE_INTRO)[0]
71 fatal_msgs.append(message)
72 return fatal_msgs

81 # We want to test all the code we can

83 # Duplicate code takes too long and is relatively safe

96 readable_messages: str = output.getvalue()

100 print(f"Successfully primed {package_name}.")
101 else:
102 fatal_msgs = self._filter_fatal_errors(messages)
103 if fatal_msgs:

105 f"Encountered fatal errors while priming {package_name} !\n"
