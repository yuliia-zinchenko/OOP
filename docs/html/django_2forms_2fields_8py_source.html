706class ImageField(FileField):
707 default_validators = [validators.validate_image_file_extension]
708 default_error_messages = {
709 "invalid_image": _(
710 "Upload a valid image. The file you uploaded was either not an "
711 "image or a corrupted image."
712 ),
713 }
714
715 def to_python(self, data):
716 """
717 Check that the file-upload field data contains a valid image (GIF, JPG,
718 PNG, etc. -- whatever Pillow supports).
719 """
720 f = super().to_python(data)
721 if f is None:
722 return None
723
724 from PIL import Image
725
726 # We need to get a file object for Pillow. We might have a path or we might
727 # have to read the data into memory.
728 if hasattr(data, "temporary_file_path"):
729 file = data.temporary_file_path()
730 else:
731 if hasattr(data, "read"):
732 file = BytesIO(data.read())
733 else:
734 file = BytesIO(data["content"])
735
736 try:
737 # load() could spot a truncated JPEG, but it loads the entire
738 # image in memory, which is a DoS vector. See #3848 and #18520.
739 image = Image.open(file)
740 # verify() must be called immediately after the constructor.
741 image.verify()
742
743 # Annotating so subclasses can reuse it for their own validation
744 f.image = image
745 # Pillow doesn't detect the MIME type of all formats. In those
746 # cases, content_type will be None.
747 f.content_type = Image.MIME.get(image.format)
748 except Exception as exc:
749 # Pillow doesn't recognize it as an image.
750 raise ValidationError(
751 self.error_messages["invalid_image"],
752 code="invalid_image",
753 ) from exc
754 if hasattr(f, "seek") and callable(f.seek):
755 f.seek(0)
756 return f

759 attrs = super().widget_attrs(widget)
760 if isinstance(widget, FileInput) and "accept" not in widget.attrs:
761 attrs.setdefault("accept", "image/*")

1070class MultiValueField(Field):
1071 """
1072 Aggregate the logic of multiple Fields.
1073
1074 Its clean() method takes a "decompressed" list of values, which are then
1075 cleaned into a single value according to self.fields. Each value in
1076 this list is cleaned by the corresponding field -- the first value is
1077 cleaned by the first field, the second value is cleaned by the second
1078 field, etc. Once all fields are cleaned, the list of clean values is
1079 "compressed" into a single value.
1080
1081 Subclasses should not have to implement clean(). Instead, they must
1082 implement compress(), which takes a list of valid values and returns a
1083 "compressed" version of those values -- a single value.
1084
1085 You'll probably want to use this with MultiWidget.
1086 """
1087
1088 default_error_messages = {
1089 "invalid": _("Enter a list of values."),
1090 "incomplete": _("Enter a complete value."),
1091 }
1092
1093 def __init__(self, fields, *, require_all_fields=True, **kwargs):
1094 self.require_all_fields = require_all_fields
1095 super().__init__(**kwargs)
1096 for f in fields:
1097 f.error_messages.setdefault("incomplete", self.error_messages["incomplete"])
1098 if self.disabled:
1099 f.disabled = True
1100 if self.require_all_fields:
1101 # Set 'required' to False on the individual fields, because the
1102 # required validation will be handled by MultiValueField, not
1103 # by those individual fields.
1104 f.required = False
1105 self.fields = fields
1106
1107 def __deepcopy__(self, memo):
1108 result = super().__deepcopy__(memo)
1109 result.fields = tuple(x.__deepcopy__(memo) for x in self.fields)
1110 return result
1111
1112 def validate(self, value):
1113 pass
1114
1115 def clean(self, value):
1116 """
1117 Validate every value in the given list. A value is validated against
1118 the corresponding Field in self.fields.
1119
1120 For example, if this MultiValueField was instantiated with
1121 fields=(DateField(), TimeField()), clean() would call
1122 DateField.clean(value[0]) and TimeField.clean(value[1]).
1123 """
1124 clean_data = []
1125 errors = []
1126 if self.disabled and not isinstance(value, list):
1127 value = self.widget.decompress(value)
1128 if not value or isinstance(value, (list, tuple)):
1129 if not value or not [v for v in value if v not in self.empty_values]:
1130 if self.required:
1131 raise ValidationError(
1132 self.error_messages["required"], code="required"
1133 )
1134 else:
1135 return self.compress([])
1136 else:
1137 raise ValidationError(self.error_messages["invalid"], code="invalid")
1138 for i, field in enumerate(self.fields):
1139 try:
1140 field_value = value[i]
1141 except IndexError:
1142 field_value = None
1143 if field_value in self.empty_values:
1144 if self.require_all_fields:
1145 # Raise a 'required' error if the MultiValueField is
1146 # required and any field is empty.
1147 if self.required:
1148 raise ValidationError(
1149 self.error_messages["required"], code="required"
1150 )
1151 elif field.required:
1152 # Otherwise, add an 'incomplete' error to the list of
1153 # collected errors and skip field cleaning, if a required
1154 # field is empty.
1155 if field.error_messages["incomplete"] not in errors:
1156 errors.append(field.error_messages["incomplete"])
1157 continue
1158 try:
1159 clean_data.append(field.clean(field_value))
1160 except ValidationError as e:
1161 # Collect all validation errors in a single list, which we'll
1162 # raise at the end of clean(), rather than raising a single
1163 # exception for the first error we encounter. Skip duplicates.
1164 errors.extend(m for m in e.error_list if m not in errors)
1165 if errors:
1166 raise ValidationError(errors)
1167
1168 out = self.compress(clean_data)
1169 self.validate(out)
1170 self.run_validators(out)
1171 return out
1172
1173 def compress(self, data_list):
1174 """
1175 Return a single value for the given list of values. The values can be
1176 assumed to be valid.
1177
1178 For example, if this MultiValueField was instantiated with
1179 fields=(DateField(), TimeField()), this might return a datetime
1180 object created by combining the date and time in data_list.
1181 """
1182 raise NotImplementedError("Subclasses must implement this method.")
1183
1184 def has_changed(self, initial, data):
1185 if self.disabled:
1186 return False
1187 if initial is None:
1188 initial = ["" for x in range(0, len(data))]
1189 else:
1190 if not isinstance(initial, list):
1191 initial = self.widget.decompress(initial)
1192 for field, initial, data in zip(self.fields, initial, data):
1193 try:
1194 initial = field.to_python(initial)
1195 except ValidationError:
1196 return True
1197 if field.has_changed(initial, data):
1198 return True
1199 return False
