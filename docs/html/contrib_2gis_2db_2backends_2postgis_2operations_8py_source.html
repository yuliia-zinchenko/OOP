28 def __init__(self, geography=False, raster=False, **kwargs):
29 # Only a subset of the operators and functions are available for the
30 # geography type. Lookups that don't support geography will be cast to
31 # geometry.
32 self.geography = geography
33 # Only a subset of the operators and functions are available for the
34 # raster type. Lookups that don't support raster will be converted to
35 # polygons. If the raster argument is set to BILATERAL, then the
36 # operator cannot handle mixed geom-raster lookups.
37 self.raster = raster
38 super().__init__(**kwargs)

40 def as_sql(self, connection, lookup, template_params, *args):
41 template_params = self.check_raster(lookup, template_params)
42 template_params = self.check_geography(lookup, template_params)
43 return super().as_sql(connection, lookup, template_params, *args)

45 def check_raster(self, lookup, template_params):
46 spheroid = lookup.rhs_params and lookup.rhs_params[-1] == "spheroid"
47
48 # Check which input is a raster.
49 lhs_is_raster = lookup.lhs.field.geom_type == "RASTER"
50 rhs_is_raster = isinstance(lookup.rhs, GDALRaster)
51
52 # Look for band indices and inject them if provided.
53 if lookup.band_lhs is not None and lhs_is_raster:
54 if not self.func:
55 raise ValueError(
56 "Band indices are not allowed for this operator, it works on bbox "
57 "only."
58 )
59 template_params["lhs"] = "%s, %s" % (
60 template_params["lhs"],
61 lookup.band_lhs,
62 )
63
64 if lookup.band_rhs is not None and rhs_is_raster:
65 if not self.func:
66 raise ValueError(
67 "Band indices are not allowed for this operator, it works on bbox "
68 "only."
69 )
70 template_params["rhs"] = "%s, %s" % (
71 template_params["rhs"],
72 lookup.band_rhs,
73 )
74
75 # Convert rasters to polygons if necessary.
76 if not self.raster or spheroid:
77 # Operators without raster support.
78 if lhs_is_raster:
79 template_params["lhs"] = "ST_Polygon(%s)" % template_params["lhs"]
80 if rhs_is_raster:
81 template_params["rhs"] = "ST_Polygon(%s)" % template_params["rhs"]
82 elif self.raster == BILATERAL:
83 # Operators with raster support but don't support mixed (rast-geom)
84 # lookups.
85 if lhs_is_raster and not rhs_is_raster:
86 template_params["lhs"] = "ST_Polygon(%s)" % template_params["lhs"]
87 elif rhs_is_raster and not lhs_is_raster:
88 template_params["rhs"] = "ST_Polygon(%s)" % template_params["rhs"]
89
90 return template_params

92 def check_geography(self, lookup, template_params):
93 """Convert geography fields to geometry types, if necessary."""
94 if lookup.lhs.output_field.geography and not self.geography:
95 template_params["lhs"] += "::geometry"
96 return template_params

183 @cached_property
184 def spatial_version(self):
185 """Determine the version of the PostGIS library."""
186 # Trying to get the PostGIS version because the function
187 # signatures will depend on the version used. The cost
188 # here is a database query to determine the version, which
189 # can be mitigated by setting `POSTGIS_VERSION` with a 3-tuple
190 # comprising user-supplied values for the major, minor, and
191 # subminor revision of PostGIS.
192 if hasattr(settings, "POSTGIS_VERSION"):
193 version = settings.POSTGIS_VERSION
194 else:
195 # Run a basic query to check the status of the connection so we're
196 # sure we only raise the error below if the problem comes from
197 # PostGIS and not from PostgreSQL itself (see #24862).
198 self._get_postgis_func("version")
199
200 try:
201 vtup = self.postgis_version_tuple()
202 except ProgrammingError:
203 raise ImproperlyConfigured(
204 'Cannot determine PostGIS version for database "%s" '
205 'using command "SELECT postgis_lib_version()". '
206 "GeoDjango requires at least PostGIS version 3.0. "
207 "Was the database created from a spatial database "
208 "template?" % self.connection.settings_dict["NAME"]
209 )
210 version = vtup[1:]
211 return version

213 def convert_extent(self, box):
214 """
215 Return a 4-tuple extent for the `Extent` aggregate by converting
216 the bounding box text returned by PostGIS (`box` argument), for
217 example: "BOX(-90.0 30.0, -85.0 40.0)".
218 """
219 if box is None:
220 return None
221 ll, ur = box[4:-1].split(",")
222 xmin, ymin = map(float, ll.split())
223 xmax, ymax = map(float, ur.split())
224 return (xmin, ymin, xmax, ymax)

226 def convert_extent3d(self, box3d):
227 """
228 Return a 6-tuple extent for the `Extent3D` aggregate by converting
229 the 3d bounding-box text returned by PostGIS (`box3d` argument), for
230 example: "BOX3D(-90.0 30.0 1, -85.0 40.0 2)".
231 """
232 if box3d is None:
233 return None
234 ll, ur = box3d[6:-1].split(",")
235 xmin, ymin, zmin = map(float, ll.split())
236 xmax, ymax, zmax = map(float, ur.split())
237 return (xmin, ymin, zmin, xmax, ymax, zmax)

262 def get_distance(self, f, dist_val, lookup_type):
263 """
264 Retrieve the distance parameters for the given geometry field,
265 distance lookup value, and the distance lookup type.
266
267 This is the most complex implementation of the spatial backends due to
268 what is supported on geodetic geometry columns vs. what's available on
269 projected geometry columns. In addition, it has to take into account
270 the geography column type.
271 """
272 # Getting the distance parameter
273 value = dist_val[0]
274
275 # Shorthand boolean flags.
276 geodetic = f.geodetic(self.connection)
277 geography = f.geography
278
279 if isinstance(value, Distance):
280 if geography:
281 dist_param = value.m
282 elif geodetic:
283 if lookup_type == "dwithin":
284 raise ValueError(
285 "Only numeric values of degree units are "
286 "allowed on geographic DWithin queries."
287 )
288 dist_param = value.m
289 else:
290 dist_param = getattr(
291 value, Distance.unit_attname(f.units_name(self.connection))
292 )
293 else:
294 # Assuming the distance is in the units of the field.
295 dist_param = value
296
297 return [dist_param]

365 def proj_version_tuple(self):
366 """
367 Return the version of PROJ used by PostGIS as a tuple of the
368 major, minor, and subminor release numbers.
369 """
370 proj_regex = re.compile(r"(\d+)\.(\d+)\.(\d+)")
371 proj_ver_str = self.postgis_proj_version()
372 m = proj_regex.search(proj_ver_str)
373 if m:
374 return tuple(map(int, m.groups()))
