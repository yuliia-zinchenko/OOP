210def get_access_path(key: str | Literal[0], parts: list[tuple[bool, str]]) -> str:
211 """Given a list of format specifiers, returns
212 the final access path (e.g. a.b.c[0][1]).
213 """
214 path = []
215 for is_attribute, specifier in parts:
216 if is_attribute:
217 path.append(f".{specifier}")
218 else:
219 path.append(f"[{specifier!r}]")
220 return str(key) + "".join(path)

223def arg_matches_format_type(
224 arg_type: SuccessfulInferenceResult, format_type: str
225) -> bool:
226 if format_type in "sr":
227 # All types can be printed with %s and %r
228 return True
229 if isinstance(arg_type, astroid.Instance):
230 arg_type = arg_type.pytype()
231 if arg_type == "builtins.str":
232 return format_type == "c"
233 if arg_type == "builtins.float":
234 return format_type in "deEfFgGn%"
235 if arg_type == "builtins.int":
236 # Integers allow all types
237 return True
238 return False
239 return True

251 @only_required_for_messages(
252 "bad-format-character",
253 "truncated-format-string",
254 "mixed-format-string",
255 "bad-format-string-key",
256 "missing-format-string-key",
257 "unused-format-string-key",

259 "format-needs-mapping",

264 def visit_binop(self, node: nodes.BinOp) -> None:
265 if node.op != "%":
266 return
267 left = node.left
268 args = node.right
269
270 if not (isinstance(left, nodes.Const) and isinstance(left.value, str)):
271 return
272 format_string = left.value
273 try:
274 (
275 required_keys,
276 required_num_args,
277 required_key_types,
278 required_arg_types,
279 ) = utils.parse_format_string(format_string)
280 except utils.UnsupportedFormatCharacter as exc:
281 formatted = format_string[exc.index]
282 self.add_message(
283 "bad-format-character",
284 node=node,
285 args=(formatted, ord(formatted), exc.index),
286 )
287 return
288 except utils.IncompleteFormatString:
289 self.add_message("truncated-format-string", node=node)
290 return
291 if not required_keys and not required_num_args:
292 self.add_message("format-string-without-interpolation", node=node)
293 return
294 if required_keys and required_num_args:
295 # The format string uses both named and unnamed format
296 # specifiers.
297 self.add_message("mixed-format-string", node=node)
298 elif required_keys:
299 # The format string uses only named format specifiers.
300 # Check that the RHS of the % operator is a mapping object
301 # that contains precisely the set of keys required by the
302 # format string.
303 if isinstance(args, nodes.Dict):
304 keys = set()
305 unknown_keys = False
306 for k, _ in args.items:
307 if isinstance(k, nodes.Const):
308 key = k.value
309 if isinstance(key, str):
310 keys.add(key)
311 else:
312 self.add_message(
313 "bad-format-string-key", node=node, args=key
314 )
315 else:
316 # One of the keys was something other than a
317 # constant. Since we can't tell what it is,
318 # suppress checks for missing keys in the
319 # dictionary.
320 unknown_keys = True
321 if not unknown_keys:
322 for key in required_keys:
323 if key not in keys:
324 self.add_message(
325 "missing-format-string-key", node=node, args=key
326 )
327 for key in keys:
328 if key not in required_keys:
329 self.add_message(
330 "unused-format-string-key", node=node, args=key
331 )
332 for key, arg in args.items:
333 if not isinstance(key, nodes.Const):
334 continue
335 format_type = required_key_types.get(key.value, None)
336 arg_type = utils.safe_infer(arg)
337 if (
338 format_type is not None
339 and arg_type
340 and not isinstance(arg_type, util.UninferableBase)
341 and not arg_matches_format_type(arg_type, format_type)
342 ):
343 self.add_message(
344 "bad-string-format-type",
345 node=node,
346 args=(arg_type.pytype(), format_type),
347 )
348 elif isinstance(args, (OTHER_NODES, nodes.Tuple)):
349 type_name = type(args).__name__
350 self.add_message("format-needs-mapping", node=node, args=type_name)
351 # else:
352 # The RHS of the format specifier is a name or
353 # expression. It may be a mapping object, so
354 # there's nothing we can check.
355 else:
356 # The format string uses only unnamed format specifiers.
357 # Check that the number of arguments passed to the RHS of
358 # the % operator matches the number required by the format
359 # string.
360 args_elts = []
361 if isinstance(args, nodes.Tuple):
362 rhs_tuple = utils.safe_infer(args)
363 num_args = None
364 if isinstance(rhs_tuple, nodes.BaseContainer):
365 args_elts = rhs_tuple.elts
366 num_args = len(args_elts)
367 elif isinstance(args, (OTHER_NODES, (nodes.Dict, nodes.DictComp))):
368 args_elts = [args]
369 num_args = 1
370 elif isinstance(args, nodes.Name):
371 inferred = utils.safe_infer(args)
372 if isinstance(inferred, nodes.Tuple):
373 # The variable is a tuple, so we need to get the elements
374 # from it for further inspection
375 args_elts = inferred.elts
376 num_args = len(args_elts)
377 elif isinstance(inferred, nodes.Const):
378 args_elts = [inferred]
379 num_args = 1
380 else:
381 num_args = None
382 else:
383 # The RHS of the format specifier is an expression.
384 # It could be a tuple of unknown size, so
385 # there's nothing we can check.
386 num_args = None
387 if num_args is not None:
388 if num_args > required_num_args:
389 self.add_message("too-many-format-args", node=node)
390 elif num_args < required_num_args:
391 self.add_message("too-few-format-args", node=node)
392 for arg, format_type in zip(args_elts, required_arg_types):
393 if not arg:
394 continue
395 arg_type = utils.safe_infer(arg)
396 if (
397 arg_type
398 and not isinstance(arg_type, util.UninferableBase)
399 and not arg_matches_format_type(arg_type, format_type)
400 ):
401 self.add_message(
402 "bad-string-format-type",
403 node=node,
404 args=(arg_type.pytype(), format_type),
405 )

452 def _check_new_format(self, node: nodes.Call, func: bases.BoundMethod) -> None:
453 """Check the new string formatting."""
454 # Skip format nodes which don't have an explicit string on the
455 # left side of the format operation.
456 # We do this because our inference engine can't properly handle
457 # redefinition of the original string.
458 # Note that there may not be any left side at all, if the format method
459 # has been assigned to another variable. See issue 351. For example:
460 #
461 # fmt = 'some string {}'.format
462 # fmt('arg')
463 if isinstance(node.func, nodes.Attribute) and not isinstance(
464 node.func.expr, nodes.Const
465 ):
466 return
467 if node.starargs or node.kwargs:
468 return
469 try:
470 strnode = next(func.bound.infer())
471 except astroid.InferenceError:
472 return
473 if not (isinstance(strnode, nodes.Const) and isinstance(strnode.value, str)):
474 return
475 try:
476 call_site = astroid.arguments.CallSite.from_call(node)
477 except astroid.InferenceError:
478 return
479
480 try:
481 fields, num_args, manual_pos = utils.parse_format_method_string(
482 strnode.value
483 )
484 except utils.IncompleteFormatString:
485 self.add_message("bad-format-string", node=node)
486 return
487
488 positional_arguments = call_site.positional_arguments
489 named_arguments = call_site.keyword_arguments
490 named_fields = {field[0] for field in fields if isinstance(field[0], str)}
491 if num_args and manual_pos:
492 self.add_message("format-combined-specification", node=node)
493 return
494
495 check_args = False
496 # Consider "{[0]} {[1]}" as num_args.
497 num_args += sum(1 for field in named_fields if not field)
498 if named_fields:
499 for field in named_fields:
500 if field and field not in named_arguments:
501 self.add_message(
502 "missing-format-argument-key", node=node, args=(field,)
503 )
504 for field in named_arguments:
505 if field not in named_fields:
506 self.add_message(
507 "unused-format-string-argument", node=node, args=(field,)
508 )
509 # num_args can be 0 if manual_pos is not.
510 num_args = num_args or manual_pos
511 if positional_arguments or num_args:
512 empty = not all(field for field in named_fields)
513 if named_arguments or empty:
514 # Verify the required number of positional arguments
515 # only if the .format got at least one keyword argument.
516 # This means that the format strings accepts both
517 # positional and named fields and we should warn
518 # when one of them is missing or is extra.
519 check_args = True
520 else:
521 check_args = True
522 if check_args:
523 # num_args can be 0 if manual_pos is not.
524 num_args = num_args or manual_pos
525 if not num_args:
526 self.add_message("format-string-without-interpolation", node=node)
527 return
528 if len(positional_arguments) > num_args:
529 self.add_message("too-many-format-args", node=node)
530 elif len(positional_arguments) < num_args:
531 self.add_message("too-few-format-args", node=node)
532
533 self._detect_vacuous_formatting(node, positional_arguments)
534 self._check_new_format_specifiers(node, fields, named_arguments)

536 # pylint: disable = too-many-statements
537 def _check_new_format_specifiers(
538 self,
539 node: nodes.Call,
540 fields: list[tuple[str, list[tuple[bool, str]]]],
541 named: dict[str, SuccessfulInferenceResult],
542 ) -> None:
543 """Check attribute and index access in the format
544 string ("{0.a}" and "{0[a]}").
545 """
546 key: Literal[0] | str
547 for key, specifiers in fields:
548 # Obtain the argument. If it can't be obtained
549 # or inferred, skip this check.
550 if not key:
551 # {[0]} will have an unnamed argument, defaulting
552 # to 0. It will not be present in `named`, so use the value
553 # 0 for it.
554 key = 0
555 if isinstance(key, int):
556 try:
557 argname = utils.get_argument_from_call(node, key)
558 except utils.NoSuchArgumentError:
559 continue
560 else:
561 if key not in named:
562 continue
563 argname = named[key]
564 if argname is None or isinstance(argname, util.UninferableBase):
565 continue
566 try:
567 argument = utils.safe_infer(argname)
568 except astroid.InferenceError:
569 continue
570 if not specifiers or not argument:
571 # No need to check this key if it doesn't
572 # use attribute / item access
573 continue
574 if argument.parent and isinstance(argument.parent, nodes.Arguments):
575 # Ignore any object coming from an argument,
576 # because we can't infer its value properly.
577 continue
578 previous = argument
579 parsed: list[tuple[bool, str]] = []
580 for is_attribute, specifier in specifiers:
581 if isinstance(previous, util.UninferableBase):
582 break
583 parsed.append((is_attribute, specifier))
584 if is_attribute:
585 try:
586 previous = previous.getattr(specifier)[0]
587 except astroid.NotFoundError:
588 if (
589 hasattr(previous, "has_dynamic_getattr")
590 and previous.has_dynamic_getattr()
591 ):
592 # Don't warn if the object has a custom __getattr__
593 break
594 path = get_access_path(key, parsed)
595 self.add_message(
596 "missing-format-attribute",
597 args=(specifier, path),
598 node=node,
599 )
600 break
601 else:
602 warn_error = False
603 if hasattr(previous, "getitem"):
604 try:
605 previous = previous.getitem(nodes.Const(specifier))
606 except (
607 astroid.AstroidIndexError,
608 astroid.AstroidTypeError,
609 astroid.AttributeInferenceError,
610 ):
611 warn_error = True
612 except astroid.InferenceError:
613 break
614 if isinstance(previous, util.UninferableBase):
615 break
616 else:
617 try:
618 # Lookup __getitem__ in the current node,
619 # but skip further checks, because we can't
620 # retrieve the looked object
621 previous.getattr("__getitem__")
622 break
623 except astroid.NotFoundError:
624 warn_error = True
625 if warn_error:
626 path = get_access_path(key, parsed)
627 self.add_message(
628 "invalid-format-index", args=(specifier, path), node=node
629 )
630 break
631
632 try:
633 previous = next(previous.infer())
634 except astroid.InferenceError:
635 # can't check further if we can't infer it
636 break

727 if token_type == tokenize.ENCODING:
728 # this is always the first token processed
729 encoding = token
730 elif token_type == tokenize.STRING:
731 # 'token' is the whole un-parsed token; we can look at the start
732 # of it to see whether it's a raw or unicode string etc.
733 self.process_string_token(token, start[0], start[1])
734 # We figure the next token, ignoring comments & newlines:
735 j = i + 1
736 while j < len(tokens) and tokens[j].type in (
737 tokenize.NEWLINE,
738 tokenize.NL,
739 tokenize.COMMENT,
740 ):
741 j += 1
742 next_token = tokens[j] if j < len(tokens) else None
743 if encoding != "ascii":
744 # We convert `tokenize` character count into a byte count,
745 # to match with astroid `.col_offset`
746 start = (start[0], len(line[: start[1]].encode(encoding)))
747 self.string_tokens[start] = (str_eval(token), next_token)
748 is_parenthesized = self._is_initial_string_token(
749 i, tokens
750 ) and self._is_parenthesized(i, tokens)
751 self._parenthesized_string_tokens[start] = is_parenthesized
752
753 if self.linter.config.check_quote_consistency:
754 self.check_for_consistent_string_delimiters(tokens)

756 def _is_initial_string_token(
757 self, index: int, tokens: Sequence[tokenize.TokenInfo]
758 ) -> bool:
759 # Must NOT be preceded by a string literal
760 prev_token = self._find_prev_token(index, tokens)
761 if prev_token and prev_token.type == tokenize.STRING:
762 return False
763 # Must be followed by a string literal token.
764 next_token = self._find_next_token(index, tokens)
765 return bool(next_token and next_token.type == tokenize.STRING)

767 def _is_parenthesized(self, index: int, tokens: list[tokenize.TokenInfo]) -> bool:
768 prev_token = self._find_prev_token(
769 index, tokens, ignore=(*_PAREN_IGNORE_TOKEN_TYPES, tokenize.STRING)
770 )
771 if not prev_token or prev_token.type != tokenize.OP or prev_token[1] != "(":
772 return False
773 next_token = self._find_next_token(
774 index, tokens, ignore=(*_PAREN_IGNORE_TOKEN_TYPES, tokenize.STRING)
775 )
776 return bool(
777 next_token and next_token.type == tokenize.OP and next_token[1] == ")"
778 )

824 def check_for_consistent_string_delimiters(
825 self, tokens: Iterable[tokenize.TokenInfo]
826 ) -> None:
827 """Adds a message for each string using inconsistent quote delimiters.
828
829 Quote delimiters are used inconsistently if " and ' are mixed in a module's
830 shortstrings without having done so to avoid escaping an internal quote
831 character.
832
833 Args:
834 tokens: The tokens to be checked against for consistent usage.
835 """
836 string_delimiters: Counter[str] = collections.Counter()
837
838 inside_fstring = False # whether token is inside f-string (since 3.12)
839 target_py312 = self.linter.config.py_version >= (3, 12)
840
841 # First, figure out which quote character predominates in the module
842 for tok_type, token, _, _, _ in tokens:
843 if sys.version_info[:2] >= (3, 12):
844 # pylint: disable=no-member,useless-suppression
845 if tok_type == tokenize.FSTRING_START:
846 inside_fstring = True
847 elif tok_type == tokenize.FSTRING_END:
848 inside_fstring = False
849
850 if inside_fstring and not target_py312:
851 # skip analysis of f-string contents
852 continue
853
854 if tok_type == tokenize.STRING and _is_quote_delimiter_chosen_freely(token):
855 string_delimiters[_get_quote_delimiter(token)] += 1
856
857 if len(string_delimiters) > 1:
858 # Ties are broken arbitrarily
859 most_common_delimiter = string_delimiters.most_common(1)[0][0]
860 for tok_type, token, start, _, _ in tokens:
861 if tok_type != tokenize.STRING:
862 continue
863 quote_delimiter = _get_quote_delimiter(token)
864 if (
865 _is_quote_delimiter_chosen_freely(token)
866 and quote_delimiter != most_common_delimiter
867 ):
868 self.add_message(
869 "inconsistent-quotes", line=start[0], args=(quote_delimiter,)
870 )

872 def check_for_concatenated_strings(
873 self, elements: Sequence[nodes.NodeNG], iterable_type: str
874 ) -> None:
875 for elt in elements:
876 if not (
877 isinstance(elt, nodes.Const) and elt.pytype() in _AST_NODE_STR_TYPES
878 ):
879 continue

881 # This can happen in case of escaped newlines
882 continue

887 continue

889 # We detect string concatenation: the AST Const is the
890 # combination of 2 string tokens

892 matching_token != elt.value

897 self.linter.config.check_str_concat_over_line_jumps
898 # Allow implicitly concatenated strings in parens.
899 # See https://github.com/pylint-dev/pylint/issues/8552.
900 and not self._parenthesized_string_tokens.get(
901 (elt.lineno, elt.col_offset)

911 def process_string_token(self, token: str, start_row: int, start_col: int) -> None:
912 quote_char = None
913 for _index, char in enumerate(token):
914 if char in "'\"":
915 quote_char = char
916 break
917 if quote_char is None:
918 return
919 # pylint: disable=undefined-loop-variable
920 prefix = token[:_index].lower() # markers like u, b, r.
921 after_prefix = token[_index:]
922 # pylint: enable=undefined-loop-variable
923 # Chop off quotes
924 quote_length = (
925 3 if after_prefix[:3] == after_prefix[-3:] == 3 * quote_char else 1
926 )
927 string_body = after_prefix[quote_length:-quote_length]
928 # No special checks on raw strings at the moment.
929 if "r" not in prefix:
930 self.process_non_raw_string_token(
931 prefix,
932 string_body,
933 start_row,
934 start_col + len(prefix) + quote_length,
935 )

937 def process_non_raw_string_token(
938 self, prefix: str, string_body: str, start_row: int, string_start_col: int
939 ) -> None:
940 """Check for bad escapes in a non-raw string.
941
942 prefix: lowercase string of string prefix markers ('ur').
943 string_body: the un-parsed body of the string, not including the quote
944 marks.
945 start_row: line number in the source.
946 string_start_col: col number of the string start in the source.
947 """
948 # Walk through the string; if we see a backslash then escape the next
949 # character, and skip over it. If we see a non-escaped character,
950 # alert, and continue.
951 #
952 # Accept a backslash when it escapes a backslash, or a quote, or
953 # end-of-line, or one of the letters that introduce a special escape
954 # sequence <https://docs.python.org/reference/lexical_analysis.html>
955 #
956 index = 0
957 while True:
958 index = string_body.find("\\", index)
959 if index == -1:
960 break
961 # There must be a next character; having a backslash at the end
962 # of the string would be a SyntaxError.
963 next_char = string_body[index + 1]
964 match = string_body[index : index + 2]
965 # The column offset will vary depending on whether the string token
966 # is broken across lines. Calculate relative to the nearest line
967 # break or relative to the start of the token's line.
968 last_newline = string_body.rfind("\n", 0, index)
969 if last_newline == -1:
970 line = start_row
971 col_offset = index + string_start_col
972 else:
973 line = start_row + string_body.count("\n", 0, index)
974 col_offset = index - last_newline - 1
975 if next_char in self.UNICODE_ESCAPE_CHARACTERS:
976 if "u" in prefix:
977 pass
978 elif "b" not in prefix:
979 pass # unicode by default
980 else:
981 self.add_message(
982 "anomalous-unicode-escape-in-string",
983 line=line,
984 args=(match,),
985 col_offset=col_offset,
986 )
987 elif next_char not in self.ESCAPE_CHARACTERS:
988 self.add_message(
989 "anomalous-backslash-in-string",
990 line=line,
991 args=(match,),
992 col_offset=col_offset,
993 )
994 # Whether it was a valid escape or not, backslash followed by
995 # another character can always be consumed whole: the second
996 # character can never be the start of a new backslash escape.
997 index += 2

1021def str_eval(token: str) -> str:
1022 """Mostly replicate `ast.literal_eval(token)` manually to avoid any performance hit.
1023
1024 This supports f-strings, contrary to `ast.literal_eval`.
1025 We have to support all string literal notations:
1026 https://docs.python.org/3/reference/lexical_analysis.html#string-and-bytes-literals
1027 """
1028 if token[0:2].lower() in {"fr", "rf"}:
1029 token = token[2:]
1030 elif token[0].lower() in {"r", "u", "f"}:
1031 token = token[1:]
1032 if token[0:3] in {'"""', "'''"}:
1033 return token[3:-3]
1034 return token[1:-1]
