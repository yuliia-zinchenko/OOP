45class LazySettings(LazyObject):
46 """
47 A lazy proxy for either global Django settings or a custom settings object.
48 The user can manually configure settings prior to using them. Otherwise,
49 Django uses the settings module pointed to by DJANGO_SETTINGS_MODULE.
50 """
51
52 def _setup(self, name=None):
53 """
54 Load the settings module pointed to by the environment variable. This
55 is used the first time settings are needed, if the user hasn't
56 configured settings manually.
57 """
58 settings_module = os.environ.get(ENVIRONMENT_VARIABLE)
59 if not settings_module:
60 desc = ("setting %s" % name) if name else "settings"
61 raise ImproperlyConfigured(
62 "Requested %s, but settings are not configured. "
63 "You must either define the environment variable %s "
64 "or call settings.configure() before accessing settings."
65 % (desc, ENVIRONMENT_VARIABLE)
66 )
67
68 self._wrapped = Settings(settings_module)
69
70 def __repr__(self):
71 # Hardcode the class name as otherwise it yields 'Settings'.
72 if self._wrapped is empty:
73 return "<LazySettings [Unevaluated]>"
74 return '<LazySettings "%(settings_module)s">' % {
75 "settings_module": self._wrapped.SETTINGS_MODULE,
76 }

78 def __getattr__(self, name):
79 """Return the value of a setting and cache it in self.__dict__."""
80 if (_wrapped := self._wrapped) is empty:
81 self._setup(name)
82 _wrapped = self._wrapped
83 val = getattr(_wrapped, name)
84
85 # Special case some settings which require further modification.
86 # This is done here for performance reasons so the modified value is cached.
87 if name in {"MEDIA_URL", "STATIC_URL"} and val is not None:
88 val = self._add_script_prefix(val)
89 elif name == "SECRET_KEY" and not val:
90 raise ImproperlyConfigured("The SECRET_KEY setting must not be empty.")
91
92 self.__dict__[name] = val
93 return val

95 def __setattr__(self, name, value):
96 """
97 Set the value of setting. Clear all cached values if _wrapped changes
98 (@override_settings does this) or clear single values when set.
99 """
100 if name == "_wrapped":
101 self.__dict__.clear()
102 else:
103 self.__dict__.pop(name, None)
104 super().__setattr__(name, value)
105
106 def __delattr__(self, name):
107 """Delete a setting and clear it from cache if needed."""
108 super().__delattr__(name)
109 self.__dict__.pop(name, None)

111 def configure(self, default_settings=global_settings, **options):
112 """
113 Called to manually configure the settings. The 'default_settings'
114 parameter sets where to retrieve any unspecified values from (its
115 argument must support attribute access (__getattr__)).
116 """
117 if self._wrapped is not empty:
118 raise RuntimeError("Settings already configured.")
119 holder = UserSettingsHolder(default_settings)
120 for name, value in options.items():
121 if not name.isupper():
122 raise TypeError("Setting %r must be uppercase." % name)
123 setattr(holder, name, value)
124 self._wrapped = holder

127 def _add_script_prefix(value):
128 """
129 Add SCRIPT_NAME prefix to relative paths.
130
131 Useful when the app is being served at a subpath and manually prefixing
132 subpath to STATIC_URL and MEDIA_URL in settings is inconvenient.
133 """
134 # Don't apply prefix to absolute paths and URLs.
135 if value.startswith(("http://", "https://", "/")):
136 return value
137 from django.urls import get_script_prefix

139 return "%s%s" % (get_script_prefix(), value)
