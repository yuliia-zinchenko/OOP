153 @staticmethod
154 def get_expected_messages(stream: TextIO) -> MessageCounter:
155 """Parses a file and get expected messages.
156
157 :param stream: File-like input stream.
158 :type stream: enumerable
159 :returns: A dict mapping line,msg-symbol tuples to the count on this line.
160 :rtype: dict
161 """
162 messages: MessageCounter = Counter()
163 for i, line in enumerate(stream):
164 match = _EXPECTED_RE.search(line)
165 if match is None:
166 continue
167 line = match.group("line")
168 if line is None:
169 lineno = i + 1
170 elif line.startswith(("+", "-")):
171 lineno = i + 1 + int(line)
172 else:
173 lineno = int(line)
174
175 version = match.group("version")
176 op = match.group("op")
177 if version:
178 required = parse_python_version(version)
179 if not _OPERATORS[op](sys.version_info, required):
180 continue
181
182 for msg_id in match.group("msgs").split(","):
183 messages[lineno, msg_id.strip()] += 1
184 return messages

186 @staticmethod
187 def multiset_difference(
188 expected_entries: MessageCounter,
189 actual_entries: MessageCounter,
190 ) -> tuple[MessageCounter, dict[tuple[int, str], int]]:
191 """Takes two multisets and compares them.
192
193 A multiset is a dict with the cardinality of the key as the value.
194 """
195 missing = expected_entries.copy()
196 missing.subtract(actual_entries)
197 unexpected = {}
198 for key, value in list(missing.items()):
199 if value <= 0:
200 missing.pop(key)
201 if value < 0:
202 unexpected[key] = -value
203 return missing, unexpected

258 def error_msg_for_unequal_messages(
259 self,
260 actual_messages: MessageCounter,
261 expected_messages: MessageCounter,
262 actual_output: list[OutputLine],
263 ) -> str:
264 msg = [f'Wrong message(s) raised for "{Path(self._test_file.source).name}":']
265 missing, unexpected = self.multiset_difference(
266 expected_messages, actual_messages
267 )
268 if missing:
269 msg.append("\nExpected in testdata:")
270 msg.extend(f" {msg[0]:3}: {msg[1]}" for msg in sorted(missing))
271 if unexpected:
272 msg.append("\nUnexpected in testdata:")
273 msg.extend(f" {msg[0]:3}: {msg[1]}" for msg in sorted(unexpected))
274 error_msg = "\n".join(msg)
275 if self._config and self._config.getoption("verbose") > 0:
276 error_msg += "\n\nActual pylint output for this file:\n"
277 error_msg += "\n".join(str(o) for o in actual_output)
278 return error_msg

280 def error_msg_for_unequal_output(
281 self,
282 expected_lines: list[OutputLine],
283 received_lines: list[OutputLine],
284 ) -> str:
285 missing = set(expected_lines) - set(received_lines)
286 unexpected = set(received_lines) - set(expected_lines)
287 error_msg = f'Wrong output for "{Path(self._test_file.expected_output).name}":'
288 sort_by_line_number = operator.attrgetter("lineno")
289 if missing:
290 error_msg += "\n- Missing lines:\n"
291 for line in sorted(missing, key=sort_by_line_number):
292 error_msg += f"{line}\n"
293 if unexpected:
294 error_msg += "\n- Unexpected lines:\n"
295 for line in sorted(unexpected, key=sort_by_line_number):
296 error_msg += f"{line}\n"
297 error_msg += (
298 "\nYou can update the expected output automatically with:\n'"
299 f"python tests/test_functional.py {UPDATE_OPTION} -k "
300 f'"test_functional[{self._test_file.base}]"\'\n\n'
301 "Here's the update text in case you can't:\n"
302 )
303 expected_csv = StringIO()
304 writer = csv.writer(expected_csv, dialect="test")
305 for line in sorted(received_lines, key=sort_by_line_number):
306 writer.writerow(line.to_csv())
