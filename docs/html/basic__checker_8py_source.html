60def report_by_type_stats(
61 sect: reporter_nodes.Section,
62 stats: LinterStats,
63 old_stats: LinterStats | None,
64) -> None:
65 """Make a report of.
66
67 * percentage of different types documented
68 * percentage of different types with a bad name
69 """
70 # percentage of different types documented and/or with a bad name
71 nice_stats: dict[str, dict[str, str]] = {}
72 for node_type in ("module", "class", "method", "function"):
73 node_type = cast(Literal["function", "class", "method", "module"], node_type)
74 total = stats.get_node_count(node_type)
75 nice_stats[node_type] = {}
76 if total != 0:
77 undocumented_node = stats.get_undocumented(node_type)
78 documented = total - undocumented_node
79 percent = (documented * 100.0) / total
80 nice_stats[node_type]["percent_documented"] = f"{percent:.2f}"
81 badname_node = stats.get_bad_names(node_type)
82 percent = (badname_node * 100.0) / total
83 nice_stats[node_type]["percent_badname"] = f"{percent:.2f}"
84 lines = ["type", "number", "old number", "difference", "%documented", "%badname"]
85 for node_type in ("module", "class", "method", "function"):
86 node_type = cast(Literal["function", "class", "method", "module"], node_type)
87 new = stats.get_node_count(node_type)
88 old = old_stats.get_node_count(node_type) if old_stats else None
89 diff_str = lint_utils.diff_string(old, new) if old else None
90 lines += [
91 node_type,
92 str(new),
93 str(old) if old else "NC",
94 diff_str if diff_str else "NC",
95 nice_stats[node_type].get("percent_documented", "0"),
96 nice_stats[node_type].get("percent_badname", "0"),
97 ]
98 sect.append(reporter_nodes.Table(children=lines, cols=6, rheaders=1))

304 def _check_using_constant_test(
305 self,
306 node: nodes.If | nodes.IfExp | nodes.Comprehension,
307 test: nodes.NodeNG | None,
308 ) -> None:
309 const_nodes = (
310 nodes.Module,
311 nodes.GeneratorExp,
312 nodes.Lambda,
313 nodes.FunctionDef,
314 nodes.ClassDef,
315 astroid.bases.Generator,
316 astroid.UnboundMethod,
317 astroid.BoundMethod,
318 nodes.Module,
319 )
320 structs = (nodes.Dict, nodes.Tuple, nodes.Set, nodes.List)
321
322 # These nodes are excepted, since they are not constant
323 # values, requiring a computation to happen.
324 except_nodes = (
325 nodes.Call,
326 nodes.BinOp,
327 nodes.BoolOp,
328 nodes.UnaryOp,
329 nodes.Subscript,
330 )
331 inferred = None
332 emit = isinstance(test, (nodes.Const, *structs, *const_nodes))
333 maybe_generator_call = None
334 if not isinstance(test, except_nodes):
335 inferred = utils.safe_infer(test)
336 if isinstance(inferred, util.UninferableBase) and isinstance(
337 test, nodes.Name
338 ):
339 emit, maybe_generator_call = BasicChecker._name_holds_generator(test)
340
341 # Emit if calling a function that only returns GeneratorExp (always tests True)
342 elif isinstance(test, nodes.Call):
343 maybe_generator_call = test
344 if maybe_generator_call:
345 inferred_call = utils.safe_infer(maybe_generator_call.func)
346 if isinstance(inferred_call, nodes.FunctionDef):
347 # Can't use all(x) or not any(not x) for this condition, because it
348 # will return True for empty generators, which is not what we want.
349 all_returns_were_generator = None
350 for return_node in inferred_call._get_return_nodes_skip_functions():
351 if not isinstance(return_node.value, nodes.GeneratorExp):
352 all_returns_were_generator = False
353 break
354 all_returns_were_generator = True
355 if all_returns_were_generator:
356 self.add_message(
357 "using-constant-test", node=node, confidence=INFERENCE
358 )
359 return
360
361 if emit:
362 self.add_message("using-constant-test", node=test, confidence=INFERENCE)
363 elif isinstance(inferred, const_nodes):
364 # If the constant node is a FunctionDef or Lambda then
365 # it may be an illicit function call due to missing parentheses
366 call_inferred = None
367 try:
368 # Just forcing the generator to infer all elements.
369 # astroid.exceptions.InferenceError are false positives
370 # see https://github.com/pylint-dev/pylint/pull/8185
371 if isinstance(inferred, nodes.FunctionDef):
372 call_inferred = list(inferred.infer_call_result(node))
373 elif isinstance(inferred, nodes.Lambda):
374 call_inferred = list(inferred.infer_call_result(node))
375 except astroid.InferenceError:
376 call_inferred = None
377 if call_inferred:
378 self.add_message(
379 "missing-parentheses-for-call-in-test",
380 node=test,
381 confidence=INFERENCE,
382 )
383 self.add_message("using-constant-test", node=test, confidence=INFERENCE)

385 @staticmethod
386 def _name_holds_generator(test: nodes.Name) -> tuple[bool, nodes.Call | None]:
387 """Return whether `test` tests a name certain to hold a generator, or optionally
388 a call that should be then tested to see if *it* returns only generators.
389 """
390 assert isinstance(test, nodes.Name)
391 emit = False
392 maybe_generator_call = None
393 lookup_result = test.frame().lookup(test.name)
394 if not lookup_result:
395 return emit, maybe_generator_call
396 maybe_generator_assigned = (
397 isinstance(assign_name.parent.value, nodes.GeneratorExp)
398 for assign_name in lookup_result[1]
399 if isinstance(assign_name.parent, nodes.Assign)
400 )
401 first_item = next(maybe_generator_assigned, None)
402 if first_item is not None:
403 # Emit if this variable is certain to hold a generator
404 if all(itertools.chain((first_item,), maybe_generator_assigned)):
405 emit = True
406 # If this variable holds the result of a call, save it for next test
407 elif (
408 len(lookup_result[1]) == 1
409 and isinstance(lookup_result[1][0].parent, nodes.Assign)
410 and isinstance(lookup_result[1][0].parent.value, nodes.Call)
411 ):
412 maybe_generator_call = lookup_result[1][0].parent.value
413 return emit, maybe_generator_call

425 @utils.only_required_for_messages(
426 "pointless-statement",

432 def visit_expr(self, node: nodes.Expr) -> None:
433 """Check for various kind of statements without effect."""
434 expr = node.value
435 if isinstance(expr, nodes.Const) and isinstance(expr.value, str):
436 # treat string statement in a separated message
437 # Handle PEP-257 attribute docstrings.
438 # An attribute docstring is defined as being a string right after
439 # an assignment at the module level, class level or __init__ level.
440 scope = expr.scope()
441 if isinstance(scope, (nodes.ClassDef, nodes.Module, nodes.FunctionDef)):
442 if isinstance(scope, nodes.FunctionDef) and scope.name != "__init__":
443 pass
444 else:
445 sibling = expr.previous_sibling()
446 if (
447 sibling is not None
448 and sibling.scope() is scope
449 and isinstance(
450 sibling, (nodes.Assign, nodes.AnnAssign, nodes.TypeAlias)
451 )
452 ):
453 return
454 self.add_message("pointless-string-statement", node=node)
455 return
456
457 # Warn W0133 for exceptions that are used as statements
458 if isinstance(expr, nodes.Call):
459 name = ""
460 if isinstance(expr.func, nodes.Name):
461 name = expr.func.name
462 elif isinstance(expr.func, nodes.Attribute):
463 name = expr.func.attrname
464
465 # Heuristic: only run inference for names that begin with an uppercase char
466 # This reduces W0133's coverage, but retains acceptable runtime performance
467 # For more details, see: https://github.com/pylint-dev/pylint/issues/8073
468 inferred = utils.safe_infer(expr) if name[:1].isupper() else None
469 if isinstance(inferred, objects.ExceptionInstance):
470 self.add_message(
471 "pointless-exception-statement", node=node, confidence=INFERENCE
472 )
473 return
474
475 # Ignore if this is :
476 # * the unique child of a try/except body
477 # * a yield statement
478 # * an ellipsis (which can be used on Python 3 instead of pass)
479 # warn W0106 if we have any underlying function call (we can't predict
480 # side effects), else pointless-statement
481 if (
482 isinstance(expr, (nodes.Yield, nodes.Await))
483 or (
484 isinstance(node.parent, (nodes.Try, nodes.TryStar))
485 and node.parent.body == [node]
486 )
487 or (isinstance(expr, nodes.Const) and expr.value is Ellipsis)
488 ):
489 return
490 if isinstance(expr, nodes.NamedExpr):
491 self.add_message("named-expr-without-context", node=node, confidence=HIGH)
492 elif any(expr.nodes_of_class(nodes.Call)):
493 self.add_message(
494 "expression-not-assigned", node=node, args=expr.as_string()
495 )
496 else:
497 self.add_message("pointless-statement", node=node)

526 @utils.only_required_for_messages("unnecessary-lambda")
527 def visit_lambda(self, node: nodes.Lambda) -> None:
528 """Check whether the lambda is suspicious."""
529 # if the body of the lambda is a call expression with the same
530 # argument list as the lambda itself, then the lambda is
531 # possibly unnecessary and at least suspicious.
532 if node.args.defaults:
533 # If the arguments of the lambda include defaults, then a
534 # judgment cannot be made because there is no way to check
535 # that the defaults defined by the lambda are the same as
536 # the defaults defined by the function called in the body
537 # of the lambda.
538 return
539 call = node.body
540 if not isinstance(call, nodes.Call):
541 # The body of the lambda must be a function call expression
542 # for the lambda to be unnecessary.
543 return
544 if isinstance(node.body.func, nodes.Attribute) and isinstance(
545 node.body.func.expr, nodes.Call
546 ):
547 # Chained call, the intermediate call might
548 # return something else (but we don't check that, yet).
549 return
550
551 ordinary_args = list(node.args.args)
552 new_call_args = list(self._filter_vararg(node, call.args))
553 if node.args.kwarg:
554 if self._has_variadic_argument(call.keywords, node.args.kwarg):
555 return
556 elif call.keywords:
557 return
558
559 if node.args.vararg:
560 if self._has_variadic_argument(call.starargs, node.args.vararg):
561 return
562 elif call.starargs:
563 return
564
565 # The "ordinary" arguments must be in a correspondence such that:
566 # ordinary_args[i].name == call.args[i].name.
567 if len(ordinary_args) != len(new_call_args):
568 return
569 for arg, passed_arg in zip(ordinary_args, new_call_args):
570 if not isinstance(passed_arg, nodes.Name):
571 return
572 if arg.name != passed_arg.name:
573 return
574
575 # The lambda is necessary if it uses its parameter in the function it is
576 # calling in the lambda's body
577 # e.g. lambda foo: (func1 if foo else func2)(foo)
578 for name in call.func.nodes_of_class(nodes.Name):
579 if name.lookup(name.name)[0] is node:
580 return
581
582 self.add_message("unnecessary-lambda", line=node.fromlineno, node=node)

597 def _check_dangerous_default(self, node: nodes.FunctionDef) -> None:
598 """Check for dangerous default values as arguments."""
599
600 def is_iterable(internal_node: nodes.NodeNG) -> bool:
601 return isinstance(internal_node, (nodes.List, nodes.Set, nodes.Dict))
602
603 defaults = (node.args.defaults or []) + (node.args.kw_defaults or [])
604 for default in defaults:
605 if not default:
606 continue
607 try:
608 value = next(default.infer())
609 except astroid.InferenceError:
610 continue
611
612 if (
613 isinstance(value, astroid.Instance)
614 and value.qname() in DEFAULT_ARGUMENT_SYMBOLS
615 ):
616 if value is default:
617 msg = DEFAULT_ARGUMENT_SYMBOLS[value.qname()]
618 elif isinstance(value, astroid.Instance) or is_iterable(value):
619 # We are here in the following situation(s):
620 # * a dict/set/list/tuple call which wasn't inferred
621 # to a syntax node ({}, () etc.). This can happen
622 # when the arguments are invalid or unknown to
623 # the inference.
624 # * a variable from somewhere else, which turns out to be a list
625 # or a dict.
626 if is_iterable(default):
627 msg = value.pytype()
628 elif isinstance(default, nodes.Call):
629 msg = f"{value.name}() ({value.qname()})"
630 else:
631 msg = f"{default.as_string()} ({value.qname()})"
632 else:
633 # this argument is a name
634 msg = f"{default.as_string()} ({DEFAULT_ARGUMENT_SYMBOLS[value.qname()]})"
635 self.add_message("dangerous-default-value", node=node, args=(msg,))

800 def _check_not_in_finally(
801 self,
802 node: nodes.Break | nodes.Return,
803 node_name: str,

826 """Check that the argument to `reversed` is a sequence."""
827 try:
828 argument = utils.safe_infer(utils.get_argument_from_call(node, position=0))
829 except utils.NoSuchArgumentError:
830 pass
831 else:
832 if isinstance(argument, util.UninferableBase):
833 return
834 if argument is None:
835 # Nothing was inferred.
836 # Try to see if we have iter().
837 if isinstance(node.args[0], nodes.Call):

839 func = next(node.args[0].func.infer())

841 return

843 func, "name", None
844 ) == "iter" and utils.is_builtin_object(func):

846 return

848 if isinstance(argument, (nodes.List, nodes.Tuple)):
849 return

851 # dicts are reversible, but only from Python 3.8 onward. Prior to
852 # that, any class based on dict must explicitly provide a
853 # __reversed__ method

855 if any(
856 ancestor.name == "dict" and utils.is_builtin_object(ancestor)
857 for ancestor in itertools.chain(
858 (argument._proxied,), argument._proxied.ancestors()

861 try:
862 argument.locals[REVERSED_PROTOCOL_METHOD]
863 except KeyError:
864 self.add_message("bad-reversed-sequence", node=node)
865 return
866
867 if hasattr(argument, "getattr"):
868 # everything else is not a proper sequence for reversed()
869 for methods in REVERSED_METHODS:
870 for meth in methods:
871 try:
872 argument.getattr(meth)
873 except astroid.NotFoundError:
874 break
875 else:
876 break
877 else:
878 self.add_message("bad-reversed-sequence", node=node)

880 self.add_message("bad-reversed-sequence", node=node)

898 def _check_self_assigning_variable(self, node: nodes.Assign) -> None:
899 # Detect assigning to the same variable.
900
901 scope = node.scope()
902 scope_locals = scope.locals
903
904 rhs_names = []
905 targets = node.targets
906 if isinstance(targets[0], nodes.Tuple):
907 if len(targets) != 1:
908 # A complex assignment, so bail out early.
909 return
910 targets = targets[0].elts
911 if len(targets) == 1:
912 # Unpacking a variable into the same name.
913 return
914
915 if isinstance(node.value, nodes.Name):
916 if len(targets) != 1:
917 return
918 rhs_names = [node.value]
919 elif isinstance(node.value, nodes.Tuple):
920 rhs_count = len(node.value.elts)
921 if len(targets) != rhs_count or rhs_count == 1:
922 return
923 rhs_names = node.value.elts
924
925 for target, lhs_name in zip(targets, rhs_names):
926 if not isinstance(lhs_name, nodes.Name):
927 continue
928 if not isinstance(target, nodes.AssignName):
929 continue
930 # Check that the scope is different from a class level, which is usually
931 # a pattern to expose module level attributes as class level ones.
932 if isinstance(scope, nodes.ClassDef) and target.name in scope_locals:
933 continue
934 if target.name == lhs_name.name:
