37 def to_python(self, value):
38 """Transform the value to a Geometry object."""
39 if value in self.empty_values:
40 return None
41
42 if not isinstance(value, GEOSGeometry):
43 if hasattr(self.widget, "deserialize"):

45 value = self.widget.deserialize(value)

47 value = None

50 value = GEOSGeometry(value)
51 except (GEOSException, ValueError, TypeError):
52 value = None

55 self.error_messages["invalid_geom"], code="invalid_geom"

67 def clean(self, value):
68 """
69 Validate that the input value can be converted to a Geometry object
70 and return it. Raise a ValidationError if the value cannot be
71 instantiated as a Geometry.
72 """
73 geom = super().clean(value)
74 if geom is None:
75 return geom
76
77 # Ensuring that the geometry is of the correct type (indicated
78 # using the OGC string label).

98 def has_changed(self, initial, data):
99 """Compare geographic value of data with its initial value."""
100
101 try:
102 data = self.to_python(data)
103 initial = self.to_python(initial)
104 except ValidationError:
105 return True
106
107 # Only do a geographic comparison if both values are available
108 if initial and data:
109 data.transform(initial.srid)
110 # If the initial value was not added by the browser, the geometry
111 # provided may be slightly different, the first time it is saved.
112 # The comparison is done with a very low tolerance.
113 return not initial.equals_exact(data, tolerance=0.000001)
114 else:
115 # Check for change of state of existence
116 return bool(initial) != bool(data)
