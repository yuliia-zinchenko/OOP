31def get_setters_property_name(node: nodes.FunctionDef) -> str | None:
32 """Get the name of the property that the given node is a setter for.
33
34 :param node: The node to get the property name for.
35 :type node: str
36
37 :rtype: str or None
38 :returns: The name of the property that the node is a setter for,
39 or None if one could not be found.
40 """
41 decorators = node.decorators.nodes if node.decorators else []
42 for decorator in decorators:
43 if (
44 isinstance(decorator, nodes.Attribute)
45 and decorator.attrname == "setter"
46 and isinstance(decorator.expr, nodes.Name)
47 ):
48 return decorator.expr.name # type: ignore[no-any-return]
49 return None

52def get_setters_property(node: nodes.FunctionDef) -> nodes.FunctionDef | None:
53 """Get the property node for the given setter node.
54
55 :param node: The node to get the property for.
56 :type node: nodes.FunctionDef
57
58 :rtype: nodes.FunctionDef or None
59 :returns: The node relating to the property of the given setter node,
60 or None if one could not be found.
61 """
62 property_ = None
63
64 property_name = get_setters_property_name(node)
65 class_node = utils.node_frame_class(node)
66 if property_name and class_node:
67 class_attrs: list[nodes.FunctionDef] = class_node.getattr(node.name)
68 for attr in class_attrs:
69 if utils.decorated_with_property(attr):
70 property_ = attr
71 break
72
73 return property_

76def returns_something(return_node: nodes.Return) -> bool:
77 """Check if a return node returns a value other than None.
78
79 :param return_node: The return node to check.
80 :type return_node: astroid.Return
81
82 :rtype: bool
83 :return: True if the return node returns a value other than None,
84 False otherwise.
85 """
86 returns = return_node.value
87
88 if returns is None:
89 return False
90
91 return not (isinstance(returns, nodes.Const) and returns.value is None)

107def possible_exc_types(node: nodes.NodeNG) -> set[nodes.ClassDef]:
108 """Gets all the possible raised exception types for the given raise node.
109
110 .. note::
111
112 Caught exception types are ignored.
113
114 :param node: The raise node to find exception types for.
115
116 :returns: A list of exception types possibly raised by :param:`node`.
117 """
118 exceptions = []
119 if isinstance(node.exc, nodes.Name):
120 inferred = utils.safe_infer(node.exc)
121 if inferred:
122 exceptions = [inferred]
123 elif node.exc is None:
124 handler = node.parent
125 while handler and not isinstance(handler, nodes.ExceptHandler):
126 handler = handler.parent
127
128 if handler and handler.type:
129 try:
130 for exception in astroid.unpack_infer(handler.type):
131 if not isinstance(exception, UninferableBase):
132 exceptions.append(exception)
133 except astroid.InferenceError:
134 pass
135 else:
136 target = _get_raise_target(node)
137 if isinstance(target, nodes.ClassDef):
138 exceptions = [target]
139 elif isinstance(target, nodes.FunctionDef):
140 for ret in target.nodes_of_class(nodes.Return):
141 if ret.value is None:
142 continue
143 if ret.frame() != target:
144 # return from inner function - ignore it
145 continue
146
147 val = utils.safe_infer(ret.value)
148 if val and utils.inherit_from_std_ex(val):
149 if isinstance(val, nodes.ClassDef):
150 exceptions.append(val)
151 elif isinstance(val, astroid.Instance):
152 exceptions.append(val.getattr("__class__")[0])
153
154 try:
155 return {
156 exc
157 for exc in exceptions
158 if not utils.node_ignores_exception(node, exc.name)
159 }
160 except astroid.InferenceError:
161 return set()

168def _merge_annotations(
169 annotations: Iterable[nodes.NodeNG], comment_annotations: Iterable[nodes.NodeNG]
170) -> Iterable[nodes.NodeNG | None]:
171 for ann, comment_ann in itertools.zip_longest(annotations, comment_annotations):
172 if ann and not _is_ellipsis(ann):
173 yield ann
174 elif comment_ann and not _is_ellipsis(comment_ann):
175 yield comment_ann
176 else:
177 yield None

180def _annotations_list(args_node: nodes.Arguments) -> list[nodes.NodeNG]:
181 """Get a merged list of annotations.
182
183 The annotations can come from:
184
185 * Real type annotations.
186 * A type comment on the function.
187 * A type common on the individual argument.
188
189 :param args_node: The node to get the annotations for.
190 :returns: The annotations.
191 """
192 plain_annotations = args_node.annotations or ()
193 func_comment_annotations = args_node.parent.type_comment_args or ()
194 comment_annotations = args_node.type_comment_posonlyargs
195 comment_annotations += args_node.type_comment_args or []
196 comment_annotations += args_node.type_comment_kwonlyargs
197 return list(
198 _merge_annotations(
199 plain_annotations,
200 _merge_annotations(func_comment_annotations, comment_annotations),
201 )
202 )

205def args_with_annotation(args_node: nodes.Arguments) -> set[str]:
206 result = set()
207 annotations = _annotations_list(args_node)
208 annotation_offset = 0
209
210 if args_node.posonlyargs:
211 posonlyargs_annotations = args_node.posonlyargs_annotations
212 if not any(args_node.posonlyargs_annotations):
213 num_args = len(args_node.posonlyargs)
214 posonlyargs_annotations = annotations[
215 annotation_offset : annotation_offset + num_args
216 ]
217 annotation_offset += num_args
218
219 for arg, annotation in zip(args_node.posonlyargs, posonlyargs_annotations):
220 if annotation:
221 result.add(arg.name)
222
223 if args_node.args:
224 num_args = len(args_node.args)
225 for arg, annotation in zip(
226 args_node.args,
227 annotations[annotation_offset : annotation_offset + num_args],
228 ):
229 if annotation:
230 result.add(arg.name)
231
232 annotation_offset += num_args
233
234 if args_node.vararg:
235 if args_node.varargannotation:
236 result.add(args_node.vararg)
237 elif len(annotations) > annotation_offset and annotations[annotation_offset]:
238 result.add(args_node.vararg)
239 annotation_offset += 1
240
241 if args_node.kwonlyargs:
242 kwonlyargs_annotations = args_node.kwonlyargs_annotations
243 if not any(args_node.kwonlyargs_annotations):
244 num_args = len(args_node.kwonlyargs)
245 kwonlyargs_annotations = annotations[
246 annotation_offset : annotation_offset + num_args
247 ]
248 annotation_offset += num_args
249
250 for arg, annotation in zip(args_node.kwonlyargs, kwonlyargs_annotations):
251 if annotation:
252 result.add(arg.name)
253
254 if args_node.kwarg:
255 if args_node.kwargannotation:
256 result.add(args_node.kwarg)
257 elif len(annotations) > annotation_offset and annotations[annotation_offset]:
258 result.add(args_node.kwarg)
259 annotation_offset += 1
260
261 return result

264def docstringify(
265 docstring: nodes.Const | None, default_type: str = "default"
266) -> Docstring:
267 best_match = (0, DOCSTRING_TYPES.get(default_type, Docstring)(docstring))
268 for docstring_type in (
269 SphinxDocstring,
270 EpytextDocstring,
271 GoogleDocstring,
272 NumpyDocstring,
273 ):
274 instance = docstring_type(docstring)
275 matching_sections = instance.matching_sections()
276 if matching_sections > best_match[0]:
277 best_match = (matching_sections, instance)
278
279 return best_match[1]

339class SphinxDocstring(Docstring):
340 re_type = r"""
341 [~!.]? # Optional link style prefix
342 \w(?:\w|\.[^\.])* # Valid python name
343 """
344
345 re_simple_container_type = rf"""
346 {re_type} # a container type
347 [\‍(\[] [^\n\s]+ [\‍)\]] # with the contents of the container
348 """
349
350 re_multiple_simple_type = rf"""
351 (?:{re_simple_container_type}|{re_type})
352 (?:(?:\s+(?:of|or)\s+|\s*,\s*|\s+\|\s+)(?:{re_simple_container_type}|{re_type}))*
353 """
354
355 re_xref = rf"""
356 (?::\w+:)? # optional tag
357 `{re_type}` # what to reference
358 """
359
360 re_param_raw = rf"""
361 : # initial colon
362 (?: # Sphinx keywords
363 param|parameter|
364 arg|argument|
365 key|keyword
366 )
367 \s+ # whitespace
368
369 (?: # optional type declaration
370 ({re_type}|{re_simple_container_type})
371 \s+
372 )?
373
374 ((\\\*{{0,2}}\w+)|(\w+)) # Parameter name with potential asterisks
375 \s* # whitespace
376 : # final colon
377 """
378 re_param_in_docstring = re.compile(re_param_raw, re.X | re.S)
379
380 re_type_raw = rf"""
381 :type # Sphinx keyword
382 \s+ # whitespace
383 ({re_multiple_simple_type}) # Parameter name
384 \s* # whitespace
385 : # final colon
386 """
387 re_type_in_docstring = re.compile(re_type_raw, re.X | re.S)
388
389 re_property_type_raw = rf"""
390 :type: # Sphinx keyword
391 \s+ # whitespace
392 {re_multiple_simple_type} # type declaration
393 """
394 re_property_type_in_docstring = re.compile(re_property_type_raw, re.X | re.S)
395
396 re_raise_raw = rf"""
397 : # initial colon
398 (?: # Sphinx keyword
399 raises?|
400 except|exception
401 )
402 \s+ # whitespace
403 ({re_multiple_simple_type}) # exception type
404 \s* # whitespace
405 : # final colon
406 """
407 re_raise_in_docstring = re.compile(re_raise_raw, re.X | re.S)
408
409 re_rtype_in_docstring = re.compile(r":rtype:")
410
411 re_returns_in_docstring = re.compile(r":returns?:")
412
413 supports_yields = False
414
415 def matching_sections(self) -> int:
416 """Returns the number of matching docstring sections."""
417 return sum(
418 bool(i)
419 for i in (
420 self.re_param_in_docstring.search(self.doc),
421 self.re_raise_in_docstring.search(self.doc),
422 self.re_rtype_in_docstring.search(self.doc),
423 self.re_returns_in_docstring.search(self.doc),
424 self.re_property_type_in_docstring.search(self.doc),
425 )
426 )
427
428 def exceptions(self) -> set[str]:
429 types: set[str] = set()
430
431 for match in re.finditer(self.re_raise_in_docstring, self.doc):
432 raise_type = match.group(1)
433 types.update(_split_multiple_exc_types(raise_type))
434
435 return types

437 def has_params(self) -> bool:
438 if not self.doc:
439 return False
440
441 return self.re_param_in_docstring.search(self.doc) is not None
442
443 def has_returns(self) -> bool:
444 if not self.doc:
445 return False
446
447 return bool(self.re_returns_in_docstring.search(self.doc))
448
449 def has_rtype(self) -> bool:
450 if not self.doc:
451 return False
452
453 return bool(self.re_rtype_in_docstring.search(self.doc))
454
455 def has_property_returns(self) -> bool:
456 if not self.doc:
457 return False
458
459 # The summary line is the return doc,
460 # so the first line must not be a known directive.
461 return not self.doc.lstrip().startswith(":")
462
463 def has_property_type(self) -> bool:
464 if not self.doc:
465 return False
466
467 return bool(self.re_property_type_in_docstring.search(self.doc))
468
469 def match_param_docs(self) -> tuple[set[str], set[str]]:
470 params_with_doc = set()
471 params_with_type = set()
472
473 for match in re.finditer(self.re_param_in_docstring, self.doc):
474 name = match.group(2)
475 # Remove escape characters necessary for asterisks
476 name = name.replace("\\", "")
477 params_with_doc.add(name)
478 param_type = match.group(1)
479 if param_type is not None:
480 params_with_type.add(name)
481
482 params_with_type.update(re.findall(self.re_type_in_docstring, self.doc))
483 return params_with_doc, params_with_type

487 """Epytext is similar to Sphinx.

505
506 re_property_type_in_docstring = re.compile(
507 SphinxDocstring.re_property_type_raw.replace(":", "@", 1), re.X | re.S

509

511 SphinxDocstring.re_raise_raw.replace(":", "@", 1), re.X | re.S
512 )
513
514 re_rtype_in_docstring = re.compile(
515 r"""
516 @ # initial "at" symbol
517 (?: # Epytext keyword
518 rtype|returntype
519 )
520 : # final colon
521 """,
522 re.X | re.S,
523 )
524
525 re_returns_in_docstring = re.compile(r"@returns?:")

527 def has_property_returns(self) -> bool:
528 if not self.doc:
529 return False
530

534 # so the first line must not be a known directive.
535 return not self.doc.lstrip().startswith("@")
536
537 return False

723 types: set[str] = set()

725 entries = self._parse_section(self.re_raise_section)
726 for entry in entries:
727 match = self.re_raise_line.match(entry)
728 if not match:

738 def match_param_docs(self) -> tuple[set[str], set[str]]:
739 params_with_doc: set[str] = set()
740 params_with_type: set[str] = set()
741
742 entries = self._parse_section(self.re_param_section)
743 entries.extend(self._parse_section(self.re_keyword_param_section))
744 for entry in entries:
745 match = self.re_param_line.match(entry)
746 if not match:
747 continue
748
749 param_name = match.group(1)
750 # Remove escape characters necessary for asterisks
751 param_name = param_name.replace("\\", "")
752
753 param_type = match.group(2)
754 param_desc = match.group(3)
755
756 if param_type:
757 params_with_type.add(param_name)
758
759 if param_desc:
760 params_with_doc.add(param_name)
761
762 return params_with_doc, params_with_type

885 def match_param_docs(self) -> tuple[set[str], set[str]]:
886 """Matches parameter documentation section to parameter documentation rules."""
887 params_with_doc = set()
888 params_with_type = set()
889
890 entries = self._parse_section(self.re_param_section)
891 entries.extend(self._parse_section(self.re_keyword_param_section))
892 for entry in entries:
893 match = self.re_param_line.match(entry)
894 if not match:
895 continue
896
897 # check if parameter has description only
898 re_only_desc = re.match(r"\s*(\*{0,2}\w+)\s*:?\n\s*\w*$", entry)
899 if re_only_desc:
900 param_name = match.group("param_name")
901 param_desc = match.group("param_type")
902 param_type = None
903 else:
904 param_name = match.group("param_name")
905 param_type = match.group("param_type")
906 param_desc = match.group("param_desc")
907 # The re_param_line pattern needs to match multi-line which removes the ability
908 # to match a single line description like 'arg : a number type.'
909 # We are not trying to determine whether 'a number type' is correct typing
910 # but we do accept it as typing as it is in the place where typing
911 # should be
912 if param_type is None and re.match(r"\s*(\*{0,2}\w+)\s*:.+$", entry):
913 param_type = param_desc
914 # If the description is "" but we have a type description
915 # we consider the description to be the type
916 if not param_desc and param_type:
917 param_desc = param_type
918
919 if param_type:
920 params_with_type.add(param_name)
921
922 if param_desc:
923 params_with_doc.add(param_name)
924
925 return params_with_doc, params_with_type
