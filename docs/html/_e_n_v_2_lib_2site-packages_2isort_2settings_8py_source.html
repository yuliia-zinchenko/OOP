527 def is_supported_filetype(self, file_name: str) -> bool:

589 def is_skipped(self, file_path: Path) -> bool:
590 """Returns True if the file and/or folder should be skipped based on current settings."""
591 if self.directory and Path(self.directory) in file_path.resolve().parents:
592 file_name = os.path.relpath(file_path.resolve(), self.directory)
593 else:
594 file_name = str(file_path)
595
596 os_path = str(file_path)
597
598 normalized_path = os_path.replace("\\", "/")
599 if normalized_path[1:2] == ":":
600 normalized_path = normalized_path[2:]
601
602 for skip_path in self.skips:
603 if posixpath.abspath(normalized_path) == posixpath.abspath(
604 skip_path.replace("\\", "/")
605 ):
606 return True
607
608 position = os.path.split(file_name)
609 while position[1]:
610 if position[1] in self.skips:
611 return True
612 position = os.path.split(position[0])
613
614 for sglob in self.skip_globs:
615 if fnmatch.fnmatch(file_name, sglob) or fnmatch.fnmatch("/" + file_name, sglob):
616 return True
617
618 if not (os.path.isfile(os_path) or os.path.isdir(os_path) or os.path.islink(os_path)):
619 return True
620
621 if self.skip_gitignore:
622 if file_path.name == ".git": # pragma: no cover
623 return True
624
625 git_folder = None
626
627 file_paths = [file_path, file_path.resolve()]
628 for folder in self.git_ls_files:
629 if any(folder in path.parents for path in file_paths):
630 git_folder = folder
631 break
632 else:
633 git_folder = self._check_folder_git_ls_files(str(file_path.parent))
634
635 # git_ls_files are good files you should parse. If you're not in the allow list, skip.
636
637 if (
638 git_folder
639 and not file_path.is_dir()
640 and str(file_path.resolve()) not in self.git_ls_files[git_folder]
641 ):
642 return True
643
644 return False

827def _get_config_data(file_path: str, sections: Tuple[str, ...]) -> Dict[str, Any]:
828 settings: Dict[str, Any] = {}
829
830 if file_path.endswith(".toml"):
831 with open(file_path, "rb") as bin_config_file:
832 config = tomllib.load(bin_config_file)
833 for section in sections:
834 config_section = config
835 for key in section.split("."):
836 config_section = config_section.get(key, {})
837 settings.update(config_section)
838 else:
839 with open(file_path, encoding="utf-8") as config_file:
840 if file_path.endswith(".editorconfig"):
841 line = "\n"
842 last_position = config_file.tell()
843 while line:
844 line = config_file.readline()
845 if "[" in line:
846 config_file.seek(last_position)
847 break
848 last_position = config_file.tell()
849
850 config = configparser.ConfigParser(strict=False)
851 config.read_file(config_file)
852 for section in sections:
853 if section.startswith("*.{") and section.endswith("}"):
854 extension = section[len("*.{") : -1]
855 for config_key in config.keys():
856 if (
857 config_key.startswith("*.{")
858 and config_key.endswith("}")
859 and extension
860 in map(
861 lambda text: text.strip(),
862 config_key[len("*.{") : -1].split(","), # noqa
863 )
864 ):
865 settings.update(config.items(config_key))
866
867 elif config.has_section(section):
868 settings.update(config.items(section))
869
870 if settings:
871 settings["source"] = file_path
872
873 if file_path.endswith(".editorconfig"):
874 indent_style = settings.pop("indent_style", "").strip()
875 indent_size = settings.pop("indent_size", "").strip()
876 if indent_size == "tab":
877 indent_size = settings.pop("tab_width", "").strip()
878
879 if indent_style == "space":
880 settings["indent"] = " " * (indent_size and int(indent_size) or 4)
881
882 elif indent_style == "tab":
883 settings["indent"] = "\t" * (indent_size and int(indent_size) or 1)
884
885 max_line_length = settings.pop("max_line_length", "").strip()
886 if max_line_length and (max_line_length == "off" or max_line_length.isdigit()):
887 settings["line_length"] = (
888 float("inf") if max_line_length == "off" else int(max_line_length)
889 )
890 settings = {
891 key: value
892 for key, value in settings.items()
893 if key in _DEFAULT_SETTINGS.keys() or key.startswith(KNOWN_PREFIX)
894 }
895
896 for key, value in settings.items():
897 existing_value_type = _get_str_to_type_converter(key)
898 if existing_value_type == tuple:
899 settings[key] = tuple(_as_list(value))
900 elif existing_value_type == frozenset:
901 settings[key] = frozenset(_as_list(settings.get(key))) # type: ignore
902 elif existing_value_type == bool:
903 # Only some configuration formats support native boolean values.
904 if not isinstance(value, bool):
905 value = _as_bool(value)
906 settings[key] = value
907 elif key.startswith(KNOWN_PREFIX):
908 settings[key] = _abspaths(os.path.dirname(file_path), _as_list(value))
909 elif key == "force_grid_wrap":
910 try:
911 result = existing_value_type(value)
912 except ValueError: # backwards compatibility for true / false force grid wrap
913 result = 0 if value.lower().strip() == "false" else 2
914 settings[key] = result
915 elif key == "comment_prefix":
916 settings[key] = str(value).strip("'").strip('"')
917 else:
918 settings[key] = existing_value_type(value)
919
920 return settings
