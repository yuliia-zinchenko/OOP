121def infer_typing_typevar_or_newtype(
122 node: Call, context_itton: context.InferenceContext | None = None
123) -> Iterator[ClassDef]:
124 """Infer a typing.TypeVar(...) or typing.NewType(...) call."""
125 try:
126 func = next(node.func.infer(context=context_itton))
127 except (InferenceError, StopIteration) as exc:
128 raise UseInferenceDefault from exc
129
130 if func.qname() not in TYPING_TYPEVARS_QUALIFIED:
131 raise UseInferenceDefault
132 if not node.args:
133 raise UseInferenceDefault
134 # Cannot infer from a dynamic class name (f-string)
135 if isinstance(node.args[0], JoinedStr):
136 raise UseInferenceDefault
137
138 typename = node.args[0].as_string().strip("'")
139 try:
140 node = extract_node(TYPING_TYPE_TEMPLATE.format(typename))
141 except AstroidSyntaxError as exc:
142 raise InferenceError from exc
143 return node.infer(context=context_itton)

146def _looks_like_typing_subscript(node) -> bool:
147 """Try to figure out if a Subscript node *might* be a typing-related subscript."""
148 if isinstance(node, Name):
149 return node.name in TYPING_MEMBERS
150 if isinstance(node, Attribute):
151 return node.attrname in TYPING_MEMBERS
152 if isinstance(node, Subscript):
153 return _looks_like_typing_subscript(node.value)
154 return False

157def infer_typing_attr(
158 node: Subscript, ctx: context.InferenceContext | None = None
159) -> Iterator[ClassDef]:
160 """Infer a typing.X[...] subscript."""
161 try:
162 value = next(node.value.infer()) # type: ignore[union-attr] # value shouldn't be None for Subscript.
163 except (InferenceError, StopIteration) as exc:
164 raise UseInferenceDefault from exc
165
166 if not value.qname().startswith("typing.") or value.qname() in TYPING_ALIAS:
167 # If typing subscript belongs to an alias handle it separately.
168 raise UseInferenceDefault
169
170 if (
171 PY313_PLUS
172 and isinstance(value, FunctionDef)
173 and value.qname() == "typing.Annotated"
174 ):
175 # typing.Annotated is a FunctionDef on 3.13+
176 node._explicit_inference = lambda node, context: iter([value])
177 return iter([value])
178
179 if isinstance(value, ClassDef) and value.qname() in {
180 "typing.Generic",
181 "typing.Annotated",
182 "typing_extensions.Annotated",
183 }:
184 # typing.Generic and typing.Annotated (PY39) are subscriptable
185 # through __class_getitem__. Since astroid can't easily
186 # infer the native methods, replace them for an easy inference tip
187 func_to_add = _extract_single_node(CLASS_GETITEM_TEMPLATE)
188 value.locals["__class_getitem__"] = [func_to_add]
189 if (
190 isinstance(node.parent, ClassDef)
191 and node in node.parent.bases
192 and getattr(node.parent, "__cache", None)
193 ):
194 # node.parent.slots is evaluated and cached before the inference tip
195 # is first applied. Remove the last result to allow a recalculation of slots
196 cache = node.parent.__cache # type: ignore[attr-defined] # Unrecognized getattr
197 if cache.get(node.parent.slots) is not None:
198 del cache[node.parent.slots]
199 # Avoid re-instantiating this class every time it's seen
200 node._explicit_inference = lambda node, context: iter([value])
201 return iter([value])
202
203 node = extract_node(TYPING_TYPE_TEMPLATE.format(value.qname().split(".")[-1]))
204 return node.infer(context=ctx)

246def _looks_like_typing_alias(node: Call) -> bool:
247 """
248 Returns True if the node corresponds to a call to _alias function.

250 For example :
251
252 MutableSet = _alias(collections.abc.MutableSet, T)
253
254 :param node: call node
255 """
256 return (
257 isinstance(node.func, Name)
258 # TODO: remove _DeprecatedGenericAlias when Py3.14 min
259 and node.func.name in {"_alias", "_DeprecatedGenericAlias"}
260 and len(node.args) == 2
261 and (
262 # _alias function works also for builtins object such as list and dict
263 isinstance(node.args[0], (Attribute, Name))
264 )

268def _forbid_class_getitem_access(node: ClassDef) -> None:
269 """Disable the access to __class_getitem__ method for the node in parameters."""
270
271 def full_raiser(origin_func, attr, *args, **kwargs):
272 """
273 Raises an AttributeInferenceError in case of access to __class_getitem__ method.
274 Otherwise, just call origin_func.
275 """
276 if attr == "__class_getitem__":
277 raise AttributeInferenceError("__class_getitem__ access is not allowed")
278 return origin_func(attr, *args, **kwargs)
279
280 try:
281 node.getattr("__class_getitem__")
282 # If we are here, then we are sure to modify an object that does have
283 # __class_getitem__ method (which origin is the protocol defined in
284 # collections module) whereas the typing module considers it should not.
285 # We do not want __class_getitem__ to be found in the classdef
286 partial_raiser = partial(full_raiser, node.getattr)
287 node.getattr = partial_raiser
288 except AttributeInferenceError:
289 pass

292def infer_typing_alias(
293 node: Call, ctx: context.InferenceContext | None = None
294) -> Iterator[ClassDef]:
295 """
296 Infers the call to _alias function
297 Insert ClassDef, with same name as aliased class,
298 in mro to simulate _GenericAlias.
299
300 :param node: call node
301 :param context: inference context
302
303 # TODO: evaluate if still necessary when Py3.12 is minimum
304 """
305 if (
306 not isinstance(node.parent, Assign)
307 or not len(node.parent.targets) == 1
308 or not isinstance(node.parent.targets[0], AssignName)
309 ):
310 raise UseInferenceDefault
311 try:
312 res = next(node.args[0].infer(context=ctx))
313 except StopIteration as e:
314 raise InferenceError(node=node.args[0], context=ctx) from e
315
316 assign_name = node.parent.targets[0]
317
318 class_def = ClassDef(
319 name=assign_name.name,
320 lineno=assign_name.lineno,
321 col_offset=assign_name.col_offset,
322 parent=node.parent,
323 end_lineno=assign_name.end_lineno,
324 end_col_offset=assign_name.end_col_offset,
325 )
326 if isinstance(res, ClassDef):
327 # Only add `res` as base if it's a `ClassDef`
328 # This isn't the case for `typing.Pattern` and `typing.Match`
329 class_def.postinit(bases=[res], body=[], decorators=None)
330
331 maybe_type_var = node.args[1]
332 if isinstance(maybe_type_var, Const) and maybe_type_var.value > 0:
333 # If typing alias is subscriptable, add `__class_getitem__` to ClassDef
334 func_to_add = _extract_single_node(CLASS_GETITEM_TEMPLATE)
335 class_def.locals["__class_getitem__"] = [func_to_add]
336 else:
337 # If not, make sure that `__class_getitem__` access is forbidden.
338 # This is an issue in cases where the aliased class implements it,
339 # but the typing alias isn't subscriptable. E.g., `typing.ByteString` for PY39+
340 _forbid_class_getitem_access(class_def)
341
342 # Avoid re-instantiating this class every time it's seen
343 node._explicit_inference = lambda node, context: iter([class_def])
344 return iter([class_def])

347def _looks_like_special_alias(node: Call) -> bool:
348 """Return True if call is for Tuple or Callable alias.
349
350 In PY37 and PY38 the call is to '_VariadicGenericAlias' with 'tuple' as
351 first argument. In PY39+ it is replaced by a call to '_TupleType'.
352
353 PY37: Tuple = _VariadicGenericAlias(tuple, (), inst=False, special=True)
354 PY39: Tuple = _TupleType(tuple, -1, inst=False, name='Tuple')
355
356 PY37: Callable = _VariadicGenericAlias(collections.abc.Callable, (), special=True)
357 PY39: Callable = _CallableType(collections.abc.Callable, 2)
358 """
359 return isinstance(node.func, Name) and (
360 node.func.name == "_TupleType"
361 and isinstance(node.args[0], Name)
362 and node.args[0].name == "tuple"
363 or node.func.name == "_CallableType"
364 and isinstance(node.args[0], Attribute)
365 and node.args[0].as_string() == "collections.abc.Callable"
366 )
