41def bdf_char(
42 f: BinaryIO,
43) -> (
44 tuple[
45 str,
46 int,
47 tuple[tuple[int, int], tuple[int, int, int, int], tuple[int, int, int, int]],
48 Image.Image,
49 ]
50 | None
51):
52 # skip to STARTCHAR
53 while True:
54 s = f.readline()
55 if not s:
56 return None
57 if s[:9] == b"STARTCHAR":
58 break
59 id = s[9:].strip().decode("ascii")
60
61 # load symbol properties
62 props = {}
63 while True:
64 s = f.readline()
65 if not s or s[:6] == b"BITMAP":
66 break
67 i = s.find(b" ")
68 props[s[:i].decode("ascii")] = s[i + 1 : -1].decode("ascii")
69
70 # load bitmap
71 bitmap = bytearray()
72 while True:
73 s = f.readline()
74 if not s or s[:7] == b"ENDCHAR":
75 break
76 bitmap += s[:-1]
77
78 # The word BBX
79 # followed by the width in x (BBw), height in y (BBh),
80 # and x and y displacement (BBxoff0, BByoff0)
81 # of the lower left corner from the origin of the character.
82 width, height, x_disp, y_disp = (int(p) for p in props["BBX"].split())
83
84 # The word DWIDTH
85 # followed by the width in x and y of the character in device pixels.
86 dwx, dwy = (int(p) for p in props["DWIDTH"].split())
87
88 bbox = (
89 (dwx, dwy),
90 (x_disp, -y_disp - height, width + x_disp, -y_disp),
91 (0, 0, width, height),
92 )
93
94 try:
95 im = Image.frombytes("1", (width, height), bitmap, "hex", "1")
96 except ValueError:
97 # deal with zero-width characters
98 im = Image.new("1", (width, height))
99
100 return id, int(props["ENCODING"]), bbox, im
