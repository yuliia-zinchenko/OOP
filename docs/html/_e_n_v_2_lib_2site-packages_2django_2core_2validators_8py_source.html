127class URLValidator(RegexValidator):
128 ul = "\u00a1-\uffff" # Unicode letters range (must not be a raw string).
129
130 # IP patterns
131 ipv4_re = (
132 r"(?:0|25[0-5]|2[0-4][0-9]|1[0-9]?[0-9]?|[1-9][0-9]?)"
133 r"(?:\.(?:0|25[0-5]|2[0-4][0-9]|1[0-9]?[0-9]?|[1-9][0-9]?)){3}"
134 )
135 ipv6_re = r"\[[0-9a-f:.]+\]" # (simple regex, validated later)

137 hostname_re = DomainNameValidator.hostname_re

139 tld_re = DomainNameValidator.tld_re

141 host_re = "(" + hostname_re + domain_re + tld_re + "|localhost)"

143 regex = _lazy_re_compile(
144 r"^(?:[a-z0-9.+-]*)://" # scheme is validated separately
145 r"(?:[^\s:@/]+(?::[^\s:@/]*)?@)?" # user:pass authentication
146 r"(?:" + ipv4_re + "|" + ipv6_re + "|" + host_re + ")"
147 r"(?::[0-9]{1,5})?" # port
148 r"(?:[/?#][^\s]*)?" # resource path
149 r"\Z",
150 re.IGNORECASE,
151 )
152 message = _("Enter a valid URL.")
153 schemes = ["http", "https", "ftp", "ftps"]
154 unsafe_chars = frozenset("\t\r\n")
155 max_length = 2048
156
157 def __init__(self, schemes=None, **kwargs):
158 super().__init__(**kwargs)
159 if schemes is not None:
160 self.schemes = schemes
161
162 def __call__(self, value):
163 if not isinstance(value, str) or len(value) > self.max_length:
164 raise ValidationError(self.message, code=self.code, params={"value": value})
165 if self.unsafe_chars.intersection(value):
166 raise ValidationError(self.message, code=self.code, params={"value": value})
167 # Check if the scheme is valid.
168 scheme = value.split("://")[0].lower()
169 if scheme not in self.schemes:
170 raise ValidationError(self.message, code=self.code, params={"value": value})
171
172 # Then check full URL
173 try:
174 splitted_url = urlsplit(value)
175 except ValueError:
176 raise ValidationError(self.message, code=self.code, params={"value": value})
177 try:
178 super().__call__(value)
179 except ValidationError as e:
180 # Trivial case failed. Try for possible IDN domain
181 if value:
182 scheme, netloc, path, query, fragment = splitted_url
183 try:
184 netloc = punycode(netloc) # IDN -> ACE
185 except UnicodeError: # invalid domain part
186 raise e
187 url = urlunsplit((scheme, netloc, path, query, fragment))
188 super().__call__(url)
189 else:
190 raise
191 else:
192 # Now verify IPv6 in the netloc part
193 host_match = re.search(r"^\[(.+)\](?::[0-9]{1,5})?$", splitted_url.netloc)
194 if host_match:
195 potential_ip = host_match[1]
196 try:
197 validate_ipv6_address(potential_ip)
198 except ValidationError:
199 raise ValidationError(
200 self.message, code=self.code, params={"value": value}
201 )
202
203 # The maximum length of a full host name is 253 characters per RFC 1034
204 # section 3.1. It's defined to be 255 bytes or less, but this includes
205 # one byte for the length of the name and one byte for the trailing dot
206 # that's used to indicate absolute names in DNS.

208 raise ValidationError(self.message, code=self.code, params={"value": value})

524@deconstructible
525class DecimalValidator:
526 """
527 Validate that the input does not exceed the maximum number of digits
528 expected, otherwise raise ValidationError.
529 """
530
531 messages = {
532 "invalid": _("Enter a number."),
533 "max_digits": ngettext_lazy(
534 "Ensure that there are no more than %(max)s digit in total.",
535 "Ensure that there are no more than %(max)s digits in total.",
536 "max",
537 ),
538 "max_decimal_places": ngettext_lazy(
539 "Ensure that there are no more than %(max)s decimal place.",
540 "Ensure that there are no more than %(max)s decimal places.",
541 "max",
542 ),
543 "max_whole_digits": ngettext_lazy(
544 "Ensure that there are no more than %(max)s digit before the decimal "
545 "point.",
546 "Ensure that there are no more than %(max)s digits before the decimal "
547 "point.",
548 "max",
549 ),
550 }
551
552 def __init__(self, max_digits, decimal_places):
553 self.max_digits = max_digits
554 self.decimal_places = decimal_places
555
556 def __call__(self, value):
557 digit_tuple, exponent = value.as_tuple()[1:]
558 if exponent in {"F", "n", "N"}:
559 raise ValidationError(
560 self.messages["invalid"], code="invalid", params={"value": value}
561 )
562 if exponent >= 0:
563 digits = len(digit_tuple)
564 if digit_tuple != (0,):
565 # A positive exponent adds that many trailing zeros.
566 digits += exponent
567 decimals = 0
568 else:
569 # If the absolute value of the negative exponent is larger than the
570 # number of digits, then it's the same as the number of digits,
571 # because it'll consume all of the digits in digit_tuple and then
572 # add abs(exponent) - len(digit_tuple) leading zeros after the
573 # decimal point.
574 if abs(exponent) > len(digit_tuple):
575 digits = decimals = abs(exponent)
576 else:
577 digits = len(digit_tuple)
578 decimals = abs(exponent)
579 whole_digits = digits - decimals
580
581 if self.max_digits is not None and digits > self.max_digits:
582 raise ValidationError(
583 self.messages["max_digits"],
584 code="max_digits",
585 params={"max": self.max_digits, "value": value},
586 )
587 if self.decimal_places is not None and decimals > self.decimal_places:
588 raise ValidationError(
589 self.messages["max_decimal_places"],
590 code="max_decimal_places",
591 params={"max": self.decimal_places, "value": value},
592 )
593 if (
594 self.max_digits is not None
595 and self.decimal_places is not None
596 and whole_digits > (self.max_digits - self.decimal_places)
597 ):
598 raise ValidationError(
599 self.messages["max_whole_digits"],
600 code="max_whole_digits",
601 params={"max": (self.max_digits - self.decimal_places), "value": value},
602 )
603
604 def __eq__(self, other):
605 return (
606 isinstance(other, self.__class__)
607 and self.max_digits == other.max_digits
608 and self.decimal_places == other.decimal_places
609 )
