106def is_method_call(
107 func: bases.BoundMethod, types: tuple[str, ...] = (), methods: tuple[str, ...] = ()
108) -> bool:
109 """Determines if a BoundMethod node represents a method call.
110
111 Args:
112 func: The BoundMethod AST node to check.
113 types: Optional sequence of caller type names to restrict check.
114 methods: Optional sequence of method names to restrict check.
115
116 Returns:
117 true if the node represents a method call for the given type and
118 method names, False otherwise.
119 """
120 return (
121 isinstance(func, astroid.BoundMethod)
122 and isinstance(func.bound, astroid.Instance)
123 and (func.bound.name in types if types else True)
124 and (func.name in methods if methods else True)
125 )

158 def visit_module(self, _: nodes.Module) -> None:
159 """Clears any state left in this checker from last module checked."""
160 # The code being checked can just as easily "import logging as foo",
161 # so it is necessary to process the imports and store in this field
162 # what name the logging module is actually given.
163 self._logging_names: set[str] = set()
164 logging_mods = self.linter.config.logging_modules
165
166 self._format_style = self.linter.config.logging_format_style
167
168 self._logging_modules = set(logging_mods)
169 self._from_imports = {}
170 for logging_mod in logging_mods:
171 parts = logging_mod.rsplit(".", 1)
172 if len(parts) > 1:
173 self._from_imports[parts[0]] = parts[1]
174
175 def visit_importfrom(self, node: nodes.ImportFrom) -> None:
176 """Checks to see if a module uses a non-Python logging module."""
177 try:
178 logging_name = self._from_imports[node.modname]
179 for module, as_name in node.names:
180 if module == logging_name:
181 self._logging_names.add(as_name or module)
182 except KeyError:
183 pass
184
185 def visit_import(self, node: nodes.Import) -> None:
186 """Checks to see if this module uses Python's built-in logging."""
187 for module, as_name in node.names:
188 if module in self._logging_modules:
189 self._logging_names.add(as_name or module)

191 def visit_call(self, node: nodes.Call) -> None:
192 """Checks calls to logging methods."""

194 def is_logging_name() -> bool:
195 return (
196 isinstance(node.func, nodes.Attribute)
197 and isinstance(node.func.expr, nodes.Name)
198 and node.func.expr.name in self._logging_names
199 )

201 def is_logger_class() -> tuple[bool, str | None]:
202 for inferred in infer_all(node.func):
203 if isinstance(inferred, astroid.BoundMethod):
204 parent = inferred._proxied.parent
205 if isinstance(parent, nodes.ClassDef) and (
206 parent.qname() == "logging.Logger"
207 or any(
208 ancestor.qname() == "logging.Logger"
209 for ancestor in parent.ancestors()
210 )
211 ):
212 return True, inferred._proxied.name
213 return False, None
214
215 if is_logging_name():
216 name = node.func.attrname
217 else:
218 result, name = is_logger_class()
219 if not result:
220 return
221 self._check_log_method(node, name)

223 def _check_log_method(self, node: nodes.Call, name: str) -> None:
224 """Checks calls to logging.log(level, format, *format_args)."""
225 if name == "log":
226 if node.starargs or node.kwargs or len(node.args) < 2:
227 # Either a malformed call, star args, or double-star args. Beyond
228 # the scope of this checker.
229 return
230 format_pos: Literal[0, 1] = 1
231 elif name in CHECKED_CONVENIENCE_FUNCTIONS:
232 if node.starargs or node.kwargs or not node.args:
233 # Either no args, star args, or double-star args. Beyond the
234 # scope of this checker.
235 return
236 format_pos = 0
237 else:
238 return
239
240 format_arg = node.args[format_pos]
241 if isinstance(format_arg, nodes.BinOp):
242 binop = format_arg
243 emit = binop.op == "%"
244 if binop.op == "+" and not self._is_node_explicit_str_concatenation(binop):
245 total_number_of_strings = sum(
246 1
247 for operand in (binop.left, binop.right)
248 if self._is_operand_literal_str(utils.safe_infer(operand))
249 )
250 emit = total_number_of_strings > 0
251 if emit:
252 self.add_message(
253 "logging-not-lazy",
254 node=node,
255 args=(self._helper_string(node),),
256 )
257 elif isinstance(format_arg, nodes.Call):
258 self._check_call_func(format_arg)
259 elif isinstance(format_arg, nodes.Const):
260 self._check_format_string(node, format_pos)
261 elif isinstance(format_arg, nodes.JoinedStr):
262 if str_formatting_in_f_string(format_arg):
263 return
264 self.add_message(
265 "logging-fstring-interpolation",
266 node=node,
267 args=(self._helper_string(node),),
268 )

270 def _helper_string(self, node: nodes.Call) -> str:
271 """Create a string that lists the valid types of formatting for this node."""
272 valid_types = ["lazy %"]
273
274 if not self.linter.is_message_enabled(
275 "logging-fstring-formatting", node.fromlineno
276 ):
277 valid_types.append("fstring")
278 if not self.linter.is_message_enabled(
279 "logging-format-interpolation", node.fromlineno
280 ):
281 valid_types.append(".format()")
282 if not self.linter.is_message_enabled("logging-not-lazy", node.fromlineno):
283 valid_types.append("%")
284
285 return " or ".join(valid_types)

287 @staticmethod
288 def _is_operand_literal_str(operand: InferenceResult | None) -> bool:
289 """Return True if the operand in argument is a literal string."""
290 return isinstance(operand, nodes.Const) and operand.name == "str"

292 @staticmethod
293 def _is_node_explicit_str_concatenation(node: nodes.NodeNG) -> bool:
294 """Return True if the node represents an explicitly concatenated string."""
295 if not isinstance(node, nodes.BinOp):
296 return False
297 return (
298 LoggingChecker._is_operand_literal_str(node.left)
299 or LoggingChecker._is_node_explicit_str_concatenation(node.left)

302 or LoggingChecker._is_node_explicit_str_concatenation(node.right)

305 def _check_call_func(self, node: nodes.Call) -> None:
306 """Checks that function call is not format_string.format()."""
307 func = utils.safe_infer(node.func)
308 types = ("str", "unicode")
309 methods = ("format",)
310 if (
311 isinstance(func, astroid.BoundMethod)
312 and is_method_call(func, types, methods)
313 and not is_complex_format_str(func.bound)
314 ):
315 self.add_message(
316 "logging-format-interpolation",
317 node=node,
318 args=(self._helper_string(node),),
319 )

321 def _check_format_string(self, node: nodes.Call, format_arg: Literal[0, 1]) -> None:
322 """Checks that format string tokens match the supplied arguments.
323
324 Args:
325 node: AST node to be checked.
326 format_arg: Index of the format string in the node arguments.
327 """
328 num_args = _count_supplied_tokens(node.args[format_arg + 1 :])
329 if not num_args:
330 # If no args were supplied the string is not interpolated and can contain
331 # formatting characters - it's used verbatim. Don't check any further.
332 return
333
334 format_string = node.args[format_arg].value
335 required_num_args = 0
336 if isinstance(format_string, bytes):
337 format_string = format_string.decode()
338 if isinstance(format_string, str):
339 try:
340 if self._format_style == "old":
341 keyword_args, required_num_args, _, _ = utils.parse_format_string(
342 format_string
343 )
344 if keyword_args:
345 # Keyword checking on logging strings is complicated by
346 # special keywords - out of scope.
347 return
348 elif self._format_style == "new":
349 (
350 keyword_arguments,
351 implicit_pos_args,
352 explicit_pos_args,
353 ) = utils.parse_format_method_string(format_string)
354
355 keyword_args_cnt = len(
356 {k for k, _ in keyword_arguments if not isinstance(k, int)}
357 )
358 required_num_args = (
359 keyword_args_cnt + implicit_pos_args + explicit_pos_args
360 )
361 except utils.UnsupportedFormatCharacter as ex:
362 char = format_string[ex.index]
363 self.add_message(
364 "logging-unsupported-format",
365 node=node,
366 args=(char, ord(char), ex.index),
367 )
368 return
369 except utils.IncompleteFormatString:
370 self.add_message("logging-format-truncated", node=node)
371 return
372 if num_args > required_num_args:
373 self.add_message("logging-too-many-args", node=node, confidence=HIGH)
374 elif num_args < required_num_args:
375 self.add_message("logging-too-few-args", node=node)

378def is_complex_format_str(node: nodes.NodeNG) -> bool:
379 """Return whether the node represents a string with complex formatting specs."""
380 inferred = utils.safe_infer(node)
381 if inferred is None or not (
382 isinstance(inferred, nodes.Const) and isinstance(inferred.value, str)
383 ):
384 return True
385 try:
386 parsed = list(string.Formatter().parse(inferred.value))
387 except ValueError:
388 # This format string is invalid
389 return False
390 return any(format_spec for (_, _, format_spec, _) in parsed)

393def _count_supplied_tokens(args: list[nodes.NodeNG]) -> int:
394 """Counts the number of tokens in an args list.
395
396 The Python log functions allow for special keyword arguments: func,
397 exc_info and extra. To handle these cases correctly, we only count
398 arguments that aren't keywords.
399
400 Args:
401 args: AST nodes that are arguments for a log format string.
402
403 Returns:
404 Number of AST nodes that aren't keywords.
405 """
406 return sum(1 for arg in args if not isinstance(arg, nodes.Keyword))

409def str_formatting_in_f_string(node: nodes.JoinedStr) -> bool:
410 """Determine whether the node represents an f-string with string formatting.
411
412 For example: `f'Hello %s'`
413 """
414 # Check "%" presence first for performance.
415 return any(
416 "%" in val.value and any(x in val.value for x in MOST_COMMON_FORMATTING)
417 for val in node.values
418 if isinstance(val, nodes.Const)
419 )
