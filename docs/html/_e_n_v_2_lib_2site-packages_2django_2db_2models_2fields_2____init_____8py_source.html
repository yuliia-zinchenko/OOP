567 def deconstruct(self):
568 """
569 Return enough information to recreate the field as a 4-tuple:
570
571 * The name of the field on the model, if contribute_to_class() has
572 been run.
573 * The import path of the field, including the class, e.g.
574 django.db.models.IntegerField. This should be the most portable
575 version, so less specific may be better.
576 * A list of positional arguments.
577 * A dict of keyword arguments.
578
579 Note that the positional or keyword arguments must contain values of
580 the following types (including inner values of collection types):
581
582 * None, bool, str, int, float, complex, set, frozenset, list, tuple,
583 dict
584 * UUID
585 * datetime.datetime (naive), datetime.date
586 * top-level classes, top-level functions - will be referenced by their
587 full import path
588 * Storage instances - these have their own deconstruct() method
589
590 This is because the values here must be serialized into a text format
591 (possibly new Python code, possibly JSON) and these are the only types
592 with encoding handlers defined.
593
594 There's no need to return the exact way the field was instantiated this
595 time, just ensure that the resulting field is the same - prefer keyword
596 arguments over positional ones, and omit parameters with their default
597 values.
598 """

600 keywords = {}
601 possibles = {
602 "verbose_name": None,
603 "primary_key": False,
604 "max_length": None,
605 "unique": False,
606 "blank": False,
607 "null": False,
608 "db_index": False,
609 "default": NOT_PROVIDED,

611 "editable": True,

613 "unique_for_date": None,
614 "unique_for_month": None,
615 "unique_for_year": None,
616 "choices": None,
617 "help_text": "",
618 "db_column": None,
619 "db_comment": None,
620 "db_tablespace": None,
621 "auto_created": False,
622 "validators": [],
623 "error_messages": None,
624 }
625 attr_overrides = {
626 "unique": "_unique",
627 "error_messages": "_error_messages",
628 "validators": "_validators",
629 "verbose_name": "_verbose_name",
630 "db_tablespace": "_db_tablespace",

852 def db_type(self, connection):

858 # backend-specific data_types dictionary, looking up the field by its
859 # "internal type".

862 # which *preexisting* Django Field class it's most similar to -- i.e.,
863 # a custom field might be represented by a TEXT column type, which is
864 # the same as the TextField Django field type, which means the custom
865 # field's get_internal_type() returns 'TextField'.

867 # But the limitation of the get_internal_type() / data_types approach
868 # is that it cannot handle database column types that aren't already

871 # exactly which wacky database column type you want to use.

874 column_type = connection.data_types[self.get_internal_type()]
875 except KeyError:
876 return None
877 else:
878 # column_type is either a single-parameter function or a string.
879 if callable(column_type):
880 return column_type(data)
