29def render_field(
30 field,
31 form,
32 context,
33 template=None,
34 labelclass=None,
35 layout_object=None,
36 attrs=None,
37 template_pack=TEMPLATE_PACK,
38 extra_context=None,
39 **kwargs,
40):
41 """
42 Renders a django-crispy-forms field
43
44 :param field: Can be a string or a Layout object like `Row`. If it's a layout
45 object, we call its render method, otherwise we instantiate a BoundField
46 and render it using default template 'CRISPY_TEMPLATE_PACK/field.html'
47 The field is added to a list that the form holds called `rendered_fields`
48 to avoid double rendering fields.
49 :param form: The form/formset to which that field belongs to.
50 :template: Template used for rendering the field.
51 :layout_object: If passed, it points to the Layout object that is being rendered.
52 We use it to store its bound fields in a list called `layout_object.bound_fields`
53 :attrs: Attributes for the field's widget
54 :template_pack: Name of the template pack to be used for rendering `field`
55 :extra_context: Dictionary to be added to context, added variables by the layout object
56 """
57 added_keys = [] if extra_context is None else extra_context.keys()
58 with KeepContext(context, added_keys):
59 if field is None:
60 return SafeString("")
61
62 FAIL_SILENTLY = getattr(settings, "CRISPY_FAIL_SILENTLY", True)
63
64 if hasattr(field, "render"):
65 return field.render(form, context, template_pack=template_pack)
66
67 try:
68 # Injecting HTML attributes into field's widget, Django handles rendering these
69 bound_field = form[field]
70 field_instance = bound_field.field
71 if attrs is not None:
72 widgets = getattr(field_instance.widget, "widgets", [field_instance.widget])
73
74 # We use attrs as a dictionary later, so here we make a copy
75 list_attrs = attrs
76 if isinstance(attrs, dict):
77 list_attrs = [attrs] * len(widgets)
78
79 for index, (widget, attr) in enumerate(zip(widgets, list_attrs)):
80 if hasattr(field_instance.widget, "widgets"):
81 if "type" in attr and attr["type"] == "hidden":
82 field_instance.widget.widgets[index] = field_instance.hidden_widget(attr)
83
84 else:
85 field_instance.widget.widgets[index].attrs.update(attr)
86 else:
87 if "type" in attr and attr["type"] == "hidden":
88 field_instance.widget = field_instance.hidden_widget(attr)
89
90 else:
91 field_instance.widget.attrs.update(attr)
92
93 except KeyError:
94 if not FAIL_SILENTLY:
95 raise Exception("Could not resolve form field '%s'." % field)
96 else:
97 field_instance = None
98 logging.warning("Could not resolve form field '%s'." % field, exc_info=sys.exc_info())
99
100 if hasattr(form, "rendered_fields"):
101 if field not in form.rendered_fields:
102 form.rendered_fields.add(field)
103 else:
104 if not FAIL_SILENTLY:
105 raise Exception("A field should only be rendered once: %s" % field)
106 else:
107 logging.warning("A field should only be rendered once: %s" % field, exc_info=sys.exc_info())
108
109 if field_instance is None:
110 html = SafeString("")
111 else:
112 if template is None:
113 if form.crispy_field_template is None:
114 template = default_field_template(template_pack)
115 else: # FormHelper.field_template set
116 template = get_template(form.crispy_field_template)
117 else:
118 template = get_template(template)
119
120 # We save the Layout object's bound fields in the layout object's `bound_fields` list
121 if layout_object is not None:
122 if hasattr(layout_object, "bound_fields") and isinstance(layout_object.bound_fields, list):
123 layout_object.bound_fields.append(bound_field)
124 else:
125 layout_object.bound_fields = [bound_field]
126
127 context.update(
128 {
129 "field": bound_field,
130 "labelclass": labelclass,
131 "flat_attrs": flatatt(attrs if isinstance(attrs, dict) else {}),
132 }
133 )
134 if extra_context is not None:
135 context.update(extra_context)
136
137 html = template.render(context.flatten())
138
139 return html

142def flatatt(attrs):
143 """
144 Convert a dictionary of attributes to a single string.
145
146 Passed attributes are redirected to `django.forms.utils.flatatt()`
147 with replaced "_" (underscores) by "-" (dashes) in their names.
148 """
149 return _flatatt({k.replace("_", "-"): v for k, v in attrs.items()})

171def list_intersection(list1, list2):
172 """
173 Take the not-in-place intersection of two lists, similar to sets but preserving order.
174 Does not check unicity of list1.
175 """
176 return [item for item in list1 if item in list2]

179def list_difference(left, right):
180 """
181 Take the not-in-place difference of two lists (left - right), similar to sets but preserving order.
182 """
183 blocked = set(right)
184 difference = []
185 for item in left:
186 if item not in blocked:
187 blocked.add(item)
188 difference.append(item)
189 return difference
