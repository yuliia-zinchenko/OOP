53class EncodingChecker(BaseTokenChecker, BaseRawFileChecker):
54 """BaseChecker for encoding issues.
55
56 Checks for:
57 * warning notes in the code like FIXME, XXX
58 * encoding issues.
59 """

61 # configuration section name

95 def open(self) -> None:
96 super().open()
97
98 notes = "|".join(re.escape(note) for note in self.linter.config.notes)
99 if self.linter.config.notes_rgx:
100 regex_string = rf"#\s*({notes}|{self.linter.config.notes_rgx})(?=(:|\s|\Z))"
101 else:
102 regex_string = rf"#\s*({notes})(?=(:|\s|\Z))"
103
104 self._fixme_pattern = re.compile(regex_string, re.I)

106 def _check_encoding(
107 self, lineno: int, line: bytes, file_encoding: str
108 ) -> str | None:
109 try:
110 return line.decode(file_encoding)
111 except UnicodeDecodeError:
112 pass
113 except LookupError:
114 if (
115 line.startswith(b"#")
116 and "coding" in str(line)
117 and file_encoding in str(line)
118 ):
119 msg = f"Cannot decode using encoding '{file_encoding}', bad encoding"
120 self.add_message("syntax-error", line=lineno, args=msg)
121 return None

123 def process_module(self, node: nodes.Module) -> None:
124 """Inspect the source file to find encoding problem."""
125 encoding = node.file_encoding if node.file_encoding else "ascii"
126
127 with node.stream() as stream:
128 for lineno, line in enumerate(stream):
129 self._check_encoding(lineno + 1, line, encoding)

131 def process_tokens(self, tokens: list[tokenize.TokenInfo]) -> None:
132 """Inspect the source to find fixme problems."""
133 if not self.linter.config.notes:
134 return
135 for token_info in tokens:
136 if token_info.type != tokenize.COMMENT:
137 continue
138 comment_text = token_info.string[1:].lstrip() # trim '#' and white-spaces
139 if self._fixme_pattern.search("#" + comment_text.lower()):
140 self.add_message(
141 "fixme",
142 col_offset=token_info.start[1] + 1,
143 args=comment_text,
144 line=token_info.start[0],
145 )
