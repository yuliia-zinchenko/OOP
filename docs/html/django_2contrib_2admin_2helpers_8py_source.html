200class AdminReadonlyField:
201 def __init__(self, form, field, is_first, model_admin=None):
202 # Make self.field look a little bit like a field. This means that
203 # {{ field.name }} must be a useful class name to identify the field.
204 # For convenience, store other field-related data here too.
205 if callable(field):
206 class_name = field.__name__ if field.__name__ != "<lambda>" else ""
207 else:
208 class_name = field
209
210 if form._meta.labels and class_name in form._meta.labels:
211 label = form._meta.labels[class_name]
212 else:
213 label = label_for_field(field, form._meta.model, model_admin, form=form)
214
215 if form._meta.help_texts and class_name in form._meta.help_texts:
216 help_text = form._meta.help_texts[class_name]
217 else:
218 help_text = help_text_for_field(class_name, form._meta.model)
219
220 if field in form.fields:
221 is_hidden = form.fields[field].widget.is_hidden
222 else:
223 is_hidden = False
224
225 self.field = {
226 "name": class_name,
227 "label": label,
228 "help_text": help_text,
229 "field": field,
230 "is_hidden": is_hidden,
231 }
232 self.form = form
233 self.model_admin = model_admin
234 self.is_first = is_first
235 self.is_checkbox = False
236 self.is_readonly = True
237 self.empty_value_display = model_admin.get_empty_value_display()

242 attrs["class"] = "inline"

251 def get_admin_url(self, remote_field, remote_obj):
252 url_name = "admin:%s_%s_change" % (
253 remote_field.model._meta.app_label,
254 remote_field.model._meta.model_name,
255 )
256 try:
257 url = reverse(
258 url_name,
259 args=[quote(remote_obj.pk)],
260 current_app=self.model_admin.admin_site.name,
261 )
262 return format_html('<a href="{}">{}</a>', url, remote_obj)

266 def contents(self):
267 from django.contrib.admin.templatetags.admin_list import _boolean_icon
268
269 field, obj, model_admin = (
270 self.field["field"],
271 self.form.instance,
272 self.model_admin,
273 )
274 try:
275 f, attr, value = lookup_field(field, obj, model_admin)
276 except (AttributeError, ValueError, ObjectDoesNotExist):
277 result_repr = self.empty_value_display
278 else:
279 if field in self.form.fields:
280 widget = self.form[field].field.widget
281 # This isn't elegant but suffices for contrib.auth's
282 # ReadOnlyPasswordHashWidget.
283 if getattr(widget, "read_only", False):
284 return widget.render(field, value)
285 if f is None:
286 if getattr(attr, "boolean", False):
287 result_repr = _boolean_icon(value)
288 else:
289 if hasattr(value, "__html__"):
290 result_repr = value
291 else:
292 result_repr = linebreaksbr(value)
293 else:
294 if isinstance(f.remote_field, ManyToManyRel) and value is not None:
295 result_repr = ", ".join(map(str, value.all()))
296 elif (
297 isinstance(f.remote_field, (ForeignObjectRel, OneToOneField))
298 and value is not None
299 ):
300 result_repr = self.get_admin_url(f.remote_field, value)
301 else:
302 result_repr = display_for_field(value, f, self.empty_value_display)
303 result_repr = linebreaksbr(result_repr)
304 return conditional_escape(result_repr)
