35class RedirectURLMixin:
36 next_page = None
37 redirect_field_name = REDIRECT_FIELD_NAME
38 success_url_allowed_hosts = set()

40 def get_success_url(self):
41 return self.get_redirect_url() or self.get_default_redirect_url()

43 def get_redirect_url(self):
44 """Return the user-originating redirect URL if it's safe."""
45 redirect_to = self.request.POST.get(
46 self.redirect_field_name, self.request.GET.get(self.redirect_field_name)
47 )
48 url_is_safe = url_has_allowed_host_and_scheme(
49 url=redirect_to,
50 allowed_hosts=self.get_success_url_allowed_hosts(),
51 require_https=self.request.is_secure(),
52 )
53 return redirect_to if url_is_safe else ""

55 def get_success_url_allowed_hosts(self):
56 return {self.request.get_host(), *self.success_url_allowed_hosts}

59 """Return the default redirect URL."""

180def redirect_to_login(next, login_url=None, redirect_field_name=REDIRECT_FIELD_NAME):
181 """
182 Redirect the user to the login page, passing the given 'next' page.
183 """
184 resolved_url = resolve_url(login_url or settings.LOGIN_URL)
185
186 login_url_parts = list(urlparse(resolved_url))
187 if redirect_field_name:
188 querystring = QueryDict(login_url_parts[4], mutable=True)
189 querystring[redirect_field_name] = next
190 login_url_parts[4] = querystring.urlencode(safe="/")
191
192 return HttpResponseRedirect(urlunparse(login_url_parts))

255@method_decorator(login_not_required, name="dispatch")
256class PasswordResetConfirmView(PasswordContextMixin, FormView):
257 form_class = SetPasswordForm
258 post_reset_login = False
259 post_reset_login_backend = None
260 reset_url_token = "set-password"
261 success_url = reverse_lazy("password_reset_complete")
262 template_name = "registration/password_reset_confirm.html"
263 title = _("Enter new password")
264 token_generator = default_token_generator
265
266 @method_decorator(sensitive_post_parameters())
267 @method_decorator(never_cache)
268 def dispatch(self, *args, **kwargs):
269 if "uidb64" not in kwargs or "token" not in kwargs:
270 raise ImproperlyConfigured(
271 "The URL path must contain 'uidb64' and 'token' parameters."
272 )
273
274 self.validlink = False
275 self.user = self.get_user(kwargs["uidb64"])
276
277 if self.user is not None:
278 token = kwargs["token"]
279 if token == self.reset_url_token:
280 session_token = self.request.session.get(INTERNAL_RESET_SESSION_TOKEN)
281 if self.token_generator.check_token(self.user, session_token):
282 # If the token is valid, display the password reset form.
283 self.validlink = True
284 return super().dispatch(*args, **kwargs)
285 else:
286 if self.token_generator.check_token(self.user, token):
287 # Store the token in the session and redirect to the
288 # password reset form at a URL without the token. That
289 # avoids the possibility of leaking the token in the
290 # HTTP Referer header.
291 self.request.session[INTERNAL_RESET_SESSION_TOKEN] = token
292 redirect_url = self.request.path.replace(
293 token, self.reset_url_token
294 )
295 return HttpResponseRedirect(redirect_url)
296
297 # Display the "Password reset unsuccessful" page.
298 return self.render_to_response(self.get_context_data())

300 def get_user(self, uidb64):
301 try:
302 # urlsafe_base64_decode() decodes to bytestring
303 uid = urlsafe_base64_decode(uidb64).decode()
304 user = UserModel._default_manager.get(pk=uid)
305 except (
306 TypeError,
307 ValueError,
308 OverflowError,
309 UserModel.DoesNotExist,
310 ValidationError,
311 ):
312 user = None
313 return user
