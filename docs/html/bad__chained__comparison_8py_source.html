18IDENTITY_OP = frozenset(("is", "is not"))
19MEMBERSHIP_OP = frozenset(("in", "not in"))

22class BadChainedComparisonChecker(BaseChecker):
23 """Checks for unintentional usage of chained comparison."""
24
25 name = "bad-chained-comparison"
26 msgs = {
27 "W3601": (
28 "Suspicious %s-part chained comparison using semantically incompatible operators (%s)",
29 "bad-chained-comparison",
30 "Used when there is a chained comparison where one expression is part "
31 "of two comparisons that belong to different semantic groups "
32 '("<" does not mean the same thing as "is", chaining them in '
33 '"0 < x is None" is probably a mistake).',
34 )
35 }
36
37 def _has_diff_semantic_groups(self, operators: list[str]) -> bool:
38 # Check if comparison operators are in the same semantic group
39 for semantic_group in (COMPARISON_OP, IDENTITY_OP, MEMBERSHIP_OP):
40 if operators[0] in semantic_group:
41 group = semantic_group
42 return not all(o in group for o in operators)
43
44 def visit_compare(self, node: nodes.Compare) -> None:
45 operators = sorted({op[0] for op in node.ops})
46 if self._has_diff_semantic_groups(operators):
47 num_parts = f"{len(node.ops)}"
48 incompatibles = (
49 ", ".join(f"'{o}'" for o in operators[:-1]) + f" and '{operators[-1]}'"
50 )
51 self.add_message(
52 "bad-chained-comparison",
53 node=node,
54 args=(num_parts, incompatibles),
55 confidence=HIGH,
56 )
