325def merge_stats(stats: list[LinterStats]) -> LinterStats:
326 """Used to merge multiple stats objects into a new one when pylint is run in
327 parallel mode.
328 """
329 merged = LinterStats()
330 for stat in stats:
331 merged.bad_names["argument"] += stat.bad_names["argument"]
332 merged.bad_names["attr"] += stat.bad_names["attr"]
333 merged.bad_names["klass"] += stat.bad_names["klass"]
334 merged.bad_names["class_attribute"] += stat.bad_names["class_attribute"]
335 merged.bad_names["class_const"] += stat.bad_names["class_const"]
336 merged.bad_names["const"] += stat.bad_names["const"]
337 merged.bad_names["inlinevar"] += stat.bad_names["inlinevar"]
338 merged.bad_names["function"] += stat.bad_names["function"]
339 merged.bad_names["method"] += stat.bad_names["method"]
340 merged.bad_names["module"] += stat.bad_names["module"]
341 merged.bad_names["variable"] += stat.bad_names["variable"]
342 merged.bad_names["typevar"] += stat.bad_names["typevar"]
343 merged.bad_names["typealias"] += stat.bad_names["typealias"]
344
345 for mod_key, mod_value in stat.by_module.items():
346 merged.by_module[mod_key] = mod_value
347
348 for msg_key, msg_value in stat.by_msg.items():
349 try:
350 merged.by_msg[msg_key] += msg_value
351 except KeyError:
352 merged.by_msg[msg_key] = msg_value
353
354 merged.code_type_count["code"] += stat.code_type_count["code"]
355 merged.code_type_count["comment"] += stat.code_type_count["comment"]
356 merged.code_type_count["docstring"] += stat.code_type_count["docstring"]
357 merged.code_type_count["empty"] += stat.code_type_count["empty"]
358 merged.code_type_count["total"] += stat.code_type_count["total"]
359
360 for dep_key, dep_value in stat.dependencies.items():
361 try:
362 merged.dependencies[dep_key].update(dep_value)
363 except KeyError:
364 merged.dependencies[dep_key] = dep_value
365
366 merged.duplicated_lines["nb_duplicated_lines"] += stat.duplicated_lines[
367 "nb_duplicated_lines"
368 ]
369 merged.duplicated_lines["percent_duplicated_lines"] += stat.duplicated_lines[
370 "percent_duplicated_lines"
371 ]
372
373 merged.node_count["function"] += stat.node_count["function"]
374 merged.node_count["klass"] += stat.node_count["klass"]
375 merged.node_count["method"] += stat.node_count["method"]
376 merged.node_count["module"] += stat.node_count["module"]
377
378 merged.undocumented["function"] += stat.undocumented["function"]
379 merged.undocumented["klass"] += stat.undocumented["klass"]
380 merged.undocumented["method"] += stat.undocumented["method"]
381 merged.undocumented["module"] += stat.undocumented["module"]
382
383 merged.convention += stat.convention
384 merged.error += stat.error
385 merged.fatal += stat.fatal
386 merged.info += stat.info
387 merged.refactor += stat.refactor
388 merged.statement += stat.statement
389 merged.warning += stat.warning
390 merged.skipped += stat.skipped
391
392 merged.global_note += stat.global_note
393 return merged
