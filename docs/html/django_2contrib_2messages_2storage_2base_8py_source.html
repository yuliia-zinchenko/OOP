49class BaseStorage:
50 """
51 This is the base backend for temporary message storage.
52
53 This is not a complete class; to be a usable storage backend, it must be
54 subclassed and the two methods ``_get`` and ``_store`` overridden.
55 """

57 def __init__(self, request, *args, **kwargs):

91 def _get(self, *args, **kwargs):
92 """
93 Retrieve a list of stored messages. Return a tuple of the messages

128 def update(self, response):
129 """
130 Store all unread messages.
131
132 If the backend has yet to be iterated, store previously stored messages
133 again. Otherwise, only store messages added after the last iteration.
134 """
135 self._prepare_messages(self._queued_messages)
136 if self.used:
137 return self._store(self._queued_messages, response)
138 elif self.added_new:
139 messages = self._loaded_messages + self._queued_messages
140 return self._store(messages, response)

142 def add(self, level, message, extra_tags=""):
143 """
144 Queue a message to be stored.
145
146 The message is only queued if it contained something and its level is
147 not less than the recording level (``self.level``).
148 """
149 if not message:
150 return
151 # Check that the message level is not less than the recording level.
152 level = int(level)
153 if level < self.level:
154 return
155 # Add the message.
156 self.added_new = True
157 message = Message(level, message, extra_tags=extra_tags)
158 self._queued_messages.append(message)
