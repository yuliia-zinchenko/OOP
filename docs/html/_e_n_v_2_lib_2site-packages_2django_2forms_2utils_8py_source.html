20def flatatt(attrs):
21 """
22 Convert a dictionary of attributes to a single string.
23 The returned string will contain a leading space followed by key="value",
24 XML-style pairs. In the case of a boolean value, the key will appear
25 without a value. It is assumed that the keys do not need to be
26 XML-escaped. If the passed dictionary is empty, then return an empty
27 string.
28
29 The result is passed through 'mark_safe' (by way of 'format_html_join').
30 """
31 key_value_attrs = []
32 boolean_attrs = []
33 for attr, value in attrs.items():
34 if isinstance(value, bool):
35 if value:
36 boolean_attrs.append((attr,))
37 elif value is not None:
38 key_value_attrs.append((attr, value))
39
40 return format_html_join("", ' {}="{}"', sorted(key_value_attrs)) + format_html_join(
41 "", " {}", sorted(boolean_attrs)
42 )

213def from_current_timezone(value):
214 """
215 When time zone support is enabled, convert naive datetimes
216 entered in the current time zone to aware datetimes.
217 """
218 if settings.USE_TZ and value is not None and timezone.is_naive(value):
219 current_timezone = timezone.get_current_timezone()
220 try:
221 if timezone._datetime_ambiguous_or_imaginary(value, current_timezone):
222 raise ValueError("Ambiguous or non-existent time.")
223 return timezone.make_aware(value, current_timezone)
224 except Exception as exc:
225 raise ValidationError(
226 _(
227 "%(datetime)s couldn’t be interpreted "
228 "in time zone %(current_timezone)s; it "
229 "may be ambiguous or it may not exist."
230 ),
231 code="ambiguous_timezone",
232 params={"datetime": value, "current_timezone": current_timezone},
233 ) from exc
234 return value
