28def _find_pyproject() -> Path:
29 """Search for file pyproject.toml in the parent directories recursively.
30
31 It resolves symlinks, so if there is any symlink up in the tree, it does not respect them
32 """
33 current_dir = Path.cwd().resolve()
34 is_root = False
35 while not is_root:
36 if (current_dir / PYPROJECT_NAME).is_file():
37 return current_dir / PYPROJECT_NAME
38 is_root = (
39 current_dir == current_dir.parent
40 or (current_dir / ".git").is_dir()
41 or (current_dir / ".hg").is_dir()
42 )
43 current_dir = current_dir.parent
44
45 return current_dir

48def _toml_has_config(path: Path | str) -> bool:
49 with open(path, mode="rb") as toml_handle:
50 try:
51 content = tomllib.load(toml_handle)
52 except tomllib.TOMLDecodeError as error:
53 print(f"Failed to load '{path}': {error}")
54 return False
55 return "pylint" in content.get("tool", [])

58def _cfg_or_ini_has_config(path: Path | str) -> bool:
59 parser = configparser.ConfigParser()
60 try:
61 parser.read(path, encoding="utf-8")
62 except configparser.Error:
63 return False
64 return any(
65 section == "pylint" or section.startswith("pylint.")
66 for section in parser.sections()
67 )

70def _yield_default_files() -> Iterator[Path]:
71 """Iterate over the default config file names and see if they exist."""
72 for config_name in CONFIG_NAMES:
73 try:
74 if config_name.is_file():
75 if config_name.suffix == ".toml" and not _toml_has_config(config_name):
76 continue
77 if config_name.suffix in {
78 ".cfg",
79 ".ini",
80 } and not _cfg_or_ini_has_config(config_name):
81 continue
82
83 yield config_name.resolve()
84 except OSError:
85 pass

88def _find_project_config() -> Iterator[Path]:
89 """Traverse up the directory tree to find a config file.
90
91 Stop if no '__init__' is found and thus we are no longer in a package.
92 """
93 if Path("__init__.py").is_file():
94 curdir = Path(os.getcwd()).resolve()
95 while (curdir / "__init__.py").is_file():
96 curdir = curdir.parent
97 for rc_name in RC_NAMES:
98 rc_path = curdir / rc_name
99 if rc_path.is_file():
100 yield rc_path.resolve()

103def _find_config_in_home_or_environment() -> Iterator[Path]:
104 """Find a config file in the specified environment var or the home directory."""
105 if "PYLINTRC" in os.environ and Path(os.environ["PYLINTRC"]).exists():
106 if Path(os.environ["PYLINTRC"]).is_file():
107 yield Path(os.environ["PYLINTRC"]).resolve()
108 else:
109 try:
110 user_home = Path.home()
111 except RuntimeError:
112 # If the home directory does not exist a RuntimeError will be raised
113 user_home = None
114
115 if user_home is not None and str(user_home) not in ("~", "/root"):
116 home_rc = user_home / ".pylintrc"
117 if home_rc.is_file():
118 yield home_rc.resolve()
119
120 home_rc = user_home / ".config" / "pylintrc"
121 if home_rc.is_file():
122 yield home_rc.resolve()

125def find_default_config_files() -> Iterator[Path]:
126 """Find all possible config files."""
127 yield from _yield_default_files()

129 try:
130 yield from _find_project_config()
