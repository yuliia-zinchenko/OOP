30class NestedMinMaxChecker(BaseChecker):
31 """Multiple nested min/max calls on the same line will raise multiple messages.
32
33 This behaviour is intended as it would slow down the checker to check
34 for nested call with minimal benefits.
35 """
36
37 FUNC_NAMES = ("builtins.min", "builtins.max")
38
39 name = "nested_min_max"
40 msgs = {
41 "W3301": (
42 "Do not use nested call of '%s'; it's possible to do '%s' instead",
43 "nested-min-max",
44 "Nested calls ``min(1, min(2, 3))`` can be rewritten as ``min(1, 2, 3)``.",
45 )
46 }
47
48 @classmethod
49 def is_min_max_call(cls, node: nodes.NodeNG) -> bool:
50 if not isinstance(node, nodes.Call):
51 return False
52
53 inferred = safe_infer(node.func)
54 return (
55 isinstance(inferred, nodes.FunctionDef)
56 and inferred.qname() in cls.FUNC_NAMES
57 )
58
59 @classmethod
60 def get_redundant_calls(cls, node: nodes.Call) -> list[nodes.Call]:
61 return [
62 arg
63 for arg in node.args
64 if (
65 cls.is_min_max_call(arg)
66 and arg.func.name == node.func.name
67 # Nesting is useful for finding the maximum in a matrix.
68 # Allow: max(max([[1, 2, 3], [4, 5, 6]]))
69 # Meaning, redundant call only if parent max call has more than 1 arg.
70 and len(arg.parent.args) > 1
71 )
72 ]
73
74 @only_required_for_messages("nested-min-max")
75 def visit_call(self, node: nodes.Call) -> None:
76 if not self.is_min_max_call(node):
77 return
78
79 redundant_calls = self.get_redundant_calls(node)
80 if not redundant_calls:
81 return
82
83 fixed_node = copy.copy(node)
84 while len(redundant_calls) > 0:
85 for i, arg in enumerate(fixed_node.args):
86 # Exclude any calls with generator expressions as there is no
87 # clear better suggestion for them.
88 if isinstance(arg, nodes.Call) and any(
89 isinstance(a, nodes.GeneratorExp) for a in arg.args
90 ):
91 return
92
93 if arg in redundant_calls:
94 fixed_node.args = (
95 fixed_node.args[:i] + arg.args + fixed_node.args[i + 1 :]
96 )
97 break
98
99 redundant_calls = self.get_redundant_calls(fixed_node)
100
101 for idx, arg in enumerate(fixed_node.args):
102 if not isinstance(arg, nodes.Const):
103 if self._is_splattable_expression(arg):
104 splat_node = nodes.Starred(
105 ctx=Context.Load,
106 lineno=arg.lineno,
107 col_offset=0,
108 parent=nodes.NodeNG(
109 lineno=None,
110 col_offset=None,
111 end_lineno=None,
112 end_col_offset=None,
113 parent=None,
114 ),
115 end_lineno=0,
116 end_col_offset=0,
117 )
118 splat_node.value = arg
119 fixed_node.args = (
120 fixed_node.args[:idx]
121 + [splat_node]
122 + fixed_node.args[idx + 1 : idx]
123 )
124
125 self.add_message(
126 "nested-min-max",
127 node=node,
128 args=(node.func.name, fixed_node.as_string()),
129 confidence=INFERENCE,
130 )
131
132 def _is_splattable_expression(self, arg: nodes.NodeNG) -> bool:
133 """Returns true if expression under min/max could be converted to splat
134 expression.
135 """
136 # Support sequence addition (operator __add__)
137 if isinstance(arg, nodes.BinOp) and arg.op == "+":
138 return self._is_splattable_expression(
139 arg.left
140 ) and self._is_splattable_expression(arg.right)
141 # Support dict merge (operator __or__)
142 if isinstance(arg, nodes.BinOp) and arg.op == "|":
143 return self._is_splattable_expression(
144 arg.left
145 ) and self._is_splattable_expression(arg.right)
146
147 inferred = safe_infer(arg)
148 if inferred and inferred.pytype() in {"builtins.list", "builtins.tuple"}:
149 return True
150 if isinstance(
151 inferred or arg,
152 (
153 nodes.List,
154 nodes.Tuple,
155 nodes.Set,
156 nodes.ListComp,
157 nodes.DictComp,
158 *DICT_TYPES,
159 ),
160 ):
161 return True
162
163 return False
