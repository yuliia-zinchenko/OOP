47 def __init__(
48 self, queryset, chunked_fetch=False, chunk_size=GET_ITERATOR_CHUNK_SIZE
49 ):

54 async def _async_generator(self):
55 # Generators don't actually start running until the first time you call
56 # next() on them, so make the generator object in the async thread and
57 # then repeatedly dispatch to it in a sync thread.
58 sync_generator = self.__iter__()
59
60 def next_slice(gen):
61 return list(islice(gen, self.chunk_size))
62
63 while True:
64 chunk = await sync_to_async(next_slice)(sync_generator)
65 for item in chunk:
66 yield item
67 if len(chunk) < self.chunk_size:
68 break
69
70 # __aiter__() is a *synchronous* method that has to then return an
71 # *asynchronous* iterator/generator. Thus, nest an async generator inside
72 # it.
73 # This is a generic iterable converter for now, and is going to suffer a
74 # performance penalty on large sets of items due to the cost of crossing
75 # over the sync barrier for each chunk. Custom __aiter__() methods should
76 # be added to each Iterable subclass, but that needs some work in the
77 # Compiler first.
78 def __aiter__(self):
79 return self._async_generator()

85 def __iter__(self):

88 compiler = queryset.query.get_compiler(using=db)
89 # Execute the query. This will also fill compiler.select, klass_info,
90 # and annotations.
91 results = compiler.execute_sql(
92 chunked_fetch=self.chunked_fetch, chunk_size=self.chunk_size
93 )
94 select, klass_info, annotation_col_map = (
95 compiler.select,
96 compiler.klass_info,
97 compiler.annotation_col_map,
98 )
99 model_cls = klass_info["model"]
100 select_fields = klass_info["select_fields"]
101 model_fields_start, model_fields_end = select_fields[0], select_fields[-1] + 1
102 init_list = [
103 f[0].target.attname for f in select[model_fields_start:model_fields_end]
104 ]
105 related_populators = get_related_populators(klass_info, select, db)
106 known_related_objects = [
107 (
108 field,
109 related_objs,
110 operator.attrgetter(
111 *[
112 (
113 field.attname
114 if from_field == "self"
115 else queryset.model._meta.get_field(from_field).attname
116 )
117 for from_field in field.from_fields
118 ]
119 ),
120 )
121 for field, related_objs in queryset._known_related_objects.items()
122 ]

124 obj = model_cls.from_db(

126 )

151 Iterable that yields a model instance for each row from a raw queryset.

154 def __iter__(self):
155 # Cache some things for performance reasons outside the loop.
156 db = self.queryset.db
157 query = self.queryset.query
158 connection = connections[db]
159 compiler = connection.ops.compiler("SQLCompiler")(query, connection, db)
160 query_iterator = iter(query)
161
162 try:
163 (
164 model_init_names,
165 model_init_pos,
166 annotation_fields,
167 ) = self.queryset.resolve_model_init_order()
168 model_cls = self.queryset.model
169 if model_cls._meta.pk.attname not in model_init_names:
170 raise exceptions.FieldDoesNotExist(
171 "Raw query must include the primary key"
172 )
173 fields = [self.queryset.model_fields.get(c) for c in self.queryset.columns]
174 converters = compiler.get_converters(
175 [f.get_col(f.model._meta.db_table) if f else None for f in fields]

178 query_iterator = compiler.apply_converters(query_iterator, converters)

185 setattr(instance, column, values[pos])

767 Insert each of the instances into the database. Do *not* call
768 save() on each of the instances, do not send any pre/post_save
769 signals, and do not set the primary key attribute if it is an
770 autoincrement field (except if features.can_return_rows_from_bulk_insert=True).
771 Multi-table models are not supported.
772 """
773 # When you bulk insert you don't get the primary keys back (if it's an
774 # autoincrement, except if can_return_rows_from_bulk_insert=True), so
775 # you can't insert into the child tables which references this. There
776 # are two workarounds:
777 # 1) This could be implemented if you didn't have an autoincrement pk
778 # 2) You could do it by doing O(n) normal inserts into the parent
779 # tables to get the primary keys back and then doing a single bulk
780 # insert into the childmost table.
781 # We currently set the primary keys on the objects when using
782 # PostgreSQL via the RETURNING ID clause. It should be possible for
783 # Oracle as well, but the semantics for extracting the primary keys is
784 # trickier so it's not done yet.
785 if batch_size is not None and batch_size <= 0:
786 raise ValueError("Batch size must be a positive integer.")
787 # Check that the parents share the same concrete model with the our
788 # model to detect the inheritance pattern ConcreteGrandParent ->
789 # MultiTableParent -> ProxyChild. Simply checking self.model._meta.proxy
790 # would not identify that case as involving multiple tables.
791 for parent in self.model._meta.all_parents:
792 if parent._meta.concrete_model is not self.model._meta.concrete_model:
793 raise ValueError("Can't bulk create a multi-table inherited model")
794 if not objs:

797 if unique_fields:

799 unique_fields = [
800 self.model._meta.get_field(opts.pk.name if name == "pk" else name)
801 for name in unique_fields

804 update_fields = [self.model._meta.get_field(name) for name in update_fields]

806 ignore_conflicts,
807 update_conflicts,
808 update_fields,

810 )

813 objs = list(objs)

815 with transaction.atomic(using=self.db, savepoint=False):

817 if objs_with_pk:

819 objs_with_pk,
820 fields,
821 batch_size,
822 on_conflict=on_conflict,
823 update_fields=update_fields,
824 unique_fields=unique_fields,
825 )

827 for result, field in zip(results, opts.db_returning_fields):
828 if field != opts.pk:
829 setattr(obj_with_pk, field.attname, result)
830 for obj_with_pk in objs_with_pk:
831 obj_with_pk._state.adding = False
832 obj_with_pk._state.db = self.db
833 if objs_without_pk:
834 fields = [f for f in fields if not isinstance(f, AutoField)]
835 returned_columns = self._batched_insert(
836 objs_without_pk,
837 fields,
838 batch_size,
839 on_conflict=on_conflict,
840 update_fields=update_fields,
841 unique_fields=unique_fields,
842 )
843 connection = connections[self.db]

845 connection.features.can_return_rows_from_bulk_insert
846 and on_conflict is None

849 for obj_without_pk, results in zip(objs_without_pk, returned_columns):
850 for result, field in zip(results, opts.db_returning_fields):
851 setattr(obj_without_pk, field.attname, result)
852 obj_without_pk._state.adding = False
853 obj_without_pk._state.db = self.db

875 def bulk_update(self, objs, fields, batch_size=None):
876 """
877 Update the given fields in each of the given objects in the database.
878 """
879 if batch_size is not None and batch_size <= 0:
880 raise ValueError("Batch size must be a positive integer.")
881 if not fields:
882 raise ValueError("Field names must be given to bulk_update().")
883 objs = tuple(objs)
884 if any(obj.pk is None for obj in objs):
885 raise ValueError("All bulk_update() objects must have a primary key set.")
886 fields = [self.model._meta.get_field(name) for name in fields]
887 if any(not f.concrete or f.many_to_many for f in fields):
888 raise ValueError("bulk_update() can only be used with concrete fields.")
889 if any(f.primary_key for f in fields):
890 raise ValueError("bulk_update() cannot be used with primary key fields.")
891 if not objs:
892 return 0
893 for obj in objs:
894 obj._prepare_related_fields_for_save(
895 operation_name="bulk_update", fields=fields
896 )
897 # PK is used twice in the resulting update query, once in the filter
898 # and once in the WHEN. Each field will also have one CAST.
899 self._for_write = True
900 connection = connections[self.db]
901 max_batch_size = connection.ops.bulk_batch_size(["pk", "pk"] + fields, objs)
902 batch_size = min(batch_size, max_batch_size) if batch_size else max_batch_size
903 requires_casting = connection.features.requires_casted_case_in_updates
904 batches = (objs[i : i + batch_size] for i in range(0, len(objs), batch_size))
905 updates = []
906 for batch_objs in batches:
907 update_kwargs = {}
908 for field in fields:
909 when_statements = []
910 for obj in batch_objs:
911 attr = getattr(obj, field.attname)
912 if not hasattr(attr, "resolve_expression"):
913 attr = Value(attr, output_field=field)
914 when_statements.append(When(pk=obj.pk, then=attr))
915 case_statement = Case(*when_statements, output_field=field)
916 if requires_casting:
917 case_statement = Cast(case_statement, output_field=field)
918 update_kwargs[field.attname] = case_statement
919 updates.append(([obj.pk for obj in batch_objs], update_kwargs))
920 rows_updated = 0
921 queryset = self.using(self.db)
922 with transaction.atomic(using=self.db, savepoint=False):
923 for pks, update_kwargs in updates:
924 rows_updated += queryset.filter(pk__in=pks).update(**update_kwargs)
925 return rows_updated

2662 # in the same order model_cls.__init__ expects them, so we
2663 # can instantiate by model_cls(*row[cols_start:cols_end])

2665 # class, then we want to reuse the already selected parent
2666 # data. However, in this case the parent data isn't necessarily
2667 # in the same order that Model.__init__ expects it to be, so

2670 # in the order __init__ expects it.

2672 # model data. Used to check if a related object exists at all.
2673 # - init_list: the field attnames fetched from the database. For
2674 # deferred models this isn't the same as all attnames of the
2675 # model's fields.
2676 # - related_populators: a list of RelatedPopulator instances if
2677 # select_related() descends to related models from this model.

2679 # the object being populated and on the remote object. Usually
