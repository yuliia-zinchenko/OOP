30def _safe_infer_call_result(
31 node: nodes.FunctionDef,
32 caller: nodes.FunctionDef,
33 context: InferenceContext | None = None,
34) -> InferenceResult | None:
35 """Safely infer the return value of a function.
36
37 Returns None if inference failed or if there is some ambiguity (more than
38 one node has been inferred). Otherwise, returns inferred value.
39 """
40 try:
41 inferit = node.infer_call_result(caller, context=context)
42 value = next(inferit)
43 except astroid.InferenceError:
44 return None # inference failed
45 except StopIteration:
46 return None # no values inferred
47 try:
48 next(inferit)
49 return None # there is ambiguity on the inferred node
50 except astroid.InferenceError:
51 return None # there is some kind of ambiguity
52 except StopIteration:
53 return value

179 def visit_functiondef(self, node: nodes.FunctionDef) -> None:
180 if not node.is_method():
181 return
182
183 inferred = _safe_infer_call_result(node, node)
184 # Only want to check types that we are able to infer
185 if (
186 inferred
187 and node.name in self._protocol_map
188 and not is_function_body_ellipsis(node)
189 ):
190 self._protocol_map[node.name](node, inferred)
191
192 if node.name in PYMETHODS:
193 self._check_unexpected_method_signature(node)

197 def _check_unexpected_method_signature(self, node: nodes.FunctionDef) -> None:
198 expected_params = SPECIAL_METHODS_PARAMS[node.name]
199
200 if expected_params is None:
201 # This can support a variable number of parameters.
202 return
203 if not node.args.args and not node.args.vararg:
204 # Method has no parameter, will be caught
205 # by no-method-argument.
206 return
207
208 if decorated_with(node, ["builtins.staticmethod"]):
209 # We expect to not take in consideration self.
210 all_args = node.args.args
211 else:
212 all_args = node.args.args[1:]
213 mandatory = len(all_args) - len(node.args.defaults)
214 optional = len(node.args.defaults)
215 current_params = mandatory + optional
216
217 emit = False # If we don't know we choose a false negative
218 if isinstance(expected_params, tuple):
219 # The expected number of parameters can be any value from this
220 # tuple, although the user should implement the method
221 # to take all of them in consideration.
222 emit = mandatory not in expected_params
223 # mypy thinks that expected_params has type tuple[int, int] | int | None
224 # But at this point it must be 'tuple[int, int]' because of the type check
225 expected_params = f"between {expected_params[0]} or {expected_params[1]}" # type: ignore[assignment]
226 else:
227 # If the number of mandatory parameters doesn't
228 # suffice, the expected parameters for this
229 # function will be deduced from the optional
230 # parameters.
231 rest = expected_params - mandatory
232 if rest == 0:
233 emit = False
234 elif rest < 0:
235 emit = True
236 elif rest > 0:
237 emit = not ((optional - rest) >= 0 or node.args.vararg)
238
239 if emit:
240 verb = "was" if current_params <= 1 else "were"
241 self.add_message(
242 "unexpected-special-method-signature",
243 args=(node.name, expected_params, current_params, verb),
244 node=node,
245 )
