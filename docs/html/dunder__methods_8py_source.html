21class DunderCallChecker(BaseChecker):
22 """Check for unnecessary dunder method calls.
23
24 Docs: https://docs.python.org/3/reference/datamodel.html#basic-customization
25 We exclude names in list pylint.constants.EXTRA_DUNDER_METHODS such as
26 __index__ (see https://github.com/pylint-dev/pylint/issues/6795)
27 since these either have no alternative method of being called or
28 have a genuine use case for being called manually.
29
30 Additionally, we exclude classes that are not instantiated since these
31 might be used to access the dunder methods of a base class of an instance.
32 We also exclude dunder method calls on super() since
33 these can't be written in an alternative manner.
34 """
35
36 name = "unnecessary-dunder-call"
37 msgs = {
38 "C2801": (
39 "Unnecessarily calls dunder method %s. %s.",
40 "unnecessary-dunder-call",
41 "Used when a dunder method is manually called instead "
42 "of using the corresponding function/method/operator.",
43 ),
44 }
45 options = ()
46
47 def open(self) -> None:
48 self._dunder_methods: dict[str, str] = {}
49 for since_vers, dunder_methods in DUNDER_METHODS.items():
50 if since_vers <= self.linter.config.py_version:
51 self._dunder_methods.update(dunder_methods)
52
53 @staticmethod
54 def within_dunder_or_lambda_def(node: nodes.NodeNG) -> bool:
55 """Check if dunder method call is within a dunder method definition."""
56 parent = node.parent
57 while parent is not None:
58 if (
59 isinstance(parent, nodes.FunctionDef)
60 and parent.name.startswith("__")
61 and parent.name.endswith("__")
62 or DunderCallChecker.is_lambda_rule_exception(parent, node)
63 ):
64 return True
65 parent = parent.parent
66 return False
67
68 @staticmethod
69 def is_lambda_rule_exception(ancestor: nodes.NodeNG, node: nodes.NodeNG) -> bool:
70 return (
71 isinstance(ancestor, nodes.Lambda)
72 and node.func.attrname in UNNECESSARY_DUNDER_CALL_LAMBDA_EXCEPTIONS
73 )
74
75 def visit_call(self, node: nodes.Call) -> None:
76 """Check if method being called is an unnecessary dunder method."""
77 if (
78 isinstance(node.func, nodes.Attribute)
79 and node.func.attrname in self._dunder_methods
80 and not self.within_dunder_or_lambda_def(node)
81 and not (
82 isinstance(node.func.expr, nodes.Call)
83 and isinstance(node.func.expr.func, nodes.Name)
84 and node.func.expr.func.name == "super"
85 )
86 ):
87 inf_expr = safe_infer(node.func.expr)
88 if not (
89 inf_expr is None or isinstance(inf_expr, (Instance, UninferableBase))
90 ):
91 # Skip dunder calls to non instantiated classes.
92 return
93
94 self.add_message(
95 "unnecessary-dunder-call",
96 node=node,
97 args=(node.func.attrname, self._dunder_methods[node.func.attrname]),
98 confidence=HIGH,
99 )
