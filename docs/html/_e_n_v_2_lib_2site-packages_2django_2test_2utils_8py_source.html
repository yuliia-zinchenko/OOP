73class ContextList(list):
74 """
75 A wrapper that provides direct key access to context items contained
76 in a list of context objects.
77 """
78
79 def __getitem__(self, key):
80 if isinstance(key, str):
81 for subcontext in self:
82 if key in subcontext:
83 return subcontext[key]
84 raise KeyError(key)

260def dependency_ordered(test_databases, dependencies):
261 """
262 Reorder test_databases into an order that honors the dependencies
263 described in TEST[DEPENDENCIES].
264 """
265 ordered_test_databases = []
266 resolved_databases = set()
267
268 # Maps db signature to dependencies of all its aliases
269 dependencies_map = {}
270
271 # Check that no database depends on its own alias
272 for sig, (_, aliases) in test_databases:
273 all_deps = set()
274 for alias in aliases:
275 all_deps.update(dependencies.get(alias, []))
276 if not all_deps.isdisjoint(aliases):
277 raise ImproperlyConfigured(
278 "Circular dependency: databases %r depend on each other, "
279 "but are aliases." % aliases
280 )
281 dependencies_map[sig] = all_deps
282
283 while test_databases:
284 changed = False
285 deferred = []
286
287 # Try to find a DB that has all its dependencies met
288 for signature, (db_name, aliases) in test_databases:
289 if dependencies_map[signature].issubset(resolved_databases):
290 resolved_databases.update(aliases)
291 ordered_test_databases.append((signature, (db_name, aliases)))
292 changed = True
293 else:
294 deferred.append((signature, (db_name, aliases)))
295
296 if not changed:
297 raise ImproperlyConfigured("Circular dependency in TEST[DEPENDENCIES]")
298 test_databases = deferred
299 return ordered_test_databases

302def get_unique_databases_and_mirrors(aliases=None):
303 """
304 Figure out which databases actually need to be created.
305
306 Deduplicate entries in DATABASES that correspond the same database or are
307 configured as test mirrors.
308
309 Return two values:
310 - test_databases: ordered mapping of signatures to (name, list of aliases)
311 where all aliases share the same underlying database.
312 - mirrored_aliases: mapping of mirror aliases to original aliases.
313 """
314 if aliases is None:
315 aliases = connections
316 mirrored_aliases = {}
317 test_databases = {}
318 dependencies = {}
319 default_sig = connections[DEFAULT_DB_ALIAS].creation.test_db_signature()
320
321 for alias in connections:
322 connection = connections[alias]
323 test_settings = connection.settings_dict["TEST"]
324
325 if test_settings["MIRROR"]:
326 # If the database is marked as a test mirror, save the alias.
327 mirrored_aliases[alias] = test_settings["MIRROR"]
328 elif alias in aliases:
329 # Store a tuple with DB parameters that uniquely identify it.
330 # If we have two aliases with the same values for that tuple,
331 # we only need to create the test database once.
332 item = test_databases.setdefault(
333 connection.creation.test_db_signature(),
334 (connection.settings_dict["NAME"], []),
335 )
336 # The default database must be the first because data migrations
337 # use the default alias by default.
338 if alias == DEFAULT_DB_ALIAS:
339 item[1].insert(0, alias)
340 else:
341 item[1].append(alias)
342
343 if "DEPENDENCIES" in test_settings:
344 dependencies[alias] = test_settings["DEPENDENCIES"]
345 else:
346 if (
347 alias != DEFAULT_DB_ALIAS
348 and connection.creation.test_db_signature() != default_sig
349 ):
350 dependencies[alias] = test_settings.get(
351 "DEPENDENCIES", [DEFAULT_DB_ALIAS]
352 )
353
354 test_databases = dict(dependency_ordered(test_databases.items(), dependencies))
355 return test_databases, mirrored_aliases

384class TestContextDecorator:
385 """
386 A base class that can either be used as a context manager during tests
387 or as a test function or unittest.TestCase subclass decorator to perform
388 temporary alterations.
389
390 `attr_name`: attribute assigned the return value of enable() if used as
391 a class decorator.
392
393 `kwarg_name`: keyword argument passing the return value of enable() if
394 used as a function decorator.
395 """
396
397 def __init__(self, attr_name=None, kwarg_name=None):
398 self.attr_name = attr_name
399 self.kwarg_name = kwarg_name
400
401 def enable(self):
402 raise NotImplementedError
403
404 def disable(self):
405 raise NotImplementedError
406
407 def __enter__(self):
408 return self.enable()
409
410 def __exit__(self, exc_type, exc_value, traceback):
411 self.disable()
412
413 def decorate_class(self, cls):
414 if issubclass(cls, TestCase):
415 decorated_setUp = cls.setUp
416
417 def setUp(inner_self):
418 context = self.enable()
419 inner_self.addCleanup(self.disable)
420 if self.attr_name:
421 setattr(inner_self, self.attr_name, context)
422 decorated_setUp(inner_self)
423
424 cls.setUp = setUp
425 return cls
426 raise TypeError("Can only decorate subclasses of unittest.TestCase")
427
428 def decorate_callable(self, func):
429 if iscoroutinefunction(func):
430 # If the inner function is an async function, we must execute async
431 # as well so that the `with` statement executes at the right time.
432 @wraps(func)
433 async def inner(*args, **kwargs):
434 with self as context:
435 if self.kwarg_name:
436 kwargs[self.kwarg_name] = context
437 return await func(*args, **kwargs)
438
439 else:
440
441 @wraps(func)
442 def inner(*args, **kwargs):
443 with self as context:
444 if self.kwarg_name:
445 kwargs[self.kwarg_name] = context
446 return func(*args, **kwargs)
447
448 return inner
449
450 def __call__(self, decorated):
451 if isinstance(decorated, type):
452 return self.decorate_class(decorated)
453 elif callable(decorated):
454 return self.decorate_callable(decorated)
455 raise TypeError("Cannot decorate object of type %s" % type(decorated))

458class override_settings(TestContextDecorator):
459 """
460 Act as either a decorator or a context manager. If it's a decorator, take a
461 function and return a wrapped function. If it's a contextmanager, use it
462 with the ``with`` statement. In either event, entering/exiting are called
463 before and after, respectively, the function/block is executed.
464 """
465
466 enable_exception = None
467
468 def __init__(self, **kwargs):
469 self.options = kwargs
470 super().__init__()
471
472 def enable(self):
473 # Keep this code at the beginning to leave the settings unchanged
474 # in case it raises an exception because INSTALLED_APPS is invalid.
475 if "INSTALLED_APPS" in self.options:
476 try:
477 apps.set_installed_apps(self.options["INSTALLED_APPS"])
478 except Exception:
479 apps.unset_installed_apps()
480 raise
481 override = UserSettingsHolder(settings._wrapped)
482 for key, new_value in self.options.items():
483 setattr(override, key, new_value)
484 self.wrapped = settings._wrapped
485 settings._wrapped = override
486 for key, new_value in self.options.items():
487 try:
488 setting_changed.send(
489 sender=settings._wrapped.__class__,
490 setting=key,
491 value=new_value,
492 enter=True,
493 )
494 except Exception as exc:
495 self.enable_exception = exc
496 self.disable()
497
498 def disable(self):
499 if "INSTALLED_APPS" in self.options:
500 apps.unset_installed_apps()
501 settings._wrapped = self.wrapped
502 del self.wrapped
503 responses = []
504 for key in self.options:
505 new_value = getattr(settings, key, None)
506 responses_for_setting = setting_changed.send_robust(
507 sender=settings._wrapped.__class__,
508 setting=key,
509 value=new_value,
510 enter=False,
511 )
512 responses.extend(responses_for_setting)
513 if self.enable_exception is not None:
514 exc = self.enable_exception
515 self.enable_exception = None
516 raise exc
517 for _, response in responses:
518 if isinstance(response, Exception):
519 raise response

543class modify_settings(override_settings):
544 """
545 Like override_settings, but makes it possible to append, prepend, or remove
546 items instead of redefining the entire list.
547 """
548
549 def __init__(self, *args, **kwargs):
550 if args:
551 # Hack used when instantiating from SimpleTestCase.setUpClass.
552 assert not kwargs
553 self.operations = args[0]
554 else:
555 assert not args
556 self.operations = list(kwargs.items())
557 super(override_settings, self).__init__()
558
559 def save_options(self, test_func):
560 if test_func._modified_settings is None:
561 test_func._modified_settings = self.operations
562 else:
563 # Duplicate list to prevent subclasses from altering their parent.
564 test_func._modified_settings = (
565 list(test_func._modified_settings) + self.operations
566 )
567
568 def enable(self):
569 self.options = {}
570 for name, operations in self.operations:
571 try:
572 # When called from SimpleTestCase.setUpClass, values may be
573 # overridden several times; cumulate changes.
574 value = self.options[name]
575 except KeyError:
576 value = list(getattr(settings, name, []))
577 for action, items in operations.items():
578 # items may be a single value or an iterable.
579 if isinstance(items, str):
580 items = [items]
581 if action == "append":
582 value += [item for item in items if item not in value]
583 elif action == "prepend":
584 value = [item for item in items if item not in value] + value
585 elif action == "remove":
586 value = [item for item in value if item not in items]
587 else:
588 raise ValueError("Unsupported action: %s" % action)
589 self.options[name] = value
590 super().enable()

624def compare_xml(want, got):
625 """
626 Try to do a 'xml-comparison' of want and got. Plain string comparison
627 doesn't always work because, for example, attribute ordering should not be
628 important. Ignore comment nodes, processing instructions, document type
629 node, and leading and trailing whitespaces.
630
631 Based on https://github.com/lxml/lxml/blob/master/src/lxml/doctestcompare.py
632 """
633 _norm_whitespace_re = re.compile(r"[ \t\n][ \t\n]+")
634
635 def norm_whitespace(v):
636 return _norm_whitespace_re.sub(" ", v)
637
638 def child_text(element):
639 return "".join(
640 c.data for c in element.childNodes if c.nodeType == Node.TEXT_NODE
641 )
642
643 def children(element):
644 return [c for c in element.childNodes if c.nodeType == Node.ELEMENT_NODE]
645
646 def norm_child_text(element):
647 return norm_whitespace(child_text(element))
648
649 def attrs_dict(element):
650 return dict(element.attributes.items())
651
652 def check_element(want_element, got_element):
653 if want_element.tagName != got_element.tagName:
654 return False
655 if norm_child_text(want_element) != norm_child_text(got_element):
656 return False
657 if attrs_dict(want_element) != attrs_dict(got_element):
658 return False
659 want_children = children(want_element)
660 got_children = children(got_element)
661 if len(want_children) != len(got_children):
662 return False
663 return all(
664 check_element(want, got) for want, got in zip(want_children, got_children)
665 )
666
667 def first_node(document):
668 for node in document.childNodes:
669 if node.nodeType not in (
670 Node.COMMENT_NODE,
671 Node.DOCUMENT_TYPE_NODE,
672 Node.PROCESSING_INSTRUCTION_NODE,
673 ):
674 return node
675
676 want = want.strip().replace("\\n", "\n")
677 got = got.strip().replace("\\n", "\n")
678
679 # If the string is not a complete xml document, we may need to add a
680 # root element. This allow us to compare fragments, like "<foo/><bar/>"
681 if not want.startswith("<?xml"):
682 wrapper = "<root>%s</root>"
683 want = wrapper % want
684 got = wrapper % got
685
686 # Parse the want and got strings, and compare the parsings.
687 want_root = first_node(parseString(want))
688 got_root = first_node(parseString(got))
689
690 return check_element(want_root, got_root)
