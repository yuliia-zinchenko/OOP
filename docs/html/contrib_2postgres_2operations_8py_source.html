173 def __init__(self, name, locale, *, provider="libc", deterministic=True):
174 self.name = name
175 self.locale = locale

182 def deconstruct(self):
183 kwargs = {"name": self.name, "locale": self.locale}
184 if self.provider and self.provider != "libc":
185 kwargs["provider"] = self.provider
186 if self.deterministic is False:
187 kwargs["deterministic"] = self.deterministic
188 return (
189 self.__class__.__qualname__,
190 [],
191 kwargs,
192 )
193
194 def create_collation(self, schema_editor):
195 args = {"locale": schema_editor.quote_name(self.locale)}
196 if self.provider != "libc":
197 args["provider"] = schema_editor.quote_name(self.provider)
198 if self.deterministic is False:
199 args["deterministic"] = "false"
200 schema_editor.execute(
201 "CREATE COLLATION %(name)s (%(args)s)"
202 % {
203 "name": schema_editor.quote_name(self.name),
204 "args": ", ".join(
205 f"{option}={value}" for option, value in args.items()
206 ),
207 }
208 )

210 def remove_collation(self, schema_editor):
211 schema_editor.execute(
212 "DROP COLLATION %s" % schema_editor.quote_name(self.name),
