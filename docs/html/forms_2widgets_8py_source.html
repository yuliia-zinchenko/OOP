81 @property

148 @staticmethod
149 def merge(*lists):
150 """
151 Merge lists while trying to keep the relative order of the elements.
152 Warn if the lists have the same elements in a different relative order.
153
154 For static assets it can be important to have them included in the DOM
155 in a certain order. In JavaScript you may not be able to reference a
156 global or in CSS you might want to override a style.
157 """
158 ts = TopologicalSorter()
159 for head, *tail in filter(None, lists):
160 ts.add(head) # Ensure that the first items are included.
161 for item in tail:
162 if head != item: # Avoid circular dependency to self.
163 ts.add(item, head)
164 head = item
165 try:
166 return list(ts.static_order())
167 except CycleError:
168 warnings.warn(
169 "Detected duplicate Media files in an opposite order: {}".format(
170 ", ".join(repr(list_) for list_ in lists)
171 ),
172 MediaOrderConflictWarning,
173 )
174 return list(dict.fromkeys(chain.from_iterable(filter(None, lists))))

176 def __add__(self, other):
177 combined = Media()
178 combined._css_lists = self._css_lists[:]
179 combined._js_lists = self._js_lists[:]
180 for item in other._css_lists:
181 if item and item not in self._css_lists:
182 combined._css_lists.append(item)
183 for item in other._js_lists:
184 if item and item not in self._js_lists:
185 combined._js_lists.append(item)
186 return combined

189def media_property(cls):
190 def _media(self):
191 # Get the media property of the superclass, if it exists
192 sup_cls = super(cls, self)
193 try:
194 base = sup_cls.media
195 except AttributeError:
196 base = Media()
197
198 # Get the media definition for this class
199 definition = getattr(cls, "Media", None)
200 if definition:
201 extend = getattr(definition, "extend", True)
202 if extend:
203 if extend is True:
204 m = base
205 else:
206 m = Media()
207 for medium in extend:
208 m += base[medium]
209 return m + Media(definition)
210 return Media(definition)
211 return base
212
213 return property(_media)

885class MultiWidget(Widget):
886 """
887 A widget that is composed of multiple widgets.
888
889 In addition to the values added by Widget.get_context(), this widget
890 adds a list of subwidgets to the context as widget['subwidgets'].
891 These can be looped over and rendered like normal widgets.
892
893 You'll probably want to use this class with MultiValueField.
894 """
895
896 template_name = "django/forms/widgets/multiwidget.html"
897 use_fieldset = True
898
899 def __init__(self, widgets, attrs=None):
900 if isinstance(widgets, dict):
901 self.widgets_names = [("_%s" % name) if name else "" for name in widgets]
902 widgets = widgets.values()
903 else:
904 self.widgets_names = ["_%s" % i for i in range(len(widgets))]
905 self.widgets = [w() if isinstance(w, type) else w for w in widgets]
906 super().__init__(attrs)
907
908 @property
909 def is_hidden(self):
910 return all(w.is_hidden for w in self.widgets)
911
912 def get_context(self, name, value, attrs):
913 context = super().get_context(name, value, attrs)
914 if self.is_localized:
915 for widget in self.widgets:
916 widget.is_localized = self.is_localized
917 # value is a list/tuple of values, each corresponding to a widget
918 # in self.widgets.
919 if not isinstance(value, (list, tuple)):
920 value = self.decompress(value)
921
922 final_attrs = context["widget"]["attrs"]
923 input_type = final_attrs.pop("type", None)
924 id_ = final_attrs.get("id")
925 subwidgets = []
926 for i, (widget_name, widget) in enumerate(
927 zip(self.widgets_names, self.widgets)
928 ):
929 if input_type is not None:
930 widget.input_type = input_type
931 widget_name = name + widget_name
932 try:
933 widget_value = value[i]
934 except IndexError:
935 widget_value = None
936 if id_:
937 widget_attrs = final_attrs.copy()
938 widget_attrs["id"] = "%s_%s" % (id_, i)
939 else:
940 widget_attrs = final_attrs
941 subwidgets.append(
942 widget.get_context(widget_name, widget_value, widget_attrs)["widget"]
943 )
944 context["widget"]["subwidgets"] = subwidgets
945 return context
946
947 def id_for_label(self, id_):
948 return ""
949
950 def value_from_datadict(self, data, files, name):
951 return [
952 widget.value_from_datadict(data, files, name + widget_name)
953 for widget_name, widget in zip(self.widgets_names, self.widgets)
954 ]
955
956 def value_omitted_from_data(self, data, files, name):
957 return all(
958 widget.value_omitted_from_data(data, files, name + widget_name)
959 for widget_name, widget in zip(self.widgets_names, self.widgets)
960 )

963 """
964 Return a list of decompressed values for the given compressed value.
965 The given value can be assumed to be valid, but not necessarily
966 non-empty.
967 """
968 raise NotImplementedError("Subclasses must implement this method.")

970 def _get_media(self):
971 """
972 Media for a multiwidget is the combination of all media of the
973 subwidgets.
974 """
975 media = Media()
976 for w in self.widgets:
977 media += w.media
978 return media

980 media = property(_get_media)

982 def __deepcopy__(self, memo):
983 obj = super().__deepcopy__(memo)
984 obj.widgets = copy.deepcopy(self.widgets)
985 return obj

1140 def format_value(self, value):
1141 """
1142 Return a dict containing the year, month, and day of the current value.
1143 Use dict instead of a datetime to allow invalid dates such as February
1144 31 to display correctly.
1145 """
1146 year, month, day = None, None, None
1147 if isinstance(value, (datetime.date, datetime.datetime)):
1148 year, month, day = value.year, value.month, value.day
1149 elif isinstance(value, str):
1150 match = self.date_re.match(value)
1151 if match:
1152 # Convert any zeros in the date to empty strings to match the
1153 # empty option value.
1154 year, month, day = [int(val) or "" for val in match.groups()]
1155 else:
1156 input_format = get_format("DATE_INPUT_FORMATS")[0]
1157 try:
1158 d = datetime.datetime.strptime(value, input_format)
1159 except ValueError:
1160 pass
1161 else:
1162 year, month, day = d.year, d.month, d.day
1163 return {"year": year, "month": month, "day": day}

1168 escaped = False
1169 for char in fmt:
1170 if escaped:
1171 escaped = False
1172 elif char == "\\":
1173 escaped = True
1174 elif char in "Yy":
1175 yield "year"
1176 elif char in "bEFMmNn":
1177 yield "month"
1178 elif char in "dj":
1179 yield "day"
