21class PrivateImportChecker(BaseChecker):
22 name = "import-private-name"
23 msgs = {
24 "C2701": (
25 "Imported private %s (%s)",
26 "import-private-name",
27 "Used when a private module or object prefixed with _ is imported. "
28 "PEP8 guidance on Naming Conventions states that public attributes with "
29 "leading underscores should be considered private.",
30 ),
31 }

33 def __init__(self, linter: PyLinter) -> None:
34 BaseChecker.__init__(self, linter)

36 # A mapping of private names used as a type annotation to whether it is an acceptable import
37 self.all_used_type_annotations: dict[str, bool] = {}
38 self.populated_annotations = False

40 @utils.only_required_for_messages("import-private-name")
41 def visit_import(self, node: nodes.Import) -> None:
42 if utils.in_type_checking_block(node):
43 return
44 names = [name[0] for name in node.names]
45 private_names = self._get_private_imports(names)
46 private_names = self._get_type_annotation_names(node, private_names)
47 if private_names:
48 imported_identifier = "modules" if len(private_names) > 1 else "module"
49 private_name_string = ", ".join(private_names)
50 self.add_message(
51 "import-private-name",
52 node=node,
53 args=(imported_identifier, private_name_string),
54 confidence=HIGH,
55 )
56
57 @utils.only_required_for_messages("import-private-name")
58 def visit_importfrom(self, node: nodes.ImportFrom) -> None:
59 if utils.in_type_checking_block(node):
60 return
61 # Only check imported names if the module is external
62 if self.same_root_dir(node, node.modname):
63 return
64
65 names = [n[0] for n in node.names]
66
67 # Check the imported objects first. If they are all valid type annotations,
68 # the package can be private
69 private_names = self._get_type_annotation_names(node, names)
70 if not private_names:
71 return
72
73 # There are invalid imported objects, so check the name of the package
74 private_module_imports = self._get_private_imports([node.modname])
75 private_module_imports = self._get_type_annotation_names(
76 node, private_module_imports
77 )
78 if private_module_imports:
79 self.add_message(
80 "import-private-name",
81 node=node,
82 args=("module", private_module_imports[0]),
83 confidence=HIGH,
84 )
85 return # Do not emit messages on the objects if the package is private
86
87 private_names = self._get_private_imports(private_names)
88
89 if private_names:
90 imported_identifier = "objects" if len(private_names) > 1 else "object"
91 private_name_string = ", ".join(private_names)
92 self.add_message(
93 "import-private-name",
94 node=node,
95 args=(imported_identifier, private_name_string),
96 confidence=HIGH,
97 )

99 def _get_private_imports(self, names: list[str]) -> list[str]:
100 """Returns the private names from input names by a simple string check."""
101 return [name for name in names if self._name_is_private(name)]

103 @staticmethod
104 def _name_is_private(name: str) -> bool:
105 """Returns true if the name exists, starts with `_`, and if len(name) > 4
106 it is not a dunder, i.e. it does not begin and end with two underscores.
107 """
108 return (
109 bool(name)
110 and name[0] == "_"
111 and (len(name) <= 4 or name[1] != "_" or name[-2:] != "__")
112 )

114 def _get_type_annotation_names(
115 self, node: nodes.Import | nodes.ImportFrom, names: list[str]
116 ) -> list[str]:
117 """Removes from names any names that are used as type annotations with no other
118 illegal usages.
119 """
120 if names and not self.populated_annotations:
121 self._populate_type_annotations(node.root(), self.all_used_type_annotations)
122 self.populated_annotations = True
123
124 return [
125 n
126 for n in names
127 if n not in self.all_used_type_annotations
128 or (
129 n in self.all_used_type_annotations
130 and not self.all_used_type_annotations[n]
131 )
132 ]

134 def _populate_type_annotations(
135 self, node: nodes.LocalsDictNodeNG, all_used_type_annotations: dict[str, bool]
136 ) -> None:
137 """Adds to `all_used_type_annotations` all names ever used as a type annotation
138 in the node's (nested) scopes and whether they are only used as annotation.
139 """
140 for name in node.locals:
141 # If we find a private type annotation, make sure we do not mask illegal usages
142 private_name = None
143 # All the assignments using this variable that we might have to check for
144 # illegal usages later
145 name_assignments = []
146 for usage_node in node.locals[name]:
147 if isinstance(usage_node, nodes.AssignName) and isinstance(
148 usage_node.parent, (nodes.AnnAssign, nodes.Assign)
149 ):
150 assign_parent = usage_node.parent
151 if isinstance(assign_parent, nodes.AnnAssign):
152 name_assignments.append(assign_parent)
153 private_name = self._populate_type_annotations_annotation(
154 usage_node.parent.annotation, all_used_type_annotations
155 )
156 elif isinstance(assign_parent, nodes.Assign):
157 name_assignments.append(assign_parent)
158
159 if isinstance(usage_node, nodes.FunctionDef):
160 self._populate_type_annotations_function(
161 usage_node, all_used_type_annotations
162 )
163 if isinstance(usage_node, nodes.LocalsDictNodeNG):
164 self._populate_type_annotations(
165 usage_node, all_used_type_annotations
166 )
167 if private_name is not None:
168 # Found a new private annotation, make sure we are not accessing it elsewhere
169 all_used_type_annotations[private_name] = (
170 self._assignments_call_private_name(name_assignments, private_name)
171 )
172
173 def _populate_type_annotations_function(
174 self, node: nodes.FunctionDef, all_used_type_annotations: dict[str, bool]
175 ) -> None:
176 """Adds all names used as type annotation in the arguments and return type of
177 the function node into the dict `all_used_type_annotations`.
178 """
179 if node.args and node.args.annotations:
180 for annotation in node.args.annotations:
181 self._populate_type_annotations_annotation(
182 annotation, all_used_type_annotations
183 )
184 if node.returns:
185 self._populate_type_annotations_annotation(
186 node.returns, all_used_type_annotations
187 )

189 def _populate_type_annotations_annotation(
190 self,
191 node: nodes.Attribute | nodes.Subscript | nodes.Name | None,
192 all_used_type_annotations: dict[str, bool],
193 ) -> str | None:
194 """Handles the possibility of an annotation either being a Name, i.e. just type,
195 or a Subscript e.g. `Optional[type]` or an Attribute, e.g. `pylint.lint.linter`.
196 """
197 if isinstance(node, nodes.Name) and node.name not in all_used_type_annotations:
198 all_used_type_annotations[node.name] = True
199 return node.name # type: ignore[no-any-return]
200 if isinstance(node, nodes.Subscript): # e.g. Optional[List[str]]
201 # slice is the next nested type
202 self._populate_type_annotations_annotation(
203 node.slice, all_used_type_annotations
204 )
205 # value is the current type name: could be a Name or Attribute
206 return self._populate_type_annotations_annotation(
207 node.value, all_used_type_annotations
208 )
209 if isinstance(node, nodes.Attribute):
210 # An attribute is a type like `pylint.lint.pylinter`. node.expr is the next level
211 # up, could be another attribute
212 return self._populate_type_annotations_annotation(
213 node.expr, all_used_type_annotations
214 )
215 return None

217 @staticmethod
218 def _assignments_call_private_name(
219 assignments: list[nodes.AnnAssign | nodes.Assign], private_name: str
220 ) -> bool:
221 """Returns True if no assignments involve accessing `private_name`."""
222 if all(not assignment.value for assignment in assignments):
223 # Variable annotated but unassigned is not allowed because there may be
224 # possible illegal access elsewhere
225 return False
226 for assignment in assignments:
227 current_attribute = None
228 if isinstance(assignment.value, nodes.Call):
229 current_attribute = assignment.value.func
230 elif isinstance(assignment.value, nodes.Attribute):
231 current_attribute = assignment.value
232 elif isinstance(assignment.value, nodes.Name):
233 current_attribute = assignment.value.name
234 if not current_attribute:
235 continue
236 while isinstance(current_attribute, (nodes.Attribute, nodes.Call)):
237 if isinstance(current_attribute, nodes.Call):
238 current_attribute = current_attribute.func
239 if not isinstance(current_attribute, nodes.Name):
240 current_attribute = current_attribute.expr
241 if (
242 isinstance(current_attribute, nodes.Name)
243 and current_attribute.name == private_name
244 ):
245 return False
246 return True

248 @staticmethod
249 def same_root_dir(
250 node: nodes.Import | nodes.ImportFrom, import_mod_name: str
251 ) -> bool:
252 """Does the node's file's path contain the base name of `import_mod_name`?"""
253 if not import_mod_name: # from . import ...
254 return True
255 if node.level: # from .foo import ..., from ..bar import ...
256 return True
257
258 base_import_package = import_mod_name.split(".")[0]
259
260 return base_import_package in Path(node.root().file).parent.parts
