106def _sqlite_datetime_parse(dt, tzname=None, conn_tzname=None):
107 if dt is None:
108 return None
109 try:
110 dt = typecast_timestamp(dt)
111 except (TypeError, ValueError):
112 return None
113 if conn_tzname:
114 dt = dt.replace(tzinfo=zoneinfo.ZoneInfo(conn_tzname))
115 if tzname is not None and tzname != conn_tzname:
116 tzname, sign, offset = split_tzname_delta(tzname)
117 if offset:
118 hours, minutes = offset.split(":")
119 offset_delta = timedelta(hours=int(hours), minutes=int(minutes))
120 dt += offset_delta if sign == "+" else -offset_delta
121 # The tzname may originally be just the offset e.g. "+3:00",
122 # which becomes an empty string after splitting the sign and offset.
123 # In this case, use the conn_tzname as fallback.
124 dt = timezone.localtime(dt, zoneinfo.ZoneInfo(tzname or conn_tzname))
125 return dt

128def _sqlite_date_trunc(lookup_type, dt, tzname, conn_tzname):
129 dt = _sqlite_datetime_parse(dt, tzname, conn_tzname)
130 if dt is None:
131 return None
132 if lookup_type == "year":
133 return f"{dt.year:04d}-01-01"
134 elif lookup_type == "quarter":
135 month_in_quarter = dt.month - (dt.month - 1) % 3
136 return f"{dt.year:04d}-{month_in_quarter:02d}-01"
137 elif lookup_type == "month":
138 return f"{dt.year:04d}-{dt.month:02d}-01"
139 elif lookup_type == "week":
140 dt -= timedelta(days=dt.weekday())
141 return f"{dt.year:04d}-{dt.month:02d}-{dt.day:02d}"
142 elif lookup_type == "day":
143 return f"{dt.year:04d}-{dt.month:02d}-{dt.day:02d}"
144 raise ValueError(f"Unsupported lookup type: {lookup_type!r}")

182 dt = _sqlite_datetime_parse(dt, tzname, conn_tzname)

249def _sqlite_format_dtdelta(connector, lhs, rhs):
250 """
251 LHS and RHS can be either:
252 - An integer number of microseconds
253 - A string representing a datetime
254 - A scalar value, e.g. float
255 """
256 if connector is None or lhs is None or rhs is None:
257 return None
258 connector = connector.strip()
259 try:
260 real_lhs = _sqlite_prepare_dtdelta_param(connector, lhs)
261 real_rhs = _sqlite_prepare_dtdelta_param(connector, rhs)
262 except (ValueError, TypeError):
263 return None
264 if connector == "+":
265 # typecast_timestamp() returns a date or a datetime without timezone.
266 # It will be formatted as "%Y-%m-%d" or "%Y-%m-%d %H:%M:%S[.%f]"
267 out = str(real_lhs + real_rhs)
268 elif connector == "-":
269 out = str(real_lhs - real_rhs)
270 elif connector == "*":
271 out = real_lhs * real_rhs
272 else:
273 out = real_lhs / real_rhs
274 return out

277def _sqlite_time_diff(lhs, rhs):
278 if lhs is None or rhs is None:
279 return None
280 left = typecast_time(lhs)
281 right = typecast_time(rhs)
282 return (
283 (left.hour * 60 * 60 * 1000000)
284 + (left.minute * 60 * 1000000)
285 + (left.second * 1000000)
286 + (left.microsecond)
287 - (right.hour * 60 * 60 * 1000000)
288 - (right.minute * 60 * 1000000)
289 - (right.second * 1000000)
290 - (right.microsecond)
291 )
