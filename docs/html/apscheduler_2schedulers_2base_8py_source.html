134 def configure(self, gconfig={}, prefix="apscheduler.", **options):
135 """
136 Reconfigures the scheduler with the given options.
137
138 Can only be done when the scheduler isn't running.
139
140 :param dict gconfig: a "global" configuration dictionary whose values can be overridden by
141 keyword arguments to this method
142 :param str|unicode prefix: pick only those keys from ``gconfig`` that are prefixed with
143 this string (pass an empty string or ``None`` to use all keys)
144 :raises SchedulerAlreadyRunningError: if the scheduler is already running
145
146 """
147 if self.state != STATE_STOPPED:
148 raise SchedulerAlreadyRunningError
149
150 # If a non-empty prefix was given, strip it from the keys in the
151 # global configuration dict
152 if prefix:
153 prefixlen = len(prefix)
154 gconfig = dict(
155 (key[prefixlen:], value)
156 for key, value in gconfig.items()
157 if key.startswith(prefix)
158 )
159
160 # Create a structure from the dotted options
161 # (e.g. "a.b.c = d" -> {'a': {'b': {'c': 'd'}}})
162 config = {}
163 for key, value in gconfig.items():
164 parts = key.split(".")
165 parent = config
166 key = parts.pop(0)
167 while parts:
168 parent = parent.setdefault(key, {})
169 key = parts.pop(0)
170 parent[key] = value
171
172 # Override any options with explicit keyword arguments
173 config.update(options)
174 self._configure(config)

176 def start(self, paused=False):
177 """
178 Start the configured executors and job stores and begin processing scheduled jobs.
179
180 :param bool paused: if ``True``, don't start job processing until :meth:`resume` is called
181 :raises SchedulerAlreadyRunningError: if the scheduler is already running
182 :raises RuntimeError: if running under uWSGI with threads disabled
183
184 """
185 if self.state != STATE_STOPPED:
186 raise SchedulerAlreadyRunningError
187
188 self._check_uwsgi()
189
190 with self._executors_lock:
191 # Create a default executor if nothing else is configured
192 if "default" not in self._executors:
193 self.add_executor(self._create_default_executor(), "default")
194
195 # Start all the executors
196 for alias, executor in self._executors.items():
197 executor.start(self, alias)
198
199 with self._jobstores_lock:
200 # Create a default job store if nothing else is configured
201 if "default" not in self._jobstores:
202 self.add_jobstore(self._create_default_jobstore(), "default")
203
204 # Start all the job stores
205 for alias, store in self._jobstores.items():
206 store.start(self, alias)
207
208 # Schedule all pending jobs
209 for job, jobstore_alias, replace_existing in self._pending_jobs:
210 self._real_add_job(job, jobstore_alias, replace_existing)
211 del self._pending_jobs[:]
212
213 self.state = STATE_PAUSED if paused else STATE_RUNNING
214 self._logger.info("Scheduler started")
215 self._dispatch_event(SchedulerEvent(EVENT_SCHEDULER_STARTED))
216
217 if not paused:
218 self.wakeup()

220 @abstractmethod
221 def shutdown(self, wait=True):
222 """
223 Shuts down the scheduler, along with its executors and job stores.
224
225 Does not interrupt any currently running jobs.
226
227 :param bool wait: ``True`` to wait until all currently executing jobs have finished
228 :raises SchedulerNotRunningError: if the scheduler has not been started yet
229
230 """
231 if self.state == STATE_STOPPED:
232 raise SchedulerNotRunningError
233
234 self.state = STATE_STOPPED
235
236 # Shut down all executors
237 with self._executors_lock, self._jobstores_lock:
238 for executor in self._executors.values():
239 executor.shutdown(wait)
240
241 # Shut down all job stores
242 for jobstore in self._jobstores.values():
243 jobstore.shutdown()
244
245 self._logger.info("Scheduler has been shut down")
246 self._dispatch_event(SchedulerEvent(EVENT_SCHEDULER_SHUTDOWN))

263 def resume(self):
264 """Resume job processing in the scheduler."""
265 if self.state == STATE_STOPPED:
266 raise SchedulerNotRunningError

268 self.state = STATE_RUNNING

424 self,
425 func,
426 trigger=None,
427 args=None,
428 kwargs=None,

430 name=None,

440 """
441 add_job(func, trigger=None, args=None, kwargs=None, id=None, \
442 name=None, misfire_grace_time=undefined, coalesce=undefined, \
443 max_instances=undefined, next_run_time=undefined, \
444 jobstore='default', executor='default', \
445 replace_existing=False, **trigger_args)
446
447 Adds the given job to the job list and wakes up the scheduler if it's already running.
448
449 Any option that defaults to ``undefined`` will be replaced with the corresponding default
450 value when the job is scheduled (which happens when the scheduler is started, or
451 immediately if the scheduler is already running).
452
453 The ``func`` argument can be given either as a callable object or a textual reference in
454 the ``package.module:some.object`` format, where the first half (separated by ``:``) is an
455 importable module and the second half is a reference to the callable object, relative to
456 the module.
457
458 The ``trigger`` argument can either be:
459 #. the alias name of the trigger (e.g. ``date``, ``interval`` or ``cron``), in which case
460 any extra keyword arguments to this method are passed on to the trigger's constructor
461 #. an instance of a trigger class
462
463 :param func: callable (or a textual reference to one) to run at the given time
464 :param str|apscheduler.triggers.base.BaseTrigger trigger: trigger that determines when
465 ``func`` is called
466 :param list|tuple args: list of positional arguments to call func with
467 :param dict kwargs: dict of keyword arguments to call func with
468 :param str|unicode id: explicit identifier for the job (for modifying it later)
469 :param str|unicode name: textual description of the job
470 :param int misfire_grace_time: seconds after the designated runtime that the job is still
471 allowed to be run (or ``None`` to allow the job to run no matter how late it is)
472 :param bool coalesce: run once instead of many times if the scheduler determines that the
473 job should be run more than once in succession
474 :param int max_instances: maximum number of concurrently running instances allowed for this
475 job
476 :param datetime next_run_time: when to first run the job, regardless of the trigger (pass
477 ``None`` to add the job as paused)
478 :param str|unicode jobstore: alias of the job store to store the job in
479 :param str|unicode executor: alias of the executor to run the job with
480 :param bool replace_existing: ``True`` to replace an existing job with the same ``id``
481 (but retain the number of runs from the existing one)
482 :rtype: Job
483
484 """
485 job_kwargs = {
486 "trigger": self._create_trigger(trigger, trigger_args),
487 "executor": executor,
488 "func": func,
489 "args": tuple(args) if args is not None else (),
490 "kwargs": dict(kwargs) if kwargs is not None else {},
491 "id": id,
492 "name": name,
493 "misfire_grace_time": misfire_grace_time,
494 "coalesce": coalesce,
495 "max_instances": max_instances,
496 "next_run_time": next_run_time,
497 }
498 job_kwargs = dict(
499 (key, value) for key, value in job_kwargs.items() if value is not undefined
500 )
501 job = Job(self, **job_kwargs)
502
503 # Don't really add jobs to job stores before the scheduler is up and running
504 with self._jobstores_lock:
505 if self.state == STATE_STOPPED:
506 self._pending_jobs.append((job, jobstore, replace_existing))
507 self._logger.info(
508 "Adding job tentatively -- it will be properly scheduled when "
509 "the scheduler starts"
510 )
511 else:
512 self._real_add_job(job, jobstore, replace_existing)
513
514 return job

691 def remove_job(self, job_id, jobstore=None):
692 """
693 Removes a job, preventing it from being run any more.
694
695 :param str|unicode job_id: the identifier of the job
696 :param str|unicode jobstore: alias of the job store that contains the job
697 :raises JobLookupError: if the job was not found
698
699 """
700 jobstore_alias = None
701 with self._jobstores_lock:
702 # Check if the job is among the pending jobs
703 if self.state == STATE_STOPPED:
704 for i, (job, alias, replace_existing) in enumerate(self._pending_jobs):
705 if job.id == job_id and jobstore in (None, alias):
706 del self._pending_jobs[i]
707 jobstore_alias = alias
708 break
709 else:
710 # Otherwise, try to remove it from each store until it succeeds or we run out of
711 # stores to check
712 for alias, store in self._jobstores.items():
713 if jobstore in (None, alias):
714 try:
715 store.remove_job(job_id)
716 jobstore_alias = alias
717 break
718 except JobLookupError:
719 continue
720
721 if jobstore_alias is None:
722 raise JobLookupError(job_id)
723
724 # Notify listeners that a job has been removed
725 event = JobEvent(EVENT_JOB_REMOVED, job_id, jobstore_alias)
726 self._dispatch_event(event)
727
728 self._logger.info("Removed job %s", job_id)

1138 def _process_jobs(self):
1139 """
1140 Iterates through jobs in every jobstore, starts jobs that are due and figures out how long
1141 to wait for the next round.
1142
1143 If the ``get_due_jobs()`` call raises an exception, a new wakeup is scheduled in at least
1144 ``jobstore_retry_interval`` seconds.
1145
1146 """
1147 if self.state == STATE_PAUSED:
1148 self._logger.debug("Scheduler is paused -- not processing jobs")
1149 return None
1150
1151 self._logger.debug("Looking for jobs to run")
1152 now = datetime.now(self.timezone)
1153 next_wakeup_time = None
1154 events = []
1155
1156 with self._jobstores_lock:
1157 for jobstore_alias, jobstore in self._jobstores.items():
1158 try:
1159 due_jobs = jobstore.get_due_jobs(now)
1160 except Exception as e:
1161 # Schedule a wakeup at least in jobstore_retry_interval seconds
1162 self._logger.warning(
1163 "Error getting due jobs from job store %r: %s",
1164 jobstore_alias,
1165 e,
1166 )
1167 retry_wakeup_time = now + timedelta(
1168 seconds=self.jobstore_retry_interval
1169 )
1170 if not next_wakeup_time or next_wakeup_time > retry_wakeup_time:
1171 next_wakeup_time = retry_wakeup_time
1172
1173 continue
1174
1175 for job in due_jobs:
1176 # Look up the job's executor
1177 try:
1178 executor = self._lookup_executor(job.executor)
1179 except BaseException:
1180 self._logger.error(
1181 'Executor lookup ("%s") failed for job "%s" -- removing it from the '
1182 "job store",
1183 job.executor,
1184 job,
1185 )
1186 self.remove_job(job.id, jobstore_alias)
1187 continue
1188
1189 run_times = job._get_run_times(now)
1190 run_times = (
1191 run_times[-1:] if run_times and job.coalesce else run_times
1192 )
1193 if run_times:
1194 try:
1195 executor.submit_job(job, run_times)
1196 except MaxInstancesReachedError:
1197 self._logger.warning(
1198 'Execution of job "%s" skipped: maximum number of running '
1199 "instances reached (%d)",
1200 job,
1201 job.max_instances,
1202 )
1203 event = JobSubmissionEvent(
1204 EVENT_JOB_MAX_INSTANCES,
1205 job.id,
1206 jobstore_alias,
1207 run_times,
1208 )
1209 events.append(event)
1210 except BaseException:
1211 self._logger.exception(
1212 'Error submitting job "%s" to executor "%s"',
1213 job,
1214 job.executor,
1215 )
1216 else:
1217 event = JobSubmissionEvent(
1218 EVENT_JOB_SUBMITTED, job.id, jobstore_alias, run_times
1219 )
1220 events.append(event)
1221
1222 # Update the job if it has a next execution time.
1223 # Otherwise remove it from the job store.
1224 job_next_run = job.trigger.get_next_fire_time(
1225 run_times[-1], now
1226 )
1227 if job_next_run:
1228 job._modify(next_run_time=job_next_run)
1229 jobstore.update_job(job)
1230 else:
1231 self.remove_job(job.id, jobstore_alias)
1232
1233 # Set a new next wakeup time if there isn't one yet or
1234 # the jobstore has an even earlier one
1235 jobstore_next_run_time = jobstore.get_next_run_time()
1236 if jobstore_next_run_time and (
1237 next_wakeup_time is None
1238 or jobstore_next_run_time < next_wakeup_time
1239 ):
1240 next_wakeup_time = jobstore_next_run_time.astimezone(self.timezone)
1241
1242 # Dispatch collected events
1243 for event in events:
1244 self._dispatch_event(event)
1245
1246 # Determine the delay until this method should be called again
1247 if self.state == STATE_PAUSED:
1248 wait_seconds = None
1249 self._logger.debug("Scheduler is paused; waiting until resume() is called")
1250 elif next_wakeup_time is None:
1251 wait_seconds = None
1252 self._logger.debug("No jobs; waiting until a job is added")
1253 else:
1254 now = datetime.now(self.timezone)
1255 wait_seconds = min(
1256 max((next_wakeup_time - now).total_seconds(), 0), TIMEOUT_MAX
1257 )
1258 self._logger.debug(
1259 "Next wakeup is due at %s (in %f seconds)",
1260 next_wakeup_time,
1261 wait_seconds,
1262 )
1263
1264 return wait_seconds
