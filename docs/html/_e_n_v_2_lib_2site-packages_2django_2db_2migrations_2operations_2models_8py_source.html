999class RenameIndex(IndexOperation):
1000 """Rename an index."""
1001
1002 category = OperationCategory.ALTERATION
1003
1004 def __init__(self, model_name, new_name, old_name=None, old_fields=None):
1005 if not old_name and not old_fields:
1006 raise ValueError(
1007 "RenameIndex requires one of old_name and old_fields arguments to be "
1008 "set."
1009 )
1010 if old_name and old_fields:
1011 raise ValueError(
1012 "RenameIndex.old_name and old_fields are mutually exclusive."
1013 )
1014 self.model_name = model_name
1015 self.new_name = new_name
1016 self.old_name = old_name
1017 self.old_fields = old_fields
1018
1019 @cached_property
1020 def old_name_lower(self):
1021 return self.old_name.lower()
1022
1023 @cached_property
1024 def new_name_lower(self):
1025 return self.new_name.lower()
1026
1027 def deconstruct(self):
1028 kwargs = {
1029 "model_name": self.model_name,
1030 "new_name": self.new_name,
1031 }
1032 if self.old_name:
1033 kwargs["old_name"] = self.old_name
1034 if self.old_fields:
1035 kwargs["old_fields"] = self.old_fields
1036 return (self.__class__.__qualname__, [], kwargs)
1037
1038 def state_forwards(self, app_label, state):
1039 if self.old_fields:
1040 state.add_index(
1041 app_label,
1042 self.model_name_lower,
1043 models.Index(fields=self.old_fields, name=self.new_name),
1044 )
1045 state.remove_model_options(
1046 app_label,
1047 self.model_name_lower,
1048 AlterIndexTogether.option_name,
1049 self.old_fields,
1050 )
1051 else:
1052 state.rename_index(
1053 app_label, self.model_name_lower, self.old_name, self.new_name
1054 )
1055
1056 def database_forwards(self, app_label, schema_editor, from_state, to_state):
1057 model = to_state.apps.get_model(app_label, self.model_name)
1058 if not self.allow_migrate_model(schema_editor.connection.alias, model):
1059 return
1060
1061 if self.old_fields:
1062 from_model = from_state.apps.get_model(app_label, self.model_name)
1063 columns = [
1064 from_model._meta.get_field(field).column for field in self.old_fields
1065 ]
1066 matching_index_name = schema_editor._constraint_names(
1067 from_model,
1068 column_names=columns,
1069 index=True,
1070 unique=False,
1071 )
1072 if len(matching_index_name) != 1:
1073 raise ValueError(
1074 "Found wrong number (%s) of indexes for %s(%s)."
1075 % (
1076 len(matching_index_name),
1077 from_model._meta.db_table,
1078 ", ".join(columns),
1079 )
1080 )
1081 old_index = models.Index(
1082 fields=self.old_fields,
1083 name=matching_index_name[0],
1084 )
1085 else:
1086 from_model_state = from_state.models[app_label, self.model_name_lower]
1087 old_index = from_model_state.get_index_by_name(self.old_name)
1088 # Don't alter when the index name is not changed.
1089 if old_index.name == self.new_name:
1090 return
1091
1092 to_model_state = to_state.models[app_label, self.model_name_lower]
1093 new_index = to_model_state.get_index_by_name(self.new_name)
1094 schema_editor.rename_index(model, old_index, new_index)
1095
1096 def database_backwards(self, app_label, schema_editor, from_state, to_state):
1097 if self.old_fields:
1098 # Backward operation with unnamed index is a no-op.
1099 return
1100
1101 self.new_name_lower, self.old_name_lower = (
1102 self.old_name_lower,
1103 self.new_name_lower,
1104 )
1105 self.new_name, self.old_name = self.old_name, self.new_name
1106
1107 self.database_forwards(app_label, schema_editor, from_state, to_state)
1108
1109 self.new_name_lower, self.old_name_lower = (
1110 self.old_name_lower,
1111 self.new_name_lower,
1112 )
1113 self.new_name, self.old_name = self.old_name, self.new_name
1114
1115 def describe(self):
1116 if self.old_name:
1117 return (
1118 f"Rename index {self.old_name} on {self.model_name} to {self.new_name}"
1119 )
1120 return (
1121 f"Rename unnamed index for {self.old_fields} on {self.model_name} to "
1122 f"{self.new_name}"
1123 )
1124
1125 @property
1126 def migration_name_fragment(self):
1127 if self.old_name:

1132 self.new_name_lower,
1133 )

1136 if (
1137 isinstance(operation, RenameIndex)
