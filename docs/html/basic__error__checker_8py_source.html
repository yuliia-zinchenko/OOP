26def _get_break_loop_node(break_node: nodes.Break) -> nodes.For | nodes.While | None:
27 """Returns the loop node that holds the break node in arguments.
28
29 Args:
30 break_node (astroid.Break): the break node of interest.
31
32 Returns:
33 astroid.For or astroid.While: the loop node holding the break node.
34 """
35 loop_nodes = (nodes.For, nodes.While)
36 parent = break_node.parent
37 while not isinstance(parent, loop_nodes) or break_node in getattr(
38 parent, "orelse", []
39 ):
40 break_node = parent
41 parent = parent.parent
42 if parent is None:
43 break
44 return parent

47def _loop_exits_early(loop: nodes.For | nodes.While) -> bool:
48 """Returns true if a loop may end with a break statement.
49
50 Args:
51 loop (astroid.For, astroid.While): the loop node inspected.
52
53 Returns:
54 bool: True if the loop may end with a break statement, False otherwise.
55 """
56 loop_nodes = (nodes.For, nodes.While)
57 definition_nodes = (nodes.FunctionDef, nodes.ClassDef)
58 inner_loop_nodes: list[nodes.For | nodes.While] = [
59 _node
60 for _node in loop.nodes_of_class(loop_nodes, skip_klass=definition_nodes)
61 if _node != loop
62 ]
63 return any(
64 _node
65 for _node in loop.nodes_of_class(nodes.Break, skip_klass=definition_nodes)
66 if _get_break_loop_node(_node) not in inner_loop_nodes
67 )

79def redefined_by_decorator(node: nodes.FunctionDef) -> bool:
80 """Return True if the object is a method redefined via decorator.
81
82 For example:
83 @property
84 def x(self): return self._x
85 @x.setter
86 def x(self, value): self._x = value
87 """
88 if node.decorators:
89 for decorator in node.decorators.nodes:
90 if (
91 isinstance(decorator, nodes.Attribute)
92 and getattr(decorator.expr, "name", None) == node.name
93 ):
94 return True
95 return False

299 def _check_name_used_prior_global(self, node: nodes.FunctionDef) -> None:
300 scope_globals = {

303 for name in child.names

325 def _check_nonlocal_and_global(self, node: nodes.FunctionDef) -> None:

328 def same_scope(current: nodes.Global | nodes.Nonlocal) -> bool:
329 return current.scope() is node

333 from_iter(
334 child.names
335 for child in node.nodes_of_class(nodes.Nonlocal)
336 if same_scope(child)

344 from_iter(

428 def _check_inferred_class_is_abstract(
429 self, inferred: InferenceResult, node: nodes.Call
430 ) -> None:
431 if not isinstance(inferred, nodes.ClassDef):
432 return
433
434 klass = utils.node_frame_class(node)

504 def _check_redefinition(
505 self, redeftype: str, node: nodes.Call | nodes.FunctionDef
506 ) -> None:
507 """Check for redefinition of a function / method / class name."""
508 parent_frame = node.parent.frame()
509
510 # Ignore function stubs created for type information
511 redefinitions = [
512 i
513 for i in parent_frame.locals[node.name]
514 if not (isinstance(i.parent, nodes.AnnAssign) and i.parent.simple)
515 ]
516 defined_self = next(
517 (local for local in redefinitions if not utils.is_overload_stub(local)),
518 node,
519 )
520 if defined_self is not node and not astroid.are_exclusive(node, defined_self):
521 # Additional checks for methods which are not considered
522 # redefined, since they are already part of the base API.
523 if (
524 isinstance(parent_frame, nodes.ClassDef)
525 and node.name in REDEFINABLE_METHODS
526 ):
527 return
528
529 # Skip typing.overload() functions.
530 if utils.is_overload_stub(node):
531 return
532
533 # Exempt functions redefined on a condition.
534 if isinstance(node.parent, nodes.If):
535 # Exempt "if not <func>" cases
536 if (
537 isinstance(node.parent.test, nodes.UnaryOp)
538 and node.parent.test.op == "not"
539 and isinstance(node.parent.test.operand, nodes.Name)
540 and node.parent.test.operand.name == node.name
541 ):
542 return
543
544 # Exempt "if <func> is not None" cases
545 # pylint: disable=too-many-boolean-expressions
546 if (
547 isinstance(node.parent.test, nodes.Compare)
548 and isinstance(node.parent.test.left, nodes.Name)
549 and node.parent.test.left.name == node.name
550 and node.parent.test.ops[0][0] == "is"
551 and isinstance(node.parent.test.ops[0][1], nodes.Const)
552 and node.parent.test.ops[0][1].value is None
553 ):
554 return
555
556 # Check if we have forward references for this node.
557 try:
558 redefinition_index = redefinitions.index(node)
559 except ValueError:
560 pass
561 else:
562 for redefinition in redefinitions[:redefinition_index]:
563 inferred = utils.safe_infer(redefinition)
564 if (
565 inferred
566 and isinstance(inferred, astroid.Instance)
567 and inferred.qname() == TYPING_FORWARD_REF_QNAME
568 ):
569 return
570
571 dummy_variables_rgx = self.linter.config.dummy_variables_rgx
572 if dummy_variables_rgx and dummy_variables_rgx.match(node.name):
573 return
574 self.add_message(
575 "function-redefined",
576 node=node,
577 args=(redeftype, defined_self.fromlineno),
578 )
