112def url_params_from_lookup_dict(lookups):
113 """
114 Convert the type of lookups specified in a ForeignKey limit_choices_to
115 attribute to a dictionary of query parameters
116 """
117 params = {}
118 if lookups and hasattr(lookups, "items"):
119 for k, v in lookups.items():
120 if callable(v):
121 v = v()
122 if isinstance(v, (tuple, list)):
123 v = ",".join(str(x) for x in v)
124 elif isinstance(v, bool):
125 v = ("0", "1")[v]
126 else:
127 v = str(v)
128 params[k] = v
129 return params

474def get_select2_language():
475 lang_code = get_language()
476 supported_code = SELECT2_TRANSLATIONS.get(lang_code)
477 if supported_code is None and lang_code is not None:
478 # If 'zh-hant-tw' is not supported, try subsequent language codes i.e.
479 # 'zh-hant' and 'zh'.
480 i = None
481 while (i := lang_code.rfind("-", 0, i)) > -1:
482 if supported_code := SELECT2_TRANSLATIONS.get(lang_code[:i]):
483 return supported_code
484 return supported_code

508 def build_attrs(self, base_attrs, extra_attrs=None):
509 """
510 Set select2's AJAX attributes.
511
512 Attributes can be set using the html5 data attribute.
513 Nested attributes require a double dash as per
514 https://select2.org/configuration/data-attributes#nested-subkey-options
515 """
516 attrs = super().build_attrs(base_attrs, extra_attrs=extra_attrs)
517 attrs.setdefault("class", "")
518 attrs.update(
519 {
520 "data-ajax--cache": "true",
521 "data-ajax--delay": 250,
522 "data-ajax--type": "GET",
523 "data-ajax--url": self.get_url(),
524 "data-app-label": self.field.model._meta.app_label,
525 "data-model-name": self.field.model._meta.model_name,
526 "data-field-name": self.field.name,
527 "data-theme": "admin-autocomplete",
528 "data-allow-clear": json.dumps(not self.is_required),
529 "data-placeholder": "", # Allows clearing of the input.
530 "lang": self.i18n_name,
531 "class": attrs["class"]
532 + (" " if attrs["class"] else "")
533 + "admin-autocomplete",
534 }
535 )
536 return attrs

538 def optgroups(self, name, value, attr=None):
539 """Return selected options based on the ModelChoiceIterator."""
540 default = (None, [], 0)
541 groups = [default]
542 has_selected = False
543 selected_choices = {
544 str(v) for v in value if str(v) not in self.choices.field.empty_values
545 }
546 if not self.is_required and not self.allow_multiple_selected:
547 default[1].append(self.create_option(name, "", "", False, 0))
548 remote_model_opts = self.field.remote_field.model._meta
549 to_field_name = getattr(
550 self.field.remote_field, "field_name", remote_model_opts.pk.attname
551 )
552 to_field_name = remote_model_opts.get_field(to_field_name).attname
553 choices = (
554 (getattr(obj, to_field_name), self.choices.field.label_from_instance(obj))
555 for obj in self.choices.queryset.using(self.db).filter(
556 **{"%s__in" % to_field_name: selected_choices}
557 )
558 )
559 for option_value, option_label in choices:
560 selected = str(option_value) in value and (
561 has_selected is False or self.allow_multiple_selected
562 )
563 has_selected |= selected
564 index = len(default[1])
565 subgroup = default[1]
566 subgroup.append(
567 self.create_option(
568 name, option_value, option_label, selected_choices, index
569 )
570 )
571 return groups
