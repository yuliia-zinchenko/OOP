125 def infer(
126 self, context: InferenceContext | None = None, **kwargs: Any
127 ) -> Generator[InferenceResult]:
128 """Get a generator of the inferred values.
129
130 This is the main entry point to the inference system.
131
132 .. seealso:: :ref:`inference`
133
134 If the instance has some explicit inference function set, it will be
135 called instead of the default interface.
136
137 :returns: The inferred values.
138 :rtype: iterable
139 """
140 if context is None:
141 context = InferenceContext()
142 else:
143 context = context.extra_context.get(self, context)
144 if self._explicit_inference is not None:
145 # explicit_inference is not bound, give it self explicitly
146 try:
147 for result in self._explicit_inference(
148 self, # type: ignore[arg-type]
149 context,
150 **kwargs,
151 ):
152 context.nodes_inferred += 1
153 yield result
154 return
155 except UseInferenceDefault:
156 pass
157
158 key = (self, context.lookupname, context.callcontext, context.boundnode)
159 if key in context.inferred:
160 yield from context.inferred[key]
161 return
162
163 results = []
164
165 # Limit inference amount to help with performance issues with
166 # exponentially exploding possible results.
167 limit = AstroidManager().max_inferable_values
168 for i, result in enumerate(self._infer(context=context, **kwargs)):
169 if i >= limit or (context.nodes_inferred > context.max_inferred):
170 results.append(util.Uninferable)
171 yield util.Uninferable
172 break
173 results.append(result)
174 yield result
175 context.nodes_inferred += 1
176
177 # Cache generated results for subsequent inferences of the
178 # same node using the same context
179 context.inferred[key] = tuple(results)
180 return

182 def repr_name(self) -> str:
183 """Get a name for nice representation.

185 This is either :attr:`name`, :attr:`attrname`, or the empty string.

188 return getattr(self, "name", "") or getattr(self, "attrname", "")

191 def __str__(self) -> str:
192 rname = self.repr_name()
193 cname = type(self).__name__
194 if rname:
195 string = "%(cname)s.%(rname)s(%(fields)s)"
196 alignment = len(cname) + len(rname) + 2
197 else:
198 string = "%(cname)s(%(fields)s)"
199 alignment = len(cname) + 1
200 result = []
201 for field in self._other_fields + self._astroid_fields:
202 value = getattr(self, field, "Unknown")
203 width = 80 - len(field) - alignment
204 lines = pprint.pformat(value, indent=2, width=width).splitlines(True)
205
206 inner = [lines[0]]
207 for line in lines[1:]:
208 inner.append(" " * alignment + line)
209 result.append(f"{field}={''.join(inner)}")
210
211 return string % {
212 "cname": cname,
213 "rname": rname,
214 "fields": (",\n" + " " * alignment).join(result),
215 }

240 def get_children(self) -> Iterator[NodeNG]:
241 """Get the child nodes below this node."""
242 for field in self._astroid_fields:
243 attr = getattr(self, field)
244 if attr is None:
245 continue
246 if isinstance(attr, (list, tuple)):
247 yield from attr
248 else:
249 yield attr
250 yield from ()
251
252 def last_child(self) -> NodeNG | None:
253 """An optimized version of list(get_children())[-1]."""
254 for field in self._astroid_fields[::-1]:
255 attr = getattr(self, field)
256 if not attr: # None or empty list / tuple
257 continue
258 if isinstance(attr, (list, tuple)):
259 return attr[-1]
260 return attr
261 return None

341 def child_sequence(self, child):
342 """Search for the sequence that contains this child.
343
344 :param child: The child node to search sequences for.
345 :type child: NodeNG
346
347 :returns: The sequence containing the given child node.
348 :rtype: iterable(NodeNG)
349
350 :raises AstroidError: If no sequence could be found that contains
351 the given child.
352 """
353 for field in self._astroid_fields:
354 node_or_sequence = getattr(self, field)
355 if node_or_sequence is child:
356 return [node_or_sequence]
357 # /!\ compiler.ast Nodes have an __iter__ walking over child nodes
358 if (
359 isinstance(node_or_sequence, (tuple, list))
360 and child in node_or_sequence
361 ):
362 return node_or_sequence
363
364 msg = "Could not find %s in %s's children"
365 raise AstroidError(msg % (repr(child), repr(self)))

367 def locate_child(self, child):
368 """Find the field of this node that contains the given child.
369
370 :param child: The child node to search fields for.
371 :type child: NodeNG
372
373 :returns: A tuple of the name of the field that contains the child,
374 and the sequence or node that contains the child node.
375 :rtype: tuple(str, iterable(NodeNG) or NodeNG)
376
377 :raises AstroidError: If no field could be found that contains
378 the given child.
379 """
380 for field in self._astroid_fields:
381 node_or_sequence = getattr(self, field)
382 # /!\ compiler.ast Nodes have an __iter__ walking over child nodes
383 if child is node_or_sequence:
384 return field, child
385 if (
386 isinstance(node_or_sequence, (tuple, list))
387 and child in node_or_sequence
388 ):
389 return field, node_or_sequence
390 msg = "Could not find %s in %s's children"
391 raise AstroidError(msg % (repr(child), repr(self)))

425 @cached_property
426 def tolineno(self) -> int:
427 """The last line that this node appears on in the source code.
428
429 Can also return 0 if the line can not be determined.
430 """
431 if self.end_lineno is not None:
432 return self.end_lineno
433 if not self._astroid_fields:
434 # can't have children
435 last_child = None
436 else:
437 last_child = self.last_child()
438 if last_child is None:
439 return self.fromlineno

442 def _fixed_source_line(self) -> int:
443 """Attempt to find the line that this node appears on.
444
445 We need this method since not all nodes have :attr:`lineno` set.
446 Will return 0 if the line number can not be determined.
447 """
448 line = self.lineno
449 _node = self
450 try:
451 while line is None:
452 _node = next(_node.get_children())
453 line = _node.lineno
454 except StopIteration:
455 parent = self.parent
456 while parent and line is None:
457 line = parent.lineno
458 parent = parent.parent
459 return line or 0

621 def repr_tree(
622 self,
623 ids=False,
624 include_linenos=False,
625 ast_state=False,
626 indent=" ",
627 max_depth=0,
628 max_width=80,
629 ) -> str:
630 """Get a string representation of the AST from this node.
631
632 :param ids: If true, includes the ids with the node type names.
633 :type ids: bool
634
635 :param include_linenos: If true, includes the line numbers and
636 column offsets.
637 :type include_linenos: bool
638
639 :param ast_state: If true, includes information derived from
640 the whole AST like local and global variables.
641 :type ast_state: bool
642
643 :param indent: A string to use to indent the output string.
644 :type indent: str
645
646 :param max_depth: If set to a positive integer, won't return
647 nodes deeper than max_depth in the string.
648 :type max_depth: int
649
650 :param max_width: Attempt to format the output string to stay
651 within this number of characters, but can exceed it under some

653 :type max_width: int

655 :returns: The string representation of the AST.

659 @_singledispatch
660 def _repr_tree(node, result, done, cur_indent="", depth=1):
661 """Outputs a representation of a non-tuple/list, non-node that's
662 contained within an AST, including strings.
663 """
664 lines = pprint.pformat(
665 node, width=max(max_width - len(cur_indent), 1)
666 ).splitlines(True)
667 result.append(lines[0])
668 result.extend([cur_indent + line for line in lines[1:]])
669 return len(lines) != 1
670
671 # pylint: disable=unused-variable,useless-suppression; doesn't understand singledispatch
672 @_repr_tree.register(tuple)
673 @_repr_tree.register(list)
674 def _repr_seq(node, result, done, cur_indent="", depth=1):
675 """Outputs a representation of a sequence that's contained within an
676 AST.
677 """
678 cur_indent += indent
679 result.append("[")
680 if not node:
681 broken = False
682 elif len(node) == 1:
683 broken = _repr_tree(node[0], result, done, cur_indent, depth)
684 elif len(node) == 2:
685 broken = _repr_tree(node[0], result, done, cur_indent, depth)
686 if not broken:
687 result.append(", ")
688 else:
689 result.append(",\n")
690 result.append(cur_indent)
691 broken = _repr_tree(node[1], result, done, cur_indent, depth) or broken
692 else:
693 result.append("\n")
694 result.append(cur_indent)
695 for child in node[:-1]:
696 _repr_tree(child, result, done, cur_indent, depth)
697 result.append(",\n")
698 result.append(cur_indent)
699 _repr_tree(node[-1], result, done, cur_indent, depth)
700 broken = True
701 result.append("]")
702 return broken
703
704 # pylint: disable=unused-variable,useless-suppression; doesn't understand singledispatch
705 @_repr_tree.register(NodeNG)
706 def _repr_node(node, result, done, cur_indent="", depth=1):
707 """Outputs a strings representation of an astroid node."""
708 if node in done:
709 result.append(
710 indent + f"<Recursion on {type(node).__name__} with id={id(node)}"
711 )
712 return False
713 done.add(node)
714
715 if max_depth and depth > max_depth:
716 result.append("...")
717 return False
718 depth += 1
719 cur_indent += indent
720 if ids:
721 result.append(f"{type(node).__name__}<0x{id(node):x}>(\n")
722 else:
723 result.append(f"{type(node).__name__}(")
724 fields = []
725 if include_linenos:
726 fields.extend(("lineno", "col_offset"))
727 fields.extend(node._other_fields)
728 fields.extend(node._astroid_fields)
729 if ast_state:
730 fields.extend(node._other_other_fields)
731 if not fields:
732 broken = False
733 elif len(fields) == 1:
734 result.append(f"{fields[0]}=")
735 broken = _repr_tree(
736 getattr(node, fields[0]), result, done, cur_indent, depth
737 )
738 else:
739 result.append("\n")
740 result.append(cur_indent)
741 for field in fields[:-1]:
742 # TODO: Remove this after removal of the 'doc' attribute
743 if field == "doc":
744 continue
745 result.append(f"{field}=")
746 _repr_tree(getattr(node, field), result, done, cur_indent, depth)
747 result.append(",\n")
748 result.append(cur_indent)
749 result.append(f"{fields[-1]}=")
750 _repr_tree(getattr(node, fields[-1]), result, done, cur_indent, depth)
751 broken = True
752 result.append(")")
753 return broken
754
755 result: list[str] = []
756 _repr_tree(self, result, set())
757 return "".join(result)
