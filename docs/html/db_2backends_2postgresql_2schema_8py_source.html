83 def _create_like_index_sql(self, model, field):
84 """
85 Return the statement to create an index with varchar operator pattern
86 when the column type is 'varchar' or 'text', otherwise return None.
87 """
88 db_type = field.db_type(connection=self.connection)
89 if db_type is not None and (field.db_index or field.unique):
90 # Fields with database column types of `varchar` and `text` need
91 # a second index that specifies their operator class, which is
92 # needed when performing correct LIKE queries outside the
93 # C locale. See #12234.
94 #
95 # The same doesn't apply to array fields such as varchar[size]
96 # and text[size], so skip them.
97 if "[" in db_type:
98 return None
99 # Non-deterministic collations on Postgresql don't support indexes
100 # for operator classes varchar_pattern_ops/text_pattern_ops.
101 collation_name = getattr(field, "db_collation", None)
102 if not collation_name and field.is_relation:
103 collation_name = getattr(field.target_field, "db_collation", None)
104 if collation_name and not self._is_collation_deterministic(collation_name):
105 return None
106 if db_type.startswith("varchar"):
107 return self._create_index_sql(
108 model,
109 fields=[field],
110 suffix="_like",
111 opclasses=["varchar_pattern_ops"],
112 )
113 elif db_type.startswith("text"):
114 return self._create_index_sql(
115 model,
116 fields=[field],
117 suffix="_like",
118 opclasses=["text_pattern_ops"],
119 )
120 return None

122 def _using_sql(self, new_field, old_field):
123 using_sql = " USING %(column)s::%(type)s"
124 new_internal_type = new_field.get_internal_type()

132 elif self._field_data_type(old_field) != self._field_data_type(new_field):

134 return ""
