146def compile_json_path(key_transforms, include_root=True):
147 path = ["$"] if include_root else []
148 for key_transform in key_transforms:
149 try:
150 num = int(key_transform)
151 except ValueError: # non-integer
152 path.append(".")
153 path.append(json.dumps(key_transform))
154 else:
155 path.append("[%s]" % num)
156 return "".join(path)

196 def as_sql(self, compiler, connection, template=None):
197 # Process JSON path from the left-hand side.
198 if isinstance(self.lhs, KeyTransform):
199 lhs, lhs_params, lhs_key_transforms = self.lhs.preprocess_lhs(
200 compiler, connection
201 )
202 lhs_json_path = compile_json_path(lhs_key_transforms)
203 else:
204 lhs, lhs_params = self.process_lhs(compiler, connection)
205 lhs_json_path = "$"
206 sql = template % lhs
207 # Process JSON path from the right-hand side.
208 rhs = self.rhs
209 rhs_params = []
210 if not isinstance(rhs, (list, tuple)):
211 rhs = [rhs]
212 for key in rhs:
213 if isinstance(key, KeyTransform):
214 *_, rhs_key_transforms = key.preprocess_lhs(compiler, connection)
215 else:
216 rhs_key_transforms = [key]
217 *rhs_key_transforms, final_key = rhs_key_transforms
218 rhs_json_path = compile_json_path(rhs_key_transforms, include_root=False)
219 rhs_json_path += self.compile_json_path_final_key(final_key)
220 rhs_params.append(lhs_json_path + rhs_json_path)
221 # Add condition for each key.
222 if self.logical_operator:
223 sql = "(%s)" % self.logical_operator.join([sql] * len(rhs_params))
224 return sql, tuple(lhs_params) + tuple(rhs_params)

335class KeyTransform(Transform):
336 postgres_operator = "->"
337 postgres_nested_operator = "#>"
338
339 def __init__(self, key_name, *args, **kwargs):
340 super().__init__(*args, **kwargs)
341 self.key_name = str(key_name)
342
343 def preprocess_lhs(self, compiler, connection):
344 key_transforms = [self.key_name]
345 previous = self.lhs
346 while isinstance(previous, KeyTransform):
347 key_transforms.insert(0, previous.key_name)
348 previous = previous.lhs
349 lhs, params = compiler.compile(previous)
350 if connection.vendor == "oracle":
351 # Escape string-formatting.
352 key_transforms = [key.replace("%", "%%") for key in key_transforms]
353 return lhs, params, key_transforms
354
355 def as_mysql(self, compiler, connection):
356 lhs, params, key_transforms = self.preprocess_lhs(compiler, connection)
357 json_path = compile_json_path(key_transforms)
358 return "JSON_EXTRACT(%s, %%s)" % lhs, tuple(params) + (json_path,)
359
360 def as_oracle(self, compiler, connection):
361 lhs, params, key_transforms = self.preprocess_lhs(compiler, connection)
362 json_path = compile_json_path(key_transforms)
363 if connection.features.supports_primitives_in_json_field:
364 sql = (
365 "COALESCE(JSON_VALUE(%s, '%s'), JSON_QUERY(%s, '%s' DISALLOW SCALARS))"
366 )
367 else:
368 sql = "COALESCE(JSON_QUERY(%s, '%s'), JSON_VALUE(%s, '%s'))"
369 return sql % ((lhs, json_path) * 2), tuple(params) * 2
370
371 def as_postgresql(self, compiler, connection):
372 lhs, params, key_transforms = self.preprocess_lhs(compiler, connection)
373 if len(key_transforms) > 1:
374 sql = "(%s %s %%s)" % (lhs, self.postgres_nested_operator)
375 return sql, tuple(params) + (key_transforms,)
376 try:
377 lookup = int(self.key_name)
378 except ValueError:
379 lookup = self.key_name
380 return "(%s %s %%s)" % (lhs, self.postgres_operator), tuple(params) + (lookup,)
381
382 def as_sqlite(self, compiler, connection):
383 lhs, params, key_transforms = self.preprocess_lhs(compiler, connection)
384 json_path = compile_json_path(key_transforms)
385 datatype_values = ",".join(
386 [repr(datatype) for datatype in connection.ops.jsonfield_datatype_values]
387 )
388 return (
389 "(CASE WHEN JSON_TYPE(%s, %%s) IN (%s) "
390 "THEN JSON_TYPE(%s, %%s) ELSE JSON_EXTRACT(%s, %%s) END)"
391 ) % (lhs, datatype_values, lhs, lhs), (tuple(params) + (json_path,)) * 3
