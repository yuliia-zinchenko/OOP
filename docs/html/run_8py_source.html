40def _query_cpu() -> int | None:
41 """Try to determine number of CPUs allotted in a docker container.
42
43 This is based on discussion and copied from suggestions in
44 https://bugs.python.org/issue36054.
45 """
46 if Path("/sys/fs/cgroup/cpu.max").is_file():
47 avail_cpu = _query_cpu_cgroupv2()
48 else:
49 avail_cpu = _query_cpu_cgroupsv1()
50 return _query_cpu_handle_k8s_pods(avail_cpu)

53def _query_cpu_cgroupv2() -> int | None:
54 avail_cpu = None
55 with open("/sys/fs/cgroup/cpu.max", encoding="utf-8") as file:
56 line = file.read().rstrip()
57 fields = line.split()
58 if len(fields) == 2:
59 str_cpu_quota = fields[0]
60 cpu_period = int(fields[1])
61 # Make sure this is not in an unconstrained cgroup
62 if str_cpu_quota != "max":
63 cpu_quota = int(str_cpu_quota)
64 avail_cpu = int(cpu_quota / cpu_period)
65 return avail_cpu

68def _query_cpu_cgroupsv1() -> int | None:
69 cpu_quota, avail_cpu = None, None
70 if Path("/sys/fs/cgroup/cpu/cpu.cfs_quota_us").is_file():
71 with open("/sys/fs/cgroup/cpu/cpu.cfs_quota_us", encoding="utf-8") as file:
72 # Not useful for AWS Batch based jobs as result is -1, but works on local linux systems
73 cpu_quota = int(file.read().rstrip())
74
75 if (
76 cpu_quota
77 and cpu_quota != -1
78 and Path("/sys/fs/cgroup/cpu/cpu.cfs_period_us").is_file()
79 ):
80 with open("/sys/fs/cgroup/cpu/cpu.cfs_period_us", encoding="utf-8") as file:
81 cpu_period = int(file.read().rstrip())
82 # Divide quota by period and you should get num of allotted CPU to the container,
83 # rounded down if fractional.
84 avail_cpu = int(cpu_quota / cpu_period)
85 elif Path("/sys/fs/cgroup/cpu/cpu.shares").is_file():
86 with open("/sys/fs/cgroup/cpu/cpu.shares", encoding="utf-8") as file:
87 cpu_shares = int(file.read().rstrip())
88 # For AWS, gives correct value * 1024.
89 avail_cpu = int(cpu_shares / 1024)
90 return avail_cpu

93def _query_cpu_handle_k8s_pods(avail_cpu: int | None) -> int | None:
94 # In K8s Pods also a fraction of a single core could be available
95 # As multiprocessing is not able to run only a "fraction" of process
96 # assume we have 1 CPU available
97 if avail_cpu == 0:
98 avail_cpu = 1
99
100 return avail_cpu

103def _cpu_count() -> int:
104 """Use sched_affinity if available for virtualized or containerized
105 environments.
106 """
107 cpu_share = _query_cpu()
108 cpu_count = None
109 sched_getaffinity = getattr(os, "sched_getaffinity", None)
110 # pylint: disable=not-callable,using-constant-test,useless-suppression
111 if sched_getaffinity:
112 cpu_count = len(sched_getaffinity(0))
113 elif multiprocessing:
114 cpu_count = multiprocessing.cpu_count()
115 else:
116 cpu_count = 1
117 if sys.platform == "win32":
118 # See also https://github.com/python/cpython/issues/94242
119 cpu_count = min(cpu_count, 56) # pragma: no cover
120 if cpu_share is not None:
121 return min(cpu_share, cpu_count)
122 return cpu_count
