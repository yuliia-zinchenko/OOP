115def _augment_sys_path(additional_paths: Sequence[str]) -> list[str]:
116 original = list(sys.path)
117 changes = []
118 seen = set()
119 for additional_path in additional_paths:
120 if additional_path not in seen:
121 changes.append(additional_path)
122 seen.add(additional_path)
123
124 sys.path[:] = changes + sys.path
125 return original
