237 def visit_annassign(self, node: nodes.AnnAssign) -> None:

336 def _check_for_alternative_union_syntax(
337 self,
338 node: nodes.Name | nodes.Attribute,
339 name: str,
340 ) -> None:
341 """Check if alternative union syntax could be used.
342
343 Requires
344 - Python 3.10
345 - OR: Python 3.7+ with postponed evaluation in
346 a type annotation context
347 """
348 inferred = safe_infer(node)
349 if not (
350 isinstance(inferred, nodes.FunctionDef)
351 and inferred.qname() in {"typing.Optional", "typing.Union"}
352 or isinstance(inferred, astroid.bases.Instance)
353 and inferred.qname() == "typing._SpecialForm"
354 ):
355 return

357 return

360 node=node,

365 def _check_for_typing_alias(
366 self,
367 node: nodes.Name | nodes.Attribute,
368 ) -> None:
369 """Check if typing alias is deprecated or could be replaced.
370
371 Requires
372 - Python 3.9
373 - OR: Python 3.7+ with postponed evaluation in
374 a type annotation context
375
376 For Python 3.7+: Only emit message if change doesn't create
377 any name collisions, only ever used in a type annotation
378 context, and can safely be replaced.
379 """
380 inferred = safe_infer(node)
381 if not isinstance(inferred, nodes.ClassDef):
382 return
383 alias = DEPRECATED_TYPING_ALIASES.get(inferred.qname(), None)
384 if alias is None:
385 return
386
387 if self._py39_plus:
388 if inferred.qname() == "typing.Callable" and self._broken_callable_location(
389 node
390 ):
391 self._found_broken_callable_location = True
392 self._deprecated_typing_alias_msgs.append(
393 DeprecatedTypingAliasMsg(
394 node,
395 inferred.qname(),
396 alias.name,
397 )
398 )
399 return
400
401 # For PY37+, check for type annotation context first
402 if not is_node_in_type_annotation_context(node) and isinstance(
403 node.parent, nodes.Subscript
404 ):
405 if alias.name_collision is True:
406 self._alias_name_collisions.add(inferred.qname())
407 return
408 self._consider_using_alias_msgs.append(
409 DeprecatedTypingAliasMsg(
410 node,
411 inferred.qname(),
412 alias.name,
413 isinstance(node.parent, nodes.Subscript),
414 )
415 )

417 @only_required_for_messages("consider-using-alias", "deprecated-typing-alias")
418 def leave_module(self, node: nodes.Module) -> None:
419 """After parsing of module is complete, add messages for
420 'consider-using-alias' check.
421
422 Make sure results are safe to recommend / collision free.
423 """
424 if self._py39_plus:
425 for msg in self._deprecated_typing_alias_msgs:
426 if (
427 self._found_broken_callable_location
428 and msg.qname == "typing.Callable"
429 ):
430 continue
431 self.add_message(
432 "deprecated-typing-alias",
433 node=msg.node,
434 args=(msg.qname, msg.alias),
435 confidence=INFERENCE,
436 )
437
438 elif self._py37_plus:
439 msg_future_import = self._msg_postponed_eval_hint(node)
440 for msg in self._consider_using_alias_msgs:
441 if msg.qname in self._alias_name_collisions:
442 continue
443 self.add_message(
444 "consider-using-alias",
445 node=msg.node,
446 args=(
447 msg.qname,
448 msg.alias,
449 msg_future_import if msg.parent_subscript else "",
450 ),
451 confidence=INFERENCE,
452 )
453
454 # Clear all module cache variables
455 self._found_broken_callable_location = False
456 self._deprecated_typing_alias_msgs.clear()
457 self._alias_name_collisions.clear()
458 self._consider_using_alias_msgs.clear()

460 def _check_broken_noreturn(self, node: nodes.Name | nodes.Attribute) -> None:
461 """Check for 'NoReturn' inside compound types."""
462 if not isinstance(node.parent, nodes.BaseContainer):
463 # NoReturn not part of a Union or Callable type
464 return
465
466 if (
467 in_type_checking_block(node)
468 or is_postponed_evaluation_enabled(node)
469 and is_node_in_type_annotation_context(node)
470 ):
471 return
472
473 for inferred in node.infer():
474 # To deal with typing_extensions, don't use safe_infer
475 if (
476 isinstance(inferred, (nodes.FunctionDef, nodes.ClassDef))
477 and inferred.qname() in TYPING_NORETURN
478 # In Python 3.7 - 3.8, NoReturn is alias of '_SpecialForm'
479 or isinstance(inferred, astroid.bases.BaseInstance)
480 and isinstance(inferred._proxied, nodes.ClassDef)
481 and inferred._proxied.qname() == "typing._SpecialForm"
482 ):
483 self.add_message("broken-noreturn", node=node, confidence=INFERENCE)
484 break

486 def _check_broken_callable(self, node: nodes.Name | nodes.Attribute) -> None:
487 """Check for 'collections.abc.Callable' inside Optional and Union."""
488 inferred = safe_infer(node)
489 if not (
490 isinstance(inferred, nodes.ClassDef)
491 and inferred.qname() == "_collections_abc.Callable"
492 and self._broken_callable_location(node)
493 ):
494 return
495
496 self.add_message("broken-collections-callable", node=node, confidence=INFERENCE)

498 def _broken_callable_location(self, node: nodes.Name | nodes.Attribute) -> bool:
499 """Check if node would be a broken location for collections.abc.Callable."""
500 if (
501 in_type_checking_block(node)
502 or is_postponed_evaluation_enabled(node)
503 and is_node_in_type_annotation_context(node)
504 ):
505 return False
506
507 # Check first Callable arg is a list of arguments -> Callable[[int], None]
508 if not (
509 isinstance(node.parent, nodes.Subscript)
510 and isinstance(node.parent.slice, nodes.Tuple)
511 and len(node.parent.slice.elts) == 2
512 and isinstance(node.parent.slice.elts[0], nodes.List)
513 ):
514 return False
515
516 # Check nested inside Optional or Union
517 parent_subscript = node.parent.parent
518 if isinstance(parent_subscript, nodes.BaseContainer):
519 parent_subscript = parent_subscript.parent
520 if not (
521 isinstance(parent_subscript, nodes.Subscript)
522 and isinstance(parent_subscript.value, (nodes.Name, nodes.Attribute))
523 ):
524 return False
525
526 inferred_parent = safe_infer(parent_subscript.value)
527 if not (
528 isinstance(inferred_parent, nodes.FunctionDef)
529 and inferred_parent.qname() in {"typing.Optional", "typing.Union"}
530 or isinstance(inferred_parent, astroid.bases.Instance)
531 and inferred_parent.qname() == "typing._SpecialForm"
532 ):
533 return False
534
535 return True
