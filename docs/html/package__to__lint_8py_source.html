76 @property

101 def lazy_clone(self) -> str: # pragma: no cover
102 """Concatenates the target directory and clones the file.
103
104 Not expected to be tested as the primer won't work if it doesn't.
105 It's tested in the continuous integration primers, only the coverage
106 is not calculated on everything. If lazy clone breaks for local use
107 we'll probably notice because we'll have a fatal when launching the
108 primer locally.
109 """
110 logging.info("Lazy cloning %s", self.url)
111 if not self.clone_directory.exists():
112 return self._clone_repository()
113 return self._pull_repository()
114
115 def _clone_repository(self) -> str:
116 options: dict[str, str | int] = {
117 "url": self.url,
118 "to_path": str(self.clone_directory),
119 "branch": self.branch,
120 "depth": 1,
121 }
122 logging.info("Directory does not exists, cloning: %s", options)
123 repo = Repo.clone_from(
124 url=self.url, to_path=self.clone_directory, branch=self.branch, depth=1
125 )
126 return str(repo.head.object.hexsha)
127
128 def _pull_repository(self) -> str:
129 remote_sha1_commit = Git().ls_remote(self.url, self.branch).split("\t")[0]
130 local_sha1_commit = Repo(self.clone_directory).head.object.hexsha
131 if remote_sha1_commit != local_sha1_commit:
132 logging.info(
133 "Remote sha is '%s' while local sha is '%s': pulling new commits",
134 remote_sha1_commit,
135 local_sha1_commit,
136 )
137 try:
138 repo = Repo(self.clone_directory)
139 if repo.is_dirty():
140 raise DirtyPrimerDirectoryException(self.clone_directory)
141 origin = repo.remotes.origin
142 origin.pull()
143 except GitCommandError as e:
144 raise SystemError(
145 f"Failed to clone repository for {self.clone_directory}"
146 ) from e
147 else:
148 logging.info("Repository already up to date.")
149 return str(remote_sha1_commit)
