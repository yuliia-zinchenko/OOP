75class KnownPatternFinder(BaseFinder):
76 def __init__(self, config: Config) -> None:
77 super().__init__(config)
78
79 self.known_patterns: List[Tuple[Pattern[str], str]] = []
80 for placement in reversed(config.sections):
81 known_placement = KNOWN_SECTION_MAPPING.get(placement, placement).lower()
82 config_key = f"known_{known_placement}"
83 known_patterns = list(
84 getattr(self.config, config_key, self.config.known_other.get(known_placement, []))
85 )
86 known_patterns = [
87 pattern
88 for known_pattern in known_patterns
89 for pattern in self._parse_known_pattern(known_pattern)
90 ]
91 for known_pattern in known_patterns:
92 regexp = "^" + known_pattern.replace("*", ".*").replace("?", ".?") + "$"
93 self.known_patterns.append((re.compile(regexp), placement))

95 def _parse_known_pattern(self, pattern: str) -> List[str]:
96 """Expand pattern if identified as a directory and return found sub packages"""
97 if pattern.endswith(os.path.sep):
98 patterns = [
99 filename
100 for filename in os.listdir(os.path.join(self.config.directory, pattern))
101 if os.path.isdir(os.path.join(self.config.directory, pattern, filename))
102 ]
103 else:
104 patterns = [pattern]
105
106 return patterns

108 def find(self, module_name: str) -> Optional[str]:
109 # Try to find most specific placement instruction match (if any)
110 parts = module_name.split(".")
111 module_names_to_check = (".".join(parts[:first_k]) for first_k in range(len(parts), 0, -1))
112 for module_name_to_check in module_names_to_check:
113 for pattern, placement in self.known_patterns:
114 if pattern.match(module_name_to_check):
115 return placement

143 self.paths.append(venv_src_path)

166 def find(self, module_name: str) -> Optional[str]:
167 for prefix in self.paths:
168 package_path = "/".join((prefix, module_name.split(".")[0]))
169 path_obj = Path(package_path).resolve()
170 is_module = (
171 exists_case_sensitive(package_path + ".py")
172 or any(
173 exists_case_sensitive(package_path + ext_suffix)
174 for ext_suffix in importlib.machinery.EXTENSION_SUFFIXES
175 )
176 or exists_case_sensitive(package_path + "/__init__.py")
177 )
178 is_package = exists_case_sensitive(package_path) and os.path.isdir(package_path)
179 if is_module or is_package:
180 if (
181 "site-packages" in prefix
182 or "dist-packages" in prefix
183 or (self.virtual_env and self.virtual_env_src in prefix)
184 ):
185 return sections.THIRDPARTY
186 if os.path.normcase(prefix) == self.stdlib_lib_prefix:
187 return sections.STDLIB
188 if self.conda_env and self.conda_env in prefix:
189 return sections.THIRDPARTY
190 for src_path in self.config.src_paths:
191 if src_path in path_obj.parents and not self.config.is_skipped(path_obj):
192 return sections.FIRSTPARTY
193
194 if os.path.normcase(prefix).startswith(self.stdlib_lib_prefix):
195 return sections.STDLIB # pragma: no cover - edge case for one OS. Hard to test.
196
197 return self.config.default_section
198 return None
