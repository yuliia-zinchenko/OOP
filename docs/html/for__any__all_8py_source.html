25class ConsiderUsingAnyOrAllChecker(BaseChecker):
26 name = "consider-using-any-or-all"
27 msgs = {
28 "C0501": (
29 "`for` loop could be `%s`",
30 "consider-using-any-or-all",
31 "A for loop that checks for a condition and return a bool can be replaced with any or all.",
32 )
33 }
34
35 @only_required_for_messages("consider-using-any-or-all")
36 def visit_for(self, node: nodes.For) -> None:
37 if len(node.body) != 1: # Only If node with no Else
38 return
39 if not isinstance(node.body[0], nodes.If):
40 return
41
42 if_children = list(node.body[0].get_children())
43 if any(isinstance(child, nodes.If) for child in if_children):
44 # an if node within the if-children indicates an elif clause,
45 # suggesting complex logic.
46 return
47
48 node_after_loop = node.next_sibling()
49
50 if self._assigned_reassigned_returned(node, if_children, node_after_loop):
51 final_return_bool = node_after_loop.value.name
52 suggested_string = self._build_suggested_string(node, final_return_bool)
53 self.add_message(
54 "consider-using-any-or-all",
55 node=node,
56 args=suggested_string,
57 confidence=HIGH,
58 )
59 return
60
61 if self._if_statement_returns_bool(if_children, node_after_loop):
62 final_return_bool = node_after_loop.value.value
63 suggested_string = self._build_suggested_string(node, final_return_bool)
64 self.add_message(
65 "consider-using-any-or-all",
66 node=node,
67 args=suggested_string,
68 confidence=HIGH,
69 )
70 return

72 @staticmethod
73 def _if_statement_returns_bool(
74 if_children: list[nodes.NodeNG], node_after_loop: nodes.NodeNG
75 ) -> bool:
76 """Detect for-loop, if-statement, return pattern:
77
78 Ex:
79 def any_uneven(items):
80 for item in items:
81 if not item % 2 == 0:
82 return True
83 return False
84 """
85 if not len(if_children) == 2:
86 # The If node has only a comparison and return
87 return False
88 if not returns_bool(if_children[1]):
89 return False
90
91 # Check for terminating boolean return right after the loop
92 return returns_bool(node_after_loop)
93
94 @staticmethod
95 def _assigned_reassigned_returned(
96 node: nodes.For, if_children: list[nodes.NodeNG], node_after_loop: nodes.NodeNG
97 ) -> bool:
98 """Detect boolean-assign, for-loop, re-assign, return pattern:
99
100 Ex:
101 def check_lines(lines, max_chars):
102 long_line = False
103 for line in lines:
104 if len(line) > max_chars:
105 long_line = True
106 # no elif / else statement
107 return long_line
108 """
109 node_before_loop = node.previous_sibling()
110
111 if not assigned_bool(node_before_loop):
112 # node before loop isn't assigning to boolean
113 return False
114
115 assign_children = [x for x in if_children if isinstance(x, nodes.Assign)]
116 if not assign_children:
117 # if-nodes inside loop aren't assignments
118 return False
119
120 # We only care for the first assign node of the if-children. Otherwise it breaks the pattern.
121 first_target = assign_children[0].targets[0]
122 target_before_loop = node_before_loop.targets[0]
123
124 if not (
125 isinstance(first_target, nodes.AssignName)
126 and isinstance(target_before_loop, nodes.AssignName)
127 ):
128 return False
129
130 node_before_loop_name = node_before_loop.targets[0].name
131 return (
132 first_target.name == node_before_loop_name
133 and isinstance(node_after_loop, nodes.Return)
134 and isinstance(node_after_loop.value, nodes.Name)
135 and node_after_loop.value.name == node_before_loop_name
136 )
137
138 @staticmethod
139 def _build_suggested_string(node: nodes.For, final_return_bool: bool) -> str:
140 """When a nodes.For node can be rewritten as an any/all statement, return a
141 suggestion for that statement.
142
143 'final_return_bool' is the boolean literal returned after the for loop if all
144 conditions fail.
145 """
146 loop_var = node.target.as_string()
147 loop_iter = node.iter.as_string()
148 test_node = next(node.body[0].get_children())
149
150 if isinstance(test_node, nodes.UnaryOp) and test_node.op == "not":
151 # The condition is negated. Advance the node to the operand and modify the suggestion
152 test_node = test_node.operand
153 suggested_function = "all" if final_return_bool else "not all"
154 else:
155 suggested_function = "not any" if final_return_bool else "any"
156
157 test = test_node.as_string()
158 return f"{suggested_function}({test} for {loop_var} in {loop_iter})"
