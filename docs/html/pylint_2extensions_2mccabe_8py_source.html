54class PathGraphingAstVisitor(Mccabe_PathGraphingAstVisitor): # type: ignore[misc]
55 def __init__(self) -> None:
56 super().__init__()
57 self._bottom_counter = 0
58 self.graph: PathGraph | None = None
59
60 def default(self, node: nodes.NodeNG, *args: Any) -> None:
61 for child in node.get_children():
62 self.dispatch(child, *args)
63
64 def dispatch(self, node: nodes.NodeNG, *args: Any) -> Any:
65 self.node = node
66 klass = node.__class__
67 meth = self._cache.get(klass)
68 if meth is None:
69 class_name = klass.__name__
70 meth = getattr(self.visitor, "visit" + class_name, self.default)
71 self._cache[klass] = meth
72 return meth(node, *args)
73
74 def visitFunctionDef(self, node: nodes.FunctionDef) -> None:
75 if self.graph is not None:
76 # closure
77 pathnode = self._append_node(node)
78 self.tail = pathnode
79 self.dispatch_list(node.body)
80 bottom = f"{self._bottom_counter}"
81 self._bottom_counter += 1
82 self.graph.connect(self.tail, bottom)
83 self.graph.connect(node, bottom)
84 self.tail = bottom
85 else:
86 self.graph = PathGraph(node)
87 self.tail = node
88 self.dispatch_list(node.body)
89 self.graphs[f"{self.classname}{node.name}"] = self.graph
90 self.reset()
91
92 visitAsyncFunctionDef = visitFunctionDef
93
94 def visitSimpleStatement(self, node: _StatementNodes) -> None:
95 self._append_node(node)

109 def _append_node(self, node: _AppendableNodeT) -> _AppendableNodeT | None:
110 if not self.tail or not self.graph:
111 return None
112 self.graph.connect(self.tail, node)
113 self.tail = node
114 return node

116 def _subgraph(
117 self,
118 node: _SubGraphNodes,
119 name: str,
120 extra_blocks: Sequence[nodes.ExceptHandler] = (),
121 ) -> None:
122 """Create the subgraphs representing any `if` and `for` statements."""
123 if self.graph is None:
124 # global loop
125 self.graph = PathGraph(node)
126 self._subgraph_parse(node, node, extra_blocks)
127 self.graphs[f"{self.classname}{name}"] = self.graph
128 self.reset()
129 else:
130 self._append_node(node)
131 self._subgraph_parse(node, node, extra_blocks)
132
133 def _subgraph_parse(
134 self,
135 node: _SubGraphNodes,
136 pathnode: _SubGraphNodes,
137 extra_blocks: Sequence[nodes.ExceptHandler],
138 ) -> None:
139 """Parse the body and any `else` block of `if` and `for` statements."""
140 loose_ends = []
141 self.tail = node
142 self.dispatch_list(node.body)
143 loose_ends.append(self.tail)
144 for extra in extra_blocks:
145 self.tail = node
146 self.dispatch_list(extra.body)
147 loose_ends.append(self.tail)
148 if node.orelse:
149 self.tail = node
150 self.dispatch_list(node.orelse)
151 loose_ends.append(self.tail)
152 else:
153 loose_ends.append(node)
154 if node and self.graph:
155 bottom = f"{self._bottom_counter}"
156 self._bottom_counter += 1
157 for end in loose_ends:
158 self.graph.connect(end, bottom)
159 self.tail = bottom

162class McCabeMethodChecker(checkers.BaseChecker):
163 """Checks McCabe complexity cyclomatic threshold in methods and functions
164 to validate a too complex code.
165 """
166
167 name = "design"
168
169 msgs = {
170 "R1260": (
171 "%s is too complex. The McCabe rating is %d",
172 "too-complex",
173 "Used when a method or function is too complex based on "
174 "McCabe Complexity Cyclomatic",
175 )
176 }
177 options = (
178 (
179 "max-complexity",
180 {
181 "default": 10,
182 "type": "int",
183 "metavar": "<int>",
184 "help": "McCabe complexity cyclomatic threshold",
185 },
186 ),
187 )
188
189 @only_required_for_messages("too-complex")
190 def visit_module(self, node: nodes.Module) -> None:
191 """Visit an astroid.Module node to check too complex rating and
192 add message if is greater than max_complexity stored from options.
193 """
194 visitor = PathGraphingAstVisitor()
195 for child in node.body:
196 visitor.preorder(child, visitor)
197 for graph in visitor.graphs.values():
198 complexity = graph.complexity()
199 node = graph.root
200 if hasattr(node, "name"):
201 node_name = f"'{node.name}'"
202 else:
203 node_name = f"This '{node.__class__.__name__.lower()}'"
204 if complexity <= self.linter.config.max_complexity:
205 continue
206 self.add_message(
207 "too-complex", node=node, confidence=HIGH, args=(node_name, complexity)
208 )
