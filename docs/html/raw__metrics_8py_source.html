18def report_raw_stats(
19 sect: Section,
20 stats: LinterStats,
21 old_stats: LinterStats | None,
22) -> None:
23 """Calculate percentage of code / doc / comment / empty."""
24 total_lines = stats.code_type_count["total"]
25 sect.insert(0, Paragraph([Text(f"{total_lines} lines have been analyzed\n")]))
26 lines = ["type", "number", "%", "previous", "difference"]
27 for node_type in ("code", "docstring", "comment", "empty"):
28 node_type = cast(Literal["code", "docstring", "comment", "empty"], node_type)
29 total = stats.code_type_count[node_type]
30 percent = float(total * 100) / total_lines if total_lines else None
31 old = old_stats.code_type_count[node_type] if old_stats else None
32 diff_str = diff_string(old, total) if old else None
33 lines += [
34 node_type,
35 str(total),
36 f"{percent:.2f}" if percent is not None else "NC",
37 str(old) if old else "NC",
38 diff_str if diff_str else "NC",
39 ]
40 sect.append(Table(children=lines, cols=5, rheaders=1))

80def get_type(
81 tokens: list[tokenize.TokenInfo], start_index: int
82) -> tuple[int, int, Literal["code", "docstring", "comment", "empty"]]:
83 """Return the line type : docstring, comment, code, empty."""
84 i = start_index
85 start = tokens[i][2]
86 pos = start
87 line_type = None
88 while i < len(tokens) and tokens[i][2][0] == start[0]:
89 tok_type = tokens[i][0]
90 pos = tokens[i][3]
91 if line_type is None:
92 if tok_type == tokenize.STRING:
93 line_type = "docstring"
94 elif tok_type == tokenize.COMMENT:
95 line_type = "comment"
96 elif tok_type in JUNK:
97 pass
98 else:
99 line_type = "code"
100 i += 1
101 if line_type is None:
102 line_type = "empty"
103 elif i < len(tokens) and tokens[i][0] == tokenize.NEWLINE:
104 i += 1
105 # Mypy fails to infer the literal of line_type
106 return i, pos[0] - start[0] + 1, line_type # type: ignore[return-value]
