463class ParallelTestSuite(unittest.TestSuite):
464 """
465 Run a series of tests in parallel in several processes.
466
467 While the unittest module's documentation implies that orchestrating the
468 execution of tests is the responsibility of the test runner, in practice,
469 it appears that TestRunner classes are more concerned with formatting and
470 displaying test results.
471
472 Since there are fewer use cases for customizing TestSuite than TestRunner,
473 implementing parallelization at the level of the TestSuite improves
474 interoperability with existing custom test runners. A single instance of a
475 test runner can still collect results from all tests without being aware
476 that they have been run in parallel.
477 """
478
479 # In case someone wants to modify these in a subclass.
480 init_worker = _init_worker
481 process_setup = _process_setup_stub
482 process_setup_args = ()
483 run_subsuite = _run_subsuite
484 runner_class = RemoteTestRunner
485
486 def __init__(
487 self, subsuites, processes, failfast=False, debug_mode=False, buffer=False
488 ):
489 self.subsuites = subsuites
490 self.processes = processes
491 self.failfast = failfast
492 self.debug_mode = debug_mode
493 self.buffer = buffer
494 self.initial_settings = None
495 self.serialized_contents = None
496 self.used_aliases = None
497 super().__init__()
498
499 def run(self, result):
500 """
501 Distribute TestCases across workers.
502
503 Return an identifier of each TestCase with its result in order to use
504 imap_unordered to show results as soon as they're available.
505
506 To minimize pickling errors when getting results from workers:
507
508 - pass back numeric indexes in self.subsuites instead of tests
509 - make tracebacks picklable with tblib, if available
510
511 Even with tblib, errors may still occur for dynamically created
512 exception classes which cannot be unpickled.
513 """
514 self.initialize_suite()
515 counter = multiprocessing.Value(ctypes.c_int, 0)
516 pool = multiprocessing.Pool(
517 processes=self.processes,
518 initializer=self.init_worker.__func__,
519 initargs=[
520 counter,
521 self.initial_settings,
522 self.serialized_contents,
523 self.process_setup.__func__,
524 self.process_setup_args,
525 self.debug_mode,
526 self.used_aliases,
527 ],
528 )
529 args = [
530 (self.runner_class, index, subsuite, self.failfast, self.buffer)
531 for index, subsuite in enumerate(self.subsuites)
532 ]
533 test_results = pool.imap_unordered(self.run_subsuite.__func__, args)
534
535 while True:
536 if result.shouldStop:
537 pool.terminate()
538 break
539
540 try:
541 subsuite_index, events = test_results.next(timeout=0.1)
542 except multiprocessing.TimeoutError:
543 continue
544 except StopIteration:
545 pool.close()
546 break
547
548 tests = list(self.subsuites[subsuite_index])
549 for event in events:
550 event_name = event[0]
551 handler = getattr(result, event_name, None)
552 if handler is None:
553 continue
554 test = tests[event[1]]
555 args = event[2:]
556 handler(test, *args)
557
558 pool.join()
559
560 return result

577class Shuffler:
578 """
579 This class implements shuffling with a special consistency property.
580 Consistency means that, for a given seed and key function, if two sets of
581 items are shuffled, the resulting order will agree on the intersection of
582 the two sets. For example, if items are removed from an original set, the
583 shuffled order for the new set will be the shuffled order of the original
584 set restricted to the smaller set.
585 """
586
587 # This doesn't need to be cryptographically strong, so use what's fastest.
588 hash_algorithm = "md5"
589
590 @classmethod
591 def _hash_text(cls, text):
592 h = hashlib.new(cls.hash_algorithm, usedforsecurity=False)
593 h.update(text.encode("utf-8"))
594 return h.hexdigest()
595
596 def __init__(self, seed=None):
597 if seed is None:
598 # Limit seeds to 10 digits for simpler output.
599 seed = random.randint(0, 10**10 - 1)
600 seed_source = "generated"
601 else:
602 seed_source = "given"
603 self.seed = seed
604 self.seed_source = seed_source
605
606 @property
607 def seed_display(self):
608 return f"{self.seed!r} ({self.seed_source})"
609
610 def _hash_item(self, item, key):
611 text = "{}{}".format(self.seed, key(item))
612 return self._hash_text(text)
613
614 def shuffle(self, items, key):
615 """
616 Return a new list of the items in a shuffled order.
617
618 The `key` is a function that accepts an item in `items` and returns
619 a string unique for that item that can be viewed as a string id. The
620 order of the return value is deterministic. It depends on the seed
621 and key function but not on the original order.
622 """
623 hashes = {}
624 for item in items:
625 hashed = self._hash_item(item, key)
626 if hashed in hashes:
627 msg = "item {!r} has same hash {!r} as item {!r}".format(
628 item,
629 hashed,
630 hashes[hashed],
631 )
632 raise RuntimeError(msg)
633 hashes[hashed] = item
634 return [hashes[hashed] for hashed in sorted(hashes)]

903 test_labels = test_labels or ["."]

908 if self.top_level is not None:
909 discover_kwargs["top_level_dir"] = self.top_level

920 "Including test tag(s): %s." % ", ".join(sorted(self.tags)),

926 level=logging.DEBUG,

932 # found or that couldn't be loaded due to syntax errors.
933 test_types = (unittest.loader._FailedTest, *self.reorder_by)

935 reorder_tests(
936 all_tests,
937 test_types,
938 shuffler=self._shuffler,
939 reverse=self.reverse,
940 )

948 # basis, there's no need for more processes than TestCases.

1105def find_top_level(top_level):
1106 # Try to be a bit smarter than unittest about finding the default top-level
1107 # for a given directory path, to avoid breaking relative imports.
1108 # (Unittest's default is to set top-level equal to the path, which means
1109 # relative imports will result in "Attempted relative import in
1110 # non-package.").
1111
1112 # We'd be happy to skip this and require dotted module paths (which don't
1113 # cause this problem) instead of file paths (which do), but in the case of
1114 # a directory in the cwd, which would be equally valid if considered as a
1115 # top-level module or as a directory path, unittest unfortunately prefers
1116 # the latter.
1117 while True:
1118 init_py = os.path.join(top_level, "__init__.py")
1119 if not os.path.exists(init_py):
1120 break
1121 try_next = os.path.dirname(top_level)
1122 if try_next == top_level:
1123 # __init__.py all the way down? give up.
1124 break
1125 top_level = try_next
1126 return top_level

1133def shuffle_tests(tests, shuffler):
1134 """
1135 Return an iterator over the given tests in a shuffled order, keeping tests
1136 next to other tests of their class.
1137
1138 `tests` should be an iterable of tests.
1139 """
1140 tests_by_type = {}
1141 for _, class_tests in itertools.groupby(tests, type):
1142 class_tests = list(class_tests)
1143 test_type = type(class_tests[0])
1144 class_tests = shuffler.shuffle(class_tests, key=lambda test: test.id())
1145 tests_by_type[test_type] = class_tests
1146
1147 classes = shuffler.shuffle(tests_by_type, key=_class_shuffle_key)
1148
1149 return itertools.chain(*(tests_by_type[cls] for cls in classes))

1152def reorder_test_bin(tests, shuffler=None, reverse=False):
1153 """
1154 Return an iterator that reorders the given tests, keeping tests next to
1155 other tests of their class.
1156
1157 `tests` should be an iterable of tests that supports reversed().
1158 """
1159 if shuffler is None:
1160 if reverse:
1161 return reversed(tests)

1163 return iter(tests)

1166 if not reverse:
1167 return tests
1168 # Arguments to reversed() must be reversible.
1169 return reversed(list(tests))

1172def reorder_tests(tests, classes, reverse=False, shuffler=None):
1173 """
1174 Reorder an iterable of tests, grouping by the given TestCase classes.
1175
1176 This function also removes any duplicates and reorders so that tests of the
1177 same type are consecutive.
1178
1179 The result is returned as an iterator. `classes` is a sequence of types.
1180 Tests that are instances of `classes[0]` are grouped first, followed by
1181 instances of `classes[1]`, etc. Tests that are not instances of any of the
1182 classes are grouped last.
1183
1184 If `reverse` is True, the tests within each `classes` group are reversed,
1185 but without reversing the order of `classes` itself.
1186
1187 The `shuffler` argument is an optional instance of this module's `Shuffler`
1188 class. If provided, tests will be shuffled within each `classes` group, but
1189 keeping tests with other tests of their TestCase class. Reversing is
1190 applied after shuffling to allow reversing the same random order.
1191 """
1192 # Each bin maps TestCase class to OrderedSet of tests. This permits tests
1193 # to be grouped by TestCase class even if provided non-consecutively.
1194 bins = [defaultdict(OrderedSet) for i in range(len(classes) + 1)]
1195 *class_bins, last_bin = bins
1196
1197 for test in tests:
1198 for test_bin, test_class in zip(class_bins, classes):
1199 if isinstance(test, test_class):
1200 break
1201 else:
1202 test_bin = last_bin
1203 test_bin[type(test)].add(test)
1204
1205 for test_bin in bins:
1206 # Call list() since reorder_test_bin()'s input must support reversed().
1207 tests = list(itertools.chain.from_iterable(test_bin.values()))
1208 yield from reorder_test_bin(tests, shuffler=shuffler, reverse=reverse)

1218def test_match_tags(test, tags, exclude_tags):
