21 def __init__(self, lhs, rhs):
22 rhs, *self.rhs_params = rhs if isinstance(rhs, (list, tuple)) else [rhs]
23 super().__init__(lhs, rhs)
24 self.template_params = {}
25 self.process_rhs_params()
26
27 def process_rhs_params(self):
28 if self.rhs_params:
29 # Check if a band index was passed in the query argument.
30 if len(self.rhs_params) == (2 if self.lookup_name == "relate" else 1):
31 self.process_band_indices()
32 elif len(self.rhs_params) > 1:
33 raise ValueError("Tuple too long for lookup %s." % self.lookup_name)
34 elif isinstance(self.lhs, RasterBandTransform):
35 self.process_band_indices(only_lhs=True)
36
37 def process_band_indices(self, only_lhs=False):
38 """
39 Extract the lhs band index from the band transform class and the rhs
40 band index from the input tuple.
41 """
42 # PostGIS band indices are 1-based, so the band index needs to be
43 # increased to be consistent with the GDALRaster band indices.
44 if only_lhs:
45 self.band_rhs = 1
46 self.band_lhs = self.lhs.band_index + 1
47 return
48
49 if isinstance(self.lhs, RasterBandTransform):
50 self.band_lhs = self.lhs.band_index + 1
51 else:
52 self.band_lhs = 1
53
54 self.band_rhs, *self.rhs_params = self.rhs_params
55
56 def get_db_prep_lookup(self, value, connection):
57 # get_db_prep_lookup is called by process_rhs from super class
58 return ("%s", [connection.ops.Adapter(value)])
59
60 def process_rhs(self, compiler, connection):
61 if isinstance(self.rhs, Query):
62 # If rhs is some Query, don't touch it.
63 return super().process_rhs(compiler, connection)
64 if isinstance(self.rhs, Expression):
65 self.rhs = self.rhs.resolve_expression(compiler.query)
66 rhs, rhs_params = super().process_rhs(compiler, connection)
67 placeholder = connection.ops.get_geom_placeholder(
68 self.lhs.output_field, self.rhs, compiler
69 )
70 return placeholder % rhs, rhs_params
71
72 def get_rhs_op(self, connection, rhs):
73 # Unlike BuiltinLookup, the GIS get_rhs_op() implementation should return
74 # an object (SpatialOperator) with an as_sql() method to allow for more
75 # complex computations (where the lhs part can be mixed in).
76 return connection.ops.gis_operators[self.lookup_name]
77
78 def as_sql(self, compiler, connection):
79 lhs_sql, lhs_params = self.process_lhs(compiler, connection)
80 rhs_sql, rhs_params = self.process_rhs(compiler, connection)
81 sql_params = (*lhs_params, *rhs_params)
82
83 template_params = {
84 "lhs": lhs_sql,
85 "rhs": rhs_sql,
86 "value": "%s",
87 **self.template_params,
88 }
89 rhs_op = self.get_rhs_op(connection, rhs_sql)
90 return rhs_op.as_sql(connection, self, template_params, sql_params)

278 def process_rhs(self, compiler, connection):
279 # Check the pattern argument
280 pattern = self.rhs_params[0]
281 backend_op = connection.ops.gis_operators[self.lookup_name]
282 if hasattr(backend_op, "check_relate_argument"):
283 backend_op.check_relate_argument(pattern)
284 elif not isinstance(pattern, str) or not self.pattern_regex.match(pattern):
285 raise ValueError('Invalid intersection matrix pattern "%s".' % pattern)
286 sql, params = super().process_rhs(compiler, connection)

304 def process_rhs_params(self):

306 raise ValueError(
