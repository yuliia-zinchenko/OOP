25class ReportsHandlerMixIn:
26 """A mix-in class containing all the reports and stats manipulation
27 related methods for the main lint class.
28 """
29
30 def __init__(self) -> None:
31 self._reports: ReportsDict = collections.defaultdict(list)
32 self._reports_state: dict[str, bool] = {}
33
34 def report_order(self) -> MutableSequence[BaseChecker]:
35 """Return a list of reporters."""
36 return list(self._reports)
37
38 def register_report(
39 self, reportid: str, r_title: str, r_cb: ReportsCallable, checker: BaseChecker
40 ) -> None:
41 """Register a report.
42
43 :param reportid: The unique identifier for the report
44 :param r_title: The report's title
45 :param r_cb: The method to call to make the report
46 :param checker: The checker defining the report
47 """
48 reportid = reportid.upper()
49 self._reports[checker].append((reportid, r_title, r_cb))
50
51 def enable_report(self, reportid: str) -> None:
52 """Enable the report of the given id."""
53 reportid = reportid.upper()
54 self._reports_state[reportid] = True
55
56 def disable_report(self, reportid: str) -> None:
57 """Disable the report of the given id."""
58 reportid = reportid.upper()
59 self._reports_state[reportid] = False
60
61 def report_is_enabled(self, reportid: str) -> bool:
62 """Is the report associated to the given identifier enabled ?"""
63 return self._reports_state.get(reportid, True)
64
65 def make_reports( # type: ignore[misc] # ReportsHandlerMixIn is always mixed with PyLinter
66 self: PyLinter,
67 stats: LinterStats,
68 old_stats: LinterStats | None,
69 ) -> Section:
70 """Render registered reports."""
71 sect = Section("Report", f"{self.stats.statement} statements analysed.")
72 for checker in self.report_order():
73 for reportid, r_title, r_cb in self._reports[checker]:
74 if not self.report_is_enabled(reportid):
75 continue
76 report_sect = Section(r_title)
77 try:
78 r_cb(report_sect, stats, old_stats)
79 except EmptyReportError:
80 continue
81 report_sect.report_id = reportid
82 sect.append(report_sect)
83 return sect
