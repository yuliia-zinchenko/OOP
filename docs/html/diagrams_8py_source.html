121 def get_attrs(self, node: nodes.ClassDef) -> list[str]:
122 """Return visible attributes, possibly with class name."""
123 attrs = []
124 properties = {
125 local_name: local_node
126 for local_name, local_node in node.items()
127 if isinstance(local_node, nodes.FunctionDef)
128 and decorated_with_property(local_node)
129 }
130 for attr_name, attr_type in list(node.locals_type.items()) + list(
131 node.instance_attrs_type.items()
132 ):
133 if attr_name not in properties:
134 properties[attr_name] = attr_type
135
136 for node_name, associated_nodes in properties.items():
137 if not self.show_attr(node_name):
138 continue
139 names = self.class_names(associated_nodes)
140 if names:
141 node_name = f"{node_name} : {', '.join(names)}"
142 attrs.append(node_name)
143 return sorted(attrs)
144
145 def get_methods(self, node: nodes.ClassDef) -> list[nodes.FunctionDef]:
146 """Return visible methods."""
147 methods = [
148 m
149 for m in node.values()
150 if isinstance(m, nodes.FunctionDef)
151 and not isinstance(m, astroid.objects.Property)
152 and not decorated_with_property(m)
153 and self.show_attr(m.name)
154 ]
155 return sorted(methods, key=lambda n: n.name)

157 def add_object(self, title: str, node: nodes.ClassDef) -> None:

164 def class_names(self, nodes_lst: Iterable[nodes.NodeNG]) -> list[str]:
165 """Return class names if needed in diagram."""
166 names = []
167 for node in nodes_lst:
168 if isinstance(node, astroid.Instance):
169 node = node._proxied
170 if (
171 isinstance(
172 node, (nodes.ClassDef, nodes.Name, nodes.Subscript, nodes.BinOp)
173 )
174 and hasattr(node, "name")
175 and not self.has_node(node)
176 ):
177 if node.name not in names:
178 node_name = node.name
179 names.append(node_name)
180 # sorted to get predictable (hence testable) results
181 return sorted(
182 name
183 for name in names
184 if all(name not in other or name == other for other in names)
185 )

206 def extract_relationships(self) -> None:
207 """Extract relationships between nodes in the diagram."""
208 for obj in self.classes():
209 node = obj.node
210 obj.attrs = self.get_attrs(node)
211 obj.methods = self.get_methods(node)
212 obj.shape = "class"
213 # inheritance link
214 for par_node in node.ancestors(recurs=False):
215 try:
216 par_obj = self.object_from_node(par_node)
217 self.add_relationship(obj, par_obj, "specialization")
218 except KeyError:
219 continue
220
221 # associations & aggregations links
222 for name, values in list(node.aggregations_type.items()):
223 for value in values:
224 self.assign_association_relationship(
225 value, obj, name, "aggregation"
226 )
227
228 associations = node.associations_type.copy()
229
230 for name, values in node.locals_type.items():
231 if name not in associations:
232 associations[name] = values
233
234 for name, values in associations.items():
235 for value in values:
236 self.assign_association_relationship(
237 value, obj, name, "association"
238 )
239
240 def assign_association_relationship(
241 self, value: astroid.NodeNG, obj: ClassEntity, name: str, type_relationship: str
242 ) -> None:
243 if isinstance(value, util.UninferableBase):
244 return
245 if isinstance(value, astroid.Instance):
246 value = value._proxied
247 try:
248 associated_obj = self.object_from_node(value)
249 self.add_relationship(associated_obj, obj, type_relationship, name)
250 except KeyError:
251 return

277 def get_module(self, name: str, node: nodes.Module) -> PackageEntity:
278 """Return a module by its name, looking also for relative imports;
279 raise KeyError if not found.
280 """
281 for mod in self.modules():
282 mod_name = mod.node.name
283 if mod_name == name:
284 return mod
285 # search for fullname of relative import modules
286 package = node.root().name
287 if mod_name == f"{package}.{name}":
288 return mod
289 if mod_name == f"{package.rsplit('.', 1)[0]}.{name}":
290 return mod
291 raise KeyError(name)

293 def add_from_depend(self, node: nodes.ImportFrom, from_module: str) -> None:
294 """Add dependencies created by from-imports."""
295 mod_name = node.root().name
296 package = self.module(mod_name).node
297
298 if from_module in package.depends:
299 return
300
301 if not in_type_checking_block(node):
302 package.depends.append(from_module)
303 elif from_module not in package.type_depends:
304 package.type_depends.append(from_module)

306 def extract_relationships(self) -> None:
307 """Extract relationships between nodes in the diagram."""
308 super().extract_relationships()
309 for class_obj in self.classes():
310 # ownership
311 try:
312 mod = self.object_from_node(class_obj.node.root())
313 self.add_relationship(class_obj, mod, "ownership")
314 except KeyError:
315 continue
316 for package_obj in self.modules():
317 package_obj.shape = "package"
318 # dependencies
319 for dep_name in package_obj.node.depends:
320 try:
321 dep = self.get_module(dep_name, package_obj.node)
322 except KeyError:
323 continue
324 self.add_relationship(package_obj, dep, "depends")
325
326 for dep_name in package_obj.node.type_depends:
327 try:
328 dep = self.get_module(dep_name, package_obj.node)
329 except KeyError: # pragma: no cover
330 continue
331 self.add_relationship(package_obj, dep, "type_depends")
