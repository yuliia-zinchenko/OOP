37def check_string(result, func, cargs, offset=-1, str_result=False):
38 """
39 Check the string output returned from the given function, and free
40 the string pointer allocated by OGR. The `str_result` keyword
41 may be used when the result is the string pointer, otherwise
42 the OGR error code is assumed. The `offset` keyword may be used
43 to extract the string pointer passed in by-reference at the given
44 slice offset in the function arguments.
45 """
46 if str_result:
47 # For routines that return a string.
48 ptr = result
49 if not ptr:
50 s = None
51 else:
52 s = string_at(result)
53 else:
54 # Error-code return specified.
55 check_err(result)
56 ptr = ptr_byref(cargs, offset)
57 # Getting the string value
58 s = ptr.value
59 # Correctly freeing the allocated memory behind GDAL pointer
60 # with the VSIFree routine.
61 if ptr:
62 lgdal.VSIFree(ptr)
63 return s
