38 def filter_by_widget(self, widget_type):

106class FormHelper(DynamicLayoutHandler):
107 """
108 This class controls the form rendering behavior of the form passed to
109 the `{% crispy %}` tag. For doing so you will need to set its attributes
110 and pass the corresponding helper object to the tag::
111
112 {% crispy form form.helper %}
113
114 Let's see what attributes you can set and what form behaviors they apply to:
115
116 **form_method**: Specifies form method attribute.
117 You can set it to 'POST' or 'GET'. Defaults to 'POST'
118
119 **form_action**: Applied to the form action attribute:
120 - Can be a named url in your URLconf that can be executed via the `{% url %}` template tag. \
121 Example: 'show_my_profile'. In your URLconf you could have something like::
122
123 path('show/profile/', 'show_my_profile_view', name = 'show_my_profile')
124
125 - It can simply point to a URL '/whatever/blabla/'.
126
127 **form_id**: Generates a form id for dom identification.
128 If no id provided then no id attribute is created on the form.
129
130 **form_class**: String containing separated CSS classes to be applied
131 to form class attribute.
132
133 **form_group_wrapper_class**: String containing separated CSS classes to be applied
134 to each row of inputs.
135
136 **form_tag**: It specifies if <form></form> tags should be rendered when using a Layout.
137 If set to False it renders the form without the <form></form> tags. Defaults to True.
138
139 **form_error_title**: If a form has `non_field_errors` to display, they
140 are rendered in a div. You can set title's div with this attribute.
141 Example: "Oooops!" or "Form Errors"
142
143 **formset_error_title**: If a formset has `non_form_errors` to display, they
144 are rendered in a div. You can set title's div with this attribute.
145
146 **include_media**: Whether to automatically include form media. Set to False if
147 you want to manually include form media outside the form. Defaults to True.
148
149 Public Methods:
150
151 **add_input(input)**: You can add input buttons using this method. Inputs
152 added using this method will be rendered at the end of the form/formset.
153
154 **add_layout(layout)**: You can add a `Layout` object to `FormHelper`. The Layout
155 specifies in a simple, clean and DRY way how the form fields should be rendered.
156 You can wrap fields, order them, customize pretty much anything in the form.
157
158 Best way to add a helper to a form is adding a property named helper to the form
159 that returns customized `FormHelper` object::
160
161 from crispy_forms.helper import FormHelper
162 from crispy_forms.layout import Submit
163
164 class MyForm(forms.Form):
165 title = forms.CharField(_("Title"))
166
167 @property
168 def helper(self):
169 helper = FormHelper()
170 helper.form_id = 'this-form-rocks'
171 helper.form_class = 'search'
172 helper.add_input(Submit('save', 'save'))
173 [...]
174 return helper
175
176 You can use it in a template doing::
177
178 {% load crispy_forms_tags %}
179 {% crispy form %}
180 """

268 def render_layout(self, form, context, template_pack=TEMPLATE_PACK):
269 """
270 Returns safe html of the rendering of the layout
271 """
272 form.rendered_fields = set()
273 form.crispy_field_template = self.field_template
274
275 # This renders the specified Layout strictly
276 html = self.layout.render(form, context, template_pack=template_pack)
277
278 # Rendering some extra fields if specified
279 if self.render_unmentioned_fields or self.render_hidden_fields or self.render_required_fields:
280 fields = tuple(form.fields.keys())
281 left_fields_to_render = list_difference(fields, form.rendered_fields)
282 for field in left_fields_to_render:
283 if (
284 self.render_unmentioned_fields
285 or (self.render_hidden_fields and form.fields[field].widget.is_hidden)
286 or (self.render_required_fields and form.fields[field].widget.is_required)
287 ):
288 html += render_field(field, form, context, template_pack=template_pack)
289
290 return mark_safe(html)
