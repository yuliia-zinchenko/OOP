35 def __init__(self, dr_input):
36 """
37 Initialize an GDAL/OGR driver on either a string or integer input.
38 """
39 if isinstance(dr_input, str):
40 # If a string name of the driver was passed in
41 self.ensure_registered()
42
43 # Checking the alias dictionary (case-insensitive) to see if an
44 # alias exists for the given driver.
45 if dr_input.lower() in self._alias:
46 name = self._alias[dr_input.lower()]
47 else:
48 name = dr_input
49
50 # Attempting to get the GDAL/OGR driver by the string name.
51 driver = c_void_p(capi.get_driver_by_name(force_bytes(name)))
52 elif isinstance(dr_input, int):
53 self.ensure_registered()
54 driver = capi.get_driver(dr_input)
55 elif isinstance(dr_input, c_void_p):
56 driver = dr_input
57 else:
58 raise GDALException(
59 "Unrecognized input type for GDAL/OGR Driver: %s" % type(dr_input)
60 )
61
62 # Making sure we get a valid pointer to the OGR Driver
63 if not driver:
64 raise GDALException(
65 "Could not initialize GDAL/OGR Driver on input: %s" % dr_input
66 )
