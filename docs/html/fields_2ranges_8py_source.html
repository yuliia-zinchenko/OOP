139class ContinuousRangeField(RangeField):
140 """
141 Continuous range field. It allows specifying default bounds for list and
142 tuple inputs.
143 """
144
145 def __init__(self, *args, default_bounds=CANONICAL_RANGE_BOUNDS, **kwargs):
146 if default_bounds not in ("[)", "(]", "()", "[]"):
147 raise ValueError("default_bounds must be one of '[)', '(]', '()', or '[]'.")
148 self.default_bounds = default_bounds
149 super().__init__(*args, **kwargs)
150
151 def get_prep_value(self, value):
152 if isinstance(value, (list, tuple)):
153 return self.range_type(value[0], value[1], self.default_bounds)
154 return super().get_prep_value(value)
155
156 def formfield(self, **kwargs):
157 kwargs.setdefault("default_bounds", self.default_bounds)
158 return super().formfield(**kwargs)
159
160 def deconstruct(self):
161 name, path, args, kwargs = super().deconstruct()
162 if self.default_bounds and self.default_bounds != CANONICAL_RANGE_BOUNDS:
163 kwargs["default_bounds"] = self.default_bounds

275 def process_rhs(self, compiler, connection):
276 rhs, rhs_params = super().process_rhs(compiler, connection)
277 # Ignore precision for DecimalFields.
278 db_type = self.lhs.output_field.cast_db_type(connection).split("(")[0]
279 cast_type = self.type_mapping[db_type]
