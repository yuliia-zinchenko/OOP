28def get_internal_wsgi_application():
29 """
30 Load and return the WSGI application as configured by the user in
31 ``settings.WSGI_APPLICATION``. With the default ``startproject`` layout,
32 this will be the ``application`` object in ``projectname/wsgi.py``.
33
34 This function, and the ``WSGI_APPLICATION`` setting itself, are only useful
35 for Django's internal server (runserver); external WSGI servers should just
36 be configured to point to the correct application object directly.
37
38 If settings.WSGI_APPLICATION is not set (is ``None``), return
39 whatever ``django.core.wsgi.get_wsgi_application`` returns.
40 """
41 from django.conf import settings
42
43 app_path = getattr(settings, "WSGI_APPLICATION")
44 if app_path is None:
45 return get_wsgi_application()

48 return import_string(app_path)

50 raise ImproperlyConfigured(
51 "WSGI application '%s' could not be loaded; "
52 "Error importing module." % app_path

114class ServerHandler(simple_server.ServerHandler):

117 def __init__(self, stdin, stdout, stderr, environ, **kwargs):
118 """
119 Use a LimitedStream so that unread request data will be ignored at
120 the end of the request. WSGIRequest uses a LimitedStream but it
121 shouldn't discard the data since the upstream servers usually do this.
122 This fix applies only for testserver/runserver.
123 """
124 try:
125 content_length = int(environ.get("CONTENT_LENGTH"))
126 except (ValueError, TypeError):
127 content_length = 0
128 super().__init__(
129 LimitedStream(stdin, content_length), stdout, stderr, environ, **kwargs
130 )
131
132 def cleanup_headers(self):
133 super().cleanup_headers()
134 if (
135 self.environ["REQUEST_METHOD"] == "HEAD"
136 and "Content-Length" in self.headers
137 and str(self.headers["Content-Length"]) == "0"
138 ):
139 del self.headers["Content-Length"]
140 # HTTP/1.1 requires support for persistent connections. Send 'close' if
141 # the content length is unknown to prevent clients from reusing the
142 # connection.
143 if (
144 self.environ["REQUEST_METHOD"] != "HEAD"
145 and "Content-Length" not in self.headers
146 ):
147 self.headers["Connection"] = "close"
148 # Persistent connections require threading server.
149 elif not isinstance(self.request_handler.server, socketserver.ThreadingMixIn):
150 self.headers["Connection"] = "close"
151 # Mark the connection for closing if it's set as such above or if the
152 # application sent the header.
153 if self.headers.get("Connection") == "close":
154 self.request_handler.close_connection = True
155
156 def close(self):
157 self.get_stdin().read()
158 super().close()
159
160 def finish_response(self):
161 if self.environ["REQUEST_METHOD"] == "HEAD":
162 try:
163 deque(self.result, maxlen=0) # Consume iterator.
164 # Don't call self.finish_content() as, if the headers have not
165 # been sent and Content-Length isn't set, it'll default to "0"
166 # which will prevent omission of the Content-Length header with
167 # HEAD requests as permitted by RFC 9110 Section 9.3.2.
168 # Instead, send the headers, if not sent yet.
169 if not self.headers_sent:
170 self.send_headers()
171 finally:
172 self.close()
173 else:
174 super().finish_response()

215 def get_environ(self):
216 # Strip all headers with underscores in the name before constructing
217 # the WSGI environ. This prevents header-spoofing based on ambiguity
218 # between underscores and dashes both normalized to underscores in WSGI
219 # env vars. Nginx and Apache 2.4+ both do this as well.
220 for k in self.headers:
221 if "_" in k:
222 del self.headers[k]
223
224 return super().get_environ()
