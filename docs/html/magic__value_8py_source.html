66 def _check_constants_comparison(self, node: nodes.Compare) -> None:
67 """
68 Magic values in any side of the comparison should be avoided,
69 Detects comparisons that `comparison-of-constants` core checker cannot detect.
70 """
71 const_operands = []
72 LEFT_OPERAND = 0
73 RIGHT_OPERAND = 1
74
75 left_operand = node.left
76 const_operands.append(isinstance(left_operand, nodes.Const))
77
78 right_operand = node.ops[0][1]
79 const_operands.append(isinstance(right_operand, nodes.Const))
80
81 if all(const_operands):
82 # `comparison-of-constants` avoided
83 return
84
85 operand_value = None
86 if const_operands[LEFT_OPERAND] and self._is_magic_value(left_operand):
87 operand_value = left_operand.as_string()
88 elif const_operands[RIGHT_OPERAND] and self._is_magic_value(right_operand):
89 operand_value = right_operand.as_string()
90 if operand_value is not None:
91 self.add_message(
92 "magic-value-comparison",
93 node=node,
94 args=(operand_value),
95 confidence=HIGH,
96 )
97
98 def _is_magic_value(self, node: nodes.Const) -> bool:
99 return (not utils.is_singleton_const(node)) and (
100 node.value not in (self.valid_magic_vals)
101 )

103 @staticmethod
104 def _parse_rcfile_magic_numbers(parsed_val: str) -> float | str:
105 parsed_val = parsed_val.encode().decode("unicode_escape")
106
107 if parsed_val.startswith("'") and parsed_val.endswith("'"):
108 return parsed_val[1:-1]
109
110 is_number = regex_match(r"[-+]?\d+(\.0*)?$", parsed_val)
111 return float(parsed_val) if is_number else parsed_val
112
113 @utils.only_required_for_messages("magic-comparison")
114 def visit_compare(self, node: nodes.Compare) -> None:
115 self._check_constants_comparison(node)
