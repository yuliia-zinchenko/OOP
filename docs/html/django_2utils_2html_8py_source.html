124def format_html(format_string, *args, **kwargs):
125 """
126 Similar to str.format, but pass all arguments through conditional_escape(),
127 and call mark_safe() on the result. This function should be used instead
128 of str.format or % interpolation to build up small HTML fragments.
129 """
130 if not (args or kwargs):
131 # RemovedInDjango60Warning: when the deprecation ends, replace with:
132 # raise TypeError("args or kwargs must be provided.")
133 warnings.warn(
134 "Calling format_html() without passing args or kwargs is deprecated.",
135 RemovedInDjango60Warning,
136 stacklevel=2,
137 )
138 args_safe = map(conditional_escape, args)
139 kwargs_safe = {k: conditional_escape(v) for (k, v) in kwargs.items()}
140 return mark_safe(format_string.format(*args_safe, **kwargs_safe))

143def format_html_join(sep, format_string, args_generator):
144 """
145 A wrapper of format_html, for the common case of a group of arguments that
146 need to be formatted using the same format string, and then joined using
147 'sep'. 'sep' is also passed through conditional_escape.
148
149 'args_generator' should be an iterator that returns the sequence of 'args'
150 that will be passed to format_html.
151
152 Example:
153
154 format_html_join('\n', "<li>{} {}</li>", ((u.first_name, u.last_name)
155 for u in users))
156 """
157 return mark_safe(
158 conditional_escape(sep).join(
159 format_html(format_string, *args) for args in args_generator
160 )
161 )

164@keep_lazy_text
165def linebreaks(value, autoescape=False):
166 """Convert newlines into <p> and <br>s."""
167 value = normalize_newlines(value)
168 paras = re.split("\n{2,}", str(value))
169 if autoescape:
170 paras = ["<p>%s</p>" % escape(p).replace("\n", "<br>") for p in paras]
171 else:
172 paras = ["<p>%s</p>" % p.replace("\n", "<br>") for p in paras]
173 return "\n\n".join(paras)

205@keep_lazy_text
206def strip_tags(value):
207 """Return the given HTML with all tags stripped."""
208 # Note: in typical case this loop executes _strip_once once. Loop condition
209 # is redundant, but helps to reduce number of executions of _strip_once.
210 value = str(value)
211 while "<" in value and ">" in value:
212 new_value = _strip_once(value)
213 if value.count("<") == new_value.count("<"):
214 # _strip_once wasn't able to detect more tags.
215 break
216 value = new_value
217 return value

226def smart_urlquote(url):
227 """Quote a URL if it isn't already quoted."""
228
229 def unquote_quote(segment):
230 segment = unquote(segment)
231 # Tilde is part of RFC 3986 Section 2.3 Unreserved Characters,
232 # see also https://bugs.python.org/issue16285
233 return quote(segment, safe=RFC3986_SUBDELIMS + RFC3986_GENDELIMS + "~")
234
235 # Handle IDN before quoting.
236 try:
237 scheme, netloc, path, query, fragment = urlsplit(url)
238 except ValueError:
239 # invalid IPv6 URL (normally square brackets in hostname part).
240 return unquote_quote(url)
241
242 try:
243 netloc = punycode(netloc) # IDN -> ACE
244 except UnicodeError: # invalid domain part
245 return unquote_quote(url)
246
247 if query:
248 # Separately unquoting key/value, so as to not mix querystring separators
249 # included in query values. See #22267.
250 query_parts = [
251 (unquote(q[0]), unquote(q[1]))
252 for q in parse_qsl(query, keep_blank_values=True)
253 ]
254 # urlencode will take care of quoting
255 query = urlencode(query_parts)
256
257 path = unquote_quote(path)
258 fragment = unquote_quote(fragment)
259
260 return urlunsplit((scheme, netloc, path, query, fragment))

273class Urlizer:
274 """
275 Convert any URLs in text into clickable links.
276
277 Work on http://, https://, www. links, and also on links ending in one of
278 the original seven gTLDs (.com, .edu, .gov, .int, .mil, .net, and .org).
279 Links can have trailing punctuation (periods, commas, close-parens) and
280 leading punctuation (opening parens) and it'll still do the right thing.
281 """
282
283 trailing_punctuation_chars = ".,:;!"
284 wrapping_punctuation = [("(", ")"), ("[", "]")]

286 simple_url_re = _lazy_re_compile(r"^https?://\[?\w", re.IGNORECASE)
287 simple_url_2_re = _lazy_re_compile(
288 r"^www\.|^(?!http)\w[^@]+\.(com|edu|gov|int|mil|net|org)($|/.*)$", re.IGNORECASE
289 )
290 word_split_re = _lazy_re_compile(r"""([\s<>"']+)""")

292 mailto_template = "mailto:{local}@{domain}"
293 url_template = '<a href="{href}"{attrs}>{url}</a>'

295 def __call__(self, text, trim_url_limit=None, nofollow=False, autoescape=False):
296 """
297 If trim_url_limit is not None, truncate the URLs in the link text
298 longer than this limit to trim_url_limit - 1 characters and append an
299 ellipsis.
300
301 If nofollow is True, give the links a rel="nofollow" attribute.
302
303 If autoescape is True, autoescape the link text and URLs.
304 """
305 safe_input = isinstance(text, SafeData)
306
307 words = self.word_split_re.split(str(text))
308 return "".join(
309 [
310 self.handle_word(
311 word,
312 safe_input=safe_input,
313 trim_url_limit=trim_url_limit,
314 nofollow=nofollow,
315 autoescape=autoescape,
316 )
317 for word in words
318 ]
319 )

321 def handle_word(
322 self,
323 word,
324 *,
325 safe_input,
326 trim_url_limit=None,
327 nofollow=False,
328 autoescape=False,
329 ):
330 if "." in word or "@" in word or ":" in word:
331 # lead: Punctuation trimmed from the beginning of the word.
332 # middle: State of the word.
333 # trail: Punctuation trimmed from the end of the word.
334 lead, middle, trail = self.trim_punctuation(word)
335 # Make URL we want to point to.
336 url = None
337 nofollow_attr = ' rel="nofollow"' if nofollow else ""
338 if len(middle) <= MAX_URL_LENGTH and self.simple_url_re.match(middle):
339 url = smart_urlquote(html.unescape(middle))
340 elif len(middle) <= MAX_URL_LENGTH and self.simple_url_2_re.match(middle):
341 url = smart_urlquote("http://%s" % html.unescape(middle))
342 elif ":" not in middle and self.is_email_simple(middle):
343 local, domain = middle.rsplit("@", 1)
344 try:
345 domain = punycode(domain)
346 except UnicodeError:
347 return word
348 url = self.mailto_template.format(local=local, domain=domain)
349 nofollow_attr = ""
350 # Make link.
351 if url:
352 trimmed = self.trim_url(middle, limit=trim_url_limit)
353 if autoescape and not safe_input:
354 lead, trail = escape(lead), escape(trail)
355 trimmed = escape(trimmed)
356 middle = self.url_template.format(
357 href=escape(url),
358 attrs=nofollow_attr,
359 url=trimmed,
360 )
361 return mark_safe(f"{lead}{middle}{trail}")
362 else:
363 if safe_input:
364 return mark_safe(word)
365 elif autoescape:
366 return escape(word)
367 elif safe_input:
368 return mark_safe(word)
369 elif autoescape:
370 return escape(word)
371 return word

390 def trim_punctuation(self, word):
391 """
392 Trim trailing and wrapping punctuation from `word`. Return the items of
393 the new state.
394 """
395 # Strip all opening wrapping punctuation.
396 middle = word.lstrip(self.wrapping_punctuation_openings)
397 lead = word[: len(word) - len(middle)]
398 trail = ""
399
400 # Continue trimming until middle remains unchanged.
401 trimmed_something = True
402 counts = CountsDict(word=middle)
403 while trimmed_something and middle:
404 trimmed_something = False
405 # Trim wrapping punctuation.
406 for opening, closing in self.wrapping_punctuation:
407 if counts[opening] < counts[closing]:
408 rstripped = middle.rstrip(closing)
409 if rstripped != middle:
410 strip = counts[closing] - counts[opening]
411 trail = middle[-strip:]
412 middle = middle[:-strip]
413 trimmed_something = True
414 counts[closing] -= strip
415
416 amp = middle.rfind("&")
417 if amp == -1:
418 rstripped = middle.rstrip(self.trailing_punctuation_chars)
419 else:
420 rstripped = middle.rstrip(self.trailing_punctuation_chars_no_semicolon)
421 if rstripped != middle:
422 trail = middle[len(rstripped) :] + trail
423 middle = rstripped
424 trimmed_something = True
425
426 if self.trailing_punctuation_chars_has_semicolon and middle.endswith(";"):
427 # Only strip if not part of an HTML entity.
428 potential_entity = middle[amp:]
429 escaped = html.unescape(potential_entity)
430 if escaped == potential_entity or escaped.endswith(";"):
431 rstripped = middle.rstrip(self.trailing_punctuation_chars)
432 trail_start = len(rstripped)
433 amount_trailing_semicolons = len(middle) - len(middle.rstrip(";"))
434 if amp > -1 and amount_trailing_semicolons > 1:
435 # Leave up to most recent semicolon as might be an entity.
436 recent_semicolon = middle[trail_start:].index(";")
437 middle_semicolon_index = recent_semicolon + trail_start + 1
438 trail = middle[middle_semicolon_index:] + trail
439 middle = rstripped + middle[trail_start:middle_semicolon_index]
440 else:
441 trail = middle[trail_start:] + trail
442 middle = rstripped
443 trimmed_something = True
444
445 return lead, middle, trail

447 @staticmethod
448 def is_email_simple(value):
449 """Return True if value looks like an email address."""
450 # An @ must be in the middle of the value.
451 if "@" not in value or value.startswith("@") or value.endswith("@"):
452 return False
453 try:
454 p1, p2 = value.split("@")
455 except ValueError:
456 # value contains more than one @.
457 return False
458 # Max length for domain name labels is 63 characters per RFC 1034.
459 # Helps to avoid ReDoS vectors in the domain part.
460 if len(p2) > 63:
461 return False
462 # Dot must be in p2 (e.g. example.com)
463 if "." not in p2 or p2.startswith("."):
464 return False
465 return True

486def html_safe(klass):
487 """
488 A decorator that defines the __html__ method. This helps non-Django
489 templates to detect classes whose __str__ methods return SafeString.
490 """
491 if "__html__" in klass.__dict__:
492 raise ValueError(
493 "can't apply @html_safe to %s because it defines "
494 "__html__()." % klass.__name__
495 )
496 if "__str__" not in klass.__dict__:
497 raise ValueError(
498 "can't apply @html_safe to %s because it doesn't "
499 "define __str__()." % klass.__name__
500 )
501 klass_str = klass.__str__
502 klass.__str__ = lambda self: mark_safe(klass_str(self))
503 klass.__html__ = lambda self: str(self)
504 return klass
