364 def load_plugin_modules(self, modnames: Iterable[str], force: bool = False) -> None:
365 """Check a list of pylint plugins modules, load and register them.
366
367 If a module cannot be loaded, never try to load it again and instead
368 store the error message for later use in ``load_plugin_configuration``
369 below.
370
371 If `force` is True (useful when multiprocessing), then the plugin is
372 reloaded regardless if an entry exists in self._dynamic_plugins.
373 """
374 for modname in modnames:
375 if modname in self._dynamic_plugins and not force:
376 continue
377 try:
378 module = astroid.modutils.load_module_from_name(modname)
379 module.register(self)
380 self._dynamic_plugins[modname] = module
381 except ModuleNotFoundError as mnf_e:
382 self._dynamic_plugins[modname] = mnf_e

384 def load_plugin_configuration(self) -> None:
385 """Call the configuration hook for plugins.
386
387 This walks through the list of plugins, grabs the "load_configuration"
388 hook, if exposed, and calls it to allow plugins to configure specific
389 settings.
390
391 The result of attempting to load the plugin of the given name
392 is stored in the dynamic plugins dictionary in ``load_plugin_modules`` above.
393
394 ..note::
395 This function previously always tried to load modules again, which
396 led to some confusion and silent failure conditions as described
397 in GitHub issue #7264. Making it use the stored result is more efficient, and
398 means that we avoid the ``init-hook`` problems from before.
399 """
400 for modname, module_or_error in self._dynamic_plugins.items():
401 if isinstance(module_or_error, ModuleNotFoundError):
402 self.add_message(
403 "bad-plugin-value", args=(modname, module_or_error), line=0
404 )
405 elif hasattr(module_or_error, "load_configuration"):
406 module_or_error.load_configuration(self)
407
408 # We re-set all the dictionary values to True here to make sure the dict
409 # is pickle-able. This is only a problem in multiprocessing/parallel mode.
410 # (e.g. invoking pylint -j 2)
411 self._dynamic_plugins = {
412 modname: not isinstance(val, ModuleNotFoundError)
413 for modname, val in self._dynamic_plugins.items()
414 }

416 def _load_reporters(self, reporter_names: str) -> None:
417 """Load the reporters if they are available on _reporters."""
418 if not self._reporters:
419 return
420 sub_reporters = []
421 output_files = []
422 with contextlib.ExitStack() as stack:
423 for reporter_name in reporter_names.split(","):
424 reporter_name, *reporter_output = reporter_name.split(":", 1)
425
426 reporter = self._load_reporter_by_name(reporter_name)
427 sub_reporters.append(reporter)
428 if reporter_output:
429 output_file = stack.enter_context(
430 open(reporter_output[0], "w", encoding="utf-8")
431 )
432 reporter.out = output_file
433 output_files.append(output_file)
434
435 # Extend the lifetime of all opened output files
436 close_output_files = stack.pop_all().close
437
438 if len(sub_reporters) > 1 or output_files:
439 self.set_reporter(
440 reporters.MultiReporter(
441 sub_reporters,
442 close_output_files,
443 )
444 )
445 else:
446 self.set_reporter(sub_reporters[0])

499 def enable_fail_on_messages(self) -> None:
500 """Enable 'fail on' msgs.
501
502 Convert values in config.fail_on (which might be msg category, msg id,
503 or symbol) to specific msgs, then enable and flag them for later.
504 """
505 fail_on_vals = self.config.fail_on
506 if not fail_on_vals:
507 return
508
509 fail_on_cats = set()
510 fail_on_msgs = set()
511 for val in fail_on_vals:
512 # If value is a category, add category, else add message
513 if val in MSG_TYPES:
514 fail_on_cats.add(val)
515 else:
516 fail_on_msgs.add(val)
517
518 # For every message in every checker, if cat or msg flagged, enable check
519 for all_checkers in self._checkers.values():
520 for checker in all_checkers:
521 for msg in checker.messages:
522 if msg.msgid in fail_on_msgs or msg.symbol in fail_on_msgs:
523 # message id/symbol matched, enable and flag it
524 self.enable(msg.msgid)
525 self.fail_on_symbols.append(msg.symbol)
526 elif msg.msgid[0] in fail_on_cats:
527 # message starts with a category value, flag (but do not enable) it
528 self.fail_on_symbols.append(msg.symbol)

605 def initialize(self) -> None:
606 """Initialize linter for linting.
607
608 This method is called before any linting is done.
609 """
610 self._ignore_paths = self.config.ignore_paths
611 # initialize msgs_state now that all messages have been registered into
612 # the store
613 for msg in self.msgs_store.messages:
614 if not msg.may_be_emitted(self.config.py_version):
615 self._msgs_state[msg.msgid] = False

617 def _discover_files(self, files_or_modules: Sequence[str]) -> Iterator[str]:
618 """Discover python modules and packages in sub-directory.
619
620 Returns iterator of paths to discovered modules and packages.
621 """
622 for something in files_or_modules:
623 if os.path.isdir(something) and not os.path.isfile(
624 os.path.join(something, "__init__.py")
625 ):
626 skip_subtrees: list[str] = []
627 for root, _, files in os.walk(something):
628 if any(root.startswith(s) for s in skip_subtrees):
629 # Skip subtree of already discovered package.
630 continue
631
632 if _is_ignored_file(
633 root,
634 self.config.ignore,
635 self.config.ignore_patterns,
636 self.config.ignore_paths,
637 ):
638 skip_subtrees.append(root)
639 continue
640
641 if "__init__.py" in files:
642 skip_subtrees.append(root)
643 yield root
644 else:
645 yield from (
646 os.path.join(root, file)
647 for file in files
648 if file.endswith((".py", ".pyi"))
649 )
650 else:
651 yield something

653 def check(self, files_or_modules: Sequence[str]) -> None:
654 """Main checking entry: check a list of files or modules from their name.
655
656 files_or_modules is either a string or list of strings presenting modules to check.
657 """
658 self.initialize()
659 if self.config.recursive:
660 files_or_modules = tuple(self._discover_files(files_or_modules))
661 if self.config.from_stdin:
662 if len(files_or_modules) != 1:
663 raise exceptions.InvalidArgsError(
664 "Missing filename required for --from-stdin"
665 )
666
667 extra_packages_paths = list(
668 dict.fromkeys(
669 [
670 discover_package_path(file_or_module, self.config.source_roots)
671 for file_or_module in files_or_modules
672 ]
673 ).keys()
674 )
675
676 # TODO: Move the parallel invocation into step 3 of the checking process
677 if not self.config.from_stdin and self.config.jobs > 1:
678 original_sys_path = sys.path[:]
679 check_parallel(
680 self,
681 self.config.jobs,
682 self._iterate_file_descrs(files_or_modules),
683 extra_packages_paths,
684 )
685 sys.path = original_sys_path
686 return
687
688 # 1) Get all FileItems
689 with augmented_sys_path(extra_packages_paths):
690 if self.config.from_stdin:
691 fileitems = self._get_file_descr_from_stdin(files_or_modules[0])
692 data: str | None = _read_stdin()
693 else:
694 fileitems = self._iterate_file_descrs(files_or_modules)
695 data = None
696
697 # The contextmanager also opens all checkers and sets up the PyLinter class
698 with augmented_sys_path(extra_packages_paths):
699 with self._astroid_module_checker() as check_astroid_module:
700 # 2) Get the AST for each FileItem
701 ast_per_fileitem = self._get_asts(fileitems, data)
702
703 # 3) Lint each ast
704 self._lint_files(ast_per_fileitem, check_astroid_module)
705
706 def _get_asts(
707 self, fileitems: Iterator[FileItem], data: str | None
708 ) -> dict[FileItem, nodes.Module | None]:
709 """Get the AST for all given FileItems."""
710 ast_per_fileitem: dict[FileItem, nodes.Module | None] = {}
711
712 for fileitem in fileitems:
713 self.set_current_module(fileitem.name, fileitem.filepath)
714
715 try:
716 ast_per_fileitem[fileitem] = self.get_ast(
717 fileitem.filepath, fileitem.name, data

1029 def _check_astroid_module(
1030 self,
1031 node: nodes.Module,
1032 walker: ASTWalker,
1033 rawcheckers: list[checkers.BaseRawFileChecker],
1034 tokencheckers: list[checkers.BaseTokenChecker],
1035 ) -> bool | None:
1036 """Check given AST node with given walker and checkers.
1037
1038 :param astroid.nodes.Module node: AST node of the module to check
1039 :param pylint.utils.ast_walker.ASTWalker walker: AST walker
1040 :param list rawcheckers: List of token checkers to use
1041 :param list tokencheckers: List of raw checkers to use
1042

1044 None if the module contents could not be parsed
1045 """
1046 try:
1047 tokens = utils.tokenize_module(node)
1048 except tokenize.TokenError as ex:
1049 self.add_message(
1050 "syntax-error",
1051 line=ex.args[1][0],
1052 col_offset=ex.args[1][1],
1053 args=ex.args[0],
1054 confidence=HIGH,
1055 )

1057
1058 if not node.pure_python:
1059 self.add_message("raw-checker-failed", args=node.name)

1062 # Parse module/block level option pragma's
1063 self.process_tokens(tokens)
1064 if self._ignore_file:

1067 for raw_checker in rawcheckers:
1068 raw_checker.process_module(node)
1069 for token_checker in tokencheckers:
1070 token_checker.process_tokens(tokens)
