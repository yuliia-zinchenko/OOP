103 @staticmethod
104 def _get_method_arguments(method: nodes.FunctionDef) -> list[str]:
105 if method.args.args is None:
106 return []
107
108 first_arg = 0 if method.type in {"function", "staticmethod"} else 1
109 arguments: list[nodes.AssignName] = method.args.args[first_arg:]
110
111 annotations = dict(zip(arguments, method.args.annotations[first_arg:]))
112 for arg in arguments:
113 annotation_label = ""
114 ann = annotations.get(arg)
115 if ann:
116 annotation_label = get_annotation_label(ann)
117 annotations[arg] = annotation_label
118
119 return [
120 f"{arg.name}: {ann}" if ann else f"{arg.name}"
121 for arg, ann in annotations.items()
122 ]
