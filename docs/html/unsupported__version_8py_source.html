72 def open(self) -> None:
73 """Initialize visit variables and statistics."""
74 py_version = self.linter.config.py_version
75 self._py36_plus = py_version >= (3, 6)
76 self._py38_plus = py_version >= (3, 8)
77 self._py311_plus = py_version >= (3, 11)
78 self._py312_plus = py_version >= (3, 12)

80 @only_required_for_messages("using-f-string-in-unsupported-version")
81 def visit_joinedstr(self, node: nodes.JoinedStr) -> None:
82 """Check f-strings."""
83 if not self._py36_plus:
84 self.add_message(
85 "using-f-string-in-unsupported-version", node=node, confidence=HIGH
86 )
87
88 @only_required_for_messages("using-assignment-expression-in-unsupported-version")
89 def visit_namedexpr(self, node: nodes.JoinedStr) -> None:
90 if not self._py38_plus:
91 self.add_message(
92 "using-assignment-expression-in-unsupported-version",
93 node=node,
94 confidence=HIGH,
95 )
96
97 @only_required_for_messages("using-positional-only-args-in-unsupported-version")
98 def visit_arguments(self, node: nodes.Arguments) -> None:
99 if not self._py38_plus and node.posonlyargs:
100 self.add_message(
101 "using-positional-only-args-in-unsupported-version",
102 node=node,
103 confidence=HIGH,

106 @only_required_for_messages("using-final-decorator-in-unsupported-version")
107 def visit_decorators(self, node: nodes.Decorators) -> None:
108 """Check decorators."""
109 self._check_typing_final(node)
110
111 def _check_typing_final(self, node: nodes.Decorators) -> None:
112 """Add a message when the `typing.final` decorator is used and the
113 py-version is lower than 3.8.
114 """
115 if self._py38_plus:
116 return
117
118 decorators = []
119 for decorator in node.get_children():
120 inferred = safe_infer(decorator)
121 if inferred and inferred.qname() == "typing.final":
122 decorators.append(decorator)
123
124 for decorator in decorators or uninferable_final_decorators(node):
125 self.add_message(
126 "using-final-decorator-in-unsupported-version",
127 node=decorator,
128 confidence=HIGH,
129 )
130

135 "using-exception-groups-in-unsupported-version",
136 node=node,
137 confidence=HIGH,
138 )
139
140 @only_required_for_messages("using-exception-groups-in-unsupported-version")
141 def visit_excepthandler(self, node: nodes.ExceptHandler) -> None:
142 if (
143 not self._py311_plus
144 and isinstance(node.type, nodes.Name)
145 and node.type.name == "ExceptionGroup"
146 ):
147 self.add_message(
148 "using-exception-groups-in-unsupported-version",
149 node=node,
150 confidence=HIGH,
151 )

153 @only_required_for_messages("using-exception-groups-in-unsupported-version")
154 def visit_raise(self, node: nodes.Raise) -> None:
155 if (
156 not self._py311_plus

158 and isinstance(node.exc.func, nodes.Name)

160 ):
161 self.add_message(
162 "using-exception-groups-in-unsupported-version",
163 node=node,
164 confidence=HIGH,
165 )
