67 def _open_graph(self) -> None:

78 def emit_node(
79 self,

86 Nodes can be classes, packages, participants etc.

92 style = "filled" if color != self.DEFAULT_COLOR else "solid"

102 def _build_label_for_node(self, properties: NodeProperties) -> str:
103 if not properties.label:
104 return ""
105
106 label: str = properties.label
107 if properties.attrs is None and properties.methods is None:
108 # return a "compact" form which only displays the class name in a box
109 return label
110
111 # Add class attributes
112 attrs: list[str] = properties.attrs or []
113 attrs_string = rf"{HTMLLabels.LINEBREAK_LEFT.value}".join(
114 attr.replace("|", r"\|") for attr in attrs
115 )
116 label = rf"{{{label}|{attrs_string}{HTMLLabels.LINEBREAK_LEFT.value}|"
117
118 # Add class methods
119 methods: list[nodes.FunctionDef] = properties.methods or []
120 for func in methods:
121 args = ", ".join(self._get_method_arguments(func)).replace("|", r"\|")
122 method_name = (
123 f"<I>{func.name}</I>" if func.is_abstract() else f"{func.name}"
124 )
125 label += rf"{method_name}({args})"
126 if func.returns:
127 annotation_label = get_annotation_label(func.returns)
128 label += ": " + self._escape_annotation_label(annotation_label)
129 label += rf"{HTMLLabels.LINEBREAK_LEFT.value}"
130 label += "}"
131 return label
132
133 def _escape_annotation_label(self, annotation_label: str) -> str:
134 # Escape vertical bar characters to make them appear as a literal characters
135 # otherwise it gets treated as field separator of record-based nodes
136 annotation_label = annotation_label.replace("|", r"\|")
137
138 return annotation_label

140 def emit_edge(
141 self,
142 from_node: str,
143 to_node: str,
144 type_: EdgeType,
145 label: str | None = None,
146 ) -> None:
147 """Create an edge from one node to another to display relationships."""
148 arrowstyle = ARROWS[type_]
149 attrs = [f'{prop}="{value}"' for prop, value in arrowstyle.items()]
150 if label:
151 attrs.append(f'label="{label}"')
152 self.emit(f'"{from_node}" -> "{to_node}" [{", ".join(sorted(attrs))}];')

154 def generate(self, outputfile: str) -> None:
155 self._close_graph()
156 graphviz_extensions = ("dot", "gv")
157 name = self.title
158 if outputfile is None:
159 target = "png"
160 pdot, dot_sourcepath = tempfile.mkstemp(".gv", name)
161 ppng, outputfile = tempfile.mkstemp(".png", name)
162 os.close(pdot)
163 os.close(ppng)
164 else:
165 target = Path(outputfile).suffix.lstrip(".")
166 if not target:
167 target = "png"
168 outputfile = outputfile + "." + target
169 if target not in graphviz_extensions:
170 pdot, dot_sourcepath = tempfile.mkstemp(".gv", name)
171 os.close(pdot)
172 else:
173 dot_sourcepath = outputfile
174 with open(dot_sourcepath, "w", encoding="utf8") as outfile:
175 outfile.writelines(self.lines)
176 if target not in graphviz_extensions:
177 subprocess.run(
178 ["dot", "-T", target, dot_sourcepath, "-o", outputfile], check=True
179 )
180 os.unlink(dot_sourcepath)
