30 """
31 Stores jobs in a database table using SQLAlchemy.
32 The table will be created if it doesn't exist in the database.
33
34 Plugin alias: ``sqlalchemy``
35
36 :param str url: connection string (see
37 :ref:`SQLAlchemy documentation <sqlalchemy:database_urls>` on this)
38 :param engine: an SQLAlchemy :class:`~sqlalchemy.engine.Engine` to use instead of creating a
39 new one based on ``url``
40 :param str tablename: name of the table to store jobs in
41 :param metadata: a :class:`~sqlalchemy.schema.MetaData` instance to use instead of creating a
42 new one
43 :param int pickle_protocol: pickle protocol level to use (for serialization), defaults to the
44 highest available
45 :param str tableschema: name of the (existing) schema in the target database where the table
46 should be
47 :param dict engine_options: keyword arguments to :func:`~sqlalchemy.create_engine`
48 (ignored if ``engine`` is given)
49 """

158 def _reconstitute_job(self, job_state):
159 job_state = pickle.loads(job_state)
160 job_state["jobstore"] = self
161 job = Job.__new__(Job)
162 job.__setstate__(job_state)
163 job._scheduler = self._scheduler

167 def _get_jobs(self, *conditions):
168 jobs = []
169 selectable = select(self.jobs_t.c.id, self.jobs_t.c.job_state).order_by(
170 self.jobs_t.c.next_run_time
171 )
172 selectable = selectable.where(and_(*conditions)) if conditions else selectable
173 failed_job_ids = set()
174 with self.engine.begin() as connection:
175 for row in connection.execute(selectable):
176 try:
177 jobs.append(self._reconstitute_job(row.job_state))
178 except BaseException:
179 self._logger.exception(
180 'Unable to restore job "%s" -- removing it', row.id
181 )
182 failed_job_ids.add(row.id)
183
184 # Remove all the jobs we failed to restore
185 if failed_job_ids:
186 delete = self.jobs_t.delete().where(
187 self.jobs_t.c.id.in_(failed_job_ids)
188 )
189 connection.execute(delete)
190
191 return jobs
