244class ArrayRHSMixin:
245 def __init__(self, lhs, rhs):
246 # Don't wrap arrays that contains only None values, psycopg doesn't
247 # allow this.
248 if isinstance(rhs, (tuple, list)) and any(self._rhs_not_none_values(rhs)):
249 expressions = []
250 for value in rhs:
251 if not hasattr(value, "resolve_expression"):
252 field = lhs.output_field
253 value = Value(field.base_field.get_prep_value(value))
254 expressions.append(value)
255 rhs = Func(
256 *expressions,
257 function="ARRAY",
258 template="%(function)s[%(expressions)s]",
259 )
260 super().__init__(lhs, rhs)

262 def process_rhs(self, compiler, connection):
263 rhs, rhs_params = super().process_rhs(compiler, connection)
264 cast_type = self.lhs.output_field.cast_db_type(connection)
265 return "%s::%s" % (rhs, cast_type), rhs_params

267 def _rhs_not_none_values(self, rhs):
268 for x in rhs:
269 if isinstance(x, (list, tuple)):
270 yield from self._rhs_not_none_values(x)
271 elif x is not None:
272 yield True
