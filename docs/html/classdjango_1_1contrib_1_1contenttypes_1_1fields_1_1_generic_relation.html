405 def _get_path_info_with_parent(self, filtered_relation):
406 """
407 Return the path that joins the current model through any parent models.
408 The idea is that if you have a GFK defined on a parent model then we
409 need to join the parent model first, then the child model.
410 """
411 # With an inheritance chain ChildTag -> Tag and Tag defines the
412 # GenericForeignKey, and a TaggedItem model has a GenericRelation to
413 # ChildTag, then we need to generate a join from TaggedItem to Tag
414 # (as Tag.object_id == TaggedItem.pk), and another join from Tag to
415 # ChildTag (as that is where the relation is to). Do this by first
416 # generating a join to the parent model, then generating joins to the
417 # child models.
418 path = []
419 opts = self.remote_field.model._meta.concrete_model._meta
420 parent_opts = opts.get_field(self.object_id_field_name).model._meta
421 target = parent_opts.pk
422 path.append(
423 PathInfo(
424 from_opts=self.model._meta,
425 to_opts=parent_opts,
426 target_fields=(target,),
427 join_field=self.remote_field,
428 m2m=True,
429 direct=False,
430 filtered_relation=filtered_relation,
431 )
432 )
433 # Collect joins needed for the parent -> child chain. This is easiest
434 # to do if we collect joins for the child -> parent chain and then
435 # reverse the direction (call to reverse() and use of
436 # field.remote_field.get_path_info()).
437 parent_field_chain = []
438 while parent_opts != opts:
439 field = opts.get_ancestor_link(parent_opts.model)
440 parent_field_chain.append(field)
