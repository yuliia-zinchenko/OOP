29 def quote_value(self, value):
30 if isinstance(value, (datetime.date, datetime.time, datetime.datetime)):
31 return "'%s'" % value
32 elif isinstance(value, datetime.timedelta):
33 return "'%s'" % duration_iso_string(value)
34 elif isinstance(value, str):
35 return "'%s'" % value.replace("'", "''")
36 elif isinstance(value, (bytes, bytearray, memoryview)):
37 return "'%s'" % value.hex()
38 elif isinstance(value, bool):
39 return "1" if value else "0"
40 else:
41 return str(value)

73 def alter_field(self, model, old_field, new_field, strict=False):
74 try:
75 super().alter_field(model, old_field, new_field, strict)
76 except DatabaseError as e:
77 description = str(e)
78 # If we're changing type to an unsupported type we need a
79 # SQLite-ish workaround
80 if "ORA-22858" in description or "ORA-22859" in description:
81 self._alter_field_type_workaround(model, old_field, new_field)
82 # If an identity column is changing to a non-numeric type, drop the
83 # identity first.
84 elif "ORA-30675" in description:
85 self._drop_identity(model._meta.db_table, old_field.column)
86 self.alter_field(model, old_field, new_field, strict)
87 # If a primary key column is changing to an identity column, drop
88 # the primary key first.
89 elif "ORA-30673" in description and old_field.primary_key:
90 self._delete_primary_key(model, strict=True)
91 self._alter_field_type_workaround(model, old_field, new_field)
92 # If a collation is changing on a primary key, drop the primary key
93 # first.
94 elif "ORA-43923" in description and old_field.primary_key:
95 self._delete_primary_key(model, strict=True)
96 self.alter_field(model, old_field, new_field, strict)
97 # Restore a primary key, if needed.
98 if new_field.primary_key:
99 self.execute(self._create_primary_key_sql(model, new_field))
100 else:
101 raise

103 def _alter_field_type_workaround(self, model, old_field, new_field):
104 """
105 Oracle refuses to change from some type to other type.
106 What we need to do instead is:
107 - Add a nullable version of the desired field with a temporary name. If
108 the new column is an auto field, then the temporary column can't be
109 nullable.
110 - Update the table to transfer values from old to new
111 - Drop old column
112 - Rename the new column and possibly drop the nullable property
113 """
114 # Make a new field that's like the new one but with a temporary
115 # column name.
116 new_temp_field = copy.deepcopy(new_field)
117 new_temp_field.null = new_field.get_internal_type() not in (
118 "AutoField",
119 "BigAutoField",
120 "SmallAutoField",
121 )
122 new_temp_field.column = self._generate_temp_name(new_field.column)
123 # Add it
124 self.add_field(model, new_temp_field)
125 # Explicit data type conversion
126 # https://docs.oracle.com/en/database/oracle/oracle-database/21/sqlrf
127 # /Data-Type-Comparison-Rules.html#GUID-D0C5A47E-6F93-4C2D-9E49-4F2B86B359DD
128 new_value = self.quote_name(old_field.column)
129 old_type = old_field.db_type(self.connection)
130 if re.match("^N?CLOB", old_type):
131 new_value = "TO_CHAR(%s)" % new_value
132 old_type = "VARCHAR2"
133 if re.match("^N?VARCHAR2", old_type):
134 new_internal_type = new_field.get_internal_type()
135 if new_internal_type == "DateField":
136 new_value = "TO_DATE(%s, 'YYYY-MM-DD')" % new_value
137 elif new_internal_type == "DateTimeField":
138 new_value = "TO_TIMESTAMP(%s, 'YYYY-MM-DD HH24:MI:SS.FF')" % new_value
139 elif new_internal_type == "TimeField":
140 # TimeField are stored as TIMESTAMP with a 1900-01-01 date part.
141 new_value = "CONCAT('1900-01-01 ', %s)" % new_value
142 new_value = "TO_TIMESTAMP(%s, 'YYYY-MM-DD HH24:MI:SS.FF')" % new_value
143 # Transfer values across
144 self.execute(
145 "UPDATE %s set %s=%s"
146 % (
147 self.quote_name(model._meta.db_table),
148 self.quote_name(new_temp_field.column),
149 new_value,
150 )
151 )
152 # Drop the old field
153 self.remove_field(model, old_field)
154 # Rename and possibly make the new field NOT NULL
155 super().alter_field(model, new_temp_field, new_field)
156 # Recreate foreign key (if necessary) because the old field is not
157 # passed to the alter_field() and data types of new_temp_field and
158 # new_field always match.
159 new_type = new_field.db_type(self.connection)
160 if (
161 (old_field.primary_key and new_field.primary_key)
162 or (old_field.unique and new_field.unique)
163 ) and old_type != new_type:
164 for _, rel in _related_non_m2m_objects(new_temp_field, new_field):
165 if rel.field.db_constraint:
166 self.execute(
167 self._create_fk_sql(rel.related_model, rel.field, "_fk")
168 )
