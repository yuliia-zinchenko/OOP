43 def quote_value(self, value):
44 # The backend "mostly works" without this function and there are use
45 # cases for compiling Python without the sqlite3 libraries (e.g.
46 # security hardening).
47 try:
48 import sqlite3
49
50 value = sqlite3.adapt(value)
51 except ImportError:
52 pass
53 except sqlite3.ProgrammingError:
54 pass
55 # Manual emulation of SQLite parameter quoting
56 if isinstance(value, bool):
57 return str(int(value))
58 elif isinstance(value, (Decimal, float, int)):
59 return str(value)
60 elif isinstance(value, str):
61 return "'%s'" % value.replace("'", "''")
62 elif value is None:
63 return "NULL"
64 elif isinstance(value, (bytes, bytearray, memoryview)):
65 # Bytes are only allowed for BLOB fields, encoded as string
66 # literals containing hexadecimal data and preceded by a single "X"
67 # character.
68 return "X'%s'" % value.hex()
69 else:
70 raise ValueError(
71 "Cannot quote parameter value %r of type %s" % (value, type(value))
72 )

77 def _remake_table(
78 self, model, create_field=None, delete_field=None, alter_fields=None
79 ):
80 """
81 Shortcut to transform a model from old_model into new_model
82
83 This follows the correct procedure to perform non-rename or column
84 addition operations based on SQLite's documentation
85
86 https://www.sqlite.org/lang_altertable.html#caution
87
88 The essential steps are:
89 1. Create a table with the updated definition called "new__app_model"
90 2. Copy the data from the existing "app_model" table to the new table
91 3. Drop the "app_model" table
92 4. Rename the "new__app_model" table to "app_model"
93 5. Restore any index of the previous "app_model" table.
94 """
95
96 # Self-referential fields must be recreated rather than copied from
97 # the old model to ensure their remote_field.field_name doesn't refer
98 # to an altered field.
99 def is_self_referential(f):
100 return f.is_relation and f.remote_field.model is model
101
102 # Work out the new fields dict / mapping
103 body = {
104 f.name: f.clone() if is_self_referential(f) else f
105 for f in model._meta.local_concrete_fields
106 }
107 # Since mapping might mix column names and default values,
108 # its values must be already quoted.
109 mapping = {
110 f.column: self.quote_name(f.column)
111 for f in model._meta.local_concrete_fields
112 if f.generated is False
113 }
114 # This maps field names (not columns) for things like unique_together
115 rename_mapping = {}
116 # If any of the new or altered fields is introducing a new PK,
117 # remove the old one
118 restore_pk_field = None
119 alter_fields = alter_fields or []
120 if getattr(create_field, "primary_key", False) or any(
121 getattr(new_field, "primary_key", False) for _, new_field in alter_fields
122 ):
123 for name, field in list(body.items()):
124 if field.primary_key and not any(
125 # Do not remove the old primary key when an altered field
126 # that introduces a primary key is the same field.
127 name == new_field.name
128 for _, new_field in alter_fields
129 ):
130 field.primary_key = False
131 restore_pk_field = field
132 if field.auto_created:
133 del body[name]
134 del mapping[field.column]
135 # Add in any created fields
136 if create_field:
137 body[create_field.name] = create_field
138 # Choose a default and insert it into the copy map
139 if (
140 create_field.db_default is NOT_PROVIDED
141 and not (create_field.many_to_many or create_field.generated)
142 and create_field.concrete
143 ):
144 mapping[create_field.column] = self.prepare_default(
145 self.effective_default(create_field)
146 )
147 # Add in any altered fields
148 for alter_field in alter_fields:
149 old_field, new_field = alter_field
150 body.pop(old_field.name, None)
151 mapping.pop(old_field.column, None)
152 body[new_field.name] = new_field
153 rename_mapping[old_field.name] = new_field.name
154 if new_field.generated:
155 continue
156 if old_field.null and not new_field.null:
157 if new_field.db_default is NOT_PROVIDED:
158 default = self.prepare_default(self.effective_default(new_field))
159 else:
160 default, _ = self.db_default_sql(new_field)
161 case_sql = "coalesce(%(col)s, %(default)s)" % {
162 "col": self.quote_name(old_field.column),
163 "default": default,
164 }
165 mapping[new_field.column] = case_sql
166 else:
167 mapping[new_field.column] = self.quote_name(old_field.column)
168 # Remove any deleted fields
169 if delete_field:
170 del body[delete_field.name]
171 mapping.pop(delete_field.column, None)
172 # Remove any implicit M2M tables
173 if (
174 delete_field.many_to_many
175 and delete_field.remote_field.through._meta.auto_created
176 ):
177 return self.delete_model(delete_field.remote_field.through)
178 # Work inside a new app registry
179 apps = Apps()
180
181 # Work out the new value of unique_together, taking renames into
182 # account
183 unique_together = [
184 [rename_mapping.get(n, n) for n in unique]
185 for unique in model._meta.unique_together
186 ]
187
188 indexes = model._meta.indexes
189 if delete_field:
190 indexes = [
191 index for index in indexes if delete_field.name not in index.fields
192 ]
193
194 constraints = list(model._meta.constraints)
195
196 # Provide isolated instances of the fields to the new model body so
197 # that the existing model's internals aren't interfered with when
198 # the dummy model is constructed.
199 body_copy = copy.deepcopy(body)
200
201 # Construct a new model with the new fields to allow self referential
202 # primary key to resolve to. This model won't ever be materialized as a
203 # table and solely exists for foreign key reference resolution purposes.
204 # This wouldn't be required if the schema editor was operating on model
205 # states instead of rendered models.
206 meta_contents = {
207 "app_label": model._meta.app_label,
208 "db_table": model._meta.db_table,
209 "unique_together": unique_together,
210 "indexes": indexes,
211 "constraints": constraints,
212 "apps": apps,
213 }
214 meta = type("Meta", (), meta_contents)
215 body_copy["Meta"] = meta
216 body_copy["__module__"] = model.__module__
217 type(model._meta.object_name, model.__bases__, body_copy)
218
219 # Construct a model with a renamed table name.
220 body_copy = copy.deepcopy(body)
221 meta_contents = {
222 "app_label": model._meta.app_label,
223 "db_table": "new__%s" % strip_quotes(model._meta.db_table),
224 "unique_together": unique_together,
225 "indexes": indexes,
226 "constraints": constraints,
227 "apps": apps,
228 }
229 meta = type("Meta", (), meta_contents)
230 body_copy["Meta"] = meta
231 body_copy["__module__"] = model.__module__
232 new_model = type("New%s" % model._meta.object_name, model.__bases__, body_copy)
233
234 # Remove the automatically recreated default primary key, if it has
235 # been deleted.
236 if delete_field and delete_field.attname == new_model._meta.pk.attname:
237 auto_pk = new_model._meta.pk
238 delattr(new_model, auto_pk.attname)
239 new_model._meta.local_fields.remove(auto_pk)
240 new_model.pk = None
241
242 # Create a new table with the updated schema.
243 self.create_model(new_model)
244
245 # Copy data from the old table into the new table
246 self.execute(
247 "INSERT INTO %s (%s) SELECT %s FROM %s"
248 % (
249 self.quote_name(new_model._meta.db_table),
250 ", ".join(self.quote_name(x) for x in mapping),
251 ", ".join(mapping.values()),
252 self.quote_name(model._meta.db_table),
253 )
254 )
255
256 # Delete the old table to make way for the new
257 self.delete_model(model, handle_autom2m=False)
258
259 # Rename the new table to take way for the old
260 self.alter_db_table(
261 new_model,
262 new_model._meta.db_table,
263 model._meta.db_table,
264 )
265
266 # Run deferred SQL on correct table
267 for sql in self.deferred_sql:
268 self.execute(sql)
269 self.deferred_sql = []
270 # Fix any PK-removed field
271 if restore_pk_field:
272 restore_pk_field.primary_key = True

274 def delete_model(self, model, handle_autom2m=True):
275 if handle_autom2m:
276 super().delete_model(model)
277 else:
278 # Delete the table (and only that)
279 self.execute(

282 "table": self.quote_name(model._meta.db_table),
283 }
284 )

287 if isinstance(sql, Statement) and sql.references_table(

289 ):
290 self.deferred_sql.remove(sql)

292 def add_field(self, model, field):
293 """Create a field on a model."""
294 from django.db.models.expressions import Value
295
296 # Special-case implicit M2M tables.
297 if field.many_to_many and field.remote_field.through._meta.auto_created:
298 self.create_model(field.remote_field.through)
299 elif (
300 # Primary keys and unique fields are not supported in ALTER TABLE
301 # ADD COLUMN.
302 field.primary_key
303 or field.unique
304 or not field.null
305 # Fields with default values cannot by handled by ALTER TABLE ADD
306 # COLUMN statement because DROP DEFAULT is not supported in
307 # ALTER TABLE.
308 or self.effective_default(field) is not None
309 # Fields with non-constant defaults cannot by handled by ALTER
310 # TABLE ADD COLUMN statement.
311 or (
312 field.db_default is not NOT_PROVIDED
313 and not isinstance(field.db_default, Value)
314 )
315 ):
316 self._remake_table(model, create_field=field)
317 else:
318 super().add_field(model, field)

320 def remove_field(self, model, field):
321 """
322 Remove a field from a model. Usually involves deleting a column,
323 but for M2Ms may involve deleting a table.
324 """
325 # M2M fields are a special case
326 if field.many_to_many:
327 # For implicit M2M tables, delete the auto-created table
328 if field.remote_field.through._meta.auto_created:
329 self.delete_model(field.remote_field.through)
330 # For explicit "through" M2M fields, do nothing
331 elif (
332 self.connection.features.can_alter_table_drop_column
333 # Primary keys, unique fields, indexed fields, and foreign keys are
334 # not supported in ALTER TABLE DROP COLUMN.
335 and not field.primary_key
336 and not field.unique
337 and not field.db_index
338 and not (field.remote_field and field.db_constraint)
339 ):
340 super().remove_field(model, field)
341 # For everything else, remake.
342 else:
343 # It might not actually have a column behind it
344 if field.db_parameters(connection=self.connection)["type"] is None:
345 return
346 self._remake_table(model, delete_field=field)

361 # changed and there aren't any constraints.

363 old_field.column != new_field.column

377 # Alter by remaking table
378 self._remake_table(model, alter_fields=[(old_field, new_field)])
379 # Rebuild tables with FKs pointing to this field.
380 old_collation = old_db_params.get("collation")
381 new_collation = new_db_params.get("collation")
382 if new_field.unique and (
383 old_type != new_type or old_collation != new_collation
