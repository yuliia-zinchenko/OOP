220 def process_lhs(self, compiler, connection, lhs=None):
221 lhs_sql, params = super().process_lhs(compiler, connection, lhs)
222 field_internal_type = self.lhs.output_field.get_internal_type()
223 if (
224 hasattr(connection.ops.__class__, "field_cast_sql")
225 and connection.ops.__class__.field_cast_sql
226 is not BaseDatabaseOperations.field_cast_sql
227 ):

229 (
230 "The usage of DatabaseOperations.field_cast_sql() is deprecated. "
231 "Implement DatabaseOperations.lookup_cast() instead."

565 def get_rhs_op(self, connection, rhs):
566 # Assume we are in startswith. We need to produce SQL like:
567 # col LIKE %s, ['thevalue%']
568 # For python values we can (and should) do that directly in Python,
569 # but if the value is for example reference to other column, then
570 # we need to add the % pattern match to the lookup by something like
571 # col LIKE othercol || '%%'

577 connection.pattern_esc
