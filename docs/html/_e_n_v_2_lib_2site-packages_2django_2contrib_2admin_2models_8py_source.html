165 def get_change_message(self):
166 """
167 If self.change_message is a JSON structure, interpret it as a change
168 string, properly translated.
169 """
170 if self.change_message and self.change_message[0] == "[":
171 try:
172 change_message = json.loads(self.change_message)
173 except json.JSONDecodeError:
174 return self.change_message
175 messages = []
176 for sub_message in change_message:
177 if "added" in sub_message:
178 if sub_message["added"]:
179 sub_message["added"]["name"] = gettext(
180 sub_message["added"]["name"]
181 )
182 messages.append(
183 gettext("Added {name} “{object}”.").format(
184 **sub_message["added"]
185 )
186 )
187 else:
188 messages.append(gettext("Added."))
189
190 elif "changed" in sub_message:
191 sub_message["changed"]["fields"] = get_text_list(
192 [
193 gettext(field_name)
194 for field_name in sub_message["changed"]["fields"]
195 ],
196 gettext("and"),
197 )
198 if "name" in sub_message["changed"]:
199 sub_message["changed"]["name"] = gettext(
200 sub_message["changed"]["name"]
201 )
202 messages.append(
203 gettext("Changed {fields} for {name} “{object}”.").format(
204 **sub_message["changed"]
205 )
206 )
207 else:
208 messages.append(
209 gettext("Changed {fields}.").format(
210 **sub_message["changed"]
211 )
212 )
213
214 elif "deleted" in sub_message:
215 sub_message["deleted"]["name"] = gettext(
216 sub_message["deleted"]["name"]
217 )
218 messages.append(
219 gettext("Deleted {name} “{object}”.").format(
220 **sub_message["deleted"]
221 )
222 )
223
224 change_message = " ".join(msg[0].upper() + msg[1:] for msg in messages)
