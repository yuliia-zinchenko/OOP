111def get_system_username():
112 """
113 Return the current system user's username, or an empty string if the
114 username could not be determined.
115 """
116 try:
117 result = getpass.getuser()
118 except (ImportError, KeyError):
119 # KeyError will be raised by os.getpwuid() (called by getuser())
120 # if there is no corresponding entry in the /etc/passwd file
121 # (a very restricted chroot environment, for example).
122 return ""
123 return result

126def get_default_username(check_db=True, database=DEFAULT_DB_ALIAS):
127 """
128 Try to determine the current system user's username to use as a default.
129
130 :param check_db: If ``True``, requires that the username does not match an
131 existing ``auth.User`` (otherwise returns an empty string).
132 :param database: The database where the unique check will be performed.
133 :returns: The username, or an empty string if no username can be
134 determined or the suggested username is already taken.
135 """
136 # This file is used in apps.py, it should not trigger models import.
137 from django.contrib.auth import models as auth_app
138
139 # If the User model has been swapped out, we can't make any assumptions
140 # about the default user name.
141 if auth_app.User._meta.swapped:
142 return ""
143
144 default_username = get_system_username()
145 try:
146 default_username = (
147 unicodedata.normalize("NFKD", default_username)
148 .encode("ascii", "ignore")
149 .decode("ascii")
150 .replace(" ", "")
151 .lower()
152 )
153 except UnicodeDecodeError:
154 return ""
155
156 # Run the username validator
157 try:
158 auth_app.User._meta.get_field("username").run_validators(default_username)
159 except exceptions.ValidationError:
160 return ""
161
162 # Don't return the default username if it is already taken.
163 if check_db and default_username:
164 try:
165 auth_app.User._default_manager.db_manager(database).get(
166 username=default_username,
167 )
168 except auth_app.User.DoesNotExist:
169 pass
170 else:
171 return ""
172 return default_username
