28def _infer_dunder_doc_attribute(
29 node: nodes.Module | nodes.ClassDef | nodes.FunctionDef,
30) -> str | None:
31 # Try to see if we have a `__doc__` attribute.
32 try:
33 docstring = node["__doc__"]
34 except KeyError:
35 return None
36
37 docstring = utils.safe_infer(docstring)
38 if not docstring:
39 return None
40 if not isinstance(docstring, nodes.Const):
41 return None
42 return str(docstring.value)

115 @utils.only_required_for_messages("missing-function-docstring", "empty-docstring")
116 def visit_functiondef(self, node: nodes.FunctionDef) -> None:
117 if self.linter.config.no_docstring_rgx.match(node.name) is None:
118 ftype = "method" if node.is_method() else "function"
119 if (
120 is_property_setter(node)
121 or is_property_deleter(node)
122 or is_overload_stub(node)
123 ):
124 return
125
126 if isinstance(node.parent.frame(), nodes.ClassDef):
127 overridden = False
128 confidence = (
129 interfaces.INFERENCE
130 if utils.has_known_bases(node.parent.frame())
131 else interfaces.INFERENCE_FAILURE
132 )
133 # check if node is from a method overridden by its ancestor
134 for ancestor in node.parent.frame().ancestors():
135 if ancestor.qname() == "builtins.object":
136 continue
137 if node.name in ancestor and isinstance(
138 ancestor[node.name], nodes.FunctionDef
139 ):
140 overridden = True
141 break
142 self._check_docstring(
143 ftype, node, report_missing=not overridden, confidence=confidence # type: ignore[arg-type]
144 )
145 elif isinstance(node.parent.frame(), nodes.Module):
146 self._check_docstring(ftype, node) # type: ignore[arg-type]
147 else:
148 return

152 def _check_docstring(
153 self,
154 node_type: Literal["class", "function", "method", "module"],
155 node: nodes.Module | nodes.ClassDef | nodes.FunctionDef,
156 report_missing: bool = True,
157 confidence: interfaces.Confidence = interfaces.HIGH,
158 ) -> None:
159 """Check if the node has a non-empty docstring."""
160 docstring = node.doc_node.value if node.doc_node else None
161 if docstring is None:
162 docstring = _infer_dunder_doc_attribute(node)
163
164 if docstring is None:
165 if not report_missing:
166 return
167 lines = utils.get_node_last_lineno(node) - node.lineno
168
169 if node_type == "module" and not lines:
170 # If the module does not have a body, there's no reason
171 # to require a docstring.
172 return
173 max_lines = self.linter.config.docstring_min_length
174
175 if node_type != "module" and max_lines > -1 and lines < max_lines:
176 return
177 if node_type == "class":
178 self.linter.stats.undocumented["klass"] += 1
179 else:
180 self.linter.stats.undocumented[node_type] += 1
181 if (
182 node.body
183 and isinstance(node.body[0], nodes.Expr)
184 and isinstance(node.body[0].value, nodes.Call)
185 ):
186 # Most likely a string with a format call. Let's see.
187 func = utils.safe_infer(node.body[0].value.func)
188 if isinstance(func, astroid.BoundMethod) and isinstance(
189 func.bound, astroid.Instance
190 ):
191 # Strings.
192 if func.bound.name in {"str", "unicode", "bytes"}:
193 return
194 if node_type == "module":
195 message = "missing-module-docstring"
196 elif node_type == "class":
197 message = "missing-class-docstring"
198 else:
199 message = "missing-function-docstring"
200 self.add_message(message, node=node, confidence=confidence)
201 elif not docstring.strip():
202 if node_type == "class":
203 self.linter.stats.undocumented["klass"] += 1
204 else:
205 self.linter.stats.undocumented[node_type] += 1
206 self.add_message(
207 "empty-docstring", node=node, args=(node_type,), confidence=confidence
