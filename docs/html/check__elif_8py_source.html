21class ElseifUsedChecker(BaseTokenChecker):
22 """Checks for use of "else if" when an "elif" could be used."""

42 def process_tokens(self, tokens: list[TokenInfo]) -> None:
43 """Process tokens and look for 'if' or 'elif'."""
44 self._elifs = {
45 begin: token for _, token, begin, _, _ in tokens if token in {"elif", "if"}
46 }

51 @only_required_for_messages("else-if-used")
52 def visit_if(self, node: nodes.If) -> None:
53 """Current if node must directly follow an 'else'."""
54 if (
55 isinstance(node.parent, nodes.If)
56 and node.parent.orelse == [node]
57 and (node.lineno, node.col_offset) in self._elifs
58 and self._elifs[(node.lineno, node.col_offset)] == "if"
59 ):
60 self.add_message("else-if-used", node=node, confidence=HIGH)
