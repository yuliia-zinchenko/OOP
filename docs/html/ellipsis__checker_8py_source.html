20class EllipsisChecker(BaseChecker):
21 name = "unnecessary_ellipsis"
22 msgs = {

27 "A line of code consisting of an ellipsis is unnecessary if "
28 "there is a docstring on the preceding line or if there is a "

33 @only_required_for_messages("unnecessary-ellipsis")
34 def visit_const(self, node: nodes.Const) -> None:
35 """Check if the ellipsis constant is used unnecessarily.
36
37 Emits a warning when:
38 - A line consisting of an ellipsis is preceded by a docstring.
39 - A statement exists in the same scope as the ellipsis.
40 For example: A function consisting of an ellipsis followed by a
41 return statement on the next line.
42 """
43 if (
44 node.pytype() == "builtins.Ellipsis"
45 and isinstance(node.parent, nodes.Expr)
46 and (
47 (
48 isinstance(node.parent.parent, (nodes.ClassDef, nodes.FunctionDef))
49 and node.parent.parent.doc_node
50 )
51 or len(node.parent.parent.body) > 1
52 )
53 ):
54 self.add_message("unnecessary-ellipsis", node=node)
