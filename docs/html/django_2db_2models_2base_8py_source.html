1168 def _do_update(self, base_qs, using, pk_val, values, update_fields, forced_update):
1169 """
1170 Try to update the model. Return True if the model was updated (if an
1171 update query was done and a matching row was found in the DB).
1172 """
1173 filtered = base_qs.filter(pk=pk_val)
1174 if not values:
1175 # We can end up here when saving a model in inheritance chain where
1176 # update_fields doesn't target any field in current model. In that
1177 # case we just say the update succeeded. Another case ending up here
1178 # is a model with just PK - in that case check that the PK still
1179 # exists.
1180 return update_fields is not None or filtered.exists()
1181 if self._meta.select_on_save and not forced_update:
1182 return (
1183 filtered.exists()
1184 and
1185 # It may happen that the object is deleted from the DB right after
1186 # this check, causing the subsequent UPDATE to return zero matching
1187 # rows. The same result can occur in some rare cases when the
1188 # database returns zero despite the UPDATE being executed
1189 # successfully (a row is matched and updated). In order to
1190 # distinguish these two cases, the object's existence in the
1191 # database is again checked for if the UPDATE query returns 0.
1192 (filtered._update(values) > 0 or filtered.exists())
1193 )
1194 return filtered._update(values) > 0

1209 def _prepare_related_fields_for_save(self, operation_name, fields=None):
1210 # Ensure that a model instance without a PK hasn't been assigned to
1211 # a ForeignKey, GenericForeignKey or OneToOneField on this model. If
1212 # the field is nullable, allowing the save would result in silent data
1213 # loss.
1214 for field in self._meta.concrete_fields:
1215 if fields and field not in fields:
1216 continue
1217 # If the related field isn't cached, then an instance hasn't been
1218 # assigned and there's no need to worry about this check.
1219 if field.is_relation and field.is_cached(self):
1220 obj = getattr(self, field.name, None)
1221 if not obj:
1222 continue
1223 # A pk may have been assigned manually to a model instance not
1224 # saved to the database (or auto-generated in a case like
1225 # UUIDField), but we allow the save to proceed and rely on the
1226 # database to raise an IntegrityError if applicable. If
1227 # constraints aren't supported by the database, there's the
1228 # unavoidable risk of data corruption.
1229 if obj.pk is None:
1230 # Remove the object from a related instance cache.
1231 if not field.remote_field.multiple:
1232 field.remote_field.delete_cached_value(obj)
1233 raise ValueError(
1234 "%s() prohibited to prevent data loss due to unsaved "
1235 "related object '%s'." % (operation_name, field.name)
1236 )
1237 elif getattr(self, field.attname) in field.empty_values:
1238 # Set related object if it has been saved after an
1239 # assignment.
1240 setattr(self, field.name, obj)
1241 # If the relationship's pk/to_field was changed, clear the
1242 # cached relationship.
1243 if getattr(obj, field.target_field.attname) != getattr(
1244 self, field.attname
1245 ):
1246 field.delete_cached_value(self)
1247 # GenericForeignKeys are private.
1248 for field in self._meta.private_fields:
1249 if fields and field not in fields:
1250 continue
1251 if (
1252 field.is_relation
1253 and field.is_cached(self)
1254 and hasattr(field, "fk_field")
1255 ):
1256 obj = field.get_cached_value(self, default=None)
1257 if obj and obj.pk is None:
1258 raise ValueError(
1259 f"{operation_name}() prohibited to prevent data loss due to "
1260 f"unsaved related object '{field.name}'."
1261 )
