69def modify_sys_path() -> None:
70 """Modify sys path for execution as Python module.
71
72 Strip out the current working directory from sys.path.
73 Having the working directory in `sys.path` means that `pylint` might
74 inadvertently import user code from modules having the same name as
75 stdlib or pylint's own modules.
76 CPython issue: https://bugs.python.org/issue33053
77
78 - Remove the first entry. This will always be either "" or the working directory
79 - Remove the working directory from the second and third entries
80 if PYTHONPATH includes a ":" at the beginning or the end.
81 https://github.com/pylint-dev/pylint/issues/3636
82 Don't remove it if PYTHONPATH contains the cwd or '.' as the entry will
83 only be added once.
84 - Don't remove the working directory from the rest. It will be included
85 if pylint is installed in an editable configuration (as the last item).
86 https://github.com/pylint-dev/pylint/issues/4161
87 """
88 cwd = os.getcwd()
89 if sys.path[0] in ("", ".", cwd):
90 sys.path.pop(0)
91 env_pythonpath = os.environ.get("PYTHONPATH", "")
92 if env_pythonpath.startswith(":") and env_pythonpath not in (f":{cwd}", ":."):
93 sys.path.pop(0)
94 elif env_pythonpath.endswith(":") and env_pythonpath not in (f"{cwd}:", ".:"):
95 sys.path.pop(1)

98def _catch_valueerror(unraisable: sys.UnraisableHookArgs) -> None: # pragma: no cover
99 """Overwrite sys.unraisablehook to catch incorrect ValueError.
100
101 Python 3.12 introduced changes that sometimes cause astroid to emit ValueErrors
102 with 'generator already executing'. Fixed in Python 3.12.3 and 3.13.
103
104 https://github.com/pylint-dev/pylint/issues/9138
105 """
106 if (
107 isinstance(unraisable.exc_value, ValueError)
108 and unraisable.exc_value.args[0] == "generator already executing"
109 ):
110 return
111
112 sys.__unraisablehook__(unraisable)
113
114
115if (3, 12, 0) <= sys.version_info[:3] < (3, 12, 3):
116 sys.unraisablehook = _catch_valueerror
