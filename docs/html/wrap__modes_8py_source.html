36def _wrap_mode(function: Callable[..., str]) -> Callable[..., str]:
37 """Registers an individual wrap mode. Function name and order are significant and used for
38 creating enum.
39 """
40 _wrap_modes[function.__name__.upper()] = function
41 function.__signature__ = signature(_wrap_mode_interface) # type: ignore
42 function.__annotations__ = _wrap_mode_interface.__annotations__
43 return function

46@_wrap_mode
47def grid(**interface: Any) -> str:
48 if not interface["imports"]:
49 return ""
50
51 interface["statement"] += "(" + interface["imports"].pop(0)
52 while interface["imports"]:
53 next_import = interface["imports"].pop(0)
54 next_statement = isort.comments.add_to_line(
55 interface["comments"],
56 interface["statement"] + ", " + next_import,
57 removed=interface["remove_comments"],
58 comment_prefix=interface["comment_prefix"],
59 )
60 if (
61 len(next_statement.split(interface["line_separator"])[-1]) + 1
62 > interface["line_length"]
63 ):
64 lines = [f"{interface['white_space']}{next_import.split(' ')[0]}"]
65 for part in next_import.split(" ")[1:]:
66 new_line = f"{lines[-1]} {part}"
67 if len(new_line) + 1 > interface["line_length"]:
68 lines.append(f"{interface['white_space']}{part}")
69 else:
70 lines[-1] = new_line
71 next_import = interface["line_separator"].join(lines)
72 interface["statement"] = (
73 isort.comments.add_to_line(
74 interface["comments"],
75 f"{interface['statement']},",
76 removed=interface["remove_comments"],
77 comment_prefix=interface["comment_prefix"],
78 )
79 + f"{interface['line_separator']}{next_import}"
80 )
81 interface["comments"] = []
82 else:
83 interface["statement"] += ", " + next_import
84 return f"{interface['statement']}{',' if interface['include_trailing_comma'] else ''})"

116@_wrap_mode
117def hanging_indent(**interface: Any) -> str:
118 if not interface["imports"]:
119 return ""
120
121 line_length_limit = interface["line_length"] - 3
122
123 next_import = interface["imports"].pop(0)
124 next_statement = interface["statement"] + next_import
125 # Check for first import
126 if len(next_statement) > line_length_limit:
127 next_statement = (
128 _hanging_indent_end_line(interface["statement"])
129 + interface["line_separator"]
130 + interface["indent"]
131 + next_import
132 )
133
134 interface["statement"] = next_statement
135 while interface["imports"]:
136 next_import = interface["imports"].pop(0)
137 next_statement = interface["statement"] + ", " + next_import
138 if len(next_statement.split(interface["line_separator"])[-1]) > line_length_limit:
139 next_statement = (
140 _hanging_indent_end_line(interface["statement"] + ",")
141 + f"{interface['line_separator']}{interface['indent']}{next_import}"
142 )
143 interface["statement"] = next_statement
144
145 if interface["comments"]:
146 statement_with_comments = isort.comments.add_to_line(
147 interface["comments"],
148 interface["statement"],
149 removed=interface["remove_comments"],
150 comment_prefix=interface["comment_prefix"],
151 )
152 if len(statement_with_comments.split(interface["line_separator"])[-1]) <= (
153 line_length_limit + 2
154 ):
155 return statement_with_comments
156 return (
157 _hanging_indent_end_line(interface["statement"])
158 + str(interface["line_separator"])
159 + isort.comments.add_to_line(
160 interface["comments"],
161 interface["indent"],
162 removed=interface["remove_comments"],
163 comment_prefix=interface["comment_prefix"].lstrip(),
164 )
165 )
166 return str(interface["statement"])

185def _vertical_grid_common(need_trailing_char: bool, **interface: Any) -> str:
186 if not interface["imports"]:
187 return ""
188
189 interface["statement"] += (
190 isort.comments.add_to_line(
191 interface["comments"],
192 "(",
193 removed=interface["remove_comments"],
194 comment_prefix=interface["comment_prefix"],
195 )
196 + interface["line_separator"]
197 + interface["indent"]
198 + interface["imports"].pop(0)
199 )
200 while interface["imports"]:
201 next_import = interface["imports"].pop(0)
202 next_statement = f"{interface['statement']}, {next_import}"
203 current_line_length = len(next_statement.split(interface["line_separator"])[-1])
204 if interface["imports"] or interface["include_trailing_comma"]:
205 # We need to account for a comma after this import.
206 current_line_length += 1
207 if not interface["imports"] and need_trailing_char:
208 # We need to account for a closing ) we're going to add.
209 current_line_length += 1
210 if current_line_length > interface["line_length"]:
211 next_statement = (
212 f"{interface['statement']},{interface['line_separator']}"
213 f"{interface['indent']}{next_import}"
214 )
215 interface["statement"] = next_statement
216 if interface["include_trailing_comma"]:
217 interface["statement"] += ","
218 return str(interface["statement"])

242@_wrap_mode
243def noqa(**interface: Any) -> str:
244 _imports = ", ".join(interface["imports"])
245 retval = f"{interface['statement']}{_imports}"
246 comment_str = " ".join(interface["comments"])
247 if interface["comments"]:
248 if (
249 len(retval) + len(interface["comment_prefix"]) + 1 + len(comment_str)
250 <= interface["line_length"]
251 ):
252 return f"{retval}{interface['comment_prefix']} {comment_str}"
253 if "NOQA" in interface["comments"]:
254 return f"{retval}{interface['comment_prefix']} {comment_str}"
255 return f"{retval}{interface['comment_prefix']} NOQA {comment_str}"

258 return retval
