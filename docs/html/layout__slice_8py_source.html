40 def pre_map(self, function):
41 """
42 Iterates over layout objects pointed in `self.slice` executing `function` on them.
43 It passes `function` penultimate layout object and the position where to find last one
44 """
45 if isinstance(self.slice, slice):
46 for i in range(*self.slice.indices(len(self.layout.fields))):
47 function(self.layout, i)

49 elif isinstance(self.slice, list):

97 """
