83 def date_extract_sql(self, lookup_type, sql, params):
84 # https://www.postgresql.org/docs/current/functions-datetime.html#FUNCTIONS-DATETIME-EXTRACT
85 if lookup_type == "week_day":
86 # For consistency across backends, we return Sunday=1, Saturday=7.
87 return f"EXTRACT(DOW FROM {sql}) + 1", params
88 elif lookup_type == "iso_week_day":
89 return f"EXTRACT(ISODOW FROM {sql})", params
90 elif lookup_type == "iso_year":
91 return f"EXTRACT(ISOYEAR FROM {sql})", params
92
93 lookup_type = lookup_type.upper()
94 if not self._extract_format_re.fullmatch(lookup_type):
95 raise ValueError(f"Invalid lookup type: {lookup_type!r}")
96 return f"EXTRACT({lookup_type} FROM {sql})", params

198 def sql_flush(self, style, tables, *, reset_sequences=False, allow_cascade=False):
199 if not tables:
200 return []
201
202 # Perform a single SQL 'TRUNCATE x, y, z...;' statement. It allows us
203 # to truncate tables referenced by a foreign key in any other table.
204 sql_parts = [
205 style.SQL_KEYWORD("TRUNCATE"),
206 ", ".join(style.SQL_FIELD(self.quote_name(table)) for table in tables),
207 ]
208 if reset_sequences:
209 sql_parts.append(style.SQL_KEYWORD("RESTART IDENTITY"))
210 if allow_cascade:
211 sql_parts.append(style.SQL_KEYWORD("CASCADE"))
212 return ["%s;" % " ".join(sql_parts)]

239 def sequence_reset_sql(self, style, model_list):
240 from django.db import models
241
242 output = []
243 qn = self.quote_name
244 for model in model_list:
245 # Use `coalesce` to set the sequence for each model to the max pk
246 # value if there are records, or 1 if there are none. Set the
247 # `is_called` property (the third argument to `setval`) to true if
248 # there are records (as the max pk value is already in use),
249 # otherwise set it to false. Use pg_get_serial_sequence to get the
250 # underlying sequence name from the table name and column name.
251
252 for f in model._meta.local_fields:
253 if isinstance(f, models.AutoField):
254 output.append(
255 "%s setval(pg_get_serial_sequence('%s','%s'), "
256 "coalesce(max(%s), 1), max(%s) %s null) %s %s;"
257 % (
258 style.SQL_KEYWORD("SELECT"),
259 style.SQL_TABLE(qn(model._meta.db_table)),
260 style.SQL_FIELD(f.column),
261 style.SQL_FIELD(qn(f.column)),
262 style.SQL_FIELD(qn(f.column)),
263 style.SQL_KEYWORD("IS NOT"),
264 style.SQL_KEYWORD("FROM"),
265 style.SQL_TABLE(qn(model._meta.db_table)),
266 )
267 )
268 # Only one AutoField is allowed per model, so don't bother
269 # continuing.
270 break
271 return output

366 def explain_query_prefix(self, format=None, **options):
367 extra = {}
368 # Normalize options.
369 if options:
370 options = {
371 name.upper(): "true" if value else "false"
372 for name, value in options.items()
