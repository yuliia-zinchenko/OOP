18class MultipleTypesChecker(BaseChecker):
19 """Checks for variable type redefinition (NoneType excepted).
20
21 At a function, method, class or module scope
22
23 This rule could be improved:
24
25 - Currently, if an attribute is set to different types in 2 methods of a
26 same class, it won't be detected (see functional test)
27 - One could improve the support for inference on assignment with tuples,
28 ifexpr, etc. Also, it would be great to have support for inference on
29 str.split()
30 """
31
32 name = "multiple_types"
33 msgs = {
34 "R0204": (
35 "Redefinition of %s type from %s to %s",
36 "redefined-variable-type",
37 "Used when the type of a variable changes inside a "
38 "method or a function.",
39 )
40 }

55 def _check_and_add_messages(self) -> None:
56 assigns = self._assigns.pop()
57 for name, args in assigns.items():
58 if len(args) <= 1:
59 continue
60 orig_node, orig_type = args[0]
61 # Check if there is a type in the following nodes that would be
62 # different from orig_type.
63 for redef_node, redef_type in args[1:]:
64 if redef_type == orig_type:
65 continue
66 # if a variable is defined to several types in an if node,
67 # this is not actually redefining.
68 orig_parent = orig_node.parent
69 redef_parent = redef_node.parent
70 if isinstance(orig_parent, nodes.If):
71 if orig_parent == redef_parent:
72 if (
73 redef_node in orig_parent.orelse
74 and orig_node not in orig_parent.orelse
75 ):
76 orig_node, orig_type = redef_node, redef_type
77 continue
78 elif isinstance(
79 redef_parent, nodes.If
80 ) and redef_parent in orig_parent.nodes_of_class(nodes.If):
81 orig_node, orig_type = redef_node, redef_type
82 continue
83 orig_type = orig_type.replace("builtins.", "")
84 redef_type = redef_type.replace("builtins.", "")
85 self.add_message(
86 "redefined-variable-type",
87 node=redef_node,
88 args=(name, orig_type, redef_type),
89 )
90 break
91
92 def visit_assign(self, node: nodes.Assign) -> None:
93 # we don't handle multiple assignment nor slice assignment
94 target = node.targets[0]
95 if isinstance(target, (nodes.Tuple, nodes.Subscript)):
96 return
97 # ignore NoneType
98 if is_none(node):
99 return
100 _type = node_type(node.value)
101 if _type:
102 self._assigns[-1].setdefault(target.as_string(), []).append(
103 (node, _type.pytype())
104 )
