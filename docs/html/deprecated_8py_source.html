101 def visit_call(self, node: nodes.Call) -> None:
102 """Called when a :class:`nodes.Call` node is visited."""
103 self.check_deprecated_class_in_call(node)

108 @utils.only_required_for_messages(
109 "deprecated-module",

112 def visit_import(self, node: nodes.Import) -> None:
113 """Triggered when an import statement is seen."""
114 for name in (name for name, _ in node.names):
115 self.check_deprecated_module(node, name)
116 if "." in name:
117 # Checking deprecation for import module with class
118 mod_name, class_name = name.split(".", 1)
119 self.check_deprecated_class(node, mod_name, (class_name,))

129 @utils.only_required_for_messages("deprecated-decorator")
130 def visit_decorators(self, node: nodes.Decorators) -> None:
131 """Triggered when a decorator statement is seen."""
132 children = list(node.get_children())
133 if not children:
134 return
135 if isinstance(children[0], nodes.Call):
136 inf = safe_infer(children[0].func)
137 else:
138 inf = safe_infer(children[0])

140 if qname in self.deprecated_decorators():

213 def check_deprecated_attribute(self, node: astroid.Attribute) -> None:
214 """Checks if the attribute is deprecated."""
215 inferred_expr = safe_infer(node.expr)
216 if not isinstance(inferred_expr, (nodes.ClassDef, Instance, nodes.Module)):
217 return
218 attribute_qname = ".".join((inferred_expr.qname(), node.attrname))
219 for deprecated_name in self.deprecated_attributes():
220 if attribute_qname == deprecated_name:
221 self.add_message(
222 "deprecated-attribute",
223 node=node,
224 args=(attribute_qname,),
225 confidence=INFERENCE,
226 )
227
228 def check_deprecated_module(self, node: nodes.Import, mod_path: str | None) -> None:
229 """Checks if the module is deprecated."""
230 for mod_name in self.deprecated_modules():
231 if mod_path == mod_name or mod_path and mod_path.startswith(mod_name + "."):
232 self.add_message("deprecated-module", node=node, args=mod_path)

234 def check_deprecated_method(self, node: nodes.Call, inferred: nodes.NodeNG) -> None:
235 """Executes the checker for the given node.
236
237 This method should be called from the checker implementing this mixin.
238 """
239 # Reject nodes which aren't of interest to us.
240 if not isinstance(inferred, ACCEPTABLE_NODES):
241 return
242
243 if isinstance(node.func, nodes.Attribute):
244 func_name = node.func.attrname
245 elif isinstance(node.func, nodes.Name):
246 func_name = node.func.name
247 else:
248 # Not interested in other nodes.
249 return
250
251 qnames = {inferred.qname(), func_name}
252 if any(name in self.deprecated_methods() for name in qnames):
253 self.add_message("deprecated-method", node=node, args=(func_name,))
254 return
255 num_of_args = len(node.args)
256 kwargs = {kw.arg for kw in node.keywords} if node.keywords else {}
257 deprecated_arguments = (self.deprecated_arguments(qn) for qn in qnames)
258 for position, arg_name in chain(*deprecated_arguments):
259 if arg_name in kwargs:
260 # function was called with deprecated argument as keyword argument
261 self.add_message(
262 "deprecated-argument", node=node, args=(arg_name, func_name)
263 )
264 elif position is not None and position < num_of_args:
265 # function was called with deprecated argument as positional argument
266 self.add_message(
267 "deprecated-argument", node=node, args=(arg_name, func_name)
268 )

270 def check_deprecated_class(
271 self, node: nodes.NodeNG, mod_name: str, class_names: Iterable[str]
272 ) -> None:
273 """Checks if the class is deprecated."""
274 for class_name in class_names:
275 if class_name in self.deprecated_classes(mod_name):
276 self.add_message(
277 "deprecated-class", node=node, args=(class_name, mod_name)
278 )
