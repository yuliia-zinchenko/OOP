47def _infer_unary_op(obj: Any, op: str) -> ConstFactoryResult:
48 """Perform unary operation on `obj`, unless it is `NotImplemented`.
49
50 Can raise TypeError if operation is unsupported.
51 """
52 if obj is NotImplemented:
53 value = obj
54 else:
55 func = _UNARY_OPERATORS[op]
56 value = func(obj)
57 return nodes.const_factory(value)

101@decorators.yes_if_nothing_inferred
102def const_infer_binary_op(
103 self: nodes.Const,
104 opnode: nodes.AugAssign | nodes.BinOp,
105 operator: str,
106 other: InferenceResult,
107 context: InferenceContext,
108 _: SuccessfulInferenceResult,
109) -> Generator[ConstFactoryResult | util.UninferableBase]:
110 not_implemented = nodes.Const(NotImplemented)
111 if isinstance(other, nodes.Const):
112 if (
113 operator == "**"
114 and isinstance(self.value, (int, float))
115 and isinstance(other.value, (int, float))
116 and (self.value > 1e5 or other.value > 1e5)
117 ):
118 yield not_implemented

120 try:
121 impl = BIN_OP_IMPL[operator]
122 try:
123 yield nodes.const_factory(impl(self.value, other.value))
124 except TypeError:
125 # ArithmeticError is not enough: float >> float is a TypeError
126 yield not_implemented
127 except Exception: # pylint: disable=broad-except
128 yield util.Uninferable
129 except TypeError:
130 yield not_implemented
131 elif isinstance(self.value, str) and operator == "%":
132 # TODO(cpopa): implement string interpolation later on.
133 yield util.Uninferable
134 else:
135 yield not_implemented

138def _multiply_seq_by_int(
139 self: _TupleListNodeT,
140 opnode: nodes.AugAssign | nodes.BinOp,
141 value: int,
142 context: InferenceContext,
143) -> _TupleListNodeT:
144 node = self.__class__(parent=opnode)
145 if value > 1e8:
146 node.elts = [util.Uninferable]
147 return node
148 filtered_elts = (
149 util.safe_infer(elt, context) or util.Uninferable
150 for elt in self.elts
151 if not isinstance(elt, util.UninferableBase)
152 )
153 node.elts = list(filtered_elts) * value
154 return node

171@decorators.yes_if_nothing_inferred
172def tl_infer_binary_op(
173 self: _TupleListNodeT,
174 opnode: nodes.AugAssign | nodes.BinOp,
175 operator: str,
176 other: InferenceResult,
177 context: InferenceContext,
178 method: SuccessfulInferenceResult,
179) -> Generator[_TupleListNodeT | nodes.Const | util.UninferableBase]:
180 """Infer a binary operation on a tuple or list.
181
182 The instance on which the binary operation is performed is a tuple
183 or list. This refers to the left-hand side of the operation, so:
184 'tuple() + 1' or '[] + A()'
185 """
186 from astroid import helpers # pylint: disable=import-outside-toplevel
187
188 # For tuples and list the boundnode is no longer the tuple or list instance
189 context.boundnode = None
190 not_implemented = nodes.Const(NotImplemented)
191 if isinstance(other, self.__class__) and operator == "+":
192 node = self.__class__(parent=opnode)
193 node.elts = list(
194 itertools.chain(
195 _filter_uninferable_nodes(self.elts, context),
196 _filter_uninferable_nodes(other.elts, context),
197 )
198 )
199 yield node
200 elif isinstance(other, nodes.Const) and operator == "*":
201 if not isinstance(other.value, int):
202 yield not_implemented
203 return
204 yield _multiply_seq_by_int(self, opnode, other.value, context)
205 elif isinstance(other, bases.Instance) and operator == "*":
206 # Verify if the instance supports __index__.
207 as_index = helpers.class_instance_as_index(other)
208 if not as_index:
209 yield util.Uninferable
210 elif not isinstance(as_index.value, int): # pragma: no cover
211 # already checked by class_instance_as_index() but faster than casting
212 raise AssertionError("Please open a bug report.")
213 else:
214 yield _multiply_seq_by_int(self, opnode, as_index.value, context)
215 else:
216 yield not_implemented

233"""The assigned_stmts method is responsible to return the assigned statement
234(e.g. not inferred) according to the assignment type.
235
236The `assign_path` argument is used to record the lhs path of the original node.
237For instance if we want assigned statements for 'c' in 'a, (b,c)', assign_path
238will be [1, 1] once arrived to the Assign node.
239
240The `context` argument is the current inference context which should be given
241to any intermediary inference necessary.
242"""

245def _resolve_looppart(parts, assign_path, context):
246 """Recursive function to resolve multiple assignments on loops."""
247 assign_path = assign_path[:]
248 index = assign_path.pop(0)
249 for part in parts:
250 if isinstance(part, util.UninferableBase):
251 continue
252 if not hasattr(part, "itered"):
253 continue
254 try:
255 itered = part.itered()
256 except TypeError:
257 continue
258 try:
259 if isinstance(itered[index], (nodes.Const, nodes.Name)):
260 itered = [part]
261 except IndexError:
262 pass
263 for stmt in itered:
264 index_node = nodes.Const(index)
265 try:
266 assigned = stmt.getitem(index_node, context)
267 except (AttributeError, AstroidTypeError, AstroidIndexError):
268 continue
269 if not assign_path:
270 # we achieved to resolved the assignment path,
271 # don't infer the last part
272 yield assigned
273 elif isinstance(assigned, util.UninferableBase):
274 break
275 else:
276 # we are not yet on the last part of the path
277 # search on each possibly inferred value
278 try:
279 yield from _resolve_looppart(
280 assigned.infer(context), assign_path, context
281 )
282 except InferenceError:
283 break

348def _arguments_infer_argname(
349 self, name: str | None, context: InferenceContext
350) -> Generator[InferenceResult]:
351 # arguments information may be missing, in which case we can't do anything
352 # more
353 from astroid import arguments # pylint: disable=import-outside-toplevel
354
355 if not self.arguments:
356 yield util.Uninferable
357 return
358
359 args = [arg for arg in self.arguments if arg.name not in [self.vararg, self.kwarg]]
360 functype = self.parent.type
361 # first argument of instance/class method
362 if (
363 args
364 and getattr(self.arguments[0], "name", None) == name
365 and functype != "staticmethod"
366 ):
367 cls = self.parent.parent.scope()
368 is_metaclass = isinstance(cls, nodes.ClassDef) and cls.type == "metaclass"
369 # If this is a metaclass, then the first argument will always
370 # be the class, not an instance.
371 if context.boundnode and isinstance(context.boundnode, bases.Instance):
372 cls = context.boundnode._proxied
373 if is_metaclass or functype == "classmethod":
374 yield cls
375 return
376 if functype == "method":
377 yield cls.instantiate_class()
378 return
379
380 if context and context.callcontext:
381 callee = context.callcontext.callee
382 while hasattr(callee, "_proxied"):
383 callee = callee._proxied
384 if getattr(callee, "name", None) == self.parent.name:
385 call_site = arguments.CallSite(context.callcontext, context.extra_context)
386 yield from call_site.infer_argument(self.parent, name, context)
387 return
388
389 if name == self.vararg:
390 vararg = nodes.const_factory(())
391 vararg.parent = self
392 if not args and self.parent.name == "__init__":
393 cls = self.parent.parent.scope()
394 vararg.elts = [cls.instantiate_class()]
395 yield vararg
396 return
397 if name == self.kwarg:
398 kwarg = nodes.const_factory({})
399 kwarg.parent = self
400 yield kwarg
401 return
402 # if there is a default value, yield it. And then yield Uninferable to reflect
403 # we can't guess given argument value
404 try:
405 context = copy_context(context)
406 yield from self.default_value(name).infer(context)
407 yield util.Uninferable
408 except NoDefault:
409 yield util.Uninferable

478def _resolve_assignment_parts(parts, assign_path, context):
479 """Recursive function to resolve multiple assignments."""
480 assign_path = assign_path[:]
481 index = assign_path.pop(0)
482 for part in parts:
483 assigned = None
484 if isinstance(part, nodes.Dict):
485 # A dictionary in an iterating context
486 try:
487 assigned, _ = part.items[index]
488 except IndexError:
489 return
490
491 elif hasattr(part, "getitem"):
492 index_node = nodes.Const(index)
493 try:
494 assigned = part.getitem(index_node, context)
495 except (AstroidTypeError, AstroidIndexError):
496 return
497
498 if not assigned:
499 return
500
501 if not assign_path:
502 # we achieved to resolved the assignment path, don't infer the
503 # last part
504 yield assigned
505 elif isinstance(assigned, util.UninferableBase):
506 return
507 else:
508 # we are not yet on the last part of the path search on each
509 # possibly inferred value
510 try:
511 yield from _resolve_assignment_parts(
512 assigned.infer(context), assign_path, context
513 )
514 except InferenceError:
515 return

540def _infer_context_manager(self, mgr, context):
541 try:
542 inferred = next(mgr.infer(context=context))
543 except StopIteration as e:
544 raise InferenceError(node=mgr) from e
545 if isinstance(inferred, bases.Generator):
546 # Check if it is decorated with contextlib.contextmanager.
547 func = inferred.parent
548 if not func.decorators:
549 raise InferenceError(
550 "No decorators found on inferred generator %s", node=func
551 )
552
553 for decorator_node in func.decorators.nodes:
554 decorator = next(decorator_node.infer(context=context), None)
555 if isinstance(decorator, nodes.FunctionDef):
556 if decorator.qname() == _CONTEXTLIB_MGR:
557 break
558 else:
559 # It doesn't interest us.
560 raise InferenceError(node=func)
561 try:
562 yield next(inferred.infer_yield_types())
563 except StopIteration as e:
564 raise InferenceError(node=func) from e
565
566 elif isinstance(inferred, bases.Instance):
567 try:
568 enter = next(inferred.igetattr("__enter__", context=context))
569 except (InferenceError, AttributeInferenceError, StopIteration) as exc:
570 raise InferenceError(node=inferred) from exc
571 if not isinstance(enter, bases.BoundMethod):
572 raise InferenceError(node=enter)
573 yield from enter.infer_call_result(self, context)
574 else:
575 raise InferenceError(node=mgr)

578@decorators.raise_if_nothing_inferred
579def with_assigned_stmts(
580 self: nodes.With,
581 node: node_classes.AssignedStmtsPossibleNode = None,
582 context: InferenceContext | None = None,
583 assign_path: list[int] | None = None,
584) -> Any:
585 """Infer names and other nodes from a *with* statement.
586
587 This enables only inference for name binding in a *with* statement.
588 For instance, in the following code, inferring `func` will return
589 the `ContextManager` class, not whatever ``__enter__`` returns.
590 We are doing this intentionally, because we consider that the context
591 manager result is whatever __enter__ returns and what it is binded
592 using the ``as`` keyword.
593
594 class ContextManager(object):
595 def __enter__(self):
596 return 42
597 with ContextManager() as f:
598 pass
599
600 # ContextManager().infer() will return ContextManager
601 # f.infer() will return 42.
602
603 Arguments:
604 self: nodes.With
605 node: The target of the assignment, `as (a, b)` in `with foo as (a, b)`.
606 context: Inference context used for caching already inferred objects
607 assign_path:
608 A list of indices, where each index specifies what item to fetch from
609 the inference results.
610 """
611 try:
612 mgr = next(mgr for (mgr, vars) in self.items if vars == node)
613 except StopIteration:
614 return None
615 if assign_path is None:
616 yield from _infer_context_manager(self, mgr, context)
617 else:
618 for result in _infer_context_manager(self, mgr, context):
619 # Walk the assign_path and get the item at the final index.
620 obj = result
621 for index in assign_path:
622 if not hasattr(obj, "elts"):
623 raise InferenceError(
624 "Wrong type ({targets!r}) for {node!r} assignment",
625 node=self,
626 targets=node,
627 assign_path=assign_path,
628 context=context,
629 )
630 try:
631 obj = obj.elts[index]
632 except IndexError as exc:
633 raise InferenceError(
634 "Tried to infer a nonexistent target with index {index} "
635 "in {node!r}.",
636 node=self,
637 targets=node,
638 assign_path=assign_path,
639 context=context,
640 ) from exc
641 except TypeError as exc:
642 raise InferenceError(
643 "Tried to unpack a non-iterable value in {node!r}.",
644 node=self,
645 targets=node,
646 assign_path=assign_path,
647 context=context,
648 ) from exc
649 yield obj
650 return {
651 "node": self,
652 "unknown": node,
653 "assign_path": assign_path,
654 "context": context,
655 }

678def starred_assigned_stmts( # noqa: C901
679 self: nodes.Starred,
680 node: node_classes.AssignedStmtsPossibleNode = None,
681 context: InferenceContext | None = None,
682 assign_path: list[int] | None = None,
683) -> Any:
684 """
685 Arguments:
686 self: nodes.Starred
687 node: a node related to the current underlying Node.
688 context: Inference context used for caching already inferred objects
689 assign_path:
690 A list of indices, where each index specifies what item to fetch from
691 the inference results.
692 """

694 # pylint: disable=too-many-locals,too-many-statements
695 def _determine_starred_iteration_lookups(
696 starred: nodes.Starred, target: nodes.Tuple, lookups: list[tuple[int, int]]
697 ) -> None:
698 # Determine the lookups for the rhs of the iteration
699 itered = target.itered()
700 for index, element in enumerate(itered):
701 if (
702 isinstance(element, nodes.Starred)
703 and element.value.name == starred.value.name
704 ):
705 lookups.append((index, len(itered)))
706 break
707 if isinstance(element, nodes.Tuple):
708 lookups.append((index, len(element.itered())))
709 _determine_starred_iteration_lookups(starred, element, lookups)
710
711 stmt = self.statement()
712 if not isinstance(stmt, (nodes.Assign, nodes.For)):
713 raise InferenceError(
714 "Statement {stmt!r} enclosing {node!r} must be an Assign or For node.",
715 node=self,
716 stmt=stmt,
717 unknown=node,
718 context=context,
719 )
720
721 if context is None:
722 context = InferenceContext()
723
724 if isinstance(stmt, nodes.Assign):
725 value = stmt.value
726 lhs = stmt.targets[0]
727 if not isinstance(lhs, nodes.BaseContainer):
728 yield util.Uninferable
729 return
730
731 if sum(1 for _ in lhs.nodes_of_class(nodes.Starred)) > 1:
732 raise InferenceError(
733 "Too many starred arguments in the assignment targets {lhs!r}.",
734 node=self,
735 targets=lhs,
736 unknown=node,
737 context=context,
738 )
739
740 try:
741 rhs = next(value.infer(context))
742 except (InferenceError, StopIteration):
743 yield util.Uninferable
744 return
745 if isinstance(rhs, util.UninferableBase) or not hasattr(rhs, "itered"):
746 yield util.Uninferable
747 return
748
749 try:
750 elts = collections.deque(rhs.itered()) # type: ignore[union-attr]
751 except TypeError:
752 yield util.Uninferable
753 return
754
755 # Unpack iteratively the values from the rhs of the assignment,
756 # until the find the starred node. What will remain will
757 # be the list of values which the Starred node will represent
758 # This is done in two steps, from left to right to remove
759 # anything before the starred node and from right to left
760 # to remove anything after the starred node.
761
762 for index, left_node in enumerate(lhs.elts):
763 if not isinstance(left_node, nodes.Starred):
764 if not elts:
765 break
766 elts.popleft()
767 continue
768 lhs_elts = collections.deque(reversed(lhs.elts[index:]))
769 for right_node in lhs_elts:
770 if not isinstance(right_node, nodes.Starred):
771 if not elts:
772 break
773 elts.pop()
774 continue
775
776 # We're done unpacking.
777 packed = nodes.List(
778 ctx=Context.Store,
779 parent=self,
780 lineno=lhs.lineno,
781 col_offset=lhs.col_offset,
782 )
783 packed.postinit(elts=list(elts))
784 yield packed
785 break
786
787 if isinstance(stmt, nodes.For):
788 try:
789 inferred_iterable = next(stmt.iter.infer(context=context))
790 except (InferenceError, StopIteration):
791 yield util.Uninferable
792 return
793 if isinstance(inferred_iterable, util.UninferableBase) or not hasattr(
794 inferred_iterable, "itered"
795 ):
796 yield util.Uninferable
797 return
798 try:
799 itered = inferred_iterable.itered() # type: ignore[union-attr]
800 except TypeError:
801 yield util.Uninferable
802 return
803
804 target = stmt.target
805
806 if not isinstance(target, nodes.Tuple):
807 raise InferenceError(
808 "Could not make sense of this, the target must be a tuple",
809 context=context,
810 )
811
812 lookups: list[tuple[int, int]] = []
813 _determine_starred_iteration_lookups(self, target, lookups)
814 if not lookups:
815 raise InferenceError(
816 "Could not make sense of this, needs at least a lookup", context=context
817 )
818
819 # Make the last lookup a slice, since that what we want for a Starred node
820 last_element_index, last_element_length = lookups[-1]
821 is_starred_last = last_element_index == (last_element_length - 1)
822
823 lookup_slice = slice(
824 last_element_index,
825 None if is_starred_last else (last_element_length - last_element_index),
826 )
827 last_lookup = lookup_slice
828
829 for element in itered:
830 # We probably want to infer the potential values *for each* element in an
831 # iterable, but we can't infer a list of all values, when only a list of
832 # step values are expected:
833 #
834 # for a, *b in [...]:
835 # b
836 #
837 # *b* should now point to just the elements at that particular iteration step,
838 # which astroid can't know about.
839
840 found_element = None
841 for index, lookup in enumerate(lookups):
842 if not hasattr(element, "itered"):
843 break
844 if index + 1 is len(lookups):
845 cur_lookup: slice | int = last_lookup
846 else:
847 # Grab just the index, not the whole length
848 cur_lookup = lookup[0]
849 try:
850 itered_inner_element = element.itered()
851 element = itered_inner_element[cur_lookup]
852 except IndexError:
853 break
854 except TypeError:
855 # Most likely the itered() call failed, cannot make sense of this
856 yield util.Uninferable
857 return
858 else:
859 found_element = element
860
861 unpacked = nodes.List(
862 ctx=Context.Store,
863 parent=self,
864 lineno=self.lineno,
865 col_offset=self.col_offset,
866 )
867 unpacked.postinit(elts=found_element or [])
868 yield unpacked
869 return
870
871 yield util.Uninferable
