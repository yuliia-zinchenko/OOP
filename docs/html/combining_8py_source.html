45class AndTrigger(BaseCombiningTrigger):
46 """
47 Always returns the earliest next fire time that all the given triggers can agree on.
48 The trigger is considered to be finished when any of the given triggers has finished its
49 schedule.
50
51 Trigger alias: ``and``
52
53 .. warning:: This trigger should only be used to combine triggers that fire on
54 specific times of day, such as
55 :class:`~apscheduler.triggers.cron.CronTrigger` and
56 class:`~apscheduler.triggers.calendarinterval.CalendarIntervalTrigger`.
57 Attempting to use it with
58 :class:`~apscheduler.triggers.interval.IntervalTrigger` will likely result in
59 the scheduler hanging as it tries to find a fire time that matches exactly
60 between fire times produced by all the given triggers.
61
62 :param list triggers: triggers to combine
63 :param int|None jitter: delay the job execution by ``jitter`` seconds at most
64 """
