22class ModifiedIterationChecker(checkers.BaseChecker):
23 """Checks for modified iterators in for loops iterations.
24
25 Currently supports `for` loops for Sets, Dictionaries and Lists.
26 """
27
28 name = "modified_iteration"
29
30 msgs = {
31 "W4701": (
32 "Iterated list '%s' is being modified inside for loop body, consider iterating through a copy of it "
33 "instead.",
34 "modified-iterating-list",
35 "Emitted when items are added or removed to a list being iterated through. "
36 "Doing so can result in unexpected behaviour, that is why it is preferred to use a copy of the list.",
37 ),
38 "E4702": (
39 "Iterated dict '%s' is being modified inside for loop body, iterate through a copy of it instead.",
40 "modified-iterating-dict",
41 "Emitted when items are added or removed to a dict being iterated through. "
42 "Doing so raises a RuntimeError.",
43 ),
44 "E4703": (
45 "Iterated set '%s' is being modified inside for loop body, iterate through a copy of it instead.",
46 "modified-iterating-set",
47 "Emitted when items are added or removed to a set being iterated through. "
48 "Doing so raises a RuntimeError.",

54 @utils.only_required_for_messages(
55 "modified-iterating-list", "modified-iterating-dict", "modified-iterating-set"
56 )
57 def visit_for(self, node: nodes.For) -> None:
58 iter_obj = node.iter
59 for body_node in node.body:
60 self._modified_iterating_check_on_node_and_children(body_node, iter_obj)
61
62 def _modified_iterating_check_on_node_and_children(
63 self, body_node: nodes.NodeNG, iter_obj: nodes.NodeNG
64 ) -> None:
65 """See if node or any of its children raises modified iterating messages."""
66 self._modified_iterating_check(body_node, iter_obj)
67 for child in body_node.get_children():
68 self._modified_iterating_check_on_node_and_children(child, iter_obj)
69
70 def _modified_iterating_check(
71 self, node: nodes.NodeNG, iter_obj: nodes.NodeNG
72 ) -> None:
73 msg_id = None
74 if isinstance(node, nodes.Delete) and any(
75 self._deleted_iteration_target_cond(t, iter_obj) for t in node.targets
76 ):
77 inferred = utils.safe_infer(iter_obj)
78 if isinstance(inferred, nodes.List):
79 msg_id = "modified-iterating-list"
80 elif isinstance(inferred, nodes.Dict):
81 msg_id = "modified-iterating-dict"
82 elif isinstance(inferred, nodes.Set):
83 msg_id = "modified-iterating-set"
84 elif not isinstance(iter_obj, (nodes.Name, nodes.Attribute)):
85 pass
86 elif self._modified_iterating_list_cond(node, iter_obj):
87 msg_id = "modified-iterating-list"
88 elif self._modified_iterating_dict_cond(node, iter_obj):
89 msg_id = "modified-iterating-dict"
90 elif self._modified_iterating_set_cond(node, iter_obj):
91 msg_id = "modified-iterating-set"
92 if msg_id:
93 self.add_message(
94 msg_id,
95 node=node,
96 args=(iter_obj.repr_name(),),
97 confidence=interfaces.INFERENCE,
98 )

100 @staticmethod
101 def _is_node_expr_that_calls_attribute_name(node: nodes.NodeNG) -> bool:
102 return (
103 isinstance(node, nodes.Expr)
104 and isinstance(node.value, nodes.Call)
105 and isinstance(node.value.func, nodes.Attribute)
106 and isinstance(node.value.func.expr, nodes.Name)
107 )
108
109 @staticmethod
110 def _common_cond_list_set(
111 node: nodes.Expr,
112 iter_obj: nodes.Name | nodes.Attribute,
113 infer_val: nodes.List | nodes.Set,
114 ) -> bool:
115 iter_obj_name = (
116 iter_obj.attrname
117 if isinstance(iter_obj, nodes.Attribute)
118 else iter_obj.name
119 )
120 return (infer_val == utils.safe_infer(iter_obj)) and ( # type: ignore[no-any-return]
121 node.value.func.expr.name == iter_obj_name
122 )
123
124 @staticmethod
125 def _is_node_assigns_subscript_name(node: nodes.NodeNG) -> bool:
126 return isinstance(node, nodes.Assign) and (
127 isinstance(node.targets[0], nodes.Subscript)
128 and (isinstance(node.targets[0].value, nodes.Name))
129 )
130
131 def _modified_iterating_list_cond(
132 self, node: nodes.NodeNG, iter_obj: nodes.Name | nodes.Attribute
133 ) -> bool:
134 if not self._is_node_expr_that_calls_attribute_name(node):
135 return False
136 infer_val = utils.safe_infer(node.value.func.expr)
137 if not isinstance(infer_val, nodes.List):
138 return False
139 return (
140 self._common_cond_list_set(node, iter_obj, infer_val)
141 and node.value.func.attrname in _LIST_MODIFIER_METHODS
142 )
143
144 def _modified_iterating_dict_cond(
145 self, node: nodes.NodeNG, iter_obj: nodes.Name | nodes.Attribute
146 ) -> bool:
147 if not self._is_node_assigns_subscript_name(node):
148 return False
149 # Do not emit when merely updating the same key being iterated
150 if (
151 isinstance(iter_obj, nodes.Name)
152 and iter_obj.name == node.targets[0].value.name
153 and isinstance(iter_obj.parent.target, nodes.AssignName)
154 and isinstance(node.targets[0].slice, nodes.Name)
155 and iter_obj.parent.target.name == node.targets[0].slice.name
156 ):
157 return False
158 infer_val = utils.safe_infer(node.targets[0].value)
159 if not isinstance(infer_val, nodes.Dict):
160 return False
161 if infer_val != utils.safe_infer(iter_obj):
162 return False
163 if isinstance(iter_obj, nodes.Attribute):
164 iter_obj_name = iter_obj.attrname
165 else:
166 iter_obj_name = iter_obj.name
167 return node.targets[0].value.name == iter_obj_name # type: ignore[no-any-return]
168
169 def _modified_iterating_set_cond(
170 self, node: nodes.NodeNG, iter_obj: nodes.Name | nodes.Attribute
171 ) -> bool:
172 if not self._is_node_expr_that_calls_attribute_name(node):
173 return False
174 infer_val = utils.safe_infer(node.value.func.expr)
175 if not isinstance(infer_val, nodes.Set):
176 return False
177 return (
178 self._common_cond_list_set(node, iter_obj, infer_val)
179 and node.value.func.attrname in _SET_MODIFIER_METHODS
180 )
181
182 def _deleted_iteration_target_cond(
183 self, node: nodes.DelName, iter_obj: nodes.NodeNG
184 ) -> bool:
185 if not isinstance(node, nodes.DelName):
186 return False
187 if not isinstance(iter_obj.parent, nodes.For):
188 return False
189 if not isinstance(
190 iter_obj.parent.target, (nodes.AssignName, nodes.BaseContainer)
191 ):
192 return False
193 return any(
194 t == node.name
195 for t in utils.find_assigned_names_recursive(iter_obj.parent.target)
196 )
