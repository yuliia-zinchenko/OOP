45 # Load up a loader to get all the migration data, but don't replace

47 loader = MigrationLoader(connection, replace_migrations=False)
48
49 # Resolve command-line arguments into a migration
50 app_label, migration_name = options["app_label"], options["migration_name"]
51 # Validate app_label
52 try:
53 apps.get_app_config(app_label)
54 except LookupError as err:
55 raise CommandError(str(err))
56 if app_label not in loader.migrated_apps:
57 raise CommandError("App '%s' does not have migrations" % app_label)
58 try:
59 migration = loader.get_migration_by_prefix(app_label, migration_name)
60 except AmbiguityError:
61 raise CommandError(
62 "More than one migration matches '%s' in app '%s'. Please be more "
63 "specific." % (migration_name, app_label)
64 )
65 except KeyError:
66 raise CommandError(
67 "Cannot find a migration matching '%s' from app '%s'. Is it in "
68 "INSTALLED_APPS?" % (migration_name, app_label)
69 )
70 target = (app_label, migration.name)
71
72 # Show begin/end around output for atomic migrations, if the database
73 # supports transactional DDL.
74 self.output_transaction = (
75 migration.atomic and connection.features.can_rollback_ddl
76 )
77
78 # Make a plan that represents just the requested migrations and show SQL
79 # for it
80 plan = [(loader.graph.nodes[target], options["backwards"])]
81 sql_statements = loader.collect_sql(plan)
