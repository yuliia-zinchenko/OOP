69 def _create_comment(
70 self, all_missing_messages: PackageMessages, all_new_messages: PackageMessages
71 ) -> str:
72 comment = ""
73 for package, missing_messages in all_missing_messages.items():
74 if len(comment) >= MAX_GITHUB_COMMENT_LENGTH:
75 break
76 new_messages = all_new_messages[package]
77 if not missing_messages["messages"] and not new_messages["messages"]:
78 continue
79 comment += self._create_comment_for_package(
80 package, new_messages, missing_messages
81 )
82 comment = (
83 f"🤖 **Effect of this PR on checked open source code:** 🤖\n\n{comment}"
84 if comment
85 else (
86 "🤖 According to the primer, this change has **no effect** on the"
87 " checked open source code. 🤖🎉\n\n"
88 )
89 )
90 return self._truncate_comment(comment)

92 def _create_comment_for_package(
93 self, package: str, new_messages: PackageData, missing_messages: PackageData
94 ) -> str:
95 comment = f"\n\n**Effect on [{package}]({self.packages[package].url}):**\n"
96 # Create comment for new messages
97 count = 1
98 astroid_errors = 0
99 new_non_astroid_messages = ""
100 if new_messages["messages"]:
101 print("Now emitted:")
102 for message in new_messages["messages"]:
103 filepath = str(
104 PurePosixPath(message["path"]).relative_to(
105 self.packages[package].clone_directory
106 )
107 )
108 # Existing astroid errors may still show up as "new" because the timestamp
109 # in the message is slightly different.
110 if message["symbol"] == "astroid-error":
111 astroid_errors += 1
112 else:
113 new_non_astroid_messages += (
114 f"{count}) {message['symbol']}:\n*{message['message']}*\n"
115 f"{self.packages[package].url}/blob/{new_messages['commit']}/{filepath}#L{message['line']}\n"
116 )
117 print(message)
118 count += 1
119
120 if astroid_errors:
121 comment += (
122 f'{astroid_errors} "astroid error(s)" were found. '
123 "Please open the GitHub Actions log to see what failed or crashed.\n\n"
124 )
125 if new_non_astroid_messages:
126 comment += (
127 "The following messages are now emitted:\n\n<details>\n\n"
128 + new_non_astroid_messages
129 + "\n</details>\n\n"
130 )
131
132 # Create comment for missing messages
133 count = 1
134 if missing_messages["messages"]:
135 comment += "The following messages are no longer emitted:\n\n<details>\n\n"
136 print("No longer emitted:")
137 for message in missing_messages["messages"]:
138 comment += f"{count}) {message['symbol']}:\n*{message['message']}*\n"
139 filepath = str(
140 PurePosixPath(message["path"]).relative_to(
141 self.packages[package].clone_directory
142 )
143 )
144 assert not self.packages[package].url.endswith(
145 ".git"
146 ), "You don't need the .git at the end of the github url."
147 comment += (
148 f"{self.packages[package].url}"
149 f"/blob/{new_messages['commit']}/{filepath}#L{message['line']}\n"
150 )
151 count += 1
152 print(message)
153 if missing_messages:
154 comment += "\n</details>\n\n"
155 return comment

157 def _truncate_comment(self, comment: str) -> str:
158 """GitHub allows only a set number of characters in a comment."""
159 hash_information = (
160 f"*This comment was generated for commit {self.config.commit}*"
161 )
162 if len(comment) + len(hash_information) >= MAX_GITHUB_COMMENT_LENGTH:
163 truncation_information = (
164 f"*This comment was truncated because GitHub allows only"
165 f" {MAX_GITHUB_COMMENT_LENGTH} characters in a comment.*"
166 )
167 max_len = (
168 MAX_GITHUB_COMMENT_LENGTH
169 - len(hash_information)
170 - len(truncation_information)
171 )
172 comment = f"{comment[:max_len - 10]}...\n\n{truncation_information}\n\n"
173 comment += hash_information
174 return comment
